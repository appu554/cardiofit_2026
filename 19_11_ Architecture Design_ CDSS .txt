Final Architecture Design: Complete Clinical Decision Support System
Executive Summary
This is the definitive architectural blueprint integrating all requirements into a production-grade clinical decision support system. The design centers on three foundational pillars:
1. Knowledge Authority: 7 specialized Knowledge Base microservices
2. Computational Substrate: Runtime Layer Platform (8 integrated components)
3. Clinical Orchestration: Medication Service and Safety Gateway

Component 1: Context Gateway / Snapshot Manager
Purpose: Create immutable, versioned clinical data snapshots
Architecture:
class ContextGateway {
  private snapshotStore: SnapshotRepository;
  private queryRouter: QueryRouter;
  private versionTracker: VersionTracker;
  
  async createSnapshot(
    recipe: WorkflowRecipe,
    patientId: string
  ): Promise<ClinicalSnapshot> {
    
    // 1. Generate snapshot ID
    const snapshotId = generateULID();
    
    // 2. Capture version vector
    const versionVector = await this.versionTracker.captureVersions();
    // {
    //   postgres_kb1: "txid_992847",
    //   postgres_kb4: "txid_992848",
    //   neo4j_patient: "txid_447821",
    //   neo4j_semantic: "kernel_v2.1.3",
    //   graphdb: "2025-01-15T14:00:00Z"
    // }
    
    // 3. Route queries to data sources
    const queryPlan = this.queryRouter.plan(recipe.requiredFields);
    
    // 4. Parallel data fetching
    const dataFetches = queryPlan.queries.map(q => 
      this.executeQuery(q, versionVector)
    );
    const results = await Promise.all(dataFetches);
    
    // 5. Validate freshness
    this.validateFreshness(results, recipe.freshnessRequirements);
    
    // 6. Assemble snapshot
    const snapshot: ClinicalSnapshot = {
      id: snapshotId,
      patientId,
      createdAt: new Date(),
      expiresAt: new Date(Date.now() + recipe.ttlSeconds * 1000),
      
      // Immutable data
      data: this.assembleData(results),
      
      // Version tracking
      versionVector,
      includedFields: recipe.requiredFields,
      
      // Integrity
      checksum: this.calculateChecksum(results),
      signature: await this.signSnapshot(snapshotId, results)
    };
    
    // 7. Store with TTL
    await this.snapshotStore.save(snapshot);
    
    return snapshot;
  }
  
  async getSnapshot(snapshotId: string): Promise<ClinicalSnapshot> {
    const snapshot = await this.snapshotStore.findById(snapshotId);
    
    if (!snapshot) {
      throw new SnapshotNotFoundError(snapshotId);
    }
    
    if (snapshot.expiresAt < new Date()) {
      throw new SnapshotExpiredError(snapshotId);
    }
    
    // Verify integrity
    const checksumValid = await this.verifyChecksum(snapshot);
    if (!checksumValid) {
      throw new SnapshotCorruptedError(snapshotId);
    }
    
    return snapshot;
  }
}

Architecture:
class ApolloFederationGateway {
  private subgraphs: Map<string, GraphQLService>;
  private cache: CacheLayer;
  private circuitBreaker: CircuitBreaker;
  
  constructor() {
    this.subgraphs = new Map([
      ['kb1_dosing', new GraphQLService('http://kb1:4001/graphql')],
      ['kb2_context', new GraphQLService('http://kb2:4002/graphql')],
      ['kb3_guidelines', new GraphQLService('http://kb3:4003/graphql')],
      ['kb4_safety', new GraphQLService('http://kb4:4004/graphql')],
      ['kb5_ddi', new GraphQLService('http://kb5:4005/graphql')],
      ['kb6_formulary', new GraphQLService('http://kb6:4006/graphql')],
      ['kb7_terminology', new GraphQLService('http://kb7:4007/graphql')]
    ]);
  }
  
  async query(gql: string, variables: any, snapshotId?: string): Promise<any> {
    // 1. Parse query to identify required subgraphs
    const queryPlan = this.planQuery(gql);
    
    // 2. Check cache (L3)
    const cacheKey = this.generateCacheKey(gql, variables);
    const cached = await this.cache.get(cacheKey);
    if (cached) {
      return cached;
    }
    
    // 3. Execute parallel queries with circuit breaker
    const subqueries = queryPlan.subqueries.map(sq => 
      this.circuitBreaker.execute(() => 
        this.subgraphs.get(sq.subgraph).query(sq.query, variables)
      )
    );
    
    const results = await Promise.all(subqueries);
    
    // 4. Merge results
    const merged = this.mergeResults(results, queryPlan);
    
    // 5. Cache result
    await this.cache.set(cacheKey, merged, 3600); // 1 hour TTL
    
    return merged;
  }
  
  async getVersions(): Promise<KBVersions> {
    const versions = {};
    
    for (const [name, service] of this.subgraphs.entries()) {
      const version = await service.query(`
        query { _version { version signature } }
      `);
      versions[name] = version._version.version;
    }
    
    return versions;
  }
}

Purpose: Event-driven intelligent cache warming
Architecture:
class CachePrefetcher {
  private kafkaConsumer: KafkaConsumer;
  private patternAnalyzer: PatternAnalyzer;
  private queryExecutor: QueryExecutor;
  private cacheWriter: CacheWriter;
  
  async start() {
    // Subscribe to workflow events
    this.kafkaConsumer.subscribe([
      'workflow-events',
      'medication-events',
      'safety-events'
    ]);
    
    this.kafkaConsumer.on('message', async (msg) => {
      await this.handleEvent(msg);
    });
  }
  
  private async handleEvent(event: WorkflowEvent) {
    switch (event.type) {
      case 'patient_chart_opened':
        await this.prefetchPatientContext(event.patientId);
        break;
        
      case 'recipe_determined':
        await this.prefetchRecipeData(event.protocolId, event.patientId);
        break;
        
      case 'prescription_created':
        await this.invalidateAndRefresh(event.patientId);
        break;
    }
  }
  
  private async prefetchRecipeData(
    protocolId: string,
    patientId: string
  ) {
    // 1. Analyze protocol to predict queries
    const predictions = await this.patternAnalyzer.predict(protocolId);
    
    // 2. Execute prefetch queries
    for (const prediction of predictions) {
      const data = await this.queryExecutor.execute(
        prediction.query,
        { patientId }
      );
      
      // 3. Write to appropriate cache layer
      await this.cacheWriter.write(
        prediction.cacheKey,
        data,
        prediction.layer,
        prediction.ttl
      );
    }
  }
}

4.2 Recipe Resolver
class RecipeResolver {
  private recipeDefinitions: Map<string, RecipeTemplate>;
  
  resolveWorkflowRecipe(
    protocolId: string,
    contextNeeds: ContextNeeds
  ): WorkflowRecipe {
    const template = this.recipeDefinitions.get(protocolId);
    
    if (!template) {
      throw new RecipeNotFoundError(protocolId);
    }
    
    // Merge field requirements
    const requiredFields = this.mergeRequirements([
      template.calculationFields,
      template.safetyFields,
      template.auditFields,
      this.resolveConditionalFields(contextNeeds, template)
    ]);
    
    return {
      recipeId: `${protocolId}_${Date.now()}`,
      protocolId,
      requiredFields,
      freshnessRequirements: template.freshnessRules,
      ttlSeconds: template.ttl || 300,
      allowLiveFetch: false,
      allowedLiveFields: []
    };
  }
}

class CandidateBuilder {
  async generate(
    therapyOptions: string[],  // ["CCB", "thiazide-like"]
    snapshotData: SnapshotData
  ): Promise<Candidate[]> {
    
    const candidates: Candidate[] = [];
    
    for (const drugClass of therapyOptions) {
      // Query Neo4j semantic_mesh for drugs in class
      const drugs = await this.queryDrugsByClass(drugClass);
      
      // Filter by patient-specific factors
      const suitable = drugs.filter(drug => 
        this.isSuitable(drug, snapshotData)
      );
      
      candidates.push(...suitable.map(drug => ({
        drug,
        drugClass,
        source: 'neo4j_semantic_mesh'
      })));
    }
    
    return candidates;
  }
  
  private isSuitable(drug: Drug, snapshot: SnapshotData): boolean {
    // Basic renal safety check
    if (snapshot.labs.egfr < 30 && !drug.renalSafe) {
      return false;
    }
    
    // Known allergy
    if (snapshot.allergies.some(a => a.allergen === drug.name)) {
      return false;
    }
    
    return true;
  }
}


4.4 Rust Dose Calculator
// rust-dose-engine/src/calculator.rs


use serde::{Deserialize, Serialize};


#[derive(Debug, Deserialize)]
pub struct DosingRule {
    pub base_dose: Dose,
    pub max_dose: Dose,
    pub renal_adjustment: Vec<RenalRule>,
    pub hepatic_adjustment: Option<Vec<HepaticRule>>,
}


#[derive(Debug, Deserialize, Serialize)]
pub struct Dose {
    pub value: f64,
    pub unit: String,  // UCUM format
}


#[derive(Debug, Deserialize)]
pub struct RenalRule {
    pub egfr_threshold: String,  // "<30", "30-45", ">45"
    pub dose: Dose,
}


#[derive(Debug, Deserialize)]
pub struct PatientParams {
    pub weight: f64,
    pub age: u8,
    pub egfr: f64,
    pub creatinine: f64,
}


pub fn calculate_dose(
    rule: &DosingRule,
    params: &PatientParams,
) -> Result<Dose, DoseCalculationError> {
    
    // Start with base dose
    let mut calculated_dose = rule.base_dose.clone();
    
    // Apply renal adjustment
    if let Some(adjustment) = find_renal_adjustment(&rule.renal_adjustment, params.egfr) {
        calculated_dose = adjustment.dose.clone();
    }
    
    // Validate unit consistency
    if calculated_dose.unit != rule.base_dose.unit {
        return Err(DoseCalculationError::UnitMismatch);
    }
    
    // Ensure not exceeding max dose
    if calculated_dose.value > rule.max_dose.value {
        calculated_dose.value = rule.max_dose.value;
    }
    
    Ok(calculated_dose)
}


fn find_renal_adjustment(
    rules: &[RenalRule],
    egfr: f64,
) -> Option<&RenalRule> {
    rules.iter().find(|rule| {
        match rule.egfr_threshold.as_str() {
            "<30" => egfr < 30.0,
            "30-45" => egfr >= 30.0 && egfr < 45.0,
            ">45" => egfr >= 45.0,
            _ => false,
        }
    })
}


4.5 Scoring Engine
class ScoringEngine {
  async rank(
    candidates: DosedCandidate[],
    snapshotData: SnapshotData,
    kbData: KBData
  ): Promise<ScoredProposal[]> {
    
    const scored = await Promise.all(
      candidates.map(c => this.scoreCandidate(c, snapshotData, kbData))
    );
    
    // Sort by overall score descending
    scored.sort((a, b) => b.scores.overall - a.scores.overall);
    
    return scored;
  }
  
  private async scoreCandidate(
    candidate: DosedCandidate,
    snapshot: SnapshotData,
    kbData: KBData
  ): Promise<ScoredProposal> {
    
    const scores = {
      efficacy: await this.scoreEfficacy(candidate, snapshot, kbData),
      safety: await this.scoreSafety(candidate, snapshot, kbData),
      cost: await this.scoreCost(candidate, kbData),
      adherence: await this.scoreAdherence(candidate),
      overall: 0
    };
    
    // Weighted average
    scores.overall = (
      scores.efficacy * 0.35 +
      scores.safety * 0.35 +
      scores.cost * 0.20 +
      scores.adherence * 0.10
    );
    
    return {
      ...candidate,
      scores,
      reasoning: this.explainScores(scores)
    };
  }
  
  private async scoreEfficacy(
    candidate: DosedCandidate,
    snapshot: SnapshotData,
    kbData: KBData
  ): Promise<number> {
    // Query ClickHouse for real-world effectiveness
    const analytics = await this.queryAnalytics(
      candidate.drug.rxnorm,
      snapshot.data.conditions
    );
    
    // Combine guideline evidence grade + real-world data
    const guidelineScore = this.evidenceGradeToScore(
      kbData.guidelines.evidenceGrade
    );
    
    return (guidelineScore * 0.6) + (analytics.effectiveness * 0.4);
  }
  
  private async scoreSafety(
    candidate: DosedCandidate,
    snapshot: SnapshotData,
    kbData: KBData
  ): Promise<number> {
    let safetyScore = 1.0;
    
    // Check for interactions
    const interactions = await this.checkInteractions(
      candidate.drug,
      snapshot.data.medications
    );
    
    for (const interaction of interactions) {
      switch (interaction.severity) {
        case 'major': safetyScore -= 0.3; break;
        case 'moderate': safetyScore -= 0.1; break;
        case 'minor': safetyScore -= 0.05; break;
      }
    }
    
    // Check contraindications (already filtered, but double-check)
    const contraindications = kbData.safety.contraindications;
    if (contraindications.absolute.length > 0) {
      return 0.0;  // Should never happen if filtering correct
    }
    
    return Math.max(0, safetyScore);
  }
  
  private async scoreCost(
    candidate: DosedCandidate,
    kbData: KBData
  ): Promise<number> {
    const formulary = kbData.formulary.find(
      f => f.rxnorm === candidate.drug.rxnorm
    );
    
    if (!formulary) return 0.5;  // Unknown cost
    
    // Lower tier = higher score
    switch (formulary.tier) {
      case 1: return 1.0;
      case 2: return 0.8;
      case 3: return 0.5;
      case 4: return 0.3;
      default: return 0.0;
    }
  }
}

// Scenario: KB-4 (Patient Safety) is down


try {
  const safetyRules = await apolloFederation.query(`
    query { kb4_safety { contraindications(...) } }
  `);
} catch (ServiceUnavailableError error) {
  
  // Circuit breaker opens
  circuitBreaker.open('kb4_safety');
  
  // CRITICAL: Cannot proceed without safety checks
  throw new ClinicalSafetyError(
    'Safety validation unavailable. Cannot approve prescription.',
    {
      cause: 'kb4_unavailable',
      action: 'block_workflow',
      fallback: 'manual_safety_review'
    }
  );
}

  attempts: [
