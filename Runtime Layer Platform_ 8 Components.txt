# Runtime Layer Platform: Deep Dive into 8 Components
## Overview
The Runtime Layer is **not a single service** but a **distributed platform** consisting of 8 specialized components that work together to provide the computational substrate for clinical services.
---
## Component 1: Neo4j Dual-Stream Architecture
### 1.1 Purpose & Rationale
**Why Two Databases?**
```
Challenge: Patient data changes every second, knowledge changes weekly
Problem: Mixed workloads cause cache thrashing and query performance degradation
Solution: Separate databases with different optimization strategies
```

### 1.4 Query Patterns
#### Patient Data Queries (Fast, Temporal)
```cypher
// Query 1: Get active medications for snapshot
MATCH (p:Patient {id: $patientId})-[t:TAKING]->(m:Medication)
WHERE m.status = 'active'
  AND t.started_at <= datetime()
  AND (m.end_date IS NULL OR m.end_date >= date())
RETURN m.rxnorm, m.name, m.dose, m.dose_unit, m.frequency, t.started_at
ORDER BY t.started_at DESC
// Query 2: Get recent lab results (freshness check)
MATCH (p:Patient {id: $patientId})-[:HAD_LAB]->(lab:LabResult)
WHERE lab.loinc IN $requiredLabs
  AND lab.result_date >= date() - duration({days: $maxAgeDays})
RETURN lab.loinc, lab.name, lab.value, lab.unit, lab.result_date
ORDER BY lab.result_date DESC
// Query 3: Get active conditions with severity
MATCH (p:Patient {id: $patientId})-[:HAS_CONDITION]->(c:Condition)
WHERE c.status = 'active'
RETURN c.icd10, c.snomed, c.name, c.severity, c.onset_date
// Query 4: Medication history (for adherence analysis)
MATCH (p:Patient {id: $patientId})-[:TAKING]->(current:Medication)
OPTIONAL MATCH (current)-[:PRECEDED_BY*1..5]->(previous:Medication)
RETURN current, collect(previous) AS history
```
#### Semantic Mesh Queries (Knowledge Traversal)
```cypher
// Query 1: Find CKD-safe drugs in therapeutic class
MATCH (dc:DrugClass {name: $drugClass})<-[:BELONGS_TO]-(d:Drug)
WHERE d.renal_safe = true
  AND d.kernel_version = $kernelVersion
  AND NOT EXISTS {
    MATCH (d)-[:CONTRAINDICATED_WITH]->(:Condition {icd10: 'N18.3'})
  }
RETURN d.rxnorm, d.name, d.evidence_strength
ORDER BY d.evidence_strength DESC
LIMIT 10
// Query 2: Check drug-drug interactions
MATCH (d1:Drug {rxnorm: $newDrug})
MATCH (d2:Drug)<-[:TAKING]-(p:Patient {id: $patientId})
WHERE EXISTS {
  (d1)-[:INTERACTS_WITH]->(d2)
}
MATCH (d1)-[i:INTERACTS_WITH]->(d2)
RETURN d1.name AS drug1, d2.name AS drug2,
       i.severity, i.mechanism, i.clinical_significance
// Query 3: Find therapeutic alternatives (cached paths)
MATCH (original:Drug {rxnorm: $originalDrug})
      -[cp:CACHED_PATH {path_type: 'therapeutic_alternative'}]->
      (alternative:Drug)
WHERE cp.computation_timestamp >= datetime() - duration({hours: 24})
  AND alternative.renal_safe = true
RETURN alternative.rxnorm, alternative.name,
       cp.cost_difference, cp.efficacy_score
ORDER BY cp.efficacy_score DESC
LIMIT 5
// Query 4: Class-level contraindications (ontology reasoning)
MATCH (d:Drug {rxnorm: $drugRxnorm})-[:BELONGS_TO]->(dc:DrugClass)
MATCH (dc)-[:CONTRAINDICATED_WITH_CLASS]->(cc:Condition)
WHERE cc.icd10 IN $patientConditions
RETURN dc.name AS drug_class, cc.name AS contraindication,
       cc.reason, cc.applies_to_all
```

```typescript
// Patient Data Ingestion (Real-time)
class PatientDataIngestion {
  async handlePatientEvent(event: PatientEvent) {
    const session = driver.session({ database: 'patient_data' });
    
    try {
      switch (event.type) {
        case 'MEDICATION_PRESCRIBED':
          await this.addMedication(session, event);
          break;
        case 'LAB_RESULTED':
          await this.addLabResult(session, event);
          break;
        case 'CONDITION_DOCUMENTED':
          await this.addCondition(session, event);
          break;
      }
    } finally {
      await session.close();
    }
  }
  
  private async addMedication(session: Session, event: MedicationEvent) {
    await session.run(`
      MATCH (p:Patient {id: $patientId})
      CREATE (p)-[:TAKING {
        started_at: datetime($startedAt),
        prescribed_by: $providerId,
        indication: $indication,
        adherence: 1.0
      }]->(m:Medication {
        id: $medicationId,
        rxnorm: $rxnorm,
        name: $name,
        dose: $dose,
        dose_unit: $doseUnit,
        frequency: $frequency,
        route: $route,
        start_date: date($startDate),
        status: 'active',
        recorded_at: datetime()
      })
    `, event.data);
  }
}
// Semantic Mesh Kernel Update (Batch)
class SemanticMeshSync {
  async updateKernel(kernelVersion: string, updates: KernelUpdate[]) {
    const session = driver.session({ database: 'semantic_mesh' });
    
    try {
      // Begin transaction
      const tx = session.beginTransaction();
      
      // Mark old kernel data
      await tx.run(`
        MATCH (n)
        WHERE n.kernel_version = $oldVersion
        SET n.kernel_version = $oldVersion + '_deprecated'
      `, { oldVersion: this.currentKernel });
      
      // Insert new kernel data
      for (const update of updates) {
        await this.applyUpdate(tx, update, kernelVersion);
      }
      
      // Update version pointer
      await tx.run(`
        MERGE (v:VersionPointer {name: 'current'})
        SET v.kernel_version = $newVersion,
            v.updated_at = datetime()
      `, { newVersion: kernelVersion });
      
      await tx.commit();
      
      // Update cache strategy
      await this.updateCacheHints(kernelVersion);
      
    } catch (error) {
      await tx.rollback();
      throw error;
    } finally {
      await session.close();
    }
  }
}
```

## Component 2: CDC Pipeline (Debezium)
### 2.1 Purpose
**Change Data Capture for Knowledge Base Synchronization**
```
Problem: 7 KBs in PostgreSQL change frequently, downstream needs to know
Solution: Stream changes via Debezium → Kafka → Consumers
Guarantee: Exactly-once delivery with transaction boundaries
```

### 2.5 Consumer Implementation (Adapter Service)
```typescript
class DebeziumConsumer {
  private kafka: Kafka;
  private consumer: Consumer;
  
  async start() {
    this.consumer = this.kafka.consumer({
      groupId: 'adapter-service',
      sessionTimeout: 30000,
      heartbeatInterval: 3000
    });
    
    await this.consumer.connect();
    
    await this.consumer.subscribe({
      topics: [
        'kb.changes.dosing_rules',
        'kb.changes.safety_rules',
        'kb.changes.interactions',
        'kb.changes.terminology'
      ],
      fromBeginning: false
    });
    
    await this.consumer.run({
      eachMessage: async ({ topic, partition, message }) => {
        await this.handleChangeEvent(topic, message);
      }
    });
  }
  
  private async handleChangeEvent(topic: string, message: KafkaMessage) {
    const event = JSON.parse(message.value.toString());
    const txId = event.payload.source.txId;
    
    // Idempotency check
    if (await this.isProcessed(txId)) {
      return;
    }
    
    try {
      switch (topic) {
        case 'kb.changes.dosing_rules':
          await this.handleDosingRuleChange(event);
          break;
        case 'kb.changes.safety_rules':
          await this.handleSafetyRuleChange(event);
          break;
        case 'kb.changes.interactions':
          await this.handleInteractionChange(event);
          break;
        case 'kb.changes.terminology':
          await this.handleTerminologyChange(event);
          break;
      }
      
      // Mark as processed
      await this.markProcessed(txId);
      
      // Commit offset
      await this.consumer.commitOffsets([{
        topic,
        partition,
        offset: (parseInt(message.offset) + 1).toString()
      }]);
      
    } catch (error) {
      this.logger.error('CDC processing failed', { txId, error });
      // Dead letter queue
      await this.sendToDLQ(topic, message, error);
    }
  }
}
```

### 2.6 Transaction Boundaries & Ordering
```typescript
class TransactionCoordinator {
  private pendingTransactions: Map<string, ChangeEvent[]> = new Map();
  
  async handleChangeEvent(event: ChangeEvent) {
    const txId = event.payload.transaction.id;
    
    if (!this.pendingTransactions.has(txId)) {
      this.pendingTransactions.set(txId, []);
    }
    
    this.pendingTransactions.get(txId).push(event);
    
    // Check if transaction is complete
    if (this.isTransactionComplete(event)) {
      await this.processTransaction(txId);
    }
  }
  
  private isTransactionComplete(event: ChangeEvent): boolean {
    const tx = event.payload.transaction;
    return tx.data_collection_order === tx.total_order;
  }
  
  private async processTransaction(txId: string) {
    const events = this.pendingTransactions.get(txId);
    
    // Sort by data_collection_order
    events.sort((a, b) => 
      a.payload.transaction.data_collection_order - 
      b.payload.transaction.data_collection_order
    );
    
    // Process atomically
    await this.adapterService.applyBatch(events);
    
    // Cleanup
    this.pendingTransactions.delete(txId);
  }
}
```

```typescript
    if (status === 'FAILED') {
      // Attempt automatic restart
      await this.restartConnector();
      
      // Wait for recovery
      await this.waitForRecovery(30000); // 30s timeout
      
      // If still failed, alert
      if (await this.getConnectorStatus() === 'FAILED') {
        await this.alertOncall('Debezium connector failed to recover');
      }
    }
  }
  
  async handleReplicationSlotLag() {
    const lag = await this.getReplicationSlotLag();
    
    if (lag > 100 * 1024 * 1024) { // 100MB
      // Critical: WAL accumulation
      // Option 1: Increase connector throughput
      await this.scaleConnectorWorkers(4);
      
      // Option 2: Drop and recreate slot (last resort)
      if (lag > 1024 * 1024 * 1024) { // 1GB
        await this.recreateReplicationSlot();
      }
    }
  }
}
```

```typescript
class InteractionTransformer {
  async transform(event: ChangeEvent): Promise<CypherStatement[]> {
    const interaction = event.payload.after;
    
    // 1. Query GraphDB for class-level inferences
    const classInteractions = await this.graphDB.query(`
      PREFIX rxnorm: <http://rxnorm.nlm.nih.gov/>
      PREFIX owl: <http://www.w3.org/2002/07/owl#>
      
      SELECT ?drug1Class ?drug2Class ?inferredSeverity
      WHERE {
        rxnorm:${interaction.drug1_rxnorm} rdf:type ?drug1Class .
        rxnorm:${interaction.drug2_rxnorm} rdf:type ?drug2Class .
        ?drug1Class :mayInteractWith ?drug2Class .
        ?drug1Class :interactionSeverity ?inferredSeverity .
      }
    `);
    
    // 2. Generate Cypher statements
    const statements: CypherStatement[] = [];
    
    // Direct drug-drug interaction
    statements.push({
      query: `
        MATCH (d1:Drug {rxnorm: $drug1Rxnorm})
        MATCH (d2:Drug {rxnorm: $drug2Rxnorm})
        MERGE (d1)-[i:INTERACTS_WITH]->(d2)
        SET i.severity = $severity,
            i.mechanism = $mechanism,
            i.clinical_significance = $clinicalSignificance,
            i.evidence_grade = $evidenceGrade,
            i.source = $source,
            i.kernel_version = $kernelVersion,
            i.bidirectional = true
        MERGE (d2)-[i2:INTERACTS_WITH]->(d1)
        SET i2 = properties(i)
      `,
      params: {
        drug1Rxnorm: interaction.drug1_rxnorm,
        drug2Rxnorm: interaction.drug2_rxnorm,
        severity: interaction.severity,
        mechanism: interaction.mechanism,
        clinicalSignificance: interaction.clinical_significance,
        evidenceGrade: interaction.evidence_grade,
        source: 'KB-5',
        kernelVersion: this.kernelVersion
      }
    });
    
    // Class-level interactions (inferred)
    for (const classInt of classInteractions) {
      statements.push({
        query: `
          MATCH (dc1:DrugClass {name: $class1})
          MATCH (dc2:DrugClass {name: $class2})
          MERGE (dc1)-[ci:CONTRAINDICATED_WITH_CLASS]->(dc2)
          SET ci.reason = $reason,
              ci.inferred_severity = $inferredSeverity,
              ci.applies_to_all = false,
              ci.kernel_version = $kernelVersion
        `,
        params: {
          class1: classInt.drug1Class,
          class2: classInt.drug2Class,
          reason: `Potential ${classInt.inferredSeverity} interaction`,
          inferredSeverity: classInt.inferredSeverity,
          kernelVersion: this.kernelVersion
        }
      });
    }
    
    return statements;
  }
}
```

```typescript
class GraphDBReasoner {
  private graphdb: GraphDBRepository;
  
  async reason(facts: RDFFact[]): Promise<InferredTriple[]> {
    // 1. Insert new facts into temporary graph
    const tempGraph = `http://temp/reasoning/${uuidv4()}`;
    
    await this.graphdb.insertTriples(tempGraph, facts);
    
    // 2. Trigger OWL reasoning
    const reasoning Result = await this.graphdb.query(`
      PREFIX owl: <http://www.w3.org/2002/07/owl#>
      PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
      
      SELECT ?subject ?predicate ?object ?confidence
      FROM <${tempGraph}>
      WHERE {
        # Subsumption reasoning
        ?subject rdf:type ?class .
        ?class rdfs:subClassOf ?superClass .
        ?superClass :hasProperty ?property .
        
        # Infer that subject has property
        BIND(?subject AS ?subject)
        BIND(?property AS ?predicate)
        BIND(?superClass AS ?object)
        
        # Calculate confidence based on evidence
        OPTIONAL {
          ?subject :hasEvidence ?evidence .
          ?evidence :confidenceScore ?confidence .
        }
      }
    `);
    
    // 3. Clean up temp graph
    await this.graphdb.deleteGraph(tempGraph);
    
    return reasoningResult.results.bindings.map(b => ({
      subject: b.subject.value,
      predicate: b.predicate.value,
      object: b.object.value,
      confidence: parseFloat(b.confidence?.value || '0.8')
    }));
  }
  
  async materializeSPARQLPaths(): Promise<CachedPath[]> {
    // Pre-compute expensive SPARQL property paths
    const paths = await this.graphdb.query(`
      PREFIX rxnorm: <http://rxnorm.nlm.nih.gov/>
      
      SELECT ?drug1 ?drug2 ?pathType ?distance
      WHERE {
        # Therapeutic alternatives (same class, different ingredient)
        ?drug1 :belongsToClass ?class .
        ?drug2 :belongsToClass ?class .
        FILTER(?drug1 != ?drug2)
        
        # Calculate "distance" (cost difference, efficacy difference)
        ?drug1 :cost ?cost1 .
        ?drug2 :cost ?cost2 .
        BIND(ABS(?cost1 - ?cost2) AS ?costDiff)
        
        ?drug1 :efficacy ?eff1 .
        ?drug2 :efficacy ?eff2 .
        BIND(ABS(?eff1 - ?eff2) AS ?effDiff)
        
        BIND(?costDiff + ?effDiff AS ?distance)
        BIND("therapeutic_alternative" AS ?pathType)
      }
      ORDER BY ?drug1 ?distance
      LIMIT 1000
    `);
    
    return paths.results.bindings.map(p => ({
      drug1: p.drug1.value,
      drug2: p.drug2.value,
      pathType: p.pathType.value,
      distance: parseFloat(p.distance.value)
    }));
  }
}
```

### 3.5 Kernel Versioning Strategy
```typescript
class KernelVersionManager {
  async generateKernel(updates: GraphUpdate[]): Promise<string> {
    // Semantic versioning based on change impact
    const currentVersion = await this.getCurrentKernelVersion();
    const [major, minor, patch] = currentVersion.split('.').map(Number);
    
    const impactAnalysis = this.analyzeImpact(updates);
    
    let newVersion: string;
    
    if (impactAnalysis.breaking) {
      // Major version: Breaking changes (e.g., drug contraindication added)
      newVersion = `${major + 1}.0.0`;
    } else if (impactAnalysis.significant) {
      // Minor version: New features (e.g., new drug added)
      newVersion = `${major}.${minor + 1}.0`;
    } else {
      // Patch version: Minor updates (e.g., evidence grade change)
      newVersion = `${major}.${minor}.${patch + 1}`;
    }
    
    // Append content hash for uniqueness
    const contentHash = this.calculateContentHash(updates);
    const fullVersion = `v${newVersion}+${contentHash.slice(0, 7)}`;
    
    return fullVersion;
  }
  
  private analyzeImpact(updates: GraphUpdate[]): ImpactAnalysis {
    let breaking = false;
    let significant = false;
    
    for (const update of updates) {
      if (update.type === 'CONTRAINDICATION_ADDED' ||
          update.type === 'SAFETY_RULE_CHANGED') {
        breaking = true;
        break;
      }
      
      if (update.type === 'DRUG_ADDED' ||
          update.type === 'INTERACTION_ADDED') {
        significant = true;
      }
    }
    
    return { breaking, significant };
  }
}
```

```typescript
class CacheInvalidator {
  async invalidate(updates: GraphUpdate[], kernelVersion: string) {
    const affectedQueries = this.identifyAffectedQueries(updates);
    
    for (const query of affectedQueries) {
      // Redis invalidation
      await this.redis.del(query.cacheKey);
      
      // Emit invalidation event
      await this.kafka.send({
        topic: 'cache.invalidation',
        key: query.cacheKey,
        value: JSON.stringify({
          reason: 'kernel_update',
          old_kernel: this.previousKernel,
          new_kernel: kernelVersion,
          affected_patterns: query.patterns,
          timestamp: new Date().toISOString()
        })
      });
    }
  }
  
  private identifyAffectedQueries(updates: GraphUpdate[]): AffectedQuery[] {
    const affected: AffectedQuery[] = [];
    
    for (const update of updates) {
      switch (update.entityType) {
        case 'Drug':
          affected.push({
            cacheKey: `drug:${update.entityId}:*`,
            patterns: ['drug_details', 'therapeutic_alternatives']
          });
          break;
        case 'Interaction':
          affected.push({
            cacheKey: `interactions:*:${update.drug1}:*`,
            patterns: ['drug_interactions']
          });
          affected.push({
            cacheKey: `interactions:*:${update.drug2}:*`,
            patterns: ['drug_interactions']
          });
          break;
        case 'Contraindication':
          affected.push({
            cacheKey: `contraindications:*`,
            patterns: ['safety_checks']
          });
          break;
      }
    }
    
    return affected;
  }
}
```

4.4 Semantic Enrichment Function
public class SemanticEnrichmentFunction 
    extends RichMapFunction<PrescriptionEvent, EnrichedPrescription> {
    
    private transient Neo4jDriver neo4jDriver;
    private transient LoadingCache<String, DrugContext> drugCache;
    
    @Override
    public void open(Configuration parameters) {
        // Initialize Neo4j connection
        neo4jDriver = GraphDatabase.driver(
            "neo4j://neo4j-semantic-mesh:7687",
            AuthTokens.basic("neo4j", getPassword())
        );
        
        // Initialize local cache (per task manager)
        drugCache = CacheBuilder.newBuilder()
            .maximumSize(10000)
            .expireAfterWrite(5, TimeUnit.MINUTES)
            .build(new CacheLoader<String, DrugContext>() {
                @Override
                public DrugContext load(String rxnorm) {
                    return fetchDrugContext(rxnorm);
                }
            });
    }
    
    @Override
    public EnrichedPrescription map(PrescriptionEvent event) throws Exception {
        // 1. Fetch drug context from Neo4j semantic_mesh
        DrugContext drugContext = drugCache.get(event.getMedication().getRxnorm());
        
        // 2. Query for therapeutic alternatives
        List<Alternative> alternatives = queryTherapeuticAlternatives(
            event.getMedication().getRxnorm()
        );
        
        // 3. Check for potential interactions
        List<Interaction> interactions = checkInteractions(
            event.getPatientId(),
            event.getMedication().getRxnorm()
        );
        
        // 4. Build enriched prescription
        return EnrichedPrescription.builder()
            .prescriptionId(event.getPrescriptionId())
            .patientId(event.getPatientId())
            .medication(event.getMedication())
            .drugContext(drugContext)
            .therapeuticAlternatives(alternatives)
            .potentialInteractions(interactions)
            .snapshotReference(event.getSnapshotReference())
            .evidenceEnvelope(event.getEvidenceEnvelope())
            .enrichedAt(System.currentTimeMillis())
            .build();
    }
    
    private DrugContext fetchDrugContext(String rxnorm) {
        try (Session session = neo4jDriver.session(
            SessionConfig.forDatabase("semantic_mesh")
        )) {
            return session.readTransaction(tx -> {
                Result result = tx.run(
                    "MATCH (d:Drug {rxnorm: $rxnorm})" +
                    "OPTIONAL MATCH (d)-[:BELONGS_TO]->(dc:DrugClass)" +
                    "OPTIONAL MATCH (dc)-[:IS_A]->(tc:TherapeuticCategory)" +
                    "RETURN d, dc, tc",
                    Map.of("rxnorm", rxnorm)
                );
                
                if (!result.hasNext()) {
                    return DrugContext.empty();
                }
                
                Record record = result.next();
                Node drug = record.get("d").asNode();
                Node drugClass = record.get("dc").asNode();
                Node therapeuticCategory = record.get("tc").asNode();
                
                return DrugContext.builder()
                    .rxnorm(rxnorm)
                    .name(drug.get("name").asString())
                    .drugClass(drugClass.get("name").asString())
                    .therapeuticCategory(therapeuticCategory.get("name").asString())
                    .mechanism(drugClass.get("mechanism").asString())
                    .renalSafe(drug.get("renal_safe").asBoolean())
                    .hepaticSafe(drug.get("hepatic_safe").asBoolean())
                    .evidenceStrength(drug.get("evidence_strength").asString())
                    .build();
            });
        }
    }
    
    private List<Alternative> queryTherapeuticAlternatives(String rxnorm) {
        try (Session session = neo4jDriver.session(
            SessionConfig.forDatabase("semantic_mesh")
        )) {
            return session.readTransaction(tx -> {
                Result result = tx.run(
                    "MATCH (original:Drug {rxnorm: $rxnorm})" +
                    "      -[cp:CACHED_PATH {path_type: 'therapeutic_alternative'}]->" +
                    "      (alt:Drug) " +
                    "WHERE cp.computation_timestamp >= datetime() - duration({hours: 24}) " +
                    "RETURN alt.rxnorm AS rxnorm, alt.name AS name, " +
                    "       cp.cost_difference AS costDiff, " +
                    "       cp.efficacy_score AS efficacyScore " +
                    "ORDER BY cp.efficacy_score DESC " +
                    "LIMIT 5",
                    Map.of("rxnorm", rxnorm)
                );
                
                List<Alternative> alternatives = new ArrayList<>();
                while (result.hasNext()) {
                    Record record = result.next();
                    alternatives.add(Alternative.builder()
                        .rxnorm(record.get("rxnorm").asString())
                        .name(record.get("name").asString())
                        .costDifference(record.get("costDiff").asDouble())
                        .efficacyScore(record.get("efficacyScore").asDouble())
                        .build());
                }
                return alternatives;
            });
        }
    }
    
    private List<Interaction> checkInteractions(String patientId, String newDrugRxnorm) {
        try (Session session = neo4jDriver.session(
            SessionConfig.forDatabase("patient_data")
        )) {
            return session.readTransaction(tx -> {
                // First, get patient's current medications
                Result currentMeds = tx.run(
                    "MATCH (p:Patient {id: $patientId})-[:TAKING]->(m:Medication) " +
                    "WHERE m.status = 'active' " +
                    "RETURN m.rxnorm AS rxnorm",
                    Map.of("patientId", patientId)
                );
                
                List<String> existingDrugs = new ArrayList<>();
                while (currentMeds.hasNext()) {
                    existingDrugs.add(currentMeds.next().get("rxnorm").asString());
                }
                
                if (existingDrugs.isEmpty()) {
                    return List.of();
                }
                
                // Switch to semantic_mesh for interaction check
                Session semanticSession = neo4jDriver.session(
                    SessionConfig.forDatabase("semantic_mesh")
                );
                
                Result interactions = semanticSession.run(
                    "MATCH (d1:Drug {rxnorm: $newDrug}) " +
                    "MATCH (d2:Drug) WHERE d2.rxnorm IN $existingDrugs " +
                    "MATCH (d1)-[i:INTERACTS_WITH]->(d2) " +
                    "RETURN d1.name AS drug1, d2.name AS drug2, " +
                    "       i.severity AS severity, i.mechanism AS mechanism, " +
                    "       i.clinical_significance AS significance",
                    Map.of(
                        "newDrug", newDrugRxnorm,
                        "existingDrugs", existingDrugs
                    )
                );
                
                List<Interaction> result = new ArrayList<>();
                while (interactions.hasNext()) {
                    Record record = interactions.next();
                    result.add(Interaction.builder()
                        .drug1(record.get("drug1").asString())
                        .drug2(record.get("drug2").asString())
                        .severity(record.get("severity").asString())
                        .mechanism(record.get("mechanism").asString())
                        .clinicalSignificance(record.get("significance").asString())
                        .build());
                }
                
                semanticSession.close();
                return result;
            });
        }
    }
    
    @Override
    public void close() {
        if (neo4jDriver != null) {
            neo4jDriver.close();
        }
    }
}

4.7 State Management & Windowing
public class PatientJourneyAggregation 
    extends KeyedProcessFunction<String, PrescriptionEvent, PatientSummary> {
    
    // Keyed state: Patient's medication history
    private ValueState<List<Prescription>> medicationHistory;
    
    // Keyed state: Risk score accumulator
    private ValueState<Double> riskScore;
    
    @Override
    public void open(Configuration parameters) {
        ValueStateDescriptor<List<Prescription>> medicationDescriptor = 
            new ValueStateDescriptor<>(
                "medication-history",
                TypeInformation.of(new TypeHint<List<Prescription>>() {})
            );
        medicationHistory = getRuntimeContext().getState(medicationDescriptor);
        
        ValueStateDescriptor<Double> riskDescriptor = 
            new ValueStateDescriptor<>(
                "risk-score",
                Types.DOUBLE
            );
        riskScore = getRuntimeContext().getState(riskDescriptor);
    }
    
    @Override
    public void processElement(
        PrescriptionEvent event,
        Context ctx,
        Collector<PatientSummary> out
    ) throws Exception {
        
        // Update medication history
        List<Prescription> history = medicationHistory.value();
        if (history == null) {
            history = new ArrayList<>();
        }
        history.add(event.getMedication());
        medicationHistory.update(history);
        
        // Calculate risk score
        double currentRisk = riskScore.value() != null ? riskScore.value() : 0.0;
        double newRisk = calculateRisk(history);
        riskScore.update(newRisk);
        
        // Register timer for 30-day window
        long thirtyDaysLater = ctx.timestamp() + TimeUnit.DAYS.toMillis(30);
        ctx.timerService().registerEventTimeTimer(thirtyDaysLater);
        
        // Emit summary if risk is high
        if (newRisk > 0.7) {
            out.collect(PatientSummary.builder()
                .patientId(event.getPatientId())
                .medicationCount(history.size())
                .riskScore(newRisk)
                .timestamp(ctx.timestamp())
                .build());
        }
    }
    
    @Override
    public void onTimer(
        long timestamp,
        OnTimerContext ctx,
        Collector<PatientSummary> out
    ) throws Exception {
        // Clean up old medications from history
        List<Prescription> history = medicationHistory.value();
        if (history != null) {
            long thirtyDaysAgo = timestamp - TimeUnit.DAYS.toMillis(30);
            history.removeIf(p -> p.getPrescribedAt() < thirtyDaysAgo);
            medicationHistory.update(history);
        }
    }
    
    private double calculateRisk(List<Prescription> history) {
        // Simple risk calculation based on:
        // - Number of medications
        // - Drug-drug interactions
        // - High-risk drug classes
        double risk = 0.0;
        
        // Factor 1: Polypharmacy
        risk += Math.min(history.size() * 0.1, 0.5);
        
        // Factor 2: High-risk classes
        long highRiskCount = history.stream()
            .filter(p -> isHighRiskClass(p.getDrugClass()))
            .count();
        risk += highRiskCount * 0.15;
        
        return Math.min(risk, 1.0);
    }
}

class PatternAnalyzer {

        }
      ],

    });
    
    // Pattern: Diabetes management
    this.patterns.set('diabetes_medication', {
      protocolId: 'diabetes_medication',
      predictedQueries: [
        {
          source: 'postgres_kb1',
          query: 'SELECT * FROM dosing_rules WHERE drug_class IN (?, ?, ?)',
          params: ['biguanide', 'GLP1_agonist', 'SGLT2_inhibitor'],
          layer: 'L3',
          ttl: 3600,
          priority: 'high'
        },
        {
          source: 'neo4j_semantic_mesh',
          query: `
            MATCH (d:Drug)-[:CONTRAINDICATED_WITH]->(c:Condition)
            WHERE c.icd10 = 'E11.9'
            RETURN d.rxnorm, d.name, c.reason
          `,
          params: {},
          layer: 'L2',
          ttl: 1800,
          priority: 'critical'
        }
      ],
      estimatedHitRate: 0.88
    });
  }
  
  async predict(protocolId: string): Promise<PrefetchPattern> {
    // First, check predefined patterns
    if (this.patterns.has(protocolId)) {
      return this.patterns.get(protocolId);
    }
    
    // Fallback: Analyze historical access patterns
    const history = await this.accessHistory.getProtocolHistory(protocolId);
    
    if (history.length < 10) {
      // Not enough data, return minimal pattern
      return this.getMinimalPattern(protocolId);
    }
    
    // Build pattern from history
    const queryFrequency = this.analyzeQueryFrequency(history);
    const predictedQueries = queryFrequency
      .filter(q => q.frequency > 0.5) // Queries used in >50% of cases
      .map(q => ({
        source: q.source,
        query: q.query,
        params: q.mostCommonParams,
        layer: this.selectLayer(q.frequency),
        ttl: this.calculateTTL(q.frequency),
        priority: q.frequency > 0.9 ? 'high' : 'medium'
      }));
    
    return {
      protocolId,
      predictedQueries,
      estimatedHitRate: this.calculateHitRate(history)
    };
  }
  
  private analyzeQueryFrequency(history: AccessHistory[]): QueryFrequency[] {
    const queryMap = new Map<string, QueryStats>();
    
    for (const access of history) {
      const key = `${access.source}:${access.queryTemplate}`;
      
      if (!queryMap.has(key)) {
        queryMap.set(key, {
          source: access.source,
          query: access.queryTemplate,
          count: 0,
          totalExecutions: 0,
          paramVariations: []
        });
      }
      
      const stats = queryMap.get(key);
      stats.count++;
      stats.totalExecutions = history.length;
      stats.paramVariations.push(access.params);
    }
    
    return Array.from(queryMap.values()).map(stats => ({
      source: stats.source,
      query: stats.query,
      frequency: stats.count / stats.totalExecutions,
      mostCommonParams: this.findMostCommonParams(stats.paramVariations)
    }));
  }
  
  private selectLayer(frequency: number): CacheLayer {
    if (frequency > 0.9) return 'L1'; // Very hot data
    if (frequency > 0.7) return 'L2'; // Hot data
    return 'L3'; // Warm data
  }
  
  private calculateTTL(frequency: number): number {
    if (frequency > 0.9) return 300;    // 5 minutes
    if (frequency > 0.7) return 1800;   // 30 minutes
    return 3600; // 1 hour
  }
}


5.5 Query Executor
class QueryExecutor {
  private neo4jDriver: Neo4jDriver;
  private postgresPool: Pool;
  private clickhouseClient: ClickHouseClient;
  private circuitBreaker: CircuitBreaker;
  
  async executeAll(queries: PrefetchQuery[]): Promise<PrefetchResult[]> {
    // Execute in parallel with priority ordering
    const priorityGroups = this.groupByPriority(queries);
    
    const results: PrefetchResult[] = [];
    
    // Execute critical queries first
    if (priorityGroups.critical) {
      const criticalResults = await Promise.all(
        priorityGroups.critical.map(q => this.executeQuery(q))
      );
      results.push(...criticalResults);
    }
    
    // Then high priority
    if (priorityGroups.high) {
      const highResults = await Promise.all(
        priorityGroups.high.map(q => this.executeQuery(q))
      );
      results.push(...highResults);
    }
    
    // Finally medium priority
    if (priorityGroups.medium) {
      const mediumResults = await Promise.all(
        priorityGroups.medium.map(q => this.executeQuery(q))
      );
      results.push(...mediumResults);
    }
    
    return results;
  }
  
  private async executeQuery(query: PrefetchQuery): Promise<PrefetchResult> {
    const startTime = Date.now();
    
    try {
      let data: any;
      
      switch (query.source) {
        case 'neo4j_semantic_mesh':
          data = await this.queryNeo4j('semantic_mesh', query);
          break;
        case 'neo4j_patient_data':
          data = await this.queryNeo4j('patient_data', query);
          break;
        case 'postgres_kb1':
        case 'postgres_kb4':
          data = await this.queryPostgres(query);
          break;
        case 'clickhouse':
          data = await this.queryClickHouse(query);
          break;
        default:
          throw new Error(`Unknown source: ${query.source}`);
      }
      
      const duration = Date.now() - startTime;
      
      return {
        query,
        data,
        success: true,
        duration,
        recordCount: Array.isArray(data) ? data.length : 1
      };
      
    } catch (error) {
      const duration = Date.now() - startTime;
      
      return {
        query,
        data: null,
        success: false,
        duration,
        error: error.message
      };
    }
  }
  
  private async queryNeo4j(database: string, query: PrefetchQuery): Promise<any> {
    return this.circuitBreaker.execute(async () => {
      const session = this.neo4jDriver.session({ database });
      
      try {
        const result = await session.run(query.query, query.params);
        return result.records.map(record => record.toObject());
      } finally {
        await session.close();
      }
    }, `neo4j_${database}`);
  }
  
  private async queryPostgres(query: PrefetchQuery): Promise<any> {
    return this.circuitBreaker.execute(async () => {
      const client = await this.postgresPool.connect();
      
      try {
        const result = await client.query(query.query, query.params);
        return result.rows;
      } finally {
        client.release();
      }
    }, 'postgres');
  }
  
  private async queryClickHouse(query: PrefetchQuery): Promise<any> {
    return this.circuitBreaker.execute(async () => {
      const resultSet = await this.clickhouseClient.query({
        query: query.query,
        query_params: query.params,
        format: 'JSONEachRow'
      });
      
      return await resultSet.json();
    }, 'clickhouse');
  }
}


5.6 Cache Writer
class CacheWriter {
  private redis: Redis;
  private compressionThreshold = 1024; // 1KB
  
  async write(
    cacheKey: string,
    data: any,
    layer: CacheLayer,
    ttl: number
  ): Promise<void> {
    // Serialize data
    const serialized = JSON.stringify(data);
    
    // Compress if large
    const payload = serialized.length > this.compressionThreshold
      ? await this.compress(serialized)
      : serialized;
    
    // Add metadata
    const cacheEntry = {
      data: payload,
      compressed: serialized.length > this.compressionThreshold,
      layer,
      cachedAt: Date.now(),
      size: payload.length
    };
    
    // Write to Redis with TTL
    await this.redis.setex(
      cacheKey,
      ttl,
      JSON.stringify(cacheEntry)
    );
    
    // Track metrics
    await this.recordCacheWrite(cacheKey, layer, payload.length);
  }
  
  async writeBatch(results: PrefetchResult[]): Promise<void> {
    const pipeline = this.redis.pipeline();
    
    for (const result of results) {
      if (!result.success) continue;
      
      const cacheKey = this.generateCacheKey(result.query);
      const serialized = JSON.stringify(result.data);
      const payload = serialized.length > this.compressionThreshold
        ? await this.compress(serialized)
        : serialized;
      
      const cacheEntry = {
        data: payload,
        compressed: serialized.length > this.compressionThreshold,
        layer: result.query.layer,
        cachedAt: Date.now(),
        size: payload.length
      };
      
      pipeline.setex(
        cacheKey,
        result.query.ttl,
        JSON.stringify(cacheEntry)
      );
    }
    
    await pipeline.exec();
  }
  
  private generateCacheKey(query: PrefetchQuery): string {
    const paramHash = this.hashParams(query.params);
    return `${query.source}:${query.layer}:${paramHash}`;
  }
  
  private hashParams(params: any): string {
    return crypto
      .createHash('sha256')
      .update(JSON.stringify(params))
      .digest('hex')
      .substring(0, 16);
  }
  
  private async compress(data: string): Promise<string> {
    return new Promise((resolve, reject) => {
      zlib.gzip(Buffer.from(data), (err, compressed) => {
        if (err) reject(err);
        else resolve(compressed.toString('base64'));
      });
    });
  }
}

6.1 Purpose
Intelligent routing of queries to optimal data sources based on patterns
Challenge: Multiple data sources (Neo4j, PostgreSQL, ClickHouse, Elasticsearch)
Solution: Pattern-based routing with fallback strategies

6.2 Architecture
class QueryRouter {
  private routingRules: Map<string, RoutingRule>;
  private sourceHealthMonitor: SourceHealthMonitor;
  private queryOptimizer: QueryOptimizer;
  
  constructor() {
    this.routingRules = new Map();
    this.loadRoutingRules();
  }
  
  private loadRoutingRules() {
    // Rule: Terminology lookups → PostgreSQL (authoritative)
    this.routingRules.set('terminology_lookup', {
      pattern: /^terminology\.lookup\./,
      preferredSource: 'postgres_kb7',
      fallbackSources: ['elasticsearch'],
      cacheStrategy: { layer: 'L3', ttl: 86400 },
      reasoning: 'KB-7 is authoritative for terminology'
    });
    
    // Rule: Full-text search → Elasticsearch
    this.routingRules.set('terminology_search', {
      pattern: /^terminology\.search\./,
      preferredSource: 'elasticsearch',
      fallbackSources: ['postgres_kb7'],
      cacheStrategy: { layer: 'L2', ttl: 3600 },
      reasoning: 'Elasticsearch optimized for text search'
    });
    
    // Rule: Patient graph queries → Neo4j patient_data
    this.routingRules.set('patient_graph', {
      pattern: /^patient\.(medications|conditions|encounters)\./,
      preferredSource: 'neo4j_patient_data',
      fallbackSources: [],
      cacheStrategy: { layer: 'L1', ttl: 60 },
      reasoning: 'Real-time patient data in Neo4j'
    });
    
    // Rule: Drug interactions → Neo4j semantic_mesh
    this.routingRules.set('drug_interactions', {
      pattern: /^drug\.interactions\./,
      preferredSource: 'neo4j_semantic_mesh',
      fallbackSources: ['postgres_kb5'],
      cacheStrategy: { layer: 'L2', ttl: 1800 },
      reasoning: 'Graph traversal for interaction detection'
    });
    
    // Rule: Analytics queries → ClickHouse
    this.routingRules.set('scoring_analytics', {
      pattern: /^analytics\./,
      preferredSource: 'clickhouse',
      fallbackSources: [],
      cacheStrategy: { layer: 'L3', ttl: 7200 },
      reasoning: 'OLAP queries best on ClickHouse'
    });
    
    // Rule: Complex reasoning → GraphDB
    this.routingRules.set('complex_reasoning', {
      pattern: /^reasoning\./,
      preferredSource: 'graphdb',
      fallbackSources: ['neo4j_semantic_mesh'],
      cacheStrategy: { layer: 'L2', ttl: 3600 },
      reasoning: 'OWL inference requires GraphDB'
    });
  }
  
  async route(queryRequest: QueryRequest): Promise<QueryResult> {
    // 1. Identify routing rule
    const rule = this.findMatchingRule(queryRequest.queryPattern);
    
    if (!rule) {
      throw new Error(`No routing rule for pattern: ${queryRequest.queryPattern}`);
    }
    
    // 2. Check cache first
    const cached = await this.checkCache(queryRequest, rule.cacheStrategy);
    if (cached) {
      return {
        data: cached.data,
        source: 'cache',
        duration: 0,
        cached: true
      };
    }
    
    // 3. Check source health
    const preferredHealthy = await this.sourceHealthMonitor.isHealthy(
      rule.preferredSource
    );
    
    // 4. Route to source (with fallback)
    const source = preferredHealthy
      ? rule.preferredSource
      : await this.selectFallbackSource(rule);
    
    // 5. Execute query
    const startTime = Date.now();
    const result = await this.executeQuery(source, queryRequest);
    const duration = Date.now() - startTime;
    
    // 6. Cache result
    await this.cacheResult(queryRequest, result, rule.cacheStrategy);
    
    return {
      data: result,
      source,
      duration,
      cached: false,
      rule: rule.pattern.source
    };
  }
  
  private findMatchingRule(pattern: string): RoutingRule | null {
    for (const [name, rule] of this.routingRules.entries()) {
      if (rule.pattern.test(pattern)) {
        return rule;
      }
    }
    return null;
  }
  
  private async selectFallbackSource(rule: RoutingRule): Promise<string> {
    for (const fallback of rule.fallbackSources) {
      const healthy = await this.sourceHealthMonitor.isHealthy(fallback);
      if (healthy) {
        return fallback;
      }
    }
    
    throw new Error(`All sources unavailable for rule: ${rule.pattern}`);
  }
}


6.3 Query Optimization
class QueryOptimizer {
  async optimize(query: Query, source: string): Promise<OptimizedQuery> {
    switch (source) {
      case 'neo4j_patient_data':
      case 'neo4j_semantic_mesh':
        return this.optimizeCypher(query);
      
      case 'postgres_kb1':
      case 'postgres_kb4':
        return this.optimizeSQL(query);
      
      case 'clickhouse':
        return this.optimizeClickHouse(query);
      
      default:
        return query;
    }
  }
  
  private optimizeCypher(query: Query): OptimizedQuery {
    // 1. Use index hints
    let optimized = query.text;
    
    if (query.text.includes('MATCH (p:Patient')) {
      optimized = optimized.replace(
        'MATCH (p:Patient',
        'MATCH (p:Patient) USING INDEX p:Patient(id)'
      );
    }
    
    // 2. Limit early in traversal
    if (!query.text.includes('LIMIT') && query.expectsMultipleResults) {
      optimized += ' LIMIT 1000';
    }
    
    // 3. Use relationship direction
    optimized = optimized.replace(
      /-\[(\w+)\]-/g,
      '-[$1]->'
    );
    
    return {
      ...query,
      text: optimized,
      optimizations: ['index_hint', 'limit_added', 'directed_relationships']
    };
  }
  
  private optimizeSQL(query: Query): OptimizedQuery {
    // 1. Add query hints
    let optimized = query.text;
    
    if (query.text.toUpperCase().startsWith('SELECT')) {
      optimized = optimized.replace(
        'SELECT',
        'SELECT /*+ INDEX(medications_rxnorm_idx) */'
      );
    }
    
    // 2. Ensure WHERE clause uses indexes
    if (!query.text.includes('WHERE')) {
      console.warn('Query without WHERE clause may be slow');
    }
    
    return {
      ...query,
      text: optimized,
      optimizations: ['query_hint_added']
    };
  }
}

7.1 Detailed Implementation
class SnapshotManager {
  private versionTracker: VersionTracker;
  private queryRouter: QueryRouter;
  private snapshotStore: SnapshotStore;
  private cryptoService: CryptoService;
  
  async createSnapshot(
    recipe: WorkflowRecipe,
    patientId: string
  ): Promise<ClinicalSnapshot> {
    
    const snapshotId = this.generateSnapshotId();
    const startTime = Date.now();
    
    try {
      // Step 1: Capture version vector (10ms)
      const versionVector = await this.versionTracker.captureVersions();
      
      // Step 2: Plan queries based on recipe (5ms)
      const queryPlan = await this.queryRouter.plan(recipe.requiredFields);
      
      // Step 3: Execute queries in parallel (60ms)
      const dataFetches = queryPlan.queries.map(async (q) => {
        return {
          field: q.field,
          data: await this.queryRouter.execute(q, versionVector)
        };
      });
      
      const results = await Promise.all(dataFetches);
      
      // Step 4: Validate freshness (5ms)
      await this.validateFreshness(results, recipe.freshnessRequirements);
      
      // Step 5: Assemble snapshot data (10ms)
      const snapshotData = this.assembleData(results);
      
      // Step 6: Calculate checksum (5ms)
      const checksum = this.cryptoService.calculateChecksum(snapshotData);
      
      // Step 7: Sign snapshot (5ms)
      const signature = await this.cryptoService.signSnapshot(
        snapshotId,
        checksum,
        versionVector
      );
      
      // Step 8: Create snapshot object
      const snapshot: ClinicalSnapshot = {
        id: snapshotId,
        patientId,
        createdAt: new Date(),
        expiresAt: new Date(Date.now() + recipe.ttlSeconds * 1000),
        
        data: snapshotData,
        
        versionVector,
        includedFields: recipe.requiredFields,
        freshnessValidation: this.buildFreshnessReport(results, recipe),
        
        checksum,
        signature,
        
        metadata: {
          recipeId: recipe.recipeId,
          creationDuration: Date.now() - startTime,
          dataSourceCount: queryPlan.queries.length
        }
      };
      
      // Step 9: Store snapshot (5ms)
      await this.snapshotStore.save(snapshot);
      
      // Step 10: Emit metrics
      this.emitMetrics(snapshot);
      
      return snapshot;
      
    } catch (error) {
      this.handleSnapshotCreationError(error, patientId, recipe);
      throw error;
    }
  }
  
  async getSnapshot(snapshotId: string): Promise<ClinicalSnapshot> {
    // Fetch from store
    const snapshot = await this.snapshotStore.findById(snapshotId);
    
    if (!snapshot) {
      throw new SnapshotNotFoundError(snapshotId);
    }
    
    // Check expiry
    if (snapshot.expiresAt < new Date()) {
      throw new SnapshotExpiredError(snapshotId, snapshot.expiresAt);
    }
    
    // Verify integrity
    const checksumValid = this.cryptoService.verifyChecksum(
      snapshot.data,
      snapshot.checksum
    );
    
    if (!checksumValid) {
      throw new SnapshotCorruptedError(snapshotId);
    }
    
    // Verify signature
    const signatureValid = await this.cryptoService.verifySignature(
      snapshot.id,
      snapshot.checksum,
      snapshot.versionVector,
      snapshot.signature
    );
    
    if (!signatureValid) {
      throw new SnapshotTamperedError(snapshotId);
    }
    
    return snapshot;
  }
  
  private assembleData(results: QueryResult[]): SnapshotData {
    const data: Partial<SnapshotData> = {};
    
    for (const result of results) {
      const [category, field] = result.field.split('.');
      
      if (!data[category]) {
        data[category] = {};
      }
      
      data[category][field] = result.data;
    }
    
    return data as SnapshotData;
  }
  
  private async validateFreshness(
    results: QueryResult[],
    requirements: FreshnessRequirements
  ): Promise<void> {
    const violations: FreshnessViolation[] = [];
    
    for (const [field, requirement] of Object.entries(requirements)) {
      const result = results.find(r => r.field === field);
      
      if (!result) {
        if (requirement.required) {
          violations.push({
            field,
            reason: 'missing_data',
            required: true
          });
        }
        continue;
      }
      
      if (requirement.maxAge) {
        const age = this.calculateDataAge(result.data);
        const maxAgeMs = this.parseMaxAge(requirement.maxAge);
        
        if (age > maxAgeMs) {
          violations.push({
            field,
            reason: 'stale_data',
            actualAge: age,
            maxAge: maxAgeMs,
            required: requirement.required
          });
        }
      }
    }
    
    if (violations.some(v => v.required)) {
      throw new FreshnessViolationError(violations);
    }
  }
  
  private generateSnapshotId(): string {
    return `snap_${ulid()}`;
  }
}

class VersionTracker {
  private neo4jDriver: Neo4jDriver;
  private postgresPool: Pool;
  private graphdbClient: GraphDBClient;
  private clickhouseClient: ClickHouseClient;
  
  async captureVersions(): Promise<VersionVector> {
    const [
      postgresVersions,
      neo4jVersions,
      graphdbVersion,
      clickhouseVersion
    ] = await Promise.all([
      this.capturePostgresVersions(),
      this.captureNeo4jVersions(),
      this.captureGraphDBVersion(),
      this.captureClickHouseVersion()
    ]);
    
    return {
      ...postgresVersions,
      ...neo4jVersions,
      graphdb: graphdbVersion,
      clickhouse: clickhouseVersion,
      capturedAt: new Date().toISOString()
    };
  }
  
  private async capturePostgresVersions(): Promise<Record<string, string>> {
    const client = await this.postgresPool.connect();
    
    try {
      const result = await client.query(`
        SELECT
          'postgres_kb1' as kb,
          pg_current_xact_id()::text as txid
        UNION ALL
        SELECT 'postgres_kb2', pg_current_xact_id()::text
        UNION ALL
        SELECT 'postgres_kb3', pg_current_xact_id()::text
        UNION ALL
        SELECT 'postgres_kb4', pg_current_xact_id()::text
        UNION ALL
        SELECT 'postgres_kb5', pg_current_xact_id()::text
        UNION ALL
        SELECT 'postgres_kb6', pg_current_xact_id()::text
        UNION ALL
        SELECT 'postgres_kb7', pg_current_xact_id()::text
      `);
      
      const versions: Record<string, string> = {};
      for (const row of result.rows) {
        versions[row.kb] = `txid_${row.txid}`;
      }
      
      return versions;
      
    } finally {
      client.release();
    }
  }
  
  private async captureNeo4jVersions(): Promise<Record<string, string>> {
    const patientSession = this.neo4jDriver.session({
      database: 'patient_data'
    });
    const semanticSession = this.neo4jDriver.session({
      database: 'semantic_mesh'
    });
    
    try {
      const [patientTx, semanticKernel] = await Promise.all([
        patientSession.run('CALL dbms.getTXMetaData()'),
        semanticSession.run(`
          MATCH (v:VersionPointer {name: 'current'})
          RETURN v.kernel_version AS version
        `)
      ]);
      
      return {
        neo4j_patient: `txid_${patientTx.records[0].get('txId')}`,
        neo4j_semantic: semanticKernel.records[0].get('version')
      };
      
    } finally {
      await patientSession.close();
      await semanticSession.close();
    }
  }
  
  private async captureGraphDBVersion(): Promise<string> {
    const result = await this.graphdbClient.query(`
      SELECT ?version WHERE {
        <http://system/kernel> <http://system/version> ?version
      }
    `);
    
    return result.results.bindings[0].version.value;
  }
  
  private async captureClickHouseVersion(): Promise<string> {
    const result = await this.clickhouseClient.query({
      query: 'SELECT now() as version',
      format: 'JSONEachRow'
    });
    
    const data = await result.json();
    return data[0].version;
  }
}

class EvidenceEnvelopeGenerator {
  async generate(
    proposal: MedicationProposal,
    snapshot: ClinicalSnapshot,
    kbVersions: KBVersions,
    calculationTrace: CalculationTrace
  ): Promise<EvidenceEnvelope> {
    
    return {
      // Core identifiers
      proposalId: proposal.proposalId,
      snapshotId: snapshot.id,
      
      // Knowledge versions
      knowledgeVersions: {
        kb1_dosing: kbVersions.kb1_dosing,
        kb2_context: kbVersions.kb2_context,
        kb3_guidelines: kbVersions.kb3_guidelines,
        kb4_safety: kbVersions.kb4_safety,
        kb5_ddi: kbVersions.kb5_ddi,
        kb6_formulary: kbVersions.kb6_formulary,
        kb7_terminology: kbVersions.kb7_terminology,
        neo4j_semantic_kernel: snapshot.versionVector.neo4j_semantic
      },
      
      // Protocol used
      protocolUsed: proposal.metadata.protocolId,
      recipeId: proposal.metadata.recipeId,
      
      // Calculation trace
      inferenceChain: calculationTrace.steps.map(step => step.description),
      decisionPoints: calculationTrace.decisions,
      
      // Confidence scores
      confidenceScores: {
        overall: this.calculateOverallConfidence(calculationTrace),
        dosing: calculationTrace.dosingConfidence,
        safety: calculationTrace.safetyConfidence,
        evidence: calculationTrace.evidenceStrength
      },
      
      // Evidence sources
      guidelineReferences: calculationTrace.guidelines,
      evidenceGrades: calculationTrace.evidenceGrades,
      
      // Timestamps
      proposalGeneratedAt: proposal.createdAt,
      snapshotCreatedAt: snapshot.createdAt,
      
      // Signatures for tamper-evidence
      proposalChecksum: this.calculateProposalChecksum(proposal),
      envelopeSignature: await this.signEnvelope(proposal, snapshot)
    };
  }
  
  private calculateOverallConfidence(trace: CalculationTrace): number {
    const weights = {
      dosing: 0.35,
      safety: 0.35,
      evidence: 0.30
    };
    
    return (
      trace.dosingConfidence * weights.dosing +
      trace.safetyConfidence * weights.safety +
      trace.evidenceStrength * weights.evidence
    );
  }
  
  private calculateProposalChecksum(proposal: MedicationProposal): string {
    const canonical = this.canonicalize(proposal);
    return crypto
      .createHash('sha256')
      .update(canonical)
      .digest('hex');
  }
  
  private async signEnvelope(
    proposal: MedicationProposal,
    snapshot: ClinicalSnapshot
  ): Promise<string> {
    const payload = {
      proposal_id: proposal.proposalId,
      snapshot_id: snapshot.id,
      snapshot_checksum: snapshot.checksum,
      proposal_checksum: this.calculateProposalChecksum(proposal)
    };
    
    // Sign with HSM-backed key
    const signature = await this.cryptoService.sign(
      JSON.stringify(payload),
      'evidence-envelope-key'
    );
    
    return signature;
  }
}
