## 2. EnrichedEvent Schema Design


### Complete EnrichedEvent Structure


```java
package com.ehr.flink.model;


import java.util.*;


/**
 * EnrichedEvent is the core data structure flowing from Module 2 downstream.
 * Contains original event + full patient context + risk indicators for CEP.
 */
public class EnrichedEvent {
    
    // Original event data
    private ClinicalEvent originalEvent;
    
    // Full patient context snapshot
    private PatientSnapshot patientContext;
    
    // Enrichment metadata
    private long enrichmentTimestamp;
    private String enrichmentSource; // "FLINK_STATE", "FHIR", "NEO4J"
    
    // IMMEDIATE ALERTS (Module 2 generates these)
    // Simple threshold breaches that don't require pattern matching
    private List<SimpleAlert> immediateAlerts;
    
    // RISK INDICATORS (For CEP pattern matching)
    // Structured markers that CEP uses to build complex patterns
    private RiskIndicators riskIndicators;
    
    // CLINICAL SCORES (Calculated in Module 2)
    private Map<String, Double> clinicalScores; // MEWS, qSOFA, NEWS2, etc.
    
    // SEMANTIC ENRICHMENT (Added in Module 3)
    private List<DrugInteraction> potentialInteractions;
    private List<String> applicableProtocols;
    private List<String> contraindications;
    
    // Getters/Setters...
}


/**
 * Risk Indicators structure for CEP consumption
 */
public class RiskIndicators {
    
    // Vital sign concerns
    private boolean tachycardia;           // HR > 100
    private boolean bradycardia;           // HR < 60
    private boolean hypotension;           // SBP < 90
    private boolean hypertension;          // SBP > 140
    private boolean fever;                 // Temp > 38.0°C
    private boolean hypothermia;           // Temp < 36.0°C
    private boolean tachypnea;             // RR > 20
    private boolean hypoxia;               // SpO2 < 92%
    
    // Lab abnormalities
    private boolean elevatedLactate;       // > 2.0 mmol/L
    private boolean elevatedCreatinine;    // AKI indicator
    private boolean leukocytosis;          // WBC > 12K
    private boolean leukopenia;            // WBC < 4K
    private boolean hyperglycemia;         // Glucose > 180
    private boolean hypoglycemia;          // Glucose < 70
    
    // Medication concerns
    private boolean onVasopressors;
    private boolean onAnticoagulation;
    private boolean onInsulin;
    private boolean onSedation;
    private boolean recentMedicationChange;
    
    // Clinical context
    private boolean inICU;
    private boolean postOperative;
    private boolean immunosuppressed;
    private boolean hasChronicKidneyDisease;
    private boolean hasDiabetes;
    private boolean onDialysis;
    
    // Trend indicators (computed from sliding windows)
    private TrendDirection heartRateTrend;   // INCREASING, DECREASING, STABLE
    private TrendDirection bloodPressureTrend;
    private TrendDirection creatinineTrend;
    
    // Timing indicators
    private Duration timeSinceAdmission;
    private Duration timeSinceLastVitals;
    private Duration timeSinceLastLabs;
    
    // Getters/Setters...
}


/**
 * Simple alerts generated directly from threshold checks
 */
public class SimpleAlert {
    private String alertType;              // "HIGH_HEART_RATE", "LOW_BLOOD_PRESSURE"
    private AlertSeverity severity;        // INFO, WARNING, HIGH, CRITICAL
    private String message;
    private Map<String, Object> context;   // Additional context data
    private long timestamp;
}
```

## 3. Module 2 Alert Generation Logic


### Immediate Alert Generation in PatientContextAssembler


```java
public class PatientContextAssembler extends KeyedProcessFunction<String, ClinicalEvent, EnrichedEvent> {
    
    @Override
    public void processElement(ClinicalEvent event, Context ctx, Collector<EnrichedEvent> out) 
            throws Exception {
        
        // Update patient state
        PatientSnapshot snapshot = updatePatientState(event);
        
        // Calculate clinical scores
        Map<String, Double> scores = calculateClinicalScores(snapshot);
        
        // Generate immediate alerts (threshold-based)
        List<SimpleAlert> immediateAlerts = generateImmediateAlerts(event, snapshot, scores);
        
        // Build risk indicators for CEP
        RiskIndicators riskIndicators = buildRiskIndicators(snapshot, scores);
        
        // Create enriched event
        EnrichedEvent enriched = EnrichedEvent.builder()
            .originalEvent(event)
            .patientContext(snapshot)
            .clinicalScores(scores)
            .immediateAlerts(immediateAlerts)
            .riskIndicators(riskIndicators)
            .enrichmentTimestamp(System.currentTimeMillis())
            .build();
        
        // Emit to downstream
        out.collect(enriched);
        
        // Side output: Critical immediate alerts
        for (SimpleAlert alert : immediateAlerts) {
            if (alert.getSeverity() == AlertSeverity.CRITICAL) {
ctx.output(CRITICAL_ALERT_TAG, alert);
            }
        }
    }
    
    /**
     * Generate immediate threshold-based alerts
     */
    private List<SimpleAlert> generateImmediateAlerts(ClinicalEvent event, 
                                                     PatientSnapshot snapshot,
                                                     Map<String, Double> scores) {
        
        List<SimpleAlert> alerts = new ArrayList<>();
        
        if (event.getEventType().equals("VITAL_SIGN")) {
            Map<String, Object> payload = event.getPayload();
            
            // Critical vital sign thresholds
            Integer heartRate = (Integer) payload.get("heart_rate");
            if (heartRate != null) {
                if (heartRate > 140) {
                    alerts.add(SimpleAlert.builder()
                        .alertType("CRITICAL_TACHYCARDIA")
                        .severity(AlertSeverity.CRITICAL)
                        .message("Heart rate critically elevated: " + heartRate + " bpm")
                        .context(Map.of("heart_rate", heartRate, "threshold", 140))
                        .timestamp(event.getEventTimestamp())
                        .build());
                } else if (heartRate > 100) {
                    alerts.add(SimpleAlert.builder()
                        .alertType("TACHYCARDIA")
                        .severity(AlertSeverity.HIGH)
                        .message("Heart rate elevated: " + heartRate + " bpm")
                        .context(Map.of("heart_rate", heartRate))
                        .timestamp(event.getEventTimestamp())
                        .build());
                }
            }
            
            Integer systolicBP = (Integer) payload.get("systolic_bp");
            if (systolicBP != null && systolicBP < 90) {
                alerts.add(SimpleAlert.builder()
                    .alertType("HYPOTENSION")
                    .severity(AlertSeverity.CRITICAL)
                    .message("Blood pressure critically low: " + systolicBP + " mmHg")
                    .context(Map.of("systolic_bp", systolicBP))
                    .timestamp(event.getEventTimestamp())
                    .build());
            }
        }
        
        // Clinical score alerts
        Double mews = scores.get("MEWS");
        if (mews != null && mews >= 5.0) {
            alerts.add(SimpleAlert.builder()
                .alertType("HIGH_MEWS_SCORE")
                .severity(AlertSeverity.HIGH)
                .message("MEWS score elevated: " + mews)
                .context(Map.of("mews", mews, "threshold", 5.0))
                .timestamp(System.currentTimeMillis())
                .build());
        }
        
        return alerts;
    }
    
    /**
     * Build structured risk indicators for CEP pattern matching
     */
    private RiskIndicators buildRiskIndicators(PatientSnapshot snapshot, 
                                              Map<String, Double> scores) {
        
        RiskIndicators indicators = new RiskIndicators();
        
        // Vital sign indicators
        VitalReading latestVitals = snapshot.getLatestVitals();
        if (latestVitals != null) {
            indicators.setTachycardia(latestVitals.getHeartRate() > 100);
            indicators.setBradycardia(latestVitals.getHeartRate() < 60);
            indicators.setHypotension(latestVitals.getSystolicBP() < 90);
            indicators.setHypertension(latestVitals.getSystolicBP() > 140);
            indicators.setFever(latestVitals.getTemperature() > 38.0);
            indicators.setHypoxia(latestVitals.getSpO2() < 92);
        }
        
        // Lab indicators
        Map<String, LabResult> recentLabs = snapshot.getRecentLabs();
        if (recentLabs.containsKey("lactate")) {
            double lactate = recentLabs.get("lactate").getValue();
            indicators.setElevatedLactate(lactate > 2.0);
        }
        if (recentLabs.containsKey("creatinine")) {
            double creatinine = recentLabs.get("creatinine").getValue();
            // Compare to baseline if available
            double baseline = snapshot.getBaselineCreatinine().orElse(1.0);
            indicators.setElevatedCreatinine(creatinine > (baseline * 1.5));
        }
        if (recentLabs.containsKey("wbc")) {
            double wbc = recentLabs.get("wbc").getValue();
            indicators.setLeukocytosis(wbc > 12.0);
            indicators.setLeukopenia(wbc < 4.0);
        }
        
        // Medication indicators
        Map<String, MedicationEntry> activeMeds = snapshot.getActiveMedications();
        indicators.setOnVasopressors(activeMeds.containsKey("norepinephrine") || 
                                     activeMeds.containsKey("vasopressin"));
        indicators.setOnAnticoagulation(activeMeds.containsKey("warfarin") || 
                                       activeMeds.containsKey("heparin"));
        indicators.setOnInsulin(activeMeds.containsKey("insulin"));
        
        // Check for recent medication changes (within 24 hours)
        indicators.setRecentMedicationChange(
            activeMeds.values().stream()
                .anyMatch(med -> med.getOrderedAt() > System.currentTimeMillis() - 86400000L)
        );
        
        // Clinical context
        indicators.setInICU(snapshot.getCurrentLocation().startsWith("ICU"));
        indicators.setHasDiabetes(snapshot.getActiveConditions().stream()
            .anyMatch(cond -> cond.startsWith("E11"))); // ICD-10 Type 2 Diabetes
        indicators.setHasChronicKidneyDisease(snapshot.getActiveConditions().stream()
            .anyMatch(cond -> cond.startsWith("N18")));
        
        // Trend indicators (computed from recent vitals history)
        indicators.setHeartRateTrend(calculateHeartRateTrend(snapshot.getRecentVitals()));
        indicators.setBloodPressureTrend(calculateBPTrend(snapshot.getRecentVitals()));
        
        // Timing indicators
        indicators.setTimeSinceAdmission(
            Duration.ofMillis(System.currentTimeMillis() - snapshot.getAdmissionTime())
        );
        
        return indicators;
    }
    
    private TrendDirection calculateHeartRateTrend(List<VitalReading> vitals) {
        if (vitals.size() < 3) return TrendDirection.STABLE;
        
        // Simple linear regression on last 6 readings
        List<VitalReading> recent = vitals.subList(Math.max(0, vitals.size() - 6), vitals.size());
        double slope = calculateSlope(recent.stream()
            .mapToDouble(VitalReading::getHeartRate)
            .toArray());
        
        if (slope > 2.0) return TrendDirection.INCREASING;
        if (slope < -2.0) return TrendDirection.DECREASING;
        return TrendDirection.STABLE;
    }
}
```

## 4. CEP Pattern Integration


### Sepsis Detection Pattern with Risk Indicators


```java
package com.ehr.flink.cep.patterns;


import org.apache.flink.cep.pattern.Pattern;
import org.apache.flink.cep.pattern.conditions.SimpleCondition;


public class SepsisDetectionPattern {
    
    /**
     * Advanced sepsis detection using risk indicators from Module 2
     * 
     * Pattern: SIRS Criteria + Organ Dysfunction
     * - Fever/Hypothermia + Tachycardia + (Tachypnea OR Leukocytosis)
     * - Followed by organ dysfunction indicator (lactate OR hypotension)
     * - Within 2-hour window
     */
    public static Pattern<EnrichedEvent, ?> createSepsisPattern() {
        
        return Pattern.<EnrichedEvent>begin("sirs_criteria")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    RiskIndicators risk = event.getRiskIndicators();
                    
                    // SIRS: 2+ of the following
                    int sirsCount = 0;
                    if (risk.isFever() || risk.isHypothermia()) sirsCount++;
                    if (risk.isTachycardia()) sirsCount++;
                    if (risk.isTachypnea()) sirsCount++;
                    if (risk.isLeukocytosis() || risk.isLeukopenia()) sirsCount++;
                    
                    return sirsCount >= 2;
                }
            })
            .next("organ_dysfunction")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    RiskIndicators risk = event.getRiskIndicators();
                    
                    // Organ dysfunction: lactate >2 OR hypotension
                    return risk.isElevatedLactate() || risk.isHypotension();
                }
            })
            .within(Time.hours(2));
    }
    
    /**
     * Septic shock pattern (more severe)
     */
    public static Pattern<EnrichedEvent, ?> createSepticShockPattern() {
        
        return Pattern.<EnrichedEvent>begin("sepsis_detected")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    return event.getRiskIndicators().isElevatedLactate();
                }
            })
            .next("hypotension_despite_fluids")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    RiskIndicators risk = event.getRiskIndicators();
                    return risk.isHypotension() && risk.isOnVasopressors();
                }
            })
            .within(Time.hours(1));
    }
}
```

### Acute Kidney Injury (AKI) Detection Pattern


```java
public class AKIDetectionPattern {
    
    /**
     * AKI Pattern using KDIGO criteria
     * 
     * Pattern: Creatinine rise + Risk factors
     * - Creatinine increase ≥0.3 mg/dL in 48 hours OR
     * - Creatinine ≥1.5x baseline
     * - Combined with: hypotension OR nephrotoxic meds OR sepsis
     */
    public static Pattern<EnrichedEvent, ?> createAKIPattern() {
        
        return Pattern.<EnrichedEvent>begin("baseline_creatinine")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    return event.getOriginalEvent().getEventType().equals("LAB_RESULT") &&
                           event.getOriginalEvent().getLabType().equals("creatinine");
                }
            })
            .followedBy("elevated_creatinine")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    if (!event.getOriginalEvent().getEventType().equals("LAB_RESULT")) {
                        return false;
                    }
                    
                    RiskIndicators risk = event.getRiskIndicators();
                    return risk.isElevatedCreatinine();
                }
            })
            .followedBy("risk_factor")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    RiskIndicators risk = event.getRiskIndicators();
                    
                    // AKI risk factors
                    return risk.isHypotension() ||
                           risk.isOnVasopressors() ||
                           hasNephrotoxicMeds(event.getPatientContext()) ||
                           risk.isElevatedLactate(); // Sepsis indicator
                }
            })
            .within(Time.hours(48));
    }
    
    private static boolean hasNephrotoxicMeds(PatientSnapshot snapshot) {
        Set<String> nephrotoxicDrugs = Set.of(
            "vancomycin", "gentamicin", "nsaid", "acei", "arb", "contrast"
        );
        
        return snapshot.getActiveMedications().keySet().stream()
            .anyMatch(med -> nephrotoxicDrugs.contains(med.toLowerCase()));
    }
}
```

### Medication Adherence Pattern


```java
public class MedicationAdherencePattern {
    
    /**
     * Detect missed medication doses
     * 
     * Pattern: Scheduled medication → ABSENCE of administration event
     * Enhanced with risk indicators for criticality
     */
    public static Pattern<EnrichedEvent, ?> createAdherencePattern() {
        
        return Pattern.<EnrichedEvent>begin("medication_scheduled")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    return event.getOriginalEvent().getEventType().equals("MEDICATION_ORDER");
                }
            })
            .notFollowedBy("medication_administered")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    ClinicalEvent original = event.getOriginalEvent();
                    return original.getEventType().equals("MEDICATION_ADMINISTERED") &&
                           medicationMatches(original);
                }
            })
            .within(Time.hours(6));
    }
    
    /**
     * Critical medication adherence (insulin, anticoagulation, etc.)
     */
    public static Pattern<EnrichedEvent, ?> createCriticalMedAdherencePattern() {
        
        return Pattern.<EnrichedEvent>begin("critical_med_scheduled")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    if (!event.getOriginalEvent().getEventType().equals("MEDICATION_ORDER")) {
                        return false;
                    }
                    
                    RiskIndicators risk = event.getRiskIndicators();
                    
                    // Critical medications requiring strict adherence
                    return risk.isOnInsulin() || 
                           risk.isOnAnticoagulation() ||
                           risk.isOnVasopressors();
                }
            })
            .notFollowedBy("administration")
            .where(new SimpleCondition<EnrichedEvent>() {
                @Override
                public boolean filter(EnrichedEvent event) {
                    return event.getOriginalEvent().getEventType().equals("MEDICATION_ADMINISTERED");
                }
            })
            .within(Time.hours(2)); // Shorter window for critical meds
    }
}
```

## 5. CEP Pattern Stream Processing


### Integration of Module 2 Output with CEP


```java
public class ClinicalPatternProcessor {
    
    public static void main(String[] args) throws Exception {
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();
        
        // Input: Enriched events from Module 2
        DataStream<EnrichedEvent> enrichedStream = env
            .fromSource(kafkaSource("ehr-events-enriched"), 
                       watermarkStrategy, 
                       "Enriched Events");
        
        // Key by patient for pattern detection
        KeyedStream<EnrichedEvent, String> keyedByPatient = enrichedStream
            .keyBy(event -> event.getOriginalEvent().getPatientId());
        
        // Pattern 1: Sepsis Detection
        PatternStream<EnrichedEvent> sepsisPatterns = CEP.pattern(
            keyedByPatient,
            SepsisDetectionPattern.createSepsisPattern()
        );
        
        DataStream<ClinicalAlert> sepsisAlerts = sepsisPatterns.select(
            (Map<String, List<EnrichedEvent>> match) -> {
                List<EnrichedEvent> organDysfunction = match.get("organ_dysfunction");
                EnrichedEvent lastEvent = organDysfunction.get(organDysfunction.size() - 1);
                
                return ClinicalAlert.builder()
                    .alertType("SEPSIS_SUSPECTED")
                    .patientId(lastEvent.getOriginalEvent().getPatientId())
                    .severity(AlertSeverity.CRITICAL)
                    .confidence(0.85)
                    .evidence(buildSepsisEvidence(match))
                    .recommendedActions(List.of(
                        "Initiate sepsis bundle",
                        "Blood cultures before antibiotics",
                        "Measure serum lactate",
                        "Administer broad-spectrum antibiotics within 1 hour"
                    ))
                    .timestamp(System.currentTimeMillis())
                    .build();
            }
        );
        
        // Pattern 2: AKI Detection
        PatternStream<EnrichedEvent> akiPatterns = CEP.pattern(
            keyedByPatient,
            AKIDetectionPattern.createAKIPattern()
        );
        
        DataStream<ClinicalAlert> akiAlerts = akiPatterns.select(
            (Map<String, List<EnrichedEvent>> match) -> {
                return ClinicalAlert.builder()
                    .alertType("AKI_DETECTED")
                    .patientId(getPatientId(match))
                    .severity(AlertSeverity.HIGH)
                    .confidence(0.78)
                    .evidence(buildAKIEvidence(match))
                    .recommendedActions(List.of(
                        "Review nephrotoxic medications",
                        "Adjust drug doses for renal function",
                        "Monitor fluid status",
                        "Consider nephrology consult if stage 2+"
                    ))
                    .timestamp(System.currentTimeMillis())
                    .build();
            }
        );
        
        // Pattern 3: Medication Adherence
        PatternStream<EnrichedEvent> adherencePatterns = CEP.pattern(
            keyedByPatient,
            MedicationAdherencePattern.createCriticalMedAdherencePattern()
        );
        
        DataStream<ClinicalAlert> adherenceAlerts = adherencePatterns.select(
            (Map<String, List<EnrichedEvent>> match) -> {
                EnrichedEvent scheduledMed = match.get("critical_med_scheduled").get(0);
                
                return ClinicalAlert.builder()
                    .alertType("MEDICATION_NOT_ADMINISTERED")
                    .patientId(scheduledMed.getOriginalEvent().getPatientId())
                    .severity(AlertSeverity.HIGH)
                    .confidence(1.0)
                    .evidence(Map.of(
                        "medication", scheduledMed.getOriginalEvent().getPayload().get("drug_name"),
                        "scheduled_time", scheduledMed.getOriginalEvent().getEventTimestamp(),
                        "hours_overdue", calculateHoursOverdue(scheduledMed)
                    ))
                    .recommendedActions(List.of(
                        "Verify medication administration",
                        "Check for patient refusal or contraindication",
                        "Administer dose if still indicated"
                    ))
                    .timestamp(System.currentTimeMillis())
                    .build();
            }
        );
        
        // Union all CEP-generated alerts
        DataStream<ClinicalAlert> allCEPAlerts = sepsisAlerts
            .union(akiAlerts)
            .union(adherenceAlerts);
        
        // Send to Module 6 (Composer) for deduplication and routing
        allCEPAlerts.sinkTo(kafkaSink("ehr-cep-alerts"));
        
        env.execute("Clinical Pattern Processor");
    }
}
```

## 6. Alert Composition & Deduplication (Module 6)


### Merging Simple Alerts + CEP Alerts + ML Scores


```java
public class AlertComposer extends CoProcessFunction<SimpleAlert, ClinicalAlert, ComposedAlert> {
    
    // State for deduplication
    private MapState<String, AlertHistory> alertHistoryState;
    
    // Suppression window configuration
    private static final long SUPPRESSION_WINDOW_MS = 30 * 60 * 1000L; // 30 minutes
    
    @Override
    public void open(Configuration parameters) {
        MapStateDescriptor<String, AlertHistory> descriptor = 
            new MapStateDescriptor<>("alert-history", String.class, AlertHistory.class);
        alertHistoryState = getRuntimeContext().getMapState(descriptor);
    }
    
    @Override
    public void processElement1(SimpleAlert simpleAlert, Context ctx, Collector<ComposedAlert> out) 
            throws Exception {
        
        // Check if similar alert exists within suppression window
        String alertKey = simpleAlert.getAlertType();
        AlertHistory history = alertHistoryState.get(alertKey);
        
        if (history != null && 
            (System.currentTimeMillis() - history.getLastFiredTime()) < SUPPRESSION_WINDOW_MS) {
            // Suppress duplicate
            history.incrementSuppressedCount();
            alertHistoryState.put(alertKey, history);
            return;
        }
        
        // New alert or outside suppression window
        ComposedAlert composed = ComposedAlert.builder()
            .alertId(UUID.randomUUID().toString())
            .alertType(simpleAlert.getAlertType())
            .severity(simpleAlert.getSeverity())
            .message(simpleAlert.getMessage())
            .sources(List.of("MODULE_2_THRESHOLD"))
            .confidence(1.0) // Threshold alerts have 100% confidence
            .evidence(simpleAlert.getContext())
            .timestamp(simpleAlert.getTimestamp())
            .build();
        
        out.collect(composed);
        
        // Update history
        alertHistoryState.put(alertKey, new AlertHistory(System.currentTimeMillis(), 0));
    }
    
    @Override
    public void processElement2(ClinicalAlert cepAlert, Context ctx, Collector<ComposedAlert> out) 
            throws Exception {
        
        String alertKey = cepAlert.getAlertType() + ":" + cepAlert.getPatientId();
        AlertHistory history = alertHistoryState.get(alertKey);
        
        if (history != null && 
            (System.currentTimeMillis() - history.getLastFiredTime()) < SUPPRESSION_WINDOW_MS) {
            history.incrementSuppressedCount();
            alertHistoryState.put(alertKey, history);
            return;
        }
        
        // CEP alert - higher priority, include provenance
        ComposedAlert composed = ComposedAlert.builder()
            .alertId(UUID.randomUUID().toString())
            .alertType(cepAlert.getAlertType())
            .severity(cepAlert.getSeverity())
            .message(buildCEPMessage(cepAlert))
            .sources(List.of("CEP_PATTERN_MATCH"))
            .confidence(cepAlert.getConfidence())
            .evidence(cepAlert.getEvidence())
            .recommendedActions(cepAlert.getRecommendedActions())
            .timestamp(cepAlert.getTimestamp())
            .build();
        
        out.collect(composed);
        
        alertHistoryState.put(alertKey, new AlertHistory(System.currentTimeMillis(), 0));
```
