    @Test
    @DisplayName("E2E: Complete sepsis detection flow from raw events to critical alerts")
    void completeSepsisDetectionFlow() throws Exception {
        // Arrange - produce raw clinical events to Kafka
        String patientId = "e2e-patient-001";
        
        List<String> rawEvents = Arrays.asList(
            createRawEventJson(patientId, "VITAL_SIGN", Map.of("heart_rate", 110, "systolic_bp", 95)),
            createRawEventJson(patientId, "VITAL_SIGN", Map.of("heart_rate", 120, "systolic_bp", 85)),
            createRawEventJson(patientId, "LAB_RESULT", Map.of("lab_type", "lactate", "value", 2.5))
        );
        
        // Produce events
        for (String event : rawEvents) {
            producer.send(new ProducerRecord<>("ehr-events-raw", patientId, event)).get();
        }
        producer.flush();
        
        // Act - start Flink job
        Thread jobThread = new Thread(() -> {
            try {
                EHRIntelligenceEngineJob.main(new String[]{
                    "--kafka.bootstrap.servers", kafka.getBootstrapServers(),
                    "--checkpoint.interval", "1000"
                });
            } catch (Exception e) {
                throw new RuntimeException(e);
            }
        });
        jobThread.start();
        
        // Assert - consume critical alerts from output topic
        await().atMost(30, TimeUnit.SECONDS).until(() -> {
            consumer.poll(Duration.ofMillis(100));
            return hasAlert(consumer, "SEPSIS_SUSPECTED");
        });
        
        ConsumerRecord<String, String> alertRecord = getLastAlert(consumer);
        assertThat(alertRecord).isNotNull();
        assertThat(alertRecord.value()).contains("SEPSIS_SUSPECTED");
        assertThat(alertRecord.value()).contains("CRITICAL");
        
        jobThread.interrupt();
    }

    @Test
    @DisplayName("E2E: State recovery from checkpoint after restart")
    void stateRecoveryFromCheckpoint() throws Exception {
        // Arrange - send events to build patient state
        String patientId = "e2e-patient-003";
        
        String admission = createRawEventJson(patientId, "ADMISSION", 
            Map.of("location", "ICU-1"));
        String medication = createRawEventJson(patientId, "MEDICATION_ORDER",
            Map.of("drug_code", "insulin", "drug_name", "Insulin"));
        
        producer.send(new ProducerRecord<>("ehr-events-raw", patientId, admission)).get();
        producer.send(new ProducerRecord<>("ehr-events-raw", patientId, medication)).get();
        producer.flush();
        
        // Wait for checkpoint
        Thread.sleep(2000);
        
        // Act - simulate job restart (in real scenario, would cancel and restart from checkpoint)
        // For test purposes, send another event and verify state is maintained
        
        String vitals = createRawEventJson(patientId, "VITAL_SIGN",
            Map.of("heart_rate", 85, "systolic_bp", 120));
        producer.send(new ProducerRecord<>("ehr-events-raw", patientId, vitals)).get();
        producer.flush();
        
        // Assert - enriched event should contain previously accumulated state
        await().atMost(15, TimeUnit.SECONDS).until(() -> {
            consumer.poll(Duration.ofMillis(100));
            return hasEnrichedEvent(consumer, patientId);
        });
        
        ConsumerRecord<String, String> enriched = getLastEnrichedEvent(consumer);
        assertThat(enriched.value()).contains("ICU-1");
        assertThat(enriched.value()).contains("insulin");
    }
    
    @Test
    @DisplayName("E2E: High volume stress test with backpressure handling")
    void highVolumeStressTest() throws Exception {
        // Arrange - generate 10K events across 100 patients
        int numPatients = 100;
        int eventsPerPatient = 100;
        
        ExecutorService executorService = Executors.newFixedThreadPool(10);
        List<Future<?>> futures = new ArrayList<>();
        
        // Act - produce events in parallel
        for (int patientIdx = 0; patientIdx < numPatients; patientIdx++) {
            final String patientId = "stress-patient-" + patientIdx;
            
            Future<?> future = executorService.submit(() -> {
                try {
                    for (int eventIdx = 0; eventIdx < eventsPerPatient; eventIdx++) {
                        String event = createRawEventJson(patientId, "VITAL_SIGN",
                            Map.of("heart_rate", 70 + eventIdx % 40, "systolic_bp", 120));
                        producer.send(new ProducerRecord<>("ehr-events-raw", patientId, event));
                    }
                } catch (Exception e) {
                    throw new RuntimeException(e);
                }
            });
            futures.add(future);
        }
        
        // Wait for all producers to complete
        for (Future<?> future : futures) {
            future.get(30, TimeUnit.SECONDS);
        }
        producer.flush();
        executorService.shutdown();
        
        // Assert - verify processing completed without data loss
        await().atMost(60, TimeUnit.SECONDS).until(() -> {
            return getKafkaLag() < 100; // Acceptable lag threshold
        });
        
        assertThat(getKafkaLag()).isLessThan(100);
    }
    
    @Test
    @DisplayName("E2E: Late event handling and watermark progression")
    void lateEventHandling() throws Exception {
        // Arrange - send events out of order
        String patientId = "e2e-patient-004";
        long baseTime = System.currentTimeMillis();
        
        // Send event at T+3
        String event3 = createRawEventJsonWithTimestamp(patientId, "VITAL_SIGN", 
            baseTime + 3000, Map.of("heart_rate", 90));
        producer.send(new ProducerRecord<>("ehr-events-raw", patientId, event3)).get();
        
        // Send event at T+1 (late)
        String event1 = createRawEventJsonWithTimestamp(patientId, "VITAL_SIGN",
            baseTime + 1000, Map.of("heart_rate", 80));
        producer.send(new ProducerRecord<>("ehr-events-raw", patientId, event1)).get();
        
        // Send event at T+2 (late but within allowed lateness)
        String event2 = createRawEventJsonWithTimestamp(patientId, "VITAL_SIGN",
            baseTime + 2000, Map.of("heart_rate", 85));
        producer.send(new ProducerRecord<>("ehr-events-raw", patientId, event2)).get();
        
        producer.flush();
        
        // Assert - all events processed despite being out of order
        await().atMost(20, TimeUnit.SECONDS).until(() -> {
            return countProcessedEvents(consumer, patientId) >= 3;
        });
    }

        consumerProps.put(ConsumerConfig.BOOTSTRAP_SERVERS_CONFIG, kafka.getBootstrapServers());
        consumerProps.put(ConsumerConfig.GROUP_ID_CONFIG, "test-consumer-" + UUID.randomUUID());

            "org.apache.kafka.common.serialization.StringDeserializer");

        consumer = new KafkaConsumer<>(consumerProps);
        consumer.subscribe(Arrays.asList("ehr-alerts-critical", "ehr-events-enriched"));
    }
    
    private String createRawEventJson(String patientId, String eventType, Map<String, Object> payload) {
        return String.format(
            "{\"patientId\":\"%s\",\"eventType\":\"%s\",\"timestamp\":%d,\"payload\":%s}",
            patientId, eventType, System.currentTimeMillis(), toJson(payload)
        );
    }
    
    private String createRawEventJsonWithTimestamp(String patientId, String eventType, 
                                                   long timestamp, Map<String, Object> payload) {
        return String.format(
            "{\"patientId\":\"%s\",\"eventType\":\"%s\",\"timestamp\":%d,\"payload\":%s}",
            patientId, eventType, timestamp, toJson(payload)
        );
    }
    
    private String createSemanticMeshUpdateJson(String drug1, String drug2, 
                                               String severity, String mechanism) {
        return String.format(
            "{\"updateType\":\"DRUG_INTERACTION_ADD\",\"drug1\":\"%s\",\"drug2\":\"%s\"," +
            "\"severity\":\"%s\",\"mechanism\":\"%s\"}",
            drug1, drug2, severity, mechanism
        );
    }
    
    private String toJson(Map<String, Object> map) {
        StringBuilder json = new StringBuilder("{");
        map.forEach((k, v) -> {
            json.append("\"").append(k).append("\":");
            if (v instanceof String) {
                json.append("\"").append(v).append("\"");
            } else {
                json.append(v);
            }
            json.append(",");
        });
        if (json.length() > 1) json.setLength(json.length() - 1);
        json.append("}");
        return json.toString();
    }

    @Test
    @DisplayName("Sepsis deterioration pattern detected across multiple event types")
    void sepsisPatternDetection() throws Exception {
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();
        env.setParallelism(1);
        
        // Arrange - create event sequence that triggers sepsis pattern
        List<ClinicalEvent> events = Arrays.asList(
            ClinicalEventBuilder.vitals("patient-001", 1000L, 110, 95),  // High HR, low BP
            ClinicalEventBuilder.vitals("patient-001", 1500L, 120, 85),  // Worsening
            ClinicalEventBuilder.labResult("patient-001", 2000L, "lactate", 2.5) // Elevated lactate
        );
        
        DataStream<ClinicalEvent> stream = env.fromCollection(events)
            .assignTimestampsAndWatermarks(new AssignerWithPunctuatedWatermarks<ClinicalEvent>() {
                @Override
                public Watermark checkAndGetNextWatermark(ClinicalEvent event, long extractedTimestamp) {
                    return new Watermark(extractedTimestamp);
                }
                
                @Override
                public long extractTimestamp(ClinicalEvent event, long previousTimestamp) {
                    return event.getEventTimestamp();
                }
            });
        
        // Apply sepsis detection pattern
        Pattern<ClinicalEvent, ?> sepsisPattern = SepsisDetectionPattern.create();
        PatternStream<ClinicalEvent> patternStream = CEP.pattern(
            stream.keyBy(ClinicalEvent::getPatientId), 
            sepsisPattern
        );
        
        DataStream<ClinicalAlert> alerts = patternStream.select(match -> {
            return ClinicalAlert.builder()
                .alertType("SEPSIS_SUSPECTED")
                .patientId(match.get("vitals").get(0).getPatientId())
                .severity(AlertSeverity.CRITICAL)
                .timestamp(System.currentTimeMillis())
                .build();
        });
        
        alerts.addSink(new TestSink<>());
        
        // Act
        env.execute("Sepsis Detection Test");
        
        // Assert
        await().atMost(5, TimeUnit.SECONDS).until(() -> !TestSink.getValues().isEmpty());
        
        List<ClinicalAlert> results = TestSink.getValues();
        assertThat(results).hasSize(1);
        assertThat(results.get(0).getAlertType()).isEqualTo("SEPSIS_SUSPECTED");
        assertThat(results.get(0).getSeverity()).isEqualTo(AlertSeverity.CRITICAL);
    }
    
    @Test
    @DisplayName("Medication adherence pattern detects absence within time window")
    void medicationAdherenceAbsenceDetection() throws Exception {
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();
        env.setParallelism(1);
        
        // Arrange - medication scheduled but not administered
        List<ClinicalEvent> events = Arrays.asList(
            ClinicalEventBuilder.medicationOrder("patient-002", 1000L, "RX001", "Insulin"),
            // No administration event within 6 hours
            ClinicalEventBuilder.vitals("patient-002", 7 * 3600_000L, 80, 120) // 7 hours later
        );
        
        DataStream<ClinicalEvent> stream = env.fromCollection(events)
            .assignTimestampsAndWatermarks(new AssignerWithPunctuatedWatermarks<ClinicalEvent>() {
                @Override
                public Watermark checkAndGetNextWatermark(ClinicalEvent event, long extractedTimestamp) {
                    return new Watermark(extractedTimestamp);
                }
                
                @Override
                public long extractTimestamp(ClinicalEvent event, long previousTimestamp) {
                    return event.getEventTimestamp();
                }
            });
        
        // Create absence pattern (medication ordered but not administered)
        Pattern<ClinicalEvent, ?> adherencePattern = Pattern
            .<ClinicalEvent>begin("medication_ordered")

        PatternStream<ClinicalEvent> patternStream = CEP.pattern(

        );
        
        DataStream<ClinicalAlert> alerts = patternStream.select(match -> {
            ClinicalEvent ordered = match.get("medication_ordered").get(0);
            return ClinicalAlert.builder()
                .alertType("MEDICATION_NOT_ADMINISTERED")
                .patientId(ordered.getPatientId())
                .severity(AlertSeverity.HIGH)
                .details(Map.of("medication", ordered.getPayload().get("drug_name")))
                .timestamp(System.currentTimeMillis())
                .build();
        });
        
        alerts.addSink(new TestSink<>());
        
        // Act
        env.execute("Adherence Detection Test");

        await().atMost(5, TimeUnit.SECONDS).until(() -> !TestSink.getValues().isEmpty());

    @Test
    @DisplayName("Windowed vital sign aggregation detects deteriorating trends")
    void windowedVitalTrendDetection() throws Exception {
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();
        env.setParallelism(1);
        
        // Arrange - create declining BP trend
        List<ClinicalEvent> events = new ArrayList<>();
        long baseTime = 1000L;
        
        for (int i = 0; i < 6; i++) {
            events.add(ClinicalEventBuilder.vitals(
                "patient-003",
                baseTime + (i * 600_000L), // 10-minute intervals
                90 + i * 5, // Increasing HR
                120 - i * 10 // Decreasing BP
            ));
        }
        
        DataStream<ClinicalEvent> stream = env.fromCollection(events)
            .assignTimestampsAndWatermarks(new AssignerWithPunctuatedWatermarks<ClinicalEvent>() {
                @Override
                public Watermark checkAndGetNextWatermark(ClinicalEvent event, long extractedTimestamp) {
                    return new Watermark(extractedTimestamp);
                }
                
                @Override
                public long extractTimestamp(ClinicalEvent event, long previousTimestamp) {
                    return event.getEventTimestamp();
                }
            });
        
        // Apply sliding window for trend analysis
        DataStream<VitalTrendAlert> trends = stream
            .keyBy(ClinicalEvent::getPatientId)
            .timeWindow(
                org.apache.flink.streaming.api.windowing.time.Time.hours(1),
                org.apache.flink.streaming.api.windowing.time.Time.minutes(10)
            )
            .apply((key, window, events, out) -> {
                List<ClinicalEvent> eventList = new ArrayList<>();
                events.forEach(eventList::add);
                
                if (eventList.size() >= 3) {
                    // Calculate trend
                    double bpTrend = calculateBPTrend(eventList);
                    double hrTrend = calculateHRTrend(eventList);
                    
                    if (bpTrend < -5.0 && hrTrend > 5.0) { // Deteriorating
                        out.collect(VitalTrendAlert.builder()
                            .patientId(key)
                            .alertType("DETERIORATING_VITALS")
                            .bpTrend(bpTrend)
                            .hrTrend(hrTrend)
                            .windowStart(window.getStart())
                            .windowEnd(window.getEnd())
                            .build());
                    }
                }
            });
        
        trends.addSink(new TestSink<>());
        
        // Act
        env.execute("Vital Trend Detection Test");
        
        // Assert
        await().atMost(5, TimeUnit.SECONDS).until(() -> !TestSink.getValues().isEmpty());
        
        List<VitalTrendAlert> results = TestSink.getValues();
        assertThat(results).isNotEmpty();
        assertThat(results.get(0).getAlertType()).isEqualTo("DETERIORATING_VITALS");
        assertThat(results.get(0).getBpTrend()).isNegative();
        assertThat(results.get(0).getHrTrend()).isPositive();
    }
    
    private double calculateBPTrend(List<ClinicalEvent> events) {
        if (events.size() < 2) return 0.0;
        
        double firstBP = (Integer) events.get(0).getPayload().get("systolic_bp");
        double lastBP = (Integer) events.get(events.size() - 1).getPayload().get("systolic_bp");
        
        return (lastBP - firstBP) / events.size();
    }
    
    private double calculateHRTrend(List<ClinicalEvent> events) {
        if (events.size() < 2) return 0.0;
        
        double firstHR = (Integer) events.get(0).getPayload().get("heart_rate");
        double lastHR = (Integer) events.get(events.size() - 1).getPayload().get("heart_rate");
        
        return (lastHR - firstHR) / events.size();
    }

/**

public class TestSink<T> implements SinkFunction<T> {
    public static final ConcurrentLinkedQueue<Object> VALUES = new ConcurrentLinkedQueue<>();
    
    @Override
    public synchronized void invoke(T value, Context context) {
        VALUES.add(value);
    }
    
    public static void clear() {
        VALUES.clear();
    }
    
    @SuppressWarnings("unchecked")
    public static <T> List<T> getValues() {
        return new ArrayList<>((Collection<T>) VALUES);
    }
    
    @SuppressWarnings("unchecked")
    public static <T> T getLastValue() {
        Object[] array = VALUES.toArray();
        return array.length > 0 ? (T) array[array.length - 1] : null;
    }
}

            .receivedAt(System.currentTimeMillis())
            .source("test")
            .build();
        
        // Act
        harness.processElement(malformed, System.currentTimeMillis());
        
        // Assert - main output empty
        assertThat(harness.getOutput()).isEmpty();
        
        // DLQ has entry with error details
        List<DLQRecord> dlqRecords = harness.getSideOutput(IngestionRouter.DLQ_TAG);
        assertThat(dlqRecords).hasSize(1);
        assertThat(dlqRecords.get(0).getErrorReason()).contains("JSON");
        assertThat(dlqRecords.get(0).getOriginalPayload()).isEqualTo("{invalid-json");
    }
    
    @Test
    @DisplayName("Schema validation rejects events with missing required fields")
    void schemaValidationRejectsMissingFields() throws Exception {
        // Arrange - missing patientId
        RawEvent missingField = RawEvent.builder()
            .payload("{\"eventType\":\"VITAL_SIGN\"}")
            .receivedAt(System.currentTimeMillis())
            .source("test")
            .build();
        
        // Act
        harness.processElement(missingField, System.currentTimeMillis());
        
        // Assert
        assertThat(harness.getOutput()).isEmpty();
        List<DLQRecord> dlqRecords = harness.getSideOutput(IngestionRouter.DLQ_TAG);
        assertThat(dlqRecords).hasSize(1);
        assertThat(dlqRecords.get(0).getErrorReason()).contains("patientId");
    }
    
    @Test
    @DisplayName("Watermark generation handles out-of-order events")
    void watermarkHandlesOutOfOrderEvents() throws Exception {
        // Act - process events out of order
        harness.processElement(createRawEvent("p1", 3000L), 3000L);
        harness.processElement(createRawEvent("p1", 1000L), 1000L);
        harness.processElement(createRawEvent("p1", 2000L), 2000L);
        
        // Process watermark
        harness.processWatermark(new Watermark(3000L));
        
        // Assert - all events processed
        assertThat(harness.getOutput()).hasSize(3);
        
        // Watermark advanced correctly (with configured allowance)
        assertThat(harness.getOutput().stream()
            .filter(o -> o instanceof Watermark)
            .map(o -> ((Watermark) o).getTimestamp())
            .findFirst())
            .isPresent()
            .hasValueSatisfying(wm -> assertThat(wm).isGreaterThanOrEqualTo(1000L));
    }
    
    private RawEvent createRawEvent(String patientId, long timestamp) {
        return RawEvent.builder()
            .payload(String.format("{\"patientId\":\"%s\",\"eventType\":\"TEST\"}", patientId))
            .receivedAt(timestamp)
            .source("test")
            .build();
    }

package com.ehr.flink.tests.module2;


import com.ehr.flink.context.PatientContextAssembler;
import com.ehr.flink.model.*;
import com.ehr.flink.tests.builders.ClinicalEventBuilder;
import com.ehr.flink.tests.utils.TestSink;
import org.apache.flink.api.common.typeinfo.TypeInformation;
import org.apache.flink.streaming.api.operators.KeyedProcessOperator;
import org.apache.flink.streaming.util.KeyedOneInputStreamOperatorTestHarness;
import org.junit.jupiter.api.*;


import static org.assertj.core.api.Assertions.*;


/**
 * Tests for Module 2: Patient Context Assembly & Enrichment
 * - Patient state accumulation
 * - TTL and cleanup
 * - Clinical score calculation
 * - Timer-based expiration
 */
public class PatientContextAssemblerTest {
    
    private KeyedOneInputStreamOperatorTestHarness<String, ClinicalEvent, EnrichedEvent> harness;
    private PatientContextAssembler assembler;
    
    @BeforeEach
    void setUp() throws Exception {
        assembler = new PatientContextAssembler();
        KeyedProcessOperator<String, ClinicalEvent, EnrichedEvent> operator = 
            new KeyedProcessOperator<>(assembler);
        
        harness = new KeyedOneInputStreamOperatorTestHarness<>(
            operator,
            event -> event.getPatientId(),
            TypeInformation.of(String.class)
        );
        
        harness.setup();
        harness.open();
        TestSink.clear();
    }
    
    @AfterEach
    void tearDown() throws Exception {
        harness.close();
    }
    
    @Test
    @DisplayName("Patient state accumulates across multiple events")
    void stateAccumulatesAcrossEvents() throws Exception {
        // Arrange
        String patientId = "patient-001";
        long baseTime = 1000L;
        
        ClinicalEvent admission = ClinicalEventBuilder.admission(patientId, baseTime);
        ClinicalEvent vitals = ClinicalEventBuilder.vitals(patientId, baseTime + 1000, 92, 120);
        ClinicalEvent medication = ClinicalEventBuilder.medicationOrder(
            patientId, baseTime + 2000, "RX001", "Lisinopril"
        );
        
        // Act
        harness.processElement(admission, baseTime);
        harness.processElement(vitals, baseTime + 1000);
        harness.processElement(medication, baseTime + 2000);
        
        // Assert
        assertThat(harness.getOutput()).hasSize(3);
        
        EnrichedEvent lastEvent = (EnrichedEvent) harness.extractOutputValues().get(2);
        PatientSnapshot snapshot = lastEvent.getPatientContext();
        
        assertThat(snapshot).isNotNull();
        assertThat(snapshot.getCurrentLocation()).isEqualTo("Ward-3B");
        assertThat(snapshot.getActiveMedications()).containsKey("RX001");
        assertThat(snapshot.getRecentVitals()).hasSize(1);
        assertThat(snapshot.getRecentVitals().get(0).getHeartRate()).isEqualTo(92);
    }
    
    @Test
    @DisplayName("Clinical scores calculated and included in snapshot")
    void clinicalScoresCalculated() throws Exception {
        // Arrange
        String patientId = "patient-002";
        long timestamp = 1000L;
        
        // Create patient with conditions that trigger MEWS calculation
        ClinicalEvent vitals = ClinicalEventBuilder.vitals(patientId, timestamp, 110, 85);
        
        // Act
        harness.processElement(vitals, timestamp);
        
        // Assert
        EnrichedEvent enriched = (EnrichedEvent) harness.extractOutputValues().get(0);
        Map<String, Double> scores = enriched.getPatientContext().getClinicalScores();
        
        assertThat(scores).containsKey("MEWS");
        assertThat(scores.get("MEWS")).isGreaterThan(0.0);
    }
    
    @Test
    @DisplayName("Medication TTL triggers cleanup after expiration")
    void medicationTTLTriggersCleanup() throws Exception {
        // Arrange
        String patientId = "patient-003";
        long baseTime = 1000L;
        long ttlDuration = 30L * 24 * 3600 * 1000; // 30 days in milliseconds
        
        ClinicalEvent medication = ClinicalEventBuilder.medicationOrder(
            patientId, baseTime, "RX002", "Metformin"
        );
        
        // Act - add medication
        harness.processElement(medication, baseTime);
        
        EnrichedEvent initial = (EnrichedEvent) harness.extractOutputValues().get(0);
        assertThat(initial.getPatientContext().getActiveMedications()).containsKey("RX002");
        
        // Advance event time beyond TTL
        harness.setProcessingTime(baseTime + ttlDuration + 1000);
        harness.processWatermark(new org.apache.flink.streaming.api.watermark.Watermark(
            baseTime + ttlDuration + 1000
        ));
        
        // Add another event to trigger state access
        ClinicalEvent vitals = ClinicalEventBuilder.vitals(patientId, 
            baseTime + ttlDuration + 2000, 80, 120);
        harness.processElement(vitals, baseTime + ttlDuration + 2000);
        
        // Assert - medication should be removed
        EnrichedEvent afterTTL = (EnrichedEvent) harness.extractOutputValues().get(
            harness.extractOutputValues().size() - 1
        );
        assertThat(afterTTL.getPatientContext().getActiveMedications())
            .doesNotContainKey("RX002");
    }
    
    @Test
    @DisplayName("State isolation between different patients")
    void stateIsolatedBetweenPatients() throws Exception {
        // Arrange
        long timestamp = 1000L;
        ClinicalEvent patient1Event = ClinicalEventBuilder.vitals("patient-001", timestamp, 80, 120);
        ClinicalEvent patient2Event = ClinicalEventBuilder.vitals("patient-002", timestamp, 90, 130);
        
        // Act
        harness.processElement(patient1Event, timestamp);
        harness.processElement(patient2Event, timestamp);
        
        // Assert
        assertThat(harness.getOutput()).hasSize(2);
        
        EnrichedEvent event1 = (EnrichedEvent) harness.extractOutputValues().get(0);
        EnrichedEvent event2 = (EnrichedEvent) harness.extractOutputValues().get(1);
        
        assertThat(event1.getPatientId()).isEqualTo("patient-001");
        assertThat(event2.getPatientId()).isEqualTo("patient-002");
        
        // Vital signs should be different
        assertThat(event1.getPatientContext().getRecentVitals().get(0).getHeartRate())
            .isNotEqualTo(event2.getPatientContext().getRecentVitals().get(0).getHeartRate());
    }
    
    @Test
    @DisplayName("Circular buffer maintains fixed size for vitals history")
    void circularBufferMaintainsFixedSize() throws Exception {
        // Arrange
        String patientId = "patient-004";
        long baseTime = 1000L;
        int maxVitalsCount = 48; // 24 hours of 30-min readings
        
        // Act - add more vitals than buffer capacity
        for (int i = 0; i < maxVitalsCount + 10; i++) {
            ClinicalEvent vitals = ClinicalEventBuilder.vitals(
                patientId, 
                baseTime + (i * 1800_000L), // 30 min intervals
                80 + i, 
                120
            );
            harness.processElement(vitals, baseTime + (i * 1800_000L));
        }
        
        // Assert - buffer size should not exceed maximum
        EnrichedEvent lastEvent = (EnrichedEvent) harness.extractOutputValues().get(
            harness.extractOutputValues().size() - 1
        );
        assertThat(lastEvent.getPatientContext().getRecentVitals().size())
            .isLessThanOrEqualTo(maxVitalsCount);
    }
}

package com.ehr.flink.tests.module3;


import com.ehr.flink.semantic.SemanticMeshProcessor;
import com.ehr.flink.model.*;
import com.ehr.flink.tests.builders.ClinicalEventBuilder;
import org.apache.flink.api.common.state.MapStateDescriptor;
import org.apache.flink.api.common.typeinfo.TypeInformation;
import org.apache.flink.streaming.api.operators.co.KeyedBroadcastProcessOperator;
import org.apache.flink.streaming.util.KeyedTwoInputStreamOperatorTestHarness;
import org.junit.jupiter.api.*;


import static org.assertj.core.api.Assertions.*;


/**
 * Tests for Module 3: Semantic Mesh Integration
 * - Broadcast state distribution
 * - Drug interaction detection
 * - Clinical protocol applicability
 * - Version management and rollback
 */
public class SemanticMeshProcessorTest {
    
    private KeyedTwoInputStreamOperatorTestHarness<String, ClinicalEvent, SemanticMeshUpdate, EnrichedEvent> harness;
    private MapStateDescriptor<String, DrugInteraction> interactionDescriptor;
    private MapStateDescriptor<String, ClinicalProtocol> protocolDescriptor;
    
    @BeforeEach
    void setUp() throws Exception {
        interactionDescriptor = new MapStateDescriptor<>(
            "drug-interactions",
            String.class,
            DrugInteraction.class
        );
        
        protocolDescriptor = new MapStateDescriptor<>(
            "clinical-protocols",
            String.class,
            ClinicalProtocol.class
        );
        
        SemanticMeshProcessor processor = new SemanticMeshProcessor(
            interactionDescriptor,
            protocolDescriptor
        );
        
        KeyedBroadcastProcessOperator<String, ClinicalEvent, SemanticMeshUpdate, EnrichedEvent> operator =
            new KeyedBroadcastProcessOperator<>(processor);
        
        harness = new KeyedTwoInputStreamOperatorTestHarness<>(
            operator,
            event -> event.getPatientId(),
            update -> "broadcast-key", // All updates share same key for broadcast
            TypeInformation.of(String.class)
        );
        
        harness.setup();
        harness.open();
    }
    
    @AfterEach
    void tearDown() throws Exception {
        harness.close();
    }
    
    @Test
    @DisplayName("Broadcast update enriches subsequent events")
    void broadcastUpdateEnrichesEvents() throws Exception {
        // Arrange - medication event before interaction data
        long timestamp = 1000L;
        ClinicalEvent medEvent = ClinicalEventBuilder.medicationOrder(
            "patient-001", timestamp, "warfarin", "Warfarin"
        );
        
        // Act - process event before broadcast (no enrichment)
        harness.processElement1(medEvent, timestamp);
        
        EnrichedEvent beforeBroadcast = (EnrichedEvent) harness.extractOutputValues().get(0);
        assertThat(beforeBroadcast.getPotentialInteractions()).isEmpty();
        
        // Broadcast drug interaction data
        SemanticMeshUpdate interactionUpdate = SemanticMeshUpdate.builder()
            .updateType(UpdateType.DRUG_INTERACTION_ADD)
            .drug1("warfarin")
            .drug2("aspirin")
            .severity(InteractionSeverity.MAJOR)
            .mechanism("Increased bleeding risk")
            .timestamp(timestamp + 1000)
            .build();
        
        harness.processElement2(interactionUpdate, timestamp + 1000);
        
        // Process same medication event again
        ClinicalEvent medEvent2 = ClinicalEventBuilder.medicationOrder(
            "patient-001", timestamp + 2000, "warfarin", "Warfarin"
        );
        harness.processElement1(medEvent2, timestamp + 2000);
        
        // Assert - now enriched with interaction data
        EnrichedEvent afterBroadcast = (EnrichedEvent) harness.extractOutputValues().get(1);
        assertThat(afterBroadcast.getPotentialInteractions()).hasSize(1);
        assertThat(afterBroadcast.getPotentialInteractions().get(0).getDrug2())
            .isEqualTo("aspirin");
    }
    
    @Test
    @DisplayName("Clinical protocol applicability determined by broadcast state")
    void protocolApplicabilityDetermined() throws Exception {
        // Arrange - broadcast sepsis protocol
        SemanticMeshUpdate protocolUpdate = SemanticMeshUpdate.builder()
            .updateType(UpdateType.PROTOCOL_ADD)
            .protocolId("SEPSIS_BUNDLE_2023")
            .protocol(createSepsisProtocol())
            .timestamp(1000L)
            .build();
        
        harness.processElement2(protocolUpdate, 1000L);
        
        // Act - process event with sepsis indicators
        ClinicalEvent labEvent = ClinicalEventBuilder.labResult(
            "patient-002", 2000L, "lactate", 2.5
        );
        harness.processElement1(labEvent, 2000L);
        
        // Assert
        EnrichedEvent enriched = (EnrichedEvent) harness.extractOutputValues().get(0);
        assertThat(enriched.getApplicableProtocols()).contains("SEPSIS_BUNDLE_2023");
    }
    
    @Test
    @DisplayName("Incremental update modifies existing broadcast state")
    void incrementalUpdateModifiesState() throws Exception {
        // Arrange - add initial interaction
        SemanticMeshUpdate initial = SemanticMeshUpdate.builder()
            .updateType(UpdateType.DRUG_INTERACTION_ADD)
            .drug1("drugA")
            .drug2("drugB")
            .severity(InteractionSeverity.MODERATE)
            .timestamp(1000L)
            .build();
        
        harness.processElement2(initial, 1000L);
        
        // Act - update severity
        SemanticMeshUpdate update = SemanticMeshUpdate.builder()
            .updateType(UpdateType.DRUG_INTERACTION_UPDATE)
            .drug1("drugA")
            .drug2("drugB")
            .severity(InteractionSeverity.MAJOR)
            .timestamp(2000L)
            .build();
        
        harness.processElement2(update, 2000L);
        
        // Process event and verify updated severity
        ClinicalEvent medEvent = ClinicalEventBuilder.medicationOrder(
            "patient-003", 3000L, "drugA", "Drug A"
        );
        harness.processElement1(medEvent, 3000L);
        
        EnrichedEvent enriched = (EnrichedEvent) harness.extractOutputValues().get(0);
        assertThat(enriched.getPotentialInteractions().get(0).getSeverity())
            .isEqualTo(InteractionSeverity.MAJOR);
    }
    
    @Test
    @DisplayName("Version rollback restores previous semantic mesh state")
    void versionRollbackRestoresPreviousState() throws Exception {
        // Arrange - version 1
        SemanticMeshUpdate v1 = SemanticMeshUpdate.builder()
            .updateType(UpdateType.FULL_REFRESH)
            .version("v1.0")
            .interactions(Map.of("drugA:drugB", createInteraction("drugA", "drugB", InteractionSeverity.MODERATE)))
            .timestamp(1000L)
            .build();
        
        harness.processElement2(v1, 1000L);
        
        // Version 2 with different severity
        SemanticMeshUpdate v2 = SemanticMeshUpdate.builder()
            .updateType(UpdateType.FULL_REFRESH)
            .version("v2.0")
            .interactions(Map.of("drugA:drugB", createInteraction("drugA", "drugB", InteractionSeverity.MAJOR)))
            .timestamp(2000L)
            .build();
        
        harness.processElement2(v2, 2000L);
        
        // Act - rollback to v1
        SemanticMeshUpdate rollback = SemanticMeshUpdate.builder()
            .updateType(UpdateType.VERSION_ROLLBACK)
            .targetVersion("v1.0")
            .timestamp(3000L)
            .build();
        
        harness.processElement2(rollback, 3000L);
        
        // Assert - severity should be back to MODERATE
        ClinicalEvent medEvent = ClinicalEventBuilder.medicationOrder(
            "patient-004", 4000L, "drugA", "Drug A"
        );
        harness.processElement1(medEvent, 4000L);
        
        EnrichedEvent enriched = (EnrichedEvent) harness.extractOutputValues().get(0);
        assertThat(enriched.getPotentialInteractions().get(0).getSeverity())
            .isEqualTo(InteractionSeverity.MODERATE);
    }
    
    private ClinicalProtocol createSepsisProtocol() {
        return ClinicalProtocol.builder()
            .protocolId("SEPSIS_BUNDLE_2023")
            .name("Sepsis Bundle")
            .applicableDiagnoses(Set.of("sepsis", "severe_sepsis"))
            .build();
    }
    
    private DrugInteraction createInteraction(String drug1, String drug2, InteractionSeverity severity) {
        return DrugInteraction.builder()
            .drug1(drug1)
            .drug2(drug2)
            .severity(severity)
            .mechanism("Test mechanism")
            .build();
    }
}
