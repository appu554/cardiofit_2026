```java
package com.hospital.flink.cep;


import lombok.Data;
import java.time.Instant;


/**
 * Base clinical event for CEP processing
 */
@Data
public abstract class ClinicalEvent {
    private String patientId;
    private String eventType;
    private Instant timestamp;
    private String sourceSystem;
}


/**
 * Vital sign event
 */
@Data
public class VitalSignEvent extends ClinicalEvent {
    private VitalType vitalType;
    private double value;
    private String unit;
    private boolean isAbnormal;
    private AbnormalityLevel abnormalityLevel;
    
    public enum VitalType {
        HEART_RATE, BLOOD_PRESSURE_SYSTOLIC, BLOOD_PRESSURE_DIASTOLIC,
        RESPIRATORY_RATE, TEMPERATURE, OXYGEN_SATURATION, 
        MEAN_ARTERIAL_PRESSURE, GLASGOW_COMA_SCALE
    }
    
    public enum AbnormalityLevel {
        NORMAL, BORDERLINE, ABNORMAL, CRITICAL
    }
}


/**
 * Lab result event
 */
@Data
public class LabResultEvent extends ClinicalEvent {
    private String labCode;          // LOINC code
    private String labName;
    private double value;
    private String unit;
    private double referenceMin;
    private double referenceMax;
    private boolean isCritical;
}


/**
 * Medication event
 */
@Data
public class MedicationEvent extends ClinicalEvent {
    private MedicationEventType medicationEventType;
    private String medicationCode;   // RxNorm
    private String medicationName;
    private String dose;
    private String route;
    private Instant scheduledTime;
    private Instant actualTime;
    
    public enum MedicationEventType {
        ORDERED, ADMINISTERED, DUE, MISSED, DISCONTINUED
    }
}


/**
 * Clinical condition event
 */
@Data
public class ConditionEvent extends ClinicalEvent {
    private String conditionCode;    // ICD-10
    private String conditionName;
    private ConditionStatus status;
    private Instant onsetDate;
    
    public enum ConditionStatus {
        ACTIVE, RESOLVED, WORSENING, IMPROVING
    }
}

### **4A.2: CEP Pattern Definitions**


```java
package com.hospital.flink.cep.patterns;


import org.apache.flink.cep.pattern.Pattern;
import org.apache.flink.cep.pattern.conditions.SimpleCondition;
import org.apache.flink.streaming.api.windowing.time.Time;


/**
 * Clinical deterioration patterns using Flink CEP
 */
public class ClinicalPatterns {
    
    /**
     * PATTERN 1: Sepsis Early Warning
     * 
     * Clinical Logic:
     * - Elevated heart rate (>100 bpm)
     * - Followed by low blood pressure (<100 mmHg systolic)
     * - Followed by elevated lactate (>2.0 mmol/L)
     * Within 2-hour window
     * 
     * Clinical Significance: Early sepsis identification
     */
    public static Pattern<VitalSignEvent, ?> sepsisEarlyWarningPattern() {
        return Pattern.<VitalSignEvent>begin("tachycardia")
            .where(new SimpleCondition<VitalSignEvent>() {
                @Override
                public boolean filter(VitalSignEvent event) {
                    return event.getVitalType() == VitalType.HEART_RATE 
                        && event.getValue() > 100;
                }
            })
            .followedBy("hypotension")
            .where(new SimpleCondition<VitalSignEvent>() {
                @Override
                public boolean filter(VitalSignEvent event) {
                    return event.getVitalType() == VitalType.BLOOD_PRESSURE_SYSTOLIC 
                        && event.getValue() < 100;
                }
            })
            .within(Time.hours(2));
    }
    
    /**
     * PATTERN 2: Rapid Clinical Deterioration
     * 
     * Clinical Logic:
     * - Heart rate increases by >20 bpm
     * - Followed by respiratory rate increase >30%
     * - Followed by oxygen saturation drop >5%
     * Within 1-hour window
     * 
     * Clinical Significance: Respiratory/cardiac decompensation
     */
    public static Pattern<VitalSignEvent, ?> rapidDeteriorationPattern() {
        return Pattern.<VitalSignEvent>begin("hr_baseline")
            .where(new SimpleCondition<VitalSignEvent>() {
                @Override
                public boolean filter(VitalSignEvent event) {
                    return event.getVitalType() == VitalType.HEART_RATE;
                }
            })
            .followedBy("hr_elevated")
            .where(new SimpleCondition<VitalSignEvent>() {
                @Override
                public boolean filter(VitalSignEvent event) {
                    return event.getVitalType() == VitalType.HEART_RATE;
                }
            })
            .where((current, ctx) -> {
                VitalSignEvent baseline = ctx.getEventsForPattern("hr_baseline")
                    .iterator().next();
                return current.getValue() - baseline.getValue() > 20;
            })
            .followedBy("rr_elevated")
            .where(new SimpleCondition<VitalSignEvent>() {
                @Override
                public boolean filter(VitalSignEvent event) {
                    return event.getVitalType() == VitalType.RESPIRATORY_RATE 
                        && event.getValue() > 24;
                }
            })
            .followedBy("o2sat_decreased")
            .where(new SimpleCondition<VitalSignEvent>() {
                @Override
                public boolean filter(VitalSignEvent event) {
                    return event.getVitalType() == VitalType.OXYGEN_SATURATION 
                        && event.getValue() < 92;
                }
            })
            .within(Time.hours(1));
    }
    
    /**
     * PATTERN 3: Medication Adherence Monitoring
     * 
     * Clinical Logic:
     * - Medication ordered
     * - Medication due (scheduled time)
     * - ABSENCE of administration event within 6 hours
     * 
     * Clinical Significance: Medication error prevention
     */
    public static Pattern<MedicationEvent, ?> medicationAdherencePattern() {
        return Pattern.<MedicationEvent>begin("med_due")
            .where(new SimpleCondition<MedicationEvent>() {
                @Override
                public boolean filter(MedicationEvent event) {
                    return event.getMedicationEventType() == MedicationEventType.DUE;
                }
            })
            .notFollowedBy("med_administered")
            .where(new SimpleCondition<MedicationEvent>() {
                @Override
                public boolean filter(MedicationEvent event) {
                    return event.getMedicationEventType() == MedicationEventType.ADMINISTERED;
                }
            })
            .within(Time.hours(6));
    }
    
    /**
     * PATTERN 4: Drug-Lab Interaction Monitoring
     * 
     * Clinical Logic:
     * - ACE inhibitor started
     * - Check for potassium/creatinine lab within 48 hours
     * - Alert if labs not ordered
     * 
     * Clinical Significance: Drug safety monitoring
     */
    public static Pattern<Object, ?> drugLabMonitoringPattern() {
        return Pattern.<Object>begin("ace_inhibitor_started")
            .where(new SimpleCondition<Object>() {
                @Override
                public boolean filter(Object event) {
                    if (event instanceof MedicationEvent) {
                        MedicationEvent medEvent = (MedicationEvent) event;
                        return medEvent.getMedicationEventType() == MedicationEventType.ORDERED
                            && isACEInhibitor(medEvent.getMedicationCode());
                    }
                    return false;
                }
            })
            .notFollowedBy("renal_labs_ordered")
            .where(new SimpleCondition<Object>() {
                @Override
                public boolean filter(Object event) {
                    if (event instanceof LabResultEvent) {
                        LabResultEvent labEvent = (LabResultEvent) event;
                        return labEvent.getLabCode().equals("2160-0") // Creatinine
                            || labEvent.getLabCode().equals("2823-3"); // Potassium
                    }
                    return false;
                }
            })
            .within(Time.hours(48));
    }
    
    /**
     * PATTERN 5: Clinical Pathway Deviation
     * 
     * Clinical Logic:
     * - Sepsis condition diagnosed
     * - Blood cultures should be drawn within 1 hour
     * - Antibiotics should start within 1 hour
     * - Alert if pathway not followed
     * 
     * Clinical Significance: Sepsis bundle compliance
     */
    public static Pattern<Object, ?> sepsisPathwayPattern() {
        return Pattern.<Object>begin("sepsis_diagnosis")
            .where(new SimpleCondition<Object>() {
                @Override
                public boolean filter(Object event) {
                    if (event instanceof ConditionEvent) {
                        ConditionEvent condEvent = (ConditionEvent) event;
                        return condEvent.getConditionCode().startsWith("A41"); // Sepsis ICD-10
                    }
                    return false;
                }
            })
            .followedBy("blood_cultures_ordered")
            .where(new SimpleCondition<Object>() {
                @Override
                public boolean filter(Object event) {
                    if (event instanceof LabResultEvent) {
                        LabResultEvent labEvent = (LabResultEvent) event;
                        return labEvent.getLabName().contains("Blood Culture");
                    }
                    return false;
                }
            })
            .within(Time.hours(1))
            .followedBy("antibiotics_started")
            .where(new SimpleCondition<Object>() {
                @Override
                public boolean filter(Object event) {
                    if (event instanceof MedicationEvent) {
                        MedicationEvent medEvent = (MedicationEvent) event;
                        return medEvent.getMedicationEventType() == MedicationEventType.ADMINISTERED
                            && isAntibiotic(medEvent.getMedicationCode());
                    }
                    return false;
                }
            })
            .within(Time.hours(1));
    }
    
    /**
     * PATTERN 6: Acute Kidney Injury (AKI) Detection
     * 
     * Clinical Logic:
     * - Baseline creatinine
     * - New creatinine >50% increase from baseline
     * - Within 48 hours
     * 
     * Clinical Significance: Early AKI detection
     */
    public static Pattern<LabResultEvent, ?> akiDetectionPattern() {
        return Pattern.<LabResultEvent>begin("baseline_creatinine")
            .where(new SimpleCondition<LabResultEvent>() {
                @Override
                public boolean filter(LabResultEvent event) {
                    return event.getLabCode().equals("2160-0"); // Creatinine LOINC
                }
            })
            .followedBy("elevated_creatinine")
            .where(new SimpleCondition<LabResultEvent>() {
                @Override
                public boolean filter(LabResultEvent event) {
                    return event.getLabCode().equals("2160-0");
                }
            })
            .where((current, ctx) -> {
                LabResultEvent baseline = ctx.getEventsForPattern("baseline_creatinine")
                    .iterator().next();
                double increase = (current.getValue() - baseline.getValue()) / baseline.getValue();
                return increase > 0.50; // 50% increase
            })
            .within(Time.hours(48));
    }
    
    // Helper methods
    private static boolean isACEInhibitor(String medicationCode) {
        // RxNorm codes for ACE inhibitors
        return medicationCode.startsWith("1998") || medicationCode.startsWith("3827");
    }
    
    private static boolean isAntibiotic(String medicationCode) {
        // Simplified - check against antibiotic RxNorm codes
        return medicationCode.matches("(197|198|199|200).*");
    }
}

### **4A.3: CEP Pattern Processing Job**


```java
package com.hospital.flink.cep;


import org.apache.flink.cep.CEP;
import org.apache.flink.cep.PatternSelectFunction;
import org.apache.flink.cep.PatternStream;
import org.apache.flink.streaming.api.datastream.DataStream;
import org.apache.flink.streaming.api.environment.StreamExecutionEnvironment;
import org.apache.flink.api.common.eventtime.WatermarkStrategy;


import java.time.Duration;
import java.util.*;


/**
 * Main CEP processing job for clinical pattern detection
 */
public class ClinicalPatternDetectionJob {
    
    public static void main(String[] args) throws Exception {
        
        // Setup Flink environment
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();
        env.setParallelism(32);
        env.enableCheckpointing(60000); // 60-second checkpoints
        
        // ═══════════════════════════════════════════════════════
        // INPUT: Read enriched events from Kafka
        // ═══════════════════════════════════════════════════════
        
        DataStream<VitalSignEvent> vitalSignStream = env
            .fromSource(
                createKafkaSource("ehr-events-vitals"),
                WatermarkStrategy
                    .<VitalSignEvent>forBoundedOutOfOrderness(Duration.ofMinutes(5))
                    .withTimestampAssigner((event, timestamp) -> 
                        event.getTimestamp().toEpochMilli()),
                "vital-signs-source"
            )
            .keyBy(VitalSignEvent::getPatientId);
        
        DataStream<LabResultEvent> labResultStream = env
            .fromSource(
                createKafkaSource("ehr-events-labs"),
                WatermarkStrategy
                    .<LabResultEvent>forBoundedOutOfOrderness(Duration.ofMinutes(10))
                    .withTimestampAssigner((event, timestamp) -> 
                        event.getTimestamp().toEpochMilli()),
                "lab-results-source"
            )
            .keyBy(LabResultEvent::getPatientId);
        
        DataStream<MedicationEvent> medicationStream = env
            .fromSource(
                createKafkaSource("ehr-events-medications"),
                WatermarkStrategy
                    .<MedicationEvent>forBoundedOutOfOrderness(Duration.ofMinutes(2))
                    .withTimestampAssigner((event, timestamp) -> 
                        event.getTimestamp().toEpochMilli()),
                "medication-source"
            )
            .keyBy(MedicationEvent::getPatientId);
        
        // ═══════════════════════════════════════════════════════
        // CEP PATTERN 1: Sepsis Early Warning
        // ═══════════════════════════════════════════════════════
        
        PatternStream<VitalSignEvent> sepsisPattern = CEP.pattern(
            vitalSignStream,
            ClinicalPatterns.sepsisEarlyWarningPattern()
        );
        
        DataStream<ClinicalAlert> sepsisAlerts = sepsisPattern.select(
            new PatternSelectFunction<VitalSignEvent, ClinicalAlert>() {
                @Override
                public ClinicalAlert select(Map<String, List<VitalSignEvent>> pattern) {
                    VitalSignEvent tachycardia = pattern.get("tachycardia").get(0);
                    VitalSignEvent hypotension = pattern.get("hypotension").get(0);
                    
                    return ClinicalAlert.builder()
                        .alertId(UUID.randomUUID().toString())
                        .patientId(tachycardia.getPatientId())
                        .alertType(AlertType.SEPSIS_EARLY_WARNING)
                        .severity(AlertSeverity.CRITICAL)
                        .title("Sepsis Early Warning: Tachycardia + Hypotension")
                        .description(String.format(
                            "Patient showing early signs of sepsis:\n" +
                            "- Heart Rate: %.0f bpm (elevated)\n" +
                            "- Blood Pressure: %.0f/? mmHg (low)\n" +
                            "Time between events: %d minutes",
                            tachycardia.getValue(),
                            hypotension.getValue(),
                            Duration.between(
                                tachycardia.getTimestamp(), 
                                hypotension.getTimestamp()
                            ).toMinutes()
                        ))
                        .recommendations(Arrays.asList(
                            "Order lactate level immediately",
                            "Obtain blood cultures before antibiotics",
                            "Consider activating sepsis protocol",
                            "Notify physician"
                        ))
                        .triggerEvents(Arrays.asList(
                            tachycardia.getEventType(),
                            hypotension.getEventType()
                        ))
                        .timestamp(hypotension.getTimestamp())
                        .expiresAt(hypotension.getTimestamp().plus(Duration.ofHours(4)))
                        .build();
                }
            }
        );
        
        // ═══════════════════════════════════════════════════════
        // CEP PATTERN 2: Rapid Deterioration
        // ═══════════════════════════════════════════════════════
        
        PatternStream<VitalSignEvent> deteriorationPattern = CEP.pattern(
            vitalSignStream,
            ClinicalPatterns.rapidDeteriorationPattern()
        );
        
        DataStream<ClinicalAlert> deteriorationAlerts = deteriorationPattern.select(
            new PatternSelectFunction<VitalSignEvent, ClinicalAlert>() {
                @Override
                public ClinicalAlert select(Map<String, List<VitalSignEvent>> pattern) {
                    VitalSignEvent hrBaseline = pattern.get("hr_baseline").get(0);
                    VitalSignEvent hrElevated = pattern.get("hr_elevated").get(0);
                    VitalSignEvent rrElevated = pattern.get("rr_elevated").get(0);
                    VitalSignEvent o2Decreased = pattern.get("o2sat_decreased").get(0);
                    
                    double hrIncrease = hrElevated.getValue() - hrBaseline.getValue();
                    
                    return ClinicalAlert.builder()
                        .alertId(UUID.randomUUID().toString())
                        .patientId(hrBaseline.getPatientId())
                        .alertType(AlertType.RAPID_DETERIORATION)
                        .severity(AlertSeverity.HIGH)
                        .title("Rapid Clinical Deterioration Detected")
                        .description(String.format(
                            "Patient showing signs of clinical deterioration:\n" +
                            "- Heart Rate: %.0f → %.0f bpm (+%.0f)\n" +
                            "- Respiratory Rate: %.0f/min (elevated)\n" +
                            "- Oxygen Saturation: %.1f%% (decreased)\n" +
                            "Pattern detected over %d minutes",
                            hrBaseline.getValue(),
                            hrElevated.getValue(),
                            hrIncrease,
                            rrElevated.getValue(),
                            o2Decreased.getValue(),
                            Duration.between(
                                hrBaseline.getTimestamp(),
                                o2Decreased.getTimestamp()
                            ).toMinutes()
                        ))
                        .recommendations(Arrays.asList(
                            "Activate Rapid Response Team",
                            "Assess airway, breathing, circulation",
                            "Consider ICU transfer",
                            "Obtain arterial blood gas"
                        ))
                        .triggerEvents(Arrays.asList(
                            "HR_INCREASE", "RR_ELEVATED", "O2SAT_DECREASED"
                        ))
                        .timestamp(o2Decreased.getTimestamp())
                        .build();
                }
            }
        );
        
        // ═══════════════════════════════════════════════════════
        // CEP PATTERN 3: Medication Adherence
        // ═══════════════════════════════════════════════════════
        
        PatternStream<MedicationEvent> medAdherencePattern = CEP.pattern(
            medicationStream,
            ClinicalPatterns.medicationAdherencePattern()
        );
        
        DataStream<ClinicalAlert> medAdherenceAlerts = medAdherencePattern.select(
            new PatternSelectFunction<MedicationEvent, ClinicalAlert>() {
                @Override
                public ClinicalAlert select(Map<String, List<MedicationEvent>> pattern) {
                    MedicationEvent medDue = pattern.get("med_due").get(0);
                    
                    Duration delay = Duration.between(
                        medDue.getScheduledTime(),
                        Instant.now()
                    );
                    
                    return ClinicalAlert.builder()
                        .alertId(UUID.randomUUID().toString())
                        .patientId(medDue.getPatientId())
                        .alertType(AlertType.MEDICATION_ADHERENCE)
                        .severity(delay.toHours() > 12 ? 
                                 AlertSeverity.HIGH : AlertSeverity.MEDIUM)
                        .title("Medication Not Administered")
                        .description(String.format(
                            "Medication overdue for administration:\n" +
                            "- Medication: %s\n" +
                            "- Dose: %s %s\n" +
                            "- Scheduled: %s\n" +
                            "- Time Overdue: %d hours %d minutes",
                            medDue.getMedicationName(),
                            medDue.getDose(),
                            medDue.getRoute(),
                            medDue.getScheduledTime().toString(),
                            delay.toHours(),
                            delay.toMinutesPart()
                        ))
                        .recommendations(Arrays.asList(
                            "Verify medication order is still active",
                            "Administer medication if appropriate",
                            "Document reason for delay",
                            "Contact prescribing physician if discontinued"
                        ))
                        .timestamp(Instant.now())
                        .build();
                }
            }
        );
        
        // ═══════════════════════════════════════════════════════
        // CEP PATTERN 4: AKI Detection
        // ═══════════════════════════════════════════════════════
        
        PatternStream<LabResultEvent> akiPattern = CEP.pattern(
            labResultStream,
            ClinicalPatterns.akiDetectionPattern()
        );
        
        DataStream<ClinicalAlert> akiAlerts = akiPattern.select(
            new PatternSelectFunction<LabResultEvent, ClinicalAlert>() {
                @Override
                public ClinicalAlert select(Map<String, List<LabResultEvent>> pattern) {
                    LabResultEvent baseline = pattern.get("baseline_creatinine").get(0);
                    LabResultEvent elevated = pattern.get("elevated_creatinine").get(0);
                    
                    double percentIncrease = ((elevated.getValue() - baseline.getValue()) 
                                            / baseline.getValue()) * 100;
                    
                    return ClinicalAlert.builder()
                        .alertId(UUID.randomUUID().toString())
                        .patientId(baseline.getPatientId())
                        .alertType(AlertType.ACUTE_KIDNEY_INJURY)
                        .severity(AlertSeverity.HIGH)
                        .title("Acute Kidney Injury (AKI) Detected")
                        .description(String.format(
                            "Significant creatinine elevation detected:\n" +
                            "- Baseline Creatinine: %.2f %s\n" +
                            "- Current Creatinine: %.2f %s\n" +
                            "- Increase: %.1f%% (>50%% threshold)\n" +
                            "- Time Frame: %d hours\n\n" +
                            "Meets KDIGO AKI Stage 1 criteria",
                            baseline.getValue(),
                            baseline.getUnit(),
                            elevated.getValue(),
                            elevated.getUnit(),
                            percentIncrease,
                            Duration.between(
                                baseline.getTimestamp(),
                                elevated.getTimestamp()
                            ).toHours()
                        ))
                        .recommendations(Arrays.asList(
                            "Review medications (NSAIDs, ACE inhibitors, diuretics)",
                            "Check urine output trend",
                            "Consider nephrology consultation",
                            "Hold nephrotoxic medications",
                            "Ensure adequate hydration"
                        ))
                        .timestamp(elevated.getTimestamp())
                        .build();
                }
            }
        );
        
        // ═══════════════════════════════════════════════════════
        // COMBINE ALL ALERTS
        // ═══════════════════════════════════════════════════════
        
        DataStream<ClinicalAlert> allAlerts = sepsisAlerts
            .union(deteriorationAlerts)
            .union(medAdherenceAlerts)
            .union(akiAlerts);
        
        // ═══════════════════════════════════════════════════════
        // OUTPUT: Write alerts to Kafka
        // ═══════════════════════════════════════════════════════
        
        allAlerts.sinkTo(
            createKafkaSink("ehr-alerts-critical")
        );
        
        // Execute job
        env.execute("Clinical Pattern Detection CEP Job");
    }
}

### **4B.1: Lab Trend Analysis**


```java
package com.hospital.flink.windows;


import org.apache.flink.streaming.api.datastream.DataStream;
import org.apache.flink.streaming.api.windowing.assigners.SlidingEventTimeWindows;
import org.apache.flink.streaming.api.windowing.time.Time;
import org.apache.flink.streaming.api.functions.windowing.WindowFunction;
import org.apache.flink.util.Collector;


import java.util.*;


/**
 * Lab trend analysis using sliding windows
 */
public class LabTrendAnalyzer {
    
    /**
     * Analyze creatinine trends for AKI detection
     * Window: 48 hours, Slide: 1 hour
     */
    public static DataStream<LabTrendAlert> analyzeCreatinineTrends(
        DataStream<LabResultEvent> labStream) {
        
        return labStream
            .filter(lab -> lab.getLabCode().equals("2160-0")) // Creatinine
            .keyBy(LabResultEvent::getPatientId)
            .window(SlidingEventTimeWindows.of(Time.hours(48), Time.hours(1)))
            .apply(new CreatinineTrendWindowFunction());
    }
    
    /**
     * Window function to detect creatinine trends
     */
    public static class CreatinineTrendWindowFunction 
        implements WindowFunction<LabResultEvent, LabTrendAlert, String, TimeWindow> {
        
        @Override
        public void apply(String patientId,
                         TimeWindow window,
                         Iterable<LabResultEvent> values,
                         Collector<LabTrendAlert> out) {
            
            List<LabResultEvent> labResults = new ArrayList<>();
            values.forEach(labResults::add);
            
            if (labResults.size() < 2) {
                return; // Need at least 2 values for trend
            }
            
            // Sort by timestamp
            labResults.sort(Comparator.comparing(LabResultEvent::getTimestamp));
            
            // Calculate trend
            double firstValue = labResults.get(0).getValue();
            double lastValue = labResults.get(labResults.size() - 1).getValue();
            double percentChange = ((lastValue - firstValue) / firstValue) * 100;
            
            // Linear regression for trend line
            TrendAnalysis trend = calculateLinearTrend(labResults);
            
            // Check for concerning trends
            if (Math.abs(percentChange) > 25 || Math.abs(trend.getSlope()) > 0.1) {
                LabTrendAlert alert = LabTrendAlert.builder()
                    .patientId(patientId)
                    .labName("Creatinine")
                    .windowStart(window.getStart())
                    .windowEnd(window.getEnd())
                    .firstValue(firstValue)
                    .lastValue(lastValue)

            List<LabResultEvent> glucoseValues = new ArrayList<>();
            values.forEach(glucoseValues::add);
            
            if (glucoseValues.size() < 3) {
                return; // Need at least 3 readings
            }
            
            // Calculate statistics
            DoubleSummaryStatistics stats = glucoseValues.stream()
                .mapToDouble(LabResultEvent::getValue)
                .summaryStatistics();
            
            double mean = stats.getAverage();
            double min = stats.getMin();
            double max = stats.getMax();
            double range = max - min;
            
            // Calculate standard deviation (variability)
            double variance = glucoseValues.stream()
                .mapToDouble(lab -> Math.pow(lab.getValue() - mean, 2))
                .average()
                .orElse(0.0);
            double stdDev = Math.sqrt(variance);
            
            // Calculate coefficient of variation (CV)
            double cv = (stdDev / mean) * 100;
            
            // Detect concerning patterns
            boolean hasHypoglycemia = min < 70;
            boolean hasHyperglycemia = max > 200;
            boolean highVariability = cv > 36; // CV >36% indicates high variability
            
            if (hasHypoglycemia || hasHyperglycemia || highVariability) {
                String interpretation = interpretGlucosePattern(
                    mean, min, max, cv, hasHypoglycemia, hasHyperglycemia, highVariability
                );

                    .patientId(patientId)

                    .windowStart(window.getStart())
                    .windowEnd(window.getEnd())
                    .meanValue(mean)
                    .minValue(min)
                    .maxValue(max)
                    .standardDeviation(stdDev)
                    .coefficientOfVariation(cv)
                    .dataPoints(glucoseValues.size())

                VitalType type = vital.getVitalType();
                if (!latestVitals.containsKey(type) ||

                int rrScore = calculateRRScore(rr);
                mewsScore += rrScore;
                scoreBreakdown.put("Respiratory_Rate", rrScore);

                }

                }

                scoreBreakdown.put("Blood_Pressure", sbpScore);
                
                if (sbpScore >= 2) {
                    concerningVitals.add(String.format("SBP: %.0f mmHg (Score: %d)", sbp, sbpScore));
                }
            }
            
            // Temperature
            if (latestVitals.containsKey(VitalType.TEMPERATURE)) {
                double temp = latestVitals.get(VitalType.TEMPERATURE).getValue();
                int tempScore = calculateTempScore(temp);
                mewsScore += tempScore;
                scoreBreakdown.put("Temperature", tempScore);

                    concerningVitals.add(String.format("Temp: %.1f°C (Score: %d)", temp, tempScore));
                }

            if (latestVitals.containsKey(VitalType.GLASGOW_COMA_SCALE)) {

        public void apply(String patientId,
                         TimeWindow window,
                         Iterable<VitalSignEvent> values,
                         Collector<VitalVariabilityAlert> out) {
            
            // Group vitals by type
            Map<VitalType, List<Double>> vitalsByType = new HashMap<>();
            
            for (VitalSignEvent vital : values) {
                vitalsByType.computeIfAbsent(vital.getVitalType(), k -> new ArrayList<>())
                           .add(vital.getValue());
            }
            
            // Calculate coefficient of variation for each vital type
            Map<VitalType, Double> cvMap = new HashMap<>();
            List<String> unstableVitals = new ArrayList<>();
            
            for (Map.Entry<VitalType, List<Double>> entry : vitalsByType.entrySet()) {
                VitalType type = entry.getKey();
                List<Double> values = entry.getValue();
                
                if (values.size() < 4) continue; // Need multiple readings
                
                double mean = values.stream().mapToDouble(Double::doubleValue).average().orElse(0);
                double variance = values.stream()
                    .mapToDouble(v -> Math.pow(v - mean, 2))
                    .average()
                    .orElse(0);
                double stdDev = Math.sqrt(variance);
                double cv = (stdDev / mean) * 100;
                
                cvMap.put(type, cv);
                
                // Thresholds for concerning variability
                boolean isUnstable = false;
                if (type == VitalType.HEART_RATE && cv > 15) isUnstable = true;
                if (type == VitalType.BLOOD_PRESSURE_SYSTOLIC && cv > 20) isUnstable = true;
                if (type == VitalType.RESPIRATORY_RATE && cv > 25) isUnstable = true;
                if (type == VitalType.OXYGEN_SATURATION && cv > 5) isUnstable = true;
                
                if (isUnstable) {
                    unstableVitals.add(String.format(
                        "%s: CV=%.1f%% (Mean: %.1f, StdDev: %.1f)",
                        type.name(), cv, mean, stdDev
                    ));
                }
            }
            
            // Generate alert if any vital is highly variable
            if (!unstableVitals.isEmpty()) {
                VitalVariabilityAlert alert = VitalVariabilityAlert.builder()
                    .patientId(patientId)
                    .unstableVitals(unstableVitals)
                    .coefficientsOfVariation(cvMap)
                    .windowStart(window.getStart())
                    .windowEnd(window.getEnd())

                        "Patient showing hemodynamic instability with fluctuating vital signs. " +
                        "May indicate:\n" +
                        "- Sepsis or septic shock\n" +
                        "- Cardiac dysrhythmia\n" +
                        "- Inadequate pain control\n" +
                        "- Medication titration needed\n" +
                        "Recommend close monitoring and physician evaluation."
                    )
                    .timestamp(Instant.now())
                    .build();
                
                out.collect(alert);

public class AggregateRiskScoring {
    
    /**
     * Daily aggregate risk score
     * Window: 24 hours (tumbling)
     */
    public static DataStream<DailyRiskScore> calculateDailyRiskScore(
        DataStream<VitalSignEvent> vitalStream,
        DataStream<LabResultEvent> labStream,
        DataStream<MedicationEvent> medicationStream) {
        
        // Union all event streams
        DataStream<Object> allEvents = vitalStream
            .map(v -> (Object) v)
            .union(labStream.map(l -> (Object) l))
            .union(medicationStream.map(m -> (Object) m));

            .keyBy(event -> {
                if (event instanceof VitalSignEvent) {
                    return ((VitalSignEvent) event).getPatientId();
                } else if (event instanceof LabResultEvent) {
                    return ((LabResultEvent) event).getPatientId();
                } else {
                    return ((MedicationEvent) event).getPatientId();
                }
            })
            .window(TumblingEventTimeWindows.of(Time.hours(24)))
            .apply(new DailyRiskScoringWindowFunction());
    }
    
    /**
     * Daily risk scoring window function
     */
    public static class DailyRiskScoringWindowFunction 
        implements WindowFunction<Object, DailyRiskScore, String, TimeWindow> {
        
        @Override
        public void apply(String patientId,
                         TimeWindow window,
                         Iterable<Object> values,
                         Collector<DailyRiskScore> out) {
            
            // Separate events by type
            List<VitalSignEvent> vitals = new ArrayList<>();
            List<LabResultEvent> labs = new ArrayList<>();
            List<MedicationEvent> medications = new ArrayList<>();
            
            for (Object event : values) {
                if (event instanceof VitalSignEvent) {
                    vitals.add((VitalSignEvent) event);
                } else if (event instanceof LabResultEvent) {
                    labs.add((LabResultEvent) event);
                } else if (event instanceof MedicationEvent) {
                    medications.add((MedicationEvent) event);
                }
            }
            
            // Calculate component scores
            int vitalStabilityScore = calculateVitalStabilityScore(vitals);
            int labAbnormalityScore = calculateLabAbnormalityScore(labs);
            int medicationComplexityScore = calculateMedicationComplexityScore(medications);
            
            // Aggregate risk score (0-100)
            int aggregateScore = (vitalStabilityScore * 40 / 100) +
                                (labAbnormalityScore * 35 / 100) +
                                (medicationComplexityScore * 25 / 100);
            
            // Determine risk level
            RiskLevel riskLevel;
            if (aggregateScore >= 75) riskLevel = RiskLevel.CRITICAL;
            else if (aggregateScore >= 50) riskLevel = RiskLevel.HIGH;
            else if (aggregateScore >= 25) riskLevel = RiskLevel.MODERATE;
            else riskLevel = RiskLevel.LOW;
            
            DailyRiskScore dailyRisk = DailyRiskScore.builder()
                .patientId(patientId)
                .date(LocalDate.ofInstant(Instant.ofEpochMilli(window.getStart()), 
                                         ZoneId.systemDefault()))
                .aggregateRiskScore(aggregateScore)
                .riskLevel(riskLevel)
                .vitalStabilityScore(vitalStabilityScore)
                .labAbnormalityScore(labAbnormalityScore)
                .medicationComplexityScore(medicationComplexityScore)
                .vitalSignCount(vitals.size())
                .labResultCount(labs.size())
                .medicationEventCount(medications.size())
                .recommendations(generateRiskRecommendations(riskLevel, aggregateScore))
                .build();
            
            out.collect(dailyRisk);
        }
        
        private int calculateVitalStabilityScore(List<VitalSignEvent> vitals) {
            if (vitals.isEmpty()) return 0;
            
            // Count abnormal vitals
            long abnormalCount = vitals.stream()
                .filter(VitalSignEvent::isAbnormal)
                .count();
            
            // Count critical vitals
            long criticalCount = vitals.stream()
                .filter(v -> v.getAbnormalityLevel() == AbnormalityLevel.CRITICAL)
                .count();
            
            double abnormalRate = (double) abnormalCount / vitals.size();
            double criticalRate = (double) criticalCount / vitals.size();
            
            // Score (0-100): Higher = more unstable
            return (int) Math.min(100, (abnormalRate * 50) + (criticalRate * 100));
        }
        
        private int calculateLabAbnormalityScore(List<LabResultEvent> labs) {
            if (labs.isEmpty()) return 0;
            
            long criticalLabs = labs.stream()
                .filter(LabResultEvent::isCritical)
                .count();
            
            long abnormalLabs = labs.stream()
                .filter(lab -> lab.getValue() < lab.getReferenceMin() || 
                              lab.getValue() > lab.getReferenceMax())
                .count();
            
            double criticalRate = (double) criticalLabs / labs.size();
            double abnormalRate = (double) abnormalLabs / labs.size();
            
            // Score (0-100)
            return (int) Math.min(100, (abnormalRate * 40) + (criticalRate * 120));
        }
        
        private int calculateMedicationComplexityScore(List<MedicationEvent> medications) {
            // Count unique medications
            long uniqueMeds = medications.stream()
                .map(MedicationEvent::getMedicationCode)
                .distinct()
                .count();
            
            // Count high-risk medications (anticoagulants, insulin, opioids)
            long highRiskMeds = medications.stream()
                .filter(this::isHighRiskMedication)
                .map(MedicationEvent::getMedicationCode)
                .distinct()
                .count();
            
            // Count missed doses
            long missedDoses = medications.stream()
                .filter(m -> m.getMedicationEventType() == MedicationEventType.MISSED)
                .count();
            
            // Score (0-100): Complexity + missed doses
            int complexityScore = (int) Math.min(50, uniqueMeds * 5 + highRiskMeds * 10);
            int adherenceScore = (int) Math.min(50, missedDoses * 15);
            
            return complexityScore + adherenceScore;
        }

                   code.startsWith("1191");   // Heparin

    /**
     * Enrich vital signs with semantic knowledge
     */
    public static class VitalEnrichmentFunction 
        extends KeyedBroadcastProcessFunction<String, VitalSignEvent, 
                                             SemanticKnowledge, EnrichedVitalEvent> {
        
        @Override
        public void processElement(VitalSignEvent vital,
                                   ReadOnlyContext ctx,
                                   Collector<EnrichedVitalEvent> out) throws Exception {
            
            // Get current semantic mesh state
            ReadOnlyBroadcastState<String, SemanticKnowledge> meshState = 
                ctx.getBroadcastState(SEMANTIC_MESH_STATE_DESCRIPTOR);
            
            // Get vital sign thresholds
            SemanticKnowledge thresholds = meshState.get("vital_thresholds");
            
            EnrichedVitalEvent enriched = new EnrichedVitalEvent(vital);
            
            if (thresholds != null) {
                // Check against thresholds
                Map<String, Object> vitalThresholds = 
                    thresholds.getVitalThresholds().get(vital.getVitalType());
                
                if (vitalThresholds != null) {
                    double min = (double) vitalThresholds.get("min");
                    double max = (double) vitalThresholds.get("max");
                    double critical = (double) vitalThresholds.get("critical");
                    
                    // Determine abnormality level
                    if (vital.getValue() < min || vital.getValue() > max) {
                        enriched.setIsAbnormal(true);
                        
                        if (Math.abs(vital.getValue() - critical) < 
                            Math.abs(vital.getValue() - max)) {
                            enriched.setAbnormalityLevel(AbnormalityLevel.CRITICAL);
                        } else {
                            enriched.setAbnormalityLevel(AbnormalityLevel.ABNORMAL);
                        }
                    }
                    
                    enriched.setReferenceMin(min);
                    enriched.setReferenceMax(max);
                    enriched.setCriticalThreshold(critical);
                }
            }
            
            out.collect(enriched);
        }
        
        @Override
        public void processBroadcastElement(SemanticKnowledge knowledge,
                                           Context ctx,
                                           Collector<EnrichedVitalEvent> out) throws Exception {
            // Update broadcast state
            BroadcastState<String, SemanticKnowledge> meshState = 
                ctx.getBroadcastState(SEMANTIC_MESH_STATE_DESCRIPTOR);
            
            meshState.put(knowledge.getKnowledgeType(), knowledge);
        }
    }
    
    /**
     * Enrich medications with drug interaction knowledge
     */
    public static class MedicationEnrichmentFunction 
        extends KeyedBroadcastProcessFunction<String, MedicationEvent, 
                                             SemanticKnowledge, EnrichedMedicationEvent> {
        
        private final MapStateDescriptor<String, Set<String>> PATIENT_MEDS_STATE =
            new MapStateDescriptor<>("patient-medications", String.class, 
                                     new TypeHint<Set<String>>(){}.getTypeInfo());
        
        @Override
        public void processElement(MedicationEvent med,
                                   ReadOnlyContext ctx,
                                   Collector<EnrichedMedicationEvent> out) throws Exception {
            
            // Get current semantic mesh
            ReadOnlyBroadcastState<String, SemanticKnowledge> meshState = 
                ctx.getBroadcastState(SEMANTIC_MESH_STATE_DESCRIPTOR);
            
            SemanticKnowledge drugInteractions = meshState.get("drug_interactions");
            
            EnrichedMedicationEvent enriched = new EnrichedMedicationEvent(med);
            
            if (med.getMedicationEventType() == MedicationEventType.ORDERED ||
                med.getMedicationEventType() == MedicationEventType.ADMINISTERED) {
                
                // Get patient's current medications
                Set<String> currentMeds = ctx.getPartitionedState(PATIENT_MEDS_STATE)
                    .get(med.getPatientId());
                
                if (currentMeds == null) {
                    currentMeds = new HashSet<>();
                }
                
                // Check for drug interactions
                if (drugInteractions != null) {
                    List<DrugInteraction> interactions = new ArrayList<>();
                    
                    Map<String, List<String>> interactionMatrix = 
                        drugInteractions.getDrugInteractions();
                    
                    List<String> interactsWith = 
                        interactionMatrix.get(med.getMedicationCode());
                    
                    if (interactsWith != null) {
                        for (String currentMed : currentMeds) {
                            if (interactsWith.contains(currentMed)) {
                                interactions.add(new DrugInteraction(
                                    med.getMedicationCode(),
                                    currentMed,
                                    InteractionSeverity.MAJOR,
                                    "Known drug interaction - review before administration"
                                ));
                            }
                        }
                    }
                    
                    enriched.setDetectedInteractions(interactions);
                }
                
                // Update patient's medication list
                currentMeds.add(med.getMedicationCode());
                ctx.getPartitionedState(PATIENT_MEDS_STATE)
                   .put(med.getPatientId(), currentMeds);
            }
            
            out.collect(enriched);
        }
        
        @Override
        public void processBroadcastElement(SemanticKnowledge knowledge,
                                           Context ctx,
                                           Collector<EnrichedMedicationEvent> out) throws Exception {
            BroadcastState<String, SemanticKnowledge> meshState = 
                ctx.getBroadcastState(SEMANTIC_MESH_STATE_DESCRIPTOR);
            
            meshState.put(knowledge.getKnowledgeType(), knowledge);
        }
    }
    
    // ═══════════════════════════════════════════════════════
    // ALERT DEDUPLICATION FUNCTION
    // ═══════════════════════════════════════════════════════
    
    /**
     * Deduplicate and prioritize alerts per patient
     * Prevents alert fatigue by suppressing redundant alerts
     */
    public static class AlertDeduplicationFunction 
        extends KeyedProcessFunction<String, ClinicalAlert, ClinicalAlert> {
        
        private final MapStateDescriptor<String, ClinicalAlert> RECENT_ALERTS_STATE =
            new MapStateDescriptor<>("recent-alerts", String.class, ClinicalAlert.class);
        
        @Override
        public void processElement(ClinicalAlert alert,
                                   Context ctx,
                                   Collector<ClinicalAlert> out) throws Exception {
            
            MapState<String, ClinicalAlert> recentAlerts = 
                getRuntimeContext().getMapState(RECENT_ALERTS_STATE);
            
            String alertKey = alert.getAlertType().name();
            ClinicalAlert previousAlert = recentAlerts.get(alertKey);
            
            boolean shouldEmit = true;
            
            if (previousAlert != null) {
                // Check if similar alert was recently fired
                Duration timeSincePrevious = Duration.between(
                    previousAlert.getTimestamp(),
                    alert.getTimestamp()
                );
                
                // Suppression rules
                if (alert.getSeverity() == AlertSeverity.LOW && 
                    timeSincePrevious.toMinutes() < 60) {
                    shouldEmit = false; // Suppress low-severity repeats within 1 hour
                } else if (alert.getSeverity() == AlertSeverity.MEDIUM && 
                          timeSincePrevious.toMinutes() < 30) {
                    shouldEmit = false; // Suppress medium-severity repeats within 30 min
                } else if (alert.getSeverity() == AlertSeverity.HIGH && 
                          timeSincePrevious.toMinutes() < 15) {
                    shouldEmit = false; // Suppress high-severity repeats within 15 min
                }
                // CRITICAL alerts always emit (no suppression)
            }
            
            if (shouldEmit) {
                // Update state
                recentAlerts.put(alertKey, alert);
                
                // Set alert priority score for downstream routing
                int priorityScore = calculatePriorityScore(alert);
                alert.setPriorityScore(priorityScore);
                
                out.collect(alert);
                
                // Clean up old alerts (TTL = 4 hours)
                ctx.timerService().registerEventTimeTimer(
                    alert.getTimestamp().toEpochMilli() + Duration.ofHours(4).toMillis()
                );
            }
        }
        
        @Override
        public void onTimer(long timestamp,
                           OnTimerContext ctx,
                           Collector<ClinicalAlert> out) throws Exception {
            // Clean up expired alert state
            MapState<String, ClinicalAlert> recentAlerts = 
                getRuntimeContext().getMapState(RECENT_ALERTS_STATE);
            
            Iterator<Map.Entry<String, ClinicalAlert>> iterator = 
                recentAlerts.entries().iterator();
            
            while (iterator.hasNext()) {
                Map.Entry<String, ClinicalAlert> entry = iterator.next();
                if (entry.getValue().getTimestamp().toEpochMilli() < 
                    timestamp - Duration.ofHours(4).toMillis()) {
                    iterator.remove();
                }
            }
        }
        
        private int calculatePriorityScore(ClinicalAlert alert) {
            int score = 0;
            
            // Severity weight (0-40 points)
            switch (alert.getSeverity()) {
                case CRITICAL: score += 40; break;
                case HIGH: score += 30; break;
                case MEDIUM: score += 20; break;
                case LOW: score += 10; break;
            }
            
            // Alert type weight (0-30 points)
            switch (alert.getAlertType()) {
                case SEPSIS_EARLY_WARNING:
                case ACUTE_KIDNEY_INJURY:
                case RAPID_DETERIORATION:
                    score += 30;
                    break;
                case MEDICATION_ADHERENCE:
                case DRUG_LAB_MONITORING:
                    score += 20;
                    break;
                default:
                    score += 10;
            }
            
            // Time sensitivity (0-30 points)
            if (alert.getExpiresAt() != null) {
                Duration timeToExpiry = Duration.between(
                    Instant.now(),
                    alert.getExpiresAt()
                );
                
                if (timeToExpiry.toMinutes() < 15) {
                    score += 30; // Expires very soon
                } else if (timeToExpiry.toHours() < 1) {
                    score += 20; // Expires within 1 hour
                } else if (timeToExpiry.toHours() < 4) {
                    score += 10; // Expires within 4 hours
                }
            }
            
            return Math.min(100, score);
        }
    }

```java
package com.hospital.flink.loadtest;


/**
 * Load test generator for Module 4
 */
public class PatternDetectionLoadTest {
    
    public static void main(String[] args) throws Exception {
        
        // Configuration
        int patientsToSimulate = 1000;
        int eventsPerSecond = 10000;
        Duration testDuration = Duration.ofHours(2);
        
        // Kafka producer
        Properties props = new Properties();
        props.put("bootstrap.servers", "kafka-1:9092,kafka-2:9092");
        props.put("key.serializer", "org.apache.kafka.common.serialization.StringSerializer");
        props.put("value.serializer", "org.apache.kafka.common.serialization.StringSerializer");
        
        KafkaProducer<String, String> producer = new KafkaProducer<>(props);
        
        // Generate patient population
        List<Patient> patients = generatePatientPopulation(patientsToSimulate);
        
        // Event generator
        Random random = new Random();
        Instant startTime = Instant.now();
        long eventsGenerated = 0;
        
        System.out.println("Starting load test...");
        System.out.println("Patients: " + patientsToSimulate);
        System.out.println("Target rate: " + eventsPerSecond + " events/sec");
        System.out.println("Duration: " + testDuration.toHours() + " hours");
        
        while (Duration.between(startTime, Instant.now()).compareTo(testDuration) < 0) {
            
            long batchStart = System.currentTimeMillis();
            
            // Generate events for 1 second
            for (int i = 0; i < eventsPerSecond; i++) {
                Patient patient = patients.get(random.nextInt(patients.size()));
                
                // Randomly generate event type
                double eventTypeRoll = random.nextDouble();
                
                if (eventTypeRoll < 0.60) {
                    // 60% vital signs
                    VitalSignEvent vital = generateVitalSign(patient, random);
                    producer.send(new ProducerRecord<>(
                        "ehr-events-vitals",
                        patient.getPatientId(),
                        toJson(vital)
                    ));
                    
                } else if (eventTypeRoll < 0.80) {
                    // 20% lab results
                    LabResultEvent lab = generateLabResult(patient, random);
                    producer.send(new ProducerRecord<>(
                        "ehr-events-labs",
                        patient.getPatientId(),
                        toJson(lab)
                    ));
                    
                } else {
                    // 20% medication events
                    MedicationEvent med = generateMedicationEvent(patient, random);
                    producer.send(new ProducerRecord<>(
                        "ehr-events-medications",
                        patient.getPatientId(),
                        toJson(med)
                    ));
                }
                
                eventsGenerated++;
            }
            
            // Inject clinical scenarios (10% of patients)
            if (random.nextDouble() < 0.10) {
                Patient patient = patients.get(random.nextInt(patients.size()));
                injectClinicalScenario(producer, patient, random);
            }
            
            // Rate limiting
            long elapsed = System.currentTimeMillis() - batchStart;
            if (elapsed < 1000) {
                Thread.sleep(1000 - elapsed);
            }
            
            // Progress update
            if (eventsGenerated % 100000 == 0) {
                System.out.printf("Generated %,d events (%.1f hours elapsed)%n",
                                eventsGenerated,
                                Duration.between(startTime, Instant.now()).toMinutes() / 60.0);
            }
        }
        
        producer.flush();
        producer.close();
        
        System.out.println("\nLoad test complete!");
        System.out.println("Total events generated: " + String.format("%,d", eventsGenerated));
        System.out.println("Average rate: " + 
                          (eventsGenerated / testDuration.toSeconds()) + " events/sec");
    }
    
    /**
     * Inject realistic clinical scenarios
     */
    private static void injectClinicalScenario(KafkaProducer<String, String> producer,
                                               Patient patient,
                                               Random random) {
        double scenarioRoll = random.nextDouble();
        
        if (scenarioRoll < 0.30) {
            // Sepsis scenario
            injectSepsisScenario(producer, patient);
            
        } else if (scenarioRoll < 0.50) {
            // AKI scenario
            injectAKIScenario(producer, patient);
            
        } else if (scenarioRoll < 0.70) {
            // Rapid deterioration
            injectDeteriorationScenario(producer, patient);
            
        } else {
            // Medication adherence issue
            injectMedicationAdherenceScenario(producer, patient);
        }
    }
    
    private static void injectSepsisScenario(KafkaProducer<String, String> producer, 
                                            Patient patient) {
        Instant now = Instant.now();
        
        // Tachycardia
        VitalSignEvent hr = createVitalEvent(patient.getPatientId(), 
                                            VitalType.HEART_RATE, 
                                            110 + new Random().nextInt(20), 
                                            now);
        producer.send(new ProducerRecord<>("ehr-events-vitals", 
                                          patient.getPatientId(), 
                                          toJson(hr)));
        
        // Hypotension (90 minutes later)
        VitalSignEvent bp = createVitalEvent(patient.getPatientId(),
                                            VitalType.BLOOD_PRESSURE_SYSTOLIC,
                                            85 + new Random().nextInt(10),
                                            now.plus(Duration.ofMinutes(90)));
        producer.send(new ProducerRecord<>("ehr-events-vitals",
                                          patient.getPatientId(),
                                          toJson(bp)));
        
        System.out.println("⚠️  Injected sepsis scenario for patient: " + patient.getPatientId());
    }
