## 📦 **Implementation: Three-Layer Safety Net**


### **Layer 1: Instant State-Based Rules (Triage Nurse)**


**Purpose:** Catch immediate high-risk states on ANY single event  
**Technology:** Stateless function (synchronous)  
**Latency:** <10ms


```java
package com.hospital.flink.reasoning;


import org.apache.flink.streaming.api.functions.ProcessFunction;
import org.apache.flink.util.Collector;


/**
 * LAYER 1: Instant State-Based Clinical Rules
 * 
 * This is the "Triage Nurse" - catches patients who arrive in critical condition
 * Fires immediately on first event, no history needed
 * 
 * CRITICAL SAFETY FEATURE: Ensures no patient is missed due to lack of baseline
 */
public class InstantStateReasoner extends ProcessFunction<EnrichedEvent, ClinicalAlert> {
    
    @Override
    public void processElement(
        EnrichedEvent event,
        Context ctx,
        Collector<ClinicalAlert> out) throws Exception {
        
        // ═══════════════════════════════════════════════════════════
        // RULE 1: CRITICAL STATE DETECTION
        // ═══════════════════════════════════════════════════════════
        
        if (isCriticalState(event)) {
            ClinicalAlert alert = ClinicalAlert.builder()
                .alertId(generateAlertId("CRITICAL_STATE"))
                .alertType("CRITICAL_STATE_DETECTED")
                .patientId(event.getPatientId())
                .severity("CRITICAL")
                .priority(1)  // Highest priority
                .source("INSTANT_STATE_RULE")
                .message(buildCriticalStateMessage(event))
                .clinicalContext(extractClinicalContext(event))
                .recommendations(getCriticalStateRecommendations(event))
                .confidence(0.95)  // High confidence (direct measurement)
                .timestamp(System.currentTimeMillis())
                .requiresImmediateAction(true)
                .escalationRequired(true)
                .build();
            
            out.collect(alert);
            
            // Also emit metrics
            ctx.output(OutputTags.METRICS_TAG, 
                new AlertMetric("critical_state_detected", event.getPatientId()));
        }
        
        // ═══════════════════════════════════════════════════════════
        // RULE 2: HIGH-RISK STATE DETECTION
        // ═══════════════════════════════════════════════════════════
        
        else if (isHighRiskState(event)) {
            ClinicalAlert alert = ClinicalAlert.builder()
                .alertId(generateAlertId("HIGH_RISK_STATE"))
                .alertType("HIGH_RISK_STATE_DETECTED")
                .patientId(event.getPatientId())
                .severity("HIGH")
                .priority(2)
                .source("INSTANT_STATE_RULE")
                .message(buildHighRiskMessage(event))
                .clinicalContext(extractClinicalContext(event))
                .recommendations(getHighRiskRecommendations(event))
                .confidence(0.90)
                .timestamp(System.currentTimeMillis())
                .requiresImmediateAction(true)
                .build();
            
            out.collect(alert);
        }
        
        // ═══════════════════════════════════════════════════════════
        // RULE 3: SEPSIS CRITERIA (qSOFA ≥ 2)
        // ═══════════════════════════════════════════════════════════
        
        else if (meetsSepsisCriteria(event)) {
            ClinicalAlert alert = ClinicalAlert.builder()
                .alertId(generateAlertId("SEPSIS_CRITERIA"))
                .alertType("SEPSIS_CRITERIA_MET")
                .patientId(event.getPatientId())
                .severity("HIGH")
                .priority(2)
                .source("INSTANT_STATE_RULE")
                .message(buildSepsisMessage(event))
                .clinicalContext(extractClinicalContext(event))
                .recommendations(getSepsisRecommendations())
                .confidence(0.88)
                .timestamp(System.currentTimeMillis())
                .requiresImmediateAction(true)
                .build();
            
            out.collect(alert);
        }
        
        // ═══════════════════════════════════════════════════════════
        // RULE 4: RESPIRATORY FAILURE
        // ═══════════════════════════════════════════════════════════
        
        else if (hasRespiratoryFailure(event)) {
            ClinicalAlert alert = ClinicalAlert.builder()
                .alertId(generateAlertId("RESPIRATORY_FAILURE"))
                .alertType("RESPIRATORY_FAILURE")
                .patientId(event.getPatientId())
                .severity("CRITICAL")
                .priority(1)
                .source("INSTANT_STATE_RULE")
                .message(buildRespiratoryFailureMessage(event))
                .clinicalContext(extractClinicalContext(event))
                .recommendations(getRespiratoryFailureRecommendations())
                .confidence(0.92)
                .timestamp(System.currentTimeMillis())
                .requiresImmediateAction(true)
                .escalationRequired(true)
                .build();
            
            out.collect(alert);
        }
        
        // ═══════════════════════════════════════════════════════════
        // RULE 5: SHOCK STATE
        // ═══════════════════════════════════════════════════════════
        
        else if (isInShock(event)) {
            ClinicalAlert alert = ClinicalAlert.builder()
                .alertId(generateAlertId("SHOCK_STATE"))
                .alertType("SHOCK_STATE_DETECTED")
                .patientId(event.getPatientId())
                .severity("CRITICAL")
                .priority(1)
                .source("INSTANT_STATE_RULE")
                .message(buildShockMessage(event))
                .clinicalContext(extractClinicalContext(event))
                .recommendations(getShockRecommendations())
                .confidence(0.93)
                .timestamp(System.currentTimeMillis())
                .requiresImmediateAction(true)
                .escalationRequired(true)
                .build();
            
            out.collect(alert);
        }
    }
    
    /**
     * CRITICAL STATE DETECTION LOGIC
     * 
     * Triggers if ANY of:
     * - NEWS2 ≥ 10 (immediate clinical response required)
     * - qSOFA ≥ 2 (presumed sepsis)
     * - Combined acuity score ≥ 0.85
     * - Risk level = "critical"
     */
    private boolean isCriticalState(EnrichedEvent event) {
        PatientState state = event.getPatientState();
        
        return state.getNews2Score() >= 10 ||
               state.getQsofaScore() >= 2 ||
               state.getCombinedAcuityScore() >= 0.85 ||
               "critical".equalsIgnoreCase(state.getRiskLevel());
    }
    
    /**
     * HIGH-RISK STATE DETECTION LOGIC
     * 
     * Triggers if:
     * - NEWS2 = 7-9 (urgent response required)
     * - Combined acuity score ≥ 0.65
     * - Risk level = "high"
     */
    private boolean isHighRiskState(EnrichedEvent event) {
        PatientState state = event.getPatientState();
        
        int news2 = state.getNews2Score();
        double acuity = state.getCombinedAcuityScore();
        
        return (news2 >= 7 && news2 < 10) ||
               (acuity >= 0.65 && acuity < 0.85) ||
               "high".equalsIgnoreCase(state.getRiskLevel());
    }
    
    /**
     * SEPSIS CRITERIA DETECTION
     * 
     * qSOFA ≥ 2 indicates presumed sepsis
     * Components:
     * - Respiratory rate ≥ 22
     * - Altered mental status
     * - Systolic BP ≤ 100
     */
    private boolean meetsSepsisCriteria(EnrichedEvent event) {
        return event.getPatientState().getQsofaScore() >= 2;
    }
    
    /**
     * RESPIRATORY FAILURE DETECTION
     * 
     * Critical oxygen delivery failure
     */
    private boolean hasRespiratoryFailure(EnrichedEvent event) {
        Map<String, Object> vitals = event.getPatientState().getLatestVitals();
        
        double spO2 = getDoubleValue(vitals, "oxygensaturation");
        double respRate = getDoubleValue(vitals, "respiratoryrate");
        
        return spO2 <= 88 ||  // Severe hypoxemia
               respRate >= 30 ||  // Severe tachypnea
               respRate <= 8;  // Severe bradypnea
    }
    
    /**
     * SHOCK STATE DETECTION
     * 
     * Inadequate tissue perfusion
     */
    private boolean isInShock(EnrichedEvent event) {
        Map<String, Object> vitals = event.getPatientState().getLatestVitals();
        
        double systolicBP = getDoubleValue(vitals, "systolicbp");
        double heartRate = getDoubleValue(vitals, "heartrate");
        
        // Shock index (HR/SBP) > 1.0 indicates shock
        double shockIndex = heartRate / systolicBP;
        
        return systolicBP < 90 ||  // Hypotension
               shockIndex > 1.0 ||  // Shock index
               event.getPatientState().getRiskIndicators().isOnVasopressors();
    }
    
    // ═══════════════════════════════════════════════════════════
    // MESSAGE BUILDERS
    // ═══════════════════════════════════════════════════════════
    
    private String buildCriticalStateMessage(EnrichedEvent event) {
        PatientState state = event.getPatientState();
        
        return String.format(
            "CRITICAL STATE DETECTED - Patient requires immediate clinical evaluation. " +
            "NEWS2: %d, qSOFA: %d, Combined Acuity: %.2f, Risk Level: %s",
            state.getNews2Score(),
            state.getQsofaScore(),
            state.getCombinedAcuityScore(),
            state.getRiskLevel()
        );
    }
    
    private String buildHighRiskMessage(EnrichedEvent event) {
        PatientState state = event.getPatientState();
        
        return String.format(
            "HIGH-RISK STATE - Urgent clinical review required. " +
            "NEWS2: %d, Combined Acuity: %.2f",
            state.getNews2Score(),
            state.getCombinedAcuityScore()
        );
    }
    
    private String buildSepsisMessage(EnrichedEvent event) {
        PatientState state = event.getPatientState();
        
        return String.format(
            "SEPSIS CRITERIA MET - qSOFA ≥ 2 indicates presumed sepsis. " +
            "qSOFA Score: %d. Consider sepsis bundle initiation.",
            state.getQsofaScore()
        );
    }
    
    private String buildRespiratoryFailureMessage(EnrichedEvent event) {
        Map<String, Object> vitals = event.getPatientState().getLatestVitals();
        
        return String.format(
            "RESPIRATORY FAILURE - Critical oxygen delivery compromise. " +
            "SpO2: %.1f%%, Respiratory Rate: %.0f/min",
            getDoubleValue(vitals, "oxygensaturation"),
            getDoubleValue(vitals, "respiratoryrate")
        );
    }
    
    private String buildShockMessage(EnrichedEvent event) {
        Map<String, Object> vitals = event.getPatientState().getLatestVitals();
        
        double systolicBP = getDoubleValue(vitals, "systolicbp");
        double heartRate = getDoubleValue(vitals, "heartrate");
        double shockIndex = heartRate / systolicBP;
        
        return String.format(
            "SHOCK STATE - Inadequate tissue perfusion. " +
            "BP: %.0f mmHg, HR: %.0f bpm, Shock Index: %.2f",
            systolicBP, heartRate, shockIndex
        );
    }
    
    // ═══════════════════════════════════════════════════════════
    // RECOMMENDATION BUILDERS
    // ═══════════════════════════════════════════════════════════
    
    private List<String> getCriticalStateRecommendations(EnrichedEvent event) {
        List<String> recommendations = new ArrayList<>();
        
        recommendations.add("IMMEDIATE: Bedside clinical assessment required");
        recommendations.add("Consider ICU/HDU transfer");
        recommendations.add("Notify attending physician immediately");
        recommendations.add("Establish IV access if not present");
        recommendations.add("Continuous monitoring required");
        
        if (event.getPatientState().getQsofaScore() >= 2) {
            recommendations.add("SEPSIS SUSPECTED: Consider sepsis bundle");
        }
        
        return recommendations;
    }
    
    private List<String> getHighRiskRecommendations(EnrichedEvent event) {
        return Arrays.asList(
            "URGENT: Clinical review within 30 minutes",
            "Increase monitoring frequency",
            "Notify charge nurse",
            "Reassess clinical status"
        );
    }
    
    private List<String> getSepsisRecommendations() {
        return Arrays.asList(
            "Initiate sepsis bundle immediately",
            "Blood cultures x2 before antibiotics",
            "Administer broad-spectrum antibiotics within 1 hour",
            "Measure serum lactate",
            "Initiate IV fluid resuscitation if lactate ≥2 mmol/L",
            "Notify ICU team"
        );
    }
    
    private List<String> getRespiratoryFailureRecommendations() {
        return Arrays.asList(
            "CRITICAL: Assess airway, breathing, circulation",
            "Consider supplemental oxygen or escalation",
            "Prepare for possible intubation",
            "Notify respiratory therapy STAT",
            "Arterial blood gas if not recent"
        );
    }
    
    private List<String> getShockRecommendations() {
        return Arrays.asList(
            "CRITICAL: Fluid resuscitation",
            "Establish large-bore IV access",
            "Consider vasopressor support",
            "Central venous access may be needed",
            "Urgent ICU consultation"
        );
    }
    
    // Helper methods
    
    private ClinicalContext extractClinicalContext(EnrichedEvent event) {
        return ClinicalContext.builder()
            .patientId(event.getPatientId())
            .department(event.getPatientState().getDepartment())
            .news2Score(event.getPatientState().getNews2Score())
            .qsofaScore(event.getPatientState().getQsofaScore())
            .riskLevel(event.getPatientState().getRiskLevel())
            .latestVitals(event.getPatientState().getLatestVitals())
            .riskIndicators(event.getPatientState().getRiskIndicators())
            .build();
    }
    
    private String generateAlertId(String alertType) {
        return alertType + "-" + System.currentTimeMillis() + "-" + 
               UUID.randomUUID().toString().substring(0, 8);
    }
    
    private double getDoubleValue(Map<String, Object> map, String key) {
        Object value = map.get(key);
        if (value == null) return 0.0;
        if (value instanceof Number) {
            return ((Number) value).doubleValue();
        }
        return 0.0;
    }
}

## 🔧 **Integration: Reasoning Orchestrator**


```java
package com.hospital.flink.reasoning;


import org.apache.flink.streaming.api.datastream.DataStream;
import org.apache.flink.streaming.api.datastream.SingleOutputStreamOperator;


/**
 * REASONING ORCHESTRATOR
 * 
 * Coordinates all three layers of clinical intelligence:
 * 1. Instant State Rules (Triage Nurse)
 * 2. Pattern-Based CEP (ICU Monitor)
 * 3. Predictive ML (Crystal Ball)
 * 
 * Ensures comprehensive patient safety coverage
 */
public class ClinicalReasoningOrchestrator {
    
    public static DataStream<ClinicalAlert> orchestrate(
        DataStream<EnrichedEvent> enrichedEvents,
        StreamExecutionEnvironment env) {
        
        // ═══════════════════════════════════════════════════════════
        // LAYER 1: INSTANT STATE RULES (Synchronous, <10ms)
        // ═══════════════════════════════════════════════════════════
        
        SingleOutputStreamOperator<ClinicalAlert> instantAlerts = enrichedEvents
            .process(new InstantStateReasoner())
            .name("instant-state-reasoning")
            .uid("instant-state-reasoning");
        
        // ═══════════════════════════════════════════════════════════
        // LAYER 2: PATTERN-BASED CEP (Stateful, real-time)
        // ═══════════════════════════════════════════════════════════
        
        DataStream<ClinicalAlert> patternAlerts = PatternBasedReasoner
            .detectPatterns(enrichedEvents)
            .name("pattern-based-reasoning")
            .uid("pattern-based-reasoning");
        
        // ═══════════════════════════════════════════════════════════
        // LAYER 3: PREDICTIVE ML (Stateless inference, <50ms)
        // ═══════════════════════════════════════════════════════════
        
        DataStream<ClinicalAlert> mlAlerts = PredictiveMLReasoner
            .generatePredictiveAlerts(enrichedEvents)
            .name("ml-predictive-reasoning")
            .uid("ml-predictive-reasoning");
        
        // ═══════════════════════════════════════════════════════════
        // MERGE ALL ALERT STREAMS
        // ═══════════════════════════════════════════════════════════
        
        DataStream<ClinicalAlert> allAlerts = instantAlerts
            .union(patternAlerts)
            .union(mlAlerts);
        
        // ═══════════════════════════════════════════════════════════
        // DEDUPLICATION & PRIORITIZATION
        // ═══════════════════════════════════════════════════════════
        
        DataStream<ClinicalAlert> dedupedAlerts = allAlerts
            .keyBy(ClinicalAlert::getPatientId)
            .process(new AlertDeduplicationFunction())
            .name("alert-deduplication");
        
        // ═══════════════════════════════════════════════════════════
        // ALERT ENHANCEMENT
        // ═══════════════════════════════════════════════════════════
        
        DataStream<ClinicalAlert> enhancedAlerts = dedupedAlerts
            .process(new AlertEnhancementFunction())
            .name("alert-enhancement");
        
        return enhancedAlerts;
    }
}

## 📊 **Alert Deduplication Logic**


```java
/**
 * Alert Deduplication Function
 * 
 * Prevents alert storms when multiple layers fire for same patient
 * Prioritizes alerts and merges evidence
 */
public class AlertDeduplicationFunction 
    extends KeyedProcessFunction<String, ClinicalAlert, ClinicalAlert> {
    
    private transient ValueState<ClinicalAlert> lastAlertState;
    private transient MapState<String, Long> recentAlertsState;
    
    private static final long DEDUP_WINDOW_MS = 5 * 60 * 1000;  // 5 minutes
    
    @Override
    public void open(Configuration parameters) {
        ValueStateDescriptor<ClinicalAlert> lastAlertDescriptor =
            new ValueStateDescriptor<>("last-alert", ClinicalAlert.class);
        lastAlertState = getRuntimeContext().getState(lastAlertDescriptor);
        
        MapStateDescriptor<String, Long> recentAlertsDescriptor =
            new MapStateDescriptor<>("recent-alerts", String.class, Long.class);
        recentAlertsState = getRuntimeContext().getMapState(recentAlertsDescriptor);
    }
    
    @Override
    public void processElement(
        ClinicalAlert alert,
        Context ctx,
        Collector<ClinicalAlert> out) throws Exception {
        
        long now = System.currentTimeMillis();
        String alertKey = alert.getAlertType();
        
        // Check if similar alert was recently fired
        Long lastFiredTime = recentAlertsState.get(alertKey);
        
        if (lastFiredTime != null && (now - lastFiredTime) < DEDUP_WINDOW_MS) {
            // Duplicate alert within window
            
            ClinicalAlert lastAlert = lastAlertState.value();
            
            // Merge evidence from multiple sources
            if (lastAlert != null) {
                ClinicalAlert mergedAlert = mergeAlerts(lastAlert, alert);
                out.collect(mergedAlert);
                lastAlertState.update(mergedAlert);
            }
        } else {
            // New alert
            out.collect(alert);
            lastAlertState.update(alert);
            recentAlertsState.put(alertKey, now);
        }
        
        // Schedule cleanup of old alerts
        ctx.timerService().registerProcessingTimeTimer(now + DEDUP_WINDOW_MS);
    }
    
    /**
     * Merge alerts from multiple sources
     * Combines evidence and increases confidence
     */
    private ClinicalAlert mergeAlerts(ClinicalAlert existing, ClinicalAlert new) {
        
        // Create merged alert with combined evidence
        List<String> sources = new ArrayList<>();
        sources.add(existing.getSource());
        sources.add(new.getSource());
        
        double combinedConfidence = Math.min(1.0,
            existing.getConfidence() * 0.6 + new.getConfidence() * 0.4);
        
        return ClinicalAlert.builder()
            .alertId(existing.getAlertId())
            .alertType(existing.getAlertType())

    private String getHighestSeverity(String sev1, String sev2) {
        List<String> severityOrder = Arrays.asList("LOW", "MODERATE", "HIGH", "CRITICAL");
        int idx1 = severityOrder.indexOf(sev1.toUpperCase());
        int idx2 = severityOrder.indexOf(sev2.toUpperCase());
        return severityOrder.get(Math.max(idx1, idx2));
    }
    
    private List<String> mergeRecommendations(List<String> rec1, List<String> rec2) {
        Set<String> merged = new LinkedHashSet<>(rec1);
        merged.addAll(rec2);
        return new ArrayList<>(merged);

            Map.Entry<String, Long> entry = iterator.next();
