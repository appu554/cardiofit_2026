# 🚀 **Module 6: Advanced Analytics & Predictive Dashboards - Complete Design**


## 📋 **Executive Summary**


**Goal:** Build real-time analytics dashboards and predictive visualizations that aggregate insights from all previous modules, providing actionable intelligence for clinical teams, hospital administrators, and quality improvement initiatives.


**Timeline:** 3-4 weeks (120-160 hours)  
**Complexity:** High (Real-time Analytics + Complex Visualizations + Multi-user Access)  
**Prerequisites:** Modules 1-5 Complete (Especially Module 5 ML predictions operational)

## 🎯 **Module 6 Objectives**


### **What We're Building:**


1. **Real-Time Analytics Engine** (1 week)
   - Materialized views with Flink SQL
   - Time-series aggregations (1min, 5min, 1hr, 24hr windows)
   - Population health metrics
   - Predictive trend analysis


2. **Interactive Dashboards** (1 week)
   - Executive dashboard (hospital-wide KPIs)
   - Clinical dashboard (unit/ward-level)
   - Patient-level dashboard (individual risk profiles)
   - Quality metrics dashboard (compliance, outcomes)


3. **Alerting & Notification System** (1 week)
   - Multi-channel notifications (SMS, Email, Push, Pager)
   - Smart alert routing (role-based)
   - Alert escalation workflows
   - Alert fatigue mitigation


4. **Reporting & Data Export** (3-4 days)
   - Automated report generation
   - Data export APIs (CSV, JSON, HL7 FHIR)
   - Audit trail and compliance reports
   - Performance benchmarking

## 📦 **Component 6A: Real-Time Analytics Engine**


### **6A.1: Materialized Views with Flink SQL**


```java
package com.hospital.flink.analytics;


import org.apache.flink.table.api.*;
import org.apache.flink.streaming.api.environment.StreamExecutionEnvironment;
import org.apache.flink.table.api.bridge.java.StreamTableEnvironment;


/**
 * Real-time analytics using Flink SQL
 * Creates materialized views for dashboards
 */
public class Module6_AnalyticsEngine {
    
    public static void main(String[] args) throws Exception {
        
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();
        env.setParallelism(16);
        env.enableCheckpointing(60000);
        
        StreamTableEnvironment tableEnv = StreamTableEnvironment.create(env);
        
        // ═══════════════════════════════════════════════════════════
        // DEFINE SOURCE TABLES (from Kafka topics)
        // ═══════════════════════════════════════════════════════════
        
        // Raw events from Module 1-3
        tableEnv.executeSql(
            "CREATE TABLE raw_events (" +
            "  patient_id STRING," +
            "  event_type STRING," +
            "  event_timestamp TIMESTAMP(3)," +
            "  data_type STRING," +
            "  value DOUBLE," +
            "  unit STRING," +
            "  department STRING," +
            "  risk_level STRING," +
            "  WATERMARK FOR event_timestamp AS event_timestamp - INTERVAL '5' MINUTE" +
            ") WITH (" +
            "  'connector' = 'kafka'," +
            "  'topic' = 'ehr-events-enriched'," +
            "  'properties.bootstrap.servers' = 'localhost:9092'," +
            "  'properties.group.id' = 'analytics-engine'," +
            "  'format' = 'json'" +
            ")"
        );
        
        // CEP alerts from Module 4
        tableEnv.executeSql(
            "CREATE TABLE cep_alerts (" +
            "  alert_id STRING," +
            "  patient_id STRING," +
            "  pattern_name STRING," +
            "  severity STRING," +
            "  confidence DOUBLE," +
            "  alert_timestamp TIMESTAMP(3)," +
            "  department STRING," +
            "  acknowledged BOOLEAN," +
            "  WATERMARK FOR alert_timestamp AS alert_timestamp - INTERVAL '1' MINUTE" +
            ") WITH (" +
            "  'connector' = 'kafka'," +
            "  'topic' = 'ehr-alerts-cep'," +
            "  'properties.bootstrap.servers' = 'localhost:9092'," +
            "  'format' = 'json'" +
            ")"
        );
        
        // ML predictions from Module 5
        tableEnv.executeSql(
            "CREATE TABLE ml_predictions (" +
            "  prediction_id STRING," +
            "  patient_id STRING," +
            "  model_type STRING," +
            "  probability DOUBLE," +
            "  risk_category STRING," +
            "  prediction_timestamp TIMESTAMP(3)," +
            "  department STRING," +
            "  WATERMARK FOR prediction_timestamp AS prediction_timestamp - INTERVAL '2' MINUTE" +
            ") WITH (" +
            "  'connector' = 'kafka'," +
            "  'topic' = 'ehr-ml-predictions'," +
            "  'properties.bootstrap.servers' = 'localhost:9092'," +
            "  'format' = 'json'" +
            ")"
        );
        
        // ═══════════════════════════════════════════════════════════
        // VIEW 1: Real-time Patient Census
        // ═══════════════════════════════════════════════════════════
        
        tableEnv.executeSql(
            "CREATE VIEW patient_census AS " +
            "SELECT " +
            "  department," +
            "  COUNT(DISTINCT patient_id) AS active_patients," +
            "  SUM(CASE WHEN risk_level = 'HIGH' THEN 1 ELSE 0 END) AS high_risk_count," +
            "  SUM(CASE WHEN risk_level = 'CRITICAL' THEN 1 ELSE 0 END) AS critical_count," +
            "  TUMBLE_START(event_timestamp, INTERVAL '1' MINUTE) AS window_start " +
            "FROM raw_events " +
            "WHERE event_type IN ('vitals', 'labs', 'medications') " +
            "GROUP BY department, TUMBLE(event_timestamp, INTERVAL '1' MINUTE)"
        );
        
        // Sink to Kafka for dashboard consumption
        tableEnv.executeSql(
            "CREATE TABLE patient_census_sink (" +
            "  department STRING," +
            "  active_patients BIGINT," +
            "  high_risk_count BIGINT," +
            "  critical_count BIGINT," +
            "  window_start TIMESTAMP(3)" +
            ") WITH (" +
            "  'connector' = 'kafka'," +
            "  'topic' = 'analytics-patient-census'," +
            "  'properties.bootstrap.servers' = 'localhost:9092'," +
            "  'format' = 'json'" +
            ")"
        );
        
        tableEnv.executeSql(
            "INSERT INTO patient_census_sink " +
            "SELECT * FROM patient_census"
        );
        
        // ═══════════════════════════════════════════════════════════
        // VIEW 2: Alert Performance Metrics (1-minute tumbling window)
        // ═══════════════════════════════════════════════════════════
        
        tableEnv.executeSql(
            "CREATE VIEW alert_metrics AS " +
            "SELECT " +
            "  department," +
            "  pattern_name," +
            "  COUNT(*) AS alert_count," +
            "  SUM(CASE WHEN acknowledged = true THEN 1 ELSE 0 END) AS acknowledged_count," +
            "  AVG(confidence) AS avg_confidence," +
            "  SUM(CASE WHEN severity = 'CRITICAL' THEN 1 ELSE 0 END) AS critical_alerts," +
            "  TUMBLE_START(alert_timestamp, INTERVAL '1' MINUTE) AS window_start " +
            "FROM cep_alerts " +
            "GROUP BY department, pattern_name, TUMBLE(alert_timestamp, INTERVAL '1' MINUTE)"
        );
        
        tableEnv.executeSql(
            "CREATE TABLE alert_metrics_sink (" +
            "  department STRING," +
            "  pattern_name STRING," +
            "  alert_count BIGINT," +
            "  acknowledged_count BIGINT," +
            "  avg_confidence DOUBLE," +
            "  critical_alerts BIGINT," +
            "  window_start TIMESTAMP(3)" +
            ") WITH (" +
            "  'connector' = 'kafka'," +
            "  'topic' = 'analytics-alert-metrics'," +
            "  'properties.bootstrap.servers' = 'localhost:9092'," +
            "  'format' = 'json'" +
            ")"
        );
        
        tableEnv.executeSql(
            "INSERT INTO alert_metrics_sink " +
            "SELECT * FROM alert_metrics"
        );
        
        // ═══════════════════════════════════════════════════════════
        // VIEW 3: ML Model Performance (5-minute windows)
        // ═══════════════════════════════════════════════════════════
        
        tableEnv.executeSql(
            "CREATE VIEW ml_performance AS " +
            "SELECT " +
            "  model_type," +
            "  department," +
            "  COUNT(*) AS prediction_count," +
            "  AVG(probability) AS avg_probability," +
            "  SUM(CASE WHEN risk_category = 'HIGH' THEN 1 ELSE 0 END) AS high_risk_predictions," +
            "  SUM(CASE WHEN risk_category = 'CRITICAL' THEN 1 ELSE 0 END) AS critical_predictions," +
            "  TUMBLE_START(prediction_timestamp, INTERVAL '5' MINUTE) AS window_start " +
            "FROM ml_predictions " +
            "GROUP BY model_type, department, TUMBLE(prediction_timestamp, INTERVAL '5' MINUTE)"
        );
        
        tableEnv.executeSql(
            "CREATE TABLE ml_performance_sink (" +
            "  model_type STRING," +
            "  department STRING," +
            "  prediction_count BIGINT," +
            "  avg_probability DOUBLE," +
            "  high_risk_predictions BIGINT," +
            "  critical_predictions BIGINT," +
            "  window_start TIMESTAMP(3)" +
            ") WITH (" +
            "  'connector' = 'kafka'," +
            "  'topic' = 'analytics-ml-performance'," +
            "  'properties.bootstrap.servers' = 'localhost:9092'," +
            "  'format' = 'json'" +
            ")"
        );
        
        tableEnv.executeSql(
            "INSERT INTO ml_performance_sink " +
            "SELECT * FROM ml_performance"
        );
        
        // ═══════════════════════════════════════════════════════════
        // VIEW 4: Department Workload (1-hour sliding window)
        // ═══════════════════════════════════════════════════════════
        
        tableEnv.executeSql(
            "CREATE VIEW department_workload AS " +
            "SELECT " +
            "  r.department," +
            "  COUNT(DISTINCT r.patient_id) AS total_patients," +
            "  COUNT(DISTINCT a.patient_id) AS alerted_patients," +
            "  COUNT(DISTINCT m.patient_id) AS ml_flagged_patients," +
            "  HOP_START(r.event_timestamp, INTERVAL '5' MINUTE, INTERVAL '1' HOUR) AS window_start," +
            "  HOP_END(r.event_timestamp, INTERVAL '5' MINUTE, INTERVAL '1' HOUR) AS window_end " +
            "FROM raw_events r " +
            "LEFT JOIN cep_alerts a " +
            "  ON r.patient_id = a.patient_id " +
            "  AND a.alert_timestamp BETWEEN r.event_timestamp - INTERVAL '1' HOUR AND r.event_timestamp " +
            "LEFT JOIN ml_predictions m " +
            "  ON r.patient_id = m.patient_id " +
            "  AND m.prediction_timestamp BETWEEN r.event_timestamp - INTERVAL '1' HOUR AND r.event_timestamp " +
            "  AND m.risk_category IN ('HIGH', 'CRITICAL') " +
            "GROUP BY r.department, HOP(r.event_timestamp, INTERVAL '5' MINUTE, INTERVAL '1' HOUR)"
        );
        
        tableEnv.executeSql(
            "CREATE TABLE department_workload_sink (" +
            "  department STRING," +
            "  total_patients BIGINT," +
            "  alerted_patients BIGINT," +
            "  ml_flagged_patients BIGINT," +
            "  window_start TIMESTAMP(3)," +
            "  window_end TIMESTAMP(3)" +
            ") WITH (" +
            "  'connector' = 'kafka'," +
            "  'topic' = 'analytics-department-workload'," +
            "  'properties.bootstrap.servers' = 'localhost:9092'," +
            "  'format' = 'json'" +
            ")"
        );
        
        tableEnv.executeSql(
            "INSERT INTO department_workload_sink " +
            "SELECT * FROM department_workload"
        );
        
        // ═══════════════════════════════════════════════════════════
        // VIEW 5: Sepsis Surveillance Dashboard (real-time)
        // ═══════════════════════════════════════════════════════════
        
        tableEnv.executeSql(
            "CREATE VIEW sepsis_surveillance AS " +
            "SELECT " +
            "  m.patient_id," +
            "  m.probability AS sepsis_probability," +
            "  m.risk_category," +
            "  r.department," +
            "  a.alert_id AS cep_alert_id," +
            "  a.severity AS alert_severity," +
            "  CASE " +
            "    WHEN a.alert_id IS NOT NULL AND m.risk_category = 'HIGH' THEN 'DUAL_CONFIRMED' " +
            "    WHEN m.risk_category = 'HIGH' THEN 'ML_ONLY' " +
            "    WHEN a.alert_id IS NOT NULL THEN 'CEP_ONLY' " +
            "    ELSE 'LOW_RISK' " +
            "  END AS detection_method," +
            "  m.prediction_timestamp " +
            "FROM ml_predictions m " +
            "LEFT JOIN raw_events r " +
            "  ON m.patient_id = r.patient_id " +
            "  AND r.event_timestamp BETWEEN m.prediction_timestamp - INTERVAL '5' MINUTE AND m.prediction_timestamp " +
            "LEFT JOIN cep_alerts a " +
            "  ON m.patient_id = a.patient_id " +
            "  AND a.pattern_name LIKE '%sepsis%' " +
            "  AND a.alert_timestamp BETWEEN m.prediction_timestamp - INTERVAL '10' MINUTE AND m.prediction_timestamp " +
            "WHERE m.model_type = 'SEPSIS_ONSET' " +
            "  AND m.probability > 0.25"
        );
        
        tableEnv.executeSql(
            "CREATE TABLE sepsis_surveillance_sink (" +
            "  patient_id STRING," +
            "  sepsis_probability DOUBLE," +
            "  risk_category STRING," +
            "  department STRING," +
            "  cep_alert_id STRING," +
            "  alert_severity STRING," +
            "  detection_method STRING," +
            "  prediction_timestamp TIMESTAMP(3)" +
            ") WITH (" +
            "  'connector' = 'kafka'," +
            "  'topic' = 'analytics-sepsis-surveillance'," +
            "  'properties.bootstrap.servers' = 'localhost:9092'," +
            "  'format' = 'json'" +
            ")"
        );
        
        tableEnv.executeSql(
            "INSERT INTO sepsis_surveillance_sink " +
            "SELECT * FROM sepsis_surveillance"
        );
        
        // Execute all queries asynchronously
        env.execute("Module 6: Real-Time Analytics Engine");
    }
}
```

### **6A.2: Time-Series Aggregator**


```java
package com.hospital.flink.analytics;


import org.apache.flink.api.common.functions.AggregateFunction;
import org.apache.flink.streaming.api.datastream.DataStream;
import org.apache.flink.streaming.api.windowing.assigners.TumblingEventTimeWindows;
import org.apache.flink.streaming.api.windowing.time.Time;


/**
 * Time-series aggregation for metrics storage
 * Creates 1min, 5min, 1hr, 24hr rollups
 */
public class TimeSeriesAggregator {
    
    /**
     * Aggregate vital signs into 1-minute buckets
     */
    public static DataStream<VitalMetric> aggregateVitals(
        DataStream<EnrichedEvent> events) {
        
        return events
            .filter(e -> e.getEventType().equals("vitals"))
            .keyBy(e -> e.getPatientId() + "_" + e.getDataType())
            .window(TumblingEventTimeWindows.of(Time.minutes(1)))
            .aggregate(new VitalAggregateFunction())
            .name("vitals-1min-aggregation");
    }
    
    /**
     * Aggregate function for vitals
     */
    public static class VitalAggregateFunction 
        implements AggregateFunction<EnrichedEvent, VitalAccumulator, VitalMetric> {
        
        @Override
        public VitalAccumulator createAccumulator() {
            return new VitalAccumulator();
        }
        
        @Override
        public VitalAccumulator add(EnrichedEvent event, VitalAccumulator acc) {
            acc.count++;
            acc.sum += event.getValue();
            acc.min = Math.min(acc.min, event.getValue());
            acc.max = Math.max(acc.max, event.getValue());
            acc.patientId = event.getPatientId();
            acc.vitalType = event.getDataType();
            acc.windowEnd = event.getTimestamp();
            return acc;
        }
        
        @Override
        public VitalMetric getResult(VitalAccumulator acc) {
            return VitalMetric.builder()
                .patientId(acc.patientId)
                .vitalType(acc.vitalType)
                .avg(acc.sum / acc.count)
                .min(acc.min)
                .max(acc.max)
                .count(acc.count)
                .timestamp(acc.windowEnd)
                .build();
        }
        
        @Override
        public VitalAccumulator merge(VitalAccumulator a, VitalAccumulator b) {
            a.count += b.count;
            a.sum += b.sum;
            a.min = Math.min(a.min, b.min);
            a.max = Math.max(a.max, b.max);
            return a;
        }
    }
    
    @Data
    public static class VitalAccumulator {
        String patientId;
        String vitalType;
        double sum = 0.0;
        double min = Double.MAX_VALUE;
        double max = Double.MIN_VALUE;
        int count = 0;
        long windowEnd;
    }
    
    @Data
    @Builder
    public static class VitalMetric {
        String patientId;
        String vitalType;
        double avg;
        double min;
        double max;
        int count;
        long timestamp;
    }
}
```

### **6A.3: Population Health Analytics**


```java
package com.hospital.flink.analytics;


import org.apache.flink.api.common.state.*;
import org.apache.flink.configuration.Configuration;
import org.apache.flink.streaming.api.functions.KeyedProcessFunction;
import org.apache.flink.util.Collector;


import java.util.*;


/**
 * Calculate population-level health metrics
 * - Hospital-wide risk distribution
 * - Department comparisons
 * - Trend analysis
 */
public class PopulationHealthAnalytics 
    extends KeyedProcessFunction<String, MLPredictionBundle, PopulationMetrics> {
    
    // State: Patient risk profiles per department
    private transient MapState<String, PatientRiskProfile> patientRiskState;
    
    // State: Department-level statistics
    private transient ValueState<DepartmentStats> departmentStatsState;
    
    @Override
    public void open(Configuration parameters) throws Exception {
        
        MapStateDescriptor<String, PatientRiskProfile> patientRiskDescriptor = 
            new MapStateDescriptor<>(
                "patient-risk-profiles",
                String.class,
                PatientRiskProfile.class
            );
        patientRiskState = getRuntimeContext().getMapState(patientRiskDescriptor);
        
        ValueStateDescriptor<DepartmentStats> statsDescriptor = 
            new ValueStateDescriptor<>(
                "department-stats",
                DepartmentStats.class
            );
        departmentStatsState = getRuntimeContext().getState(statsDescriptor);
    }
    
    @Override
    public void processElement(
        MLPredictionBundle prediction,
        Context ctx,
        Collector<PopulationMetrics> out) throws Exception {
        
        String department = ctx.getCurrentKey();
        String patientId = prediction.getPatientId();
        
        // Update patient risk profile
        PatientRiskProfile profile = PatientRiskProfile.builder()
            .patientId(patientId)
            .department(department)
            .mortalityRisk(prediction.getMortalityPrediction().getProbability())
            .sepsisRisk(prediction.getSepsisOnsetPrediction().getProbability())
            .readmissionRisk(prediction.getReadmissionPrediction().getProbability())
            .overallRiskScore(prediction.getEnsembleRisk().getOverallRiskScore())
            .lastUpdated(System.currentTimeMillis())
            .build();
        
        patientRiskState.put(patientId, profile);
        
        // Calculate department statistics
        DepartmentStats stats = calculateDepartmentStats();
        departmentStatsState.update(stats);
        
        // Emit population metrics every minute
        ctx.timerService().registerProcessingTimeTimer(
            System.currentTimeMillis() + 60000
        );
    }
    
    @Override
    public void onTimer(
        long timestamp,
        OnTimerContext ctx,
        Collector<PopulationMetrics> out) throws Exception {
        
        DepartmentStats stats = departmentStatsState.value();
        if (stats == null) return;
        
        PopulationMetrics metrics = PopulationMetrics.builder()
            .department(ctx.getCurrentKey())
            .timestamp(timestamp)
            .totalPatients(stats.getTotalPatients())
            .highRiskPatients(stats.getHighRiskPatients())
            .criticalPatients(stats.getCriticalPatients())
            .avgMortalityRisk(stats.getAvgMortalityRisk())
            .avgSepsisRisk(stats.getAvgSepsisRisk())
            .riskDistribution(stats.getRiskDistribution())
            .trendIndicator(stats.getTrendIndicator())
            .build();
        
        out.collect(metrics);
        
        // Register next timer
        ctx.timerService().registerProcessingTimeTimer(timestamp + 60000);
    }
    
    private DepartmentStats calculateDepartmentStats() throws Exception {
        
        int totalPatients = 0;
        int highRiskCount = 0;
        int criticalCount = 0;
        double mortalitySum = 0.0;
        double sepsisSum = 0.0;
        
        Map<String, Integer> riskDistribution = new HashMap<>();
        riskDistribution.put("LOW", 0);
        riskDistribution.put("MODERATE", 0);
        riskDistribution.put("HIGH", 0);
        riskDistribution.put("CRITICAL", 0);
        
        Iterator<Map.Entry<String, PatientRiskProfile>> iterator = 
            patientRiskState.iterator();
        
        while (iterator.hasNext()) {
            Map.Entry<String, PatientRiskProfile> entry = iterator.next();
            PatientRiskProfile profile = entry.getValue();
            
            // Remove stale profiles (>24 hours old)
            if (System.currentTimeMillis() - profile.getLastUpdated() > 24 * 3600 * 1000) {
                iterator.remove();
                continue;
            }
            
            totalPatients++;
            mortalitySum += profile.getMortalityRisk();
            sepsisSum += profile.getSepsisRisk();
            
            double overallRisk = profile.getOverallRiskScore();
            if (overallRisk >= 0.75) {
                criticalCount++;
                riskDistribution.put("CRITICAL", riskDistribution.get("CRITICAL") + 1);
            } else if (overallRisk >= 0.50) {
                highRiskCount++;
                riskDistribution.put("HIGH", riskDistribution.get("HIGH") + 1);
            } else if (overallRisk >= 0.25) {
                riskDistribution.put("MODERATE", riskDistribution.get("MODERATE") + 1);
            } else {
                riskDistribution.put("LOW", riskDistribution.get("LOW") + 1);
            }
        }
        
        return DepartmentStats.builder()
            .totalPatients(totalPatients)
            .highRiskPatients(highRiskCount)
            .criticalPatients(criticalCount)
            .avgMortalityRisk(totalPatients > 0 ? mortalitySum / totalPatients : 0.0)
            .avgSepsisRisk(totalPatients > 0 ? sepsisSum / totalPatients : 0.0)
            .riskDistribution(riskDistribution)
            .trendIndicator("STABLE")  // Calculate from historical data
            .build();
    }
}


@Data
@Builder
class PatientRiskProfile {
    String patientId;
    String department;
    double mortalityRisk;
    double sepsisRisk;
    double readmissionRisk;
    double overallRiskScore;
    long lastUpdated;
}


@Data
@Builder
class DepartmentStats {
    int totalPatients;
    int highRiskPatients;
    int criticalPatients;
    double avgMortalityRisk;
    double avgSepsisRisk;
    Map<String, Integer> riskDistribution;
    String trendIndicator;
}


@Data
@Builder
class PopulationMetrics {
    String department;
    long timestamp;
    int totalPatients;
    int highRiskPatients;
    int criticalPatients;
    double avgMortalityRisk;
    double avgSepsisRisk;
    Map<String, Integer> riskDistribution;
    String trendIndicator;
}
```

package com.hospital.api.service;


import org.springframework.stereotype.Service;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.jdbc.core.JdbcTemplate;


import java.util.*;


/**
 * Service for fetching analytics data
 * Combines real-time (Redis/Kafka) + historical (PostgreSQL/TimescaleDB)
 */
@Service
public class AnalyticsDataService {
    
    private final RedisTemplate<String, Object> redisTemplate;
    private final JdbcTemplate jdbcTemplate;
    private final KafkaConsumerService kafkaConsumer;
    
    public AnalyticsDataService(
        RedisTemplate<String, Object> redisTemplate,
        JdbcTemplate jdbcTemplate,
        KafkaConsumerService kafkaConsumer) {
        this.redisTemplate = redisTemplate;
        this.jdbcTemplate = jdbcTemplate;
        this.kafkaConsumer = kafkaConsumer;
    }
    
    /**
     * Get hospital-wide KPIs
     * Sources: Redis (real-time) + PostgreSQL (historical trends)
     */
    public HospitalKPIs getHospitalKPIs() {
        
        // Fetch real-time metrics from Redis
        Map<Object, Object> realtimeMetrics = redisTemplate
            .opsForHash()
            .entries("hospital:realtime:kpis");
        
        // Fetch historical trends from PostgreSQL
        String trendQuery = 
            "SELECT " +
            "  COUNT(DISTINCT patient_id) as patient_count_trend, " +
            "  AVG(risk_score) as avg_risk_trend " +
            "FROM patient_metrics " +
            "WHERE timestamp > NOW() - INTERVAL '7 days' " +
            "GROUP BY DATE_TRUNC('day', timestamp) " +
            "ORDER BY DATE_TRUNC('day', timestamp)";
        
        List<Map<String, Object>> trends = jdbcTemplate.queryForList(trendQuery);
        
        // Fetch 30-day outcomes
        String outcomeQuery = 
            "SELECT " +
            "  COUNT(CASE WHEN died_30d = true THEN 1 END)::float / COUNT(*) as mortality_30d, " +
            "  COUNT(CASE WHEN readmitted_30d = true THEN 1 END)::float / COUNT(*) as readmission_30d, " +
            "  AVG(length_of_stay) as avg_los " +
            "FROM patient_outcomes " +
            "WHERE discharge_date > NOW() - INTERVAL '30 days'";
        
        Map<String, Object> outcomes = jdbcTemplate.queryForMap(outcomeQuery);
        
        // Fetch department summaries
        List<DepartmentSummary> departmentSummaries = getDepartmentSummaries();
        
        // Calculate trends
        String patientCensusTrend = calculateTrend(trends, "patient_count_trend");
        String riskLevelTrend = calculateTrend(trends, "avg_risk_trend");
        
        return HospitalKPIs.builder()
            .timestamp(System.currentTimeMillis())
            .totalPatients(getIntValue(realtimeMetrics, "total_patients"))
            .icuPatients(getIntValue(realtimeMetrics, "icu_patients"))
            .edPatients(getIntValue(realtimeMetrics, "ed_patients"))
            .lowRiskCount(getIntValue(realtimeMetrics, "low_risk_count"))
            .moderateRiskCount(getIntValue(realtimeMetrics, "moderate_risk_count"))
            .highRiskCount(getIntValue(realtimeMetrics, "high_risk_count"))
            .criticalRiskCount(getIntValue(realtimeMetrics, "critical_risk_count"))
            .activeAlerts(getIntValue(realtimeMetrics, "active_alerts"))
            .criticalAlerts(getIntValue(realtimeMetrics, "critical_alerts"))
            .acknowledgedRate(getDoubleValue(realtimeMetrics, "acknowledged_rate"))
            .avgResponseTimeMinutes(getDoubleValue(realtimeMetrics, "avg_response_time"))
            .mlPredictionsLast24h(getIntValue(realtimeMetrics, "ml_predictions_24h"))
            .sepsisAlertsLast24h(getIntValue(realtimeMetrics, "sepsis_alerts_24h"))
            .mortalityHighRiskCount(getIntValue(realtimeMetrics, "mortality_high_risk"))
            .mortality30d(getDoubleValue(outcomes, "mortality_30d"))
            .readmission30d(getDoubleValue(outcomes, "readmission_30d"))
            .avgLengthOfStay(getDoubleValue(outcomes, "avg_los"))
            .patientCensusTrend(patientCensusTrend)
            .riskLevelTrend(riskLevelTrend)
            .alertVolumeTrend("STABLE")  // Calculate from historical data
            .departmentSummaries(departmentSummaries)
            .build();
    }

    /**
     * Get sepsis surveillance dashboard
     */
    public SepsisSurveillance getSepsisSurveillance(String departmentId) {
        
        // Fetch active sepsis patients
        String sepsisQuery = 
            "SELECT " +
            "  p.patient_id, p.patient_name, " +
            "  s.detection_method, s.detection_time, s.sepsis_probability, " +
            "  s.bundle_initiated, s.bundle_initiated_at, " +
            "  s.antibiotics_given, s.antibiotics_given_at, " +
            "  s.fluid_bolus_given, s.current_lactate, s.current_map, " +
            "  s.on_vasopressors, s.in_icu, s.status " +
            "FROM sepsis_surveillance s " +
            "JOIN patients p ON s.patient_id = p.patient_id " +
            "WHERE p.department_id = ? " +
            "  AND s.status IN ('ACTIVE', 'IMPROVING', 'WORSENING') " +
            "ORDER BY s.sepsis_probability DESC";
        
        List<SepsisPatient> sepsisPatients = jdbcTemplate.query(
            sepsisQuery,
            new Object[]{departmentId},
            (rs, rowNum) -> SepsisPatient.builder()
                .patientId(rs.getString("patient_id"))
                .patientName(rs.getString("patient_name"))
                .detectionMethod(rs.getString("detection_method"))
                .detectionTime(rs.getLong("detection_time"))
                .sepsisProbability(rs.getDouble("sepsis_probability"))
                .bundleInitiated(rs.getBoolean("bundle_initiated"))
                .bundleInitiatedAt(rs.getLong("bundle_initiated_at"))
                .antibioticsGiven(rs.getBoolean("antibiotics_given"))
                .antibioticsGivenAt(rs.getLong("antibiotics_given_at"))
                .fluidBolusGiven(rs.getBoolean("fluid_bolus_given"))
                .currentLactate(rs.getDouble("current_lactate"))
                .currentMAP(rs.getDouble("current_map"))
                .onVasopressors(rs.getBoolean("on_vasopressors"))
                .inICU(rs.getBoolean("in_icu"))
                .status(rs.getString("status"))
                .build()
        );
        
        // Fetch bundle compliance metrics
        String complianceQuery = 
            "SELECT " +
            "  COUNT(*)::float / NULLIF(COUNT(*), 0) as bundle_compliance_rate, " +
            "  AVG(EXTRACT(EPOCH FROM (antibiotics_given_at - bundle_initiated_at))/60) as avg_time_to_abx, " +
            "  AVG(EXTRACT(EPOCH FROM (fluid_bolus_given_at - bundle_initiated_at))/60) as avg_time_to_fluids " +
            "FROM sepsis_surveillance " +
            "WHERE department_id = ? " +

```java
            "WHERE department_id = ? AND metric_type = ?";

        double benchmark = getDoubleValue(result, "national_benchmark");

        if (currentValue < benchmark * 0.8) {
            performanceLevel = "EXCELLENT";

            performanceLevel = "GOOD";
        } else {
            performanceLevel = "NEEDS_IMPROVEMENT";

    private String calculateTrend(List<Map<String, Object>> data, String field) {
        if (data.size() < 2) return "STABLE";
        
        double[] values = data.stream()
            .mapToDouble(m -> getDoubleValue(m, field))
            .toArray();
        
        // Simple linear regression slope
        double avgChange = (values[values.length - 1] - values[0]) / values.length;
        
        if (avgChange > 0.05) return "INCREASING";
        if (avgChange < -0.05) return "DECREASING";
        return "STABLE";
    }
    
    private int getIntValue(Map<?, ?> map, String key) {
        Object value = map.get(key);
        if (value == null) return 0;
        if (value instanceof Number) {
            return ((Number) value).intValue();
        }
        return 0;
    }

        Object value = map.get(key);
        if (value == null) return 0.0;
        if (value instanceof Number) {
            return ((Number) value).doubleValue();
        }
        return 0.0;
    }

        Object value = map.get(key);
        return value != null ? value.toString() : "";
```

```java
package com.hospital.notifications;


import org.apache.flink.streaming.api.functions.ProcessFunction;
import org.apache.flink.util.Collector;


import java.util.*;


/**
 * Smart notification router
 * Routes alerts to appropriate channels based on:
 * - Alert severity
 * - User role and preferences
 * - On-call schedule
 * - Alert fatigue mitigation
 */
public class NotificationRouter extends ProcessFunction<Alert, Notification> {
    
    private final NotificationRuleEngine ruleEngine;
    private final OnCallScheduleService scheduleService;
    private final UserPreferenceService preferenceService;
    private final AlertFatigueTracker fatigueTracker;
    
    public NotificationRouter(
        NotificationRuleEngine ruleEngine,
        OnCallScheduleService scheduleService,
        UserPreferenceService preferenceService,
        AlertFatigueTracker fatigueTracker) {
        this.ruleEngine = ruleEngine;
        this.scheduleService = scheduleService;
        this.preferenceService = preferenceService;
        this.fatigueTracker = fatigueTracker;
    }
    
    @Override
    public void processElement(
        Alert alert,
        Context ctx,
        Collector<Notification> out) throws Exception {
        
        // Step 1: Check alert fatigue
        if (fatigueTracker.shouldSuppress(alert)) {
            // Log suppression but don't send notification
            logSuppression(alert, "ALERT_FATIGUE");
            return;
        }
        
        // Step 2: Determine target users based on alert
        List<User> targetUsers = determineTargetUsers(alert);
        
        // Step 3: For each user, determine notification channels
        for (User user : targetUsers) {
            
            // Get user preferences
            UserPreferences prefs = preferenceService.getPreferences(user.getUserId());
            
            // Apply notification rules
            List<NotificationChannel> channels = ruleEngine.determineChannels(
                alert,
                user,
                prefs
            );
            
            // Create notifications for each channel
            for (NotificationChannel channel : channels) {
                
                Notification notification = Notification.builder()
                    .notificationId(UUID.randomUUID().toString())
                    .alertId(alert.getAlertId())
                    .userId(user.getUserId())
                    .channel(channel)
                    .priority(calculatePriority(alert, user, channel))
                    .message(formatMessage(alert, channel))
                    .createdAt(System.currentTimeMillis())
                    .status("PENDING")
                    .retryCount(0)
                    .build();
                
                out.collect(notification);
                
                // Track for fatigue monitoring
                fatigueTracker.recordNotification(user.getUserId(), alert);
            }
        }
        
        // Step 4: Check if escalation is needed
        if (requiresEscalation(alert)) {
            scheduleEscalation(alert, ctx);
        }
    }
    
    /**
     * Determine which users should receive this alert
     */
    private List<User> determineTargetUsers(Alert alert) {
        
        List<User> users = new ArrayList<>();
        
        // Critical alerts → Attending physician + Charge nurse
        if (alert.getSeverity().equals("CRITICAL")) {
            users.addAll(scheduleService.getOnCallPhysicians(alert.getDepartment()));
            users.addAll(scheduleService.getChargeNurses(alert.getDepartment()));
        }
        
        // High severity → Primary nurse + Resident
        else if (alert.getSeverity().equals("HIGH")) {
            User primaryNurse = scheduleService.getPrimaryNurse(alert.getPatientId());
            if (primaryNurse != null) users.add(primaryNurse);
            
            User resident = scheduleService.getAssignedResident(alert.getDepartment());
            if (resident != null) users.add(resident);
        }
        
        // Moderate severity → Primary nurse only
        else if (alert.getSeverity().equals("MODERATE")) {
            User primaryNurse = scheduleService.getPrimaryNurse(alert.getPatientId());
            if (primaryNurse != null) users.add(primaryNurse);
        }
        
        // ML-specific alerts → Clinical informatics team
        if (alert.getSource().equals("ML") && alert.getConfidence() > 0.8) {
            users.addAll(scheduleService.getClinicalInformaticsTeam());
        }
        
        return users;
    }
    
    /**
     * Format message based on channel type
     */
    private String formatMessage(Alert alert, NotificationChannel channel) {
        
        switch (channel) {
            case SMS:
                return formatSMSMessage(alert);
                
            case PAGER:
                return formatPagerMessage(alert);
                
            case EMAIL:
                return formatEmailMessage(alert);
                
            case PUSH_NOTIFICATION:
                return formatPushMessage(alert);
                
            case IN_APP:
                return alert.getMessage();  // Full message
                
            default:
                return alert.getMessage();
        }
    }
    
    /**
     * SMS format: Short, critical info only
     * Example: "CRITICAL: PAT-001 Sepsis Alert (92%) - ICU Bed 5"
     */
    private String formatSMSMessage(Alert alert) {
        return String.format(
            "%s: %s %s (%.0f%%) - %s",
            alert.getSeverity(),
            alert.getPatientId(),
            alert.getAlertType(),
            alert.getConfidence() * 100,
            alert.getDepartment() + " " + alert.getRoom()
        );
    }
    
    /**
     * Pager format: Ultra-short, alphanumeric only
     * Example: "CRIT PAT-001 SEPSIS ICU-5"
     */
    private String formatPagerMessage(Alert alert) {
        String severity = alert.getSeverity().substring(0, 4).toUpperCase();
        String alertType = alert.getAlertType().replaceAll("[^A-Z0-9]", "").substring(0, 6);
        
        return String.format(
            "%s %s %s %s",
            severity,
            alert.getPatientId(),
            alertType,
            alert.getRoom()
        );
    }
    
    /**
     * Email format: Full details with recommendations
     */
    private String formatEmailMessage(Alert alert) {
        StringBuilder sb = new StringBuilder();
        
        sb.append("Alert: ").append(alert.getAlertType()).append("\n");
        sb.append("Severity: ").append(alert.getSeverity()).append("\n");
        sb.append("Patient: ").append(alert.getPatientId()).append("\n");
        sb.append("Location: ").append(alert.getDepartment())
          .append(" - ").append(alert.getRoom()).append("\n\n");
        
        sb.append("Details:\n");
        sb.append(alert.getMessage()).append("\n\n");
        
        if (alert.getRecommendations() != null && !alert.getRecommendations().isEmpty()) {
            sb.append("Recommended Actions:\n");
            for (String rec : alert.getRecommendations()) {
                sb.append("- ").append(rec).append("\n");
            }
        }
        
        sb.append("\nConfidence: ").append(String.format("%.1f%%", alert.getConfidence() * 100));
        
        return sb.toString();
    }
    
    /**
     * Push notification format: Concise but actionable
     */
    private String formatPushMessage(Alert alert) {
        return String.format(
            "%s Alert: %s in %s",
            alert.getSeverity(),
            alert.getAlertType(),
            alert.getDepartment()
        );
    }
    
    /**
     * Calculate notification priority (1-5, 1 = highest)
     */
    private int calculatePriority(Alert alert, User user, NotificationChannel channel) {
        
        int basePriority;
        
        // Base priority from alert severity
        switch (alert.getSeverity()) {
            case "CRITICAL":
                basePriority = 1;
                break;
            case "HIGH":
                basePriority = 2;
                break;
            case "MODERATE":
                basePriority = 3;
                break;
            default:
                basePriority = 4;
        }
        
        // Adjust based on user role
        if (user.getRole().equals("ATTENDING")) {
            basePriority = Math.max(1, basePriority - 1);
        }
        
        // Adjust based on channel urgency
        if (channel == NotificationChannel.PAGER) {
            basePriority = 1;  // Pagers are always highest priority
        }
        
        return basePriority;
    }
    
    /**
     * Check if alert requires escalation
     */
    private boolean requiresEscalation(Alert alert) {
        return alert.getSeverity().equals("CRITICAL") && 
               alert.getConfidence() > 0.9;
    }

     * Schedule escalation timer
     */
    private void scheduleEscalation(Alert alert, Context ctx) {
        // If alert not acknowledged in 5 minutes, escalate
        long escalationTime = System.currentTimeMillis() + (5 * 60 * 1000);
        ctx.timerService().registerProcessingTimeTimer(escalationTime);
    }
    
    private void logSuppression(Alert alert, String reason) {
        System.out.println(String.format(
            "Suppressed alert %s for patient %s. Reason: %s",
            alert.getAlertId(),
            alert.getPatientId(),
            reason
        ));
    }
```

### **6C.2: Alert Fatigue Mitigation**


```java
package com.hospital.notifications;


import org.apache.flink.api.common.state.*;
import org.apache.flink.configuration.Configuration;
import org.apache.flink.streaming.api.functions.KeyedProcessFunction;


import java.util.*;


/**
 * Tracks alert volume and implements fatigue mitigation
 * - Rate limiting per user
 * - Duplicate suppression
 * - Alert bundling
 * - Smart throttling
 */
public class AlertFatigueTracker {
    
    private final int MAX_ALERTS_PER_HOUR = 20;
    private final long DUPLICATE_WINDOW_MS = 5 * 60 * 1000;  // 5 minutes
    private final int BUNDLE_THRESHOLD = 3;  // Bundle if 3+ similar alerts
    
    private Map<String, UserAlertHistory> userHistories = new HashMap<>();
    
    /**
     * Check if alert should be suppressed due to fatigue
     */
    public boolean shouldSuppress(Alert alert) {
        
        String userId = alert.getAssignedUserId();
        if (userId == null) return false;
        
        UserAlertHistory history = userHistories.computeIfAbsent(
            userId,
            k -> new UserAlertHistory()
        );
        
        // Check rate limit
        if (history.getAlertsInLastHour() >= MAX_ALERTS_PER_HOUR) {
            // Only allow CRITICAL alerts through
            if (!alert.getSeverity().equals("CRITICAL")) {
                return true;
            }
        }
        
        // Check for duplicates
        if (isDuplicate(alert, history)) {
            return true;
        }
        
        // Check if should be bundled
        if (shouldBundle(alert, history)) {
            history.addToBundleQueue(alert);
            return true;  // Suppress individual, will send bundle later
        }
        
        return false;
    }
    
    /**
     * Record notification sent
     */
    public void recordNotification(String userId, Alert alert) {
        UserAlertHistory history = userHistories.computeIfAbsent(
            userId,
            k -> new UserAlertHistory()
        );
        
        history.addAlert(alert);
    }
    
    /**
     * Check if alert is duplicate of recent alert
     */
    private boolean isDuplicate(Alert alert, UserAlertHistory history) {
        
        long now = System.currentTimeMillis();
        
        return history.getRecentAlerts().stream()
            .filter(a -> now - a.getCreatedAt() < DUPLICATE_WINDOW_MS)
            .anyMatch(a -> 
                a.getPatientId().equals(alert.getPatientId()) &&
                a.getAlertType().equals(alert.getAlertType()) &&
                a.getSeverity().equals(alert.getSeverity())
            );
    }
    
    /**
     * Check if alert should be bundled with similar alerts
     */
    private boolean shouldBundle(Alert alert, UserAlertHistory history) {
        
        // Don't bundle CRITICAL alerts
        if (alert.getSeverity().equals("CRITICAL")) {
            return false;
        }
        
        // Check if there are similar alerts in bundle queue
        long similarCount = history.getBundleQueue().stream()
            .filter(a -> a.getAlertType().equals(alert.getAlertType()))
            .count();
        
        return similarCount >= BUNDLE_THRESHOLD - 1;
    }
    
    /**
     * Create bundled alert from multiple similar alerts
     */
    public Alert createBundledAlert(List<Alert> alerts) {
        
        if (alerts.isEmpty()) return null;
        
        Alert first = alerts.get(0);
        
        String bundledMessage = String.format(
            "BUNDLED: %d similar %s alerts",
            alerts.size(),
            first.getAlertType()
        );
        
        List<String> patientIds = alerts.stream()
            .map(Alert::getPatientId)
            .distinct()
            .collect(Collectors.toList());
        
        return Alert.builder()
            .alertId("BUNDLE-" + UUID.randomUUID().toString())
            .alertType("BUNDLED_" + first.getAlertType())
            .severity(first.getSeverity())
            .message(bundledMessage)
            .patientIds(patientIds)
            .bundledAlertCount(alerts.size())
            .createdAt(System.currentTimeMillis())
            .build();
    }
}


/**
 * Tracks alert history per user
 */
@Data
class UserAlertHistory {
    private List<Alert> recentAlerts = new ArrayList<>();
    private List<Alert> bundleQueue = new ArrayList<>();
    private long lastCleanup = System.currentTimeMillis();
    
    public void addAlert(Alert alert) {
        recentAlerts.add(alert);
        cleanup();
    }
    
    public void addToBundleQueue(Alert alert) {
        bundleQueue.add(alert);
        cleanup();
    }
    
    public int getAlertsInLastHour() {
        long oneHourAgo = System.currentTimeMillis() - (60 * 60 * 1000);
        return (int) recentAlerts.stream()
            .filter(a -> a.getCreatedAt() > oneHourAgo)
            .count();
    }
    
    private void cleanup() {
        long now = System.currentTimeMillis();
        
        // Remove alerts older than 1 hour
        if (now - lastCleanup > 60 * 60 * 1000) {
            long oneHourAgo = now - (60 * 60 * 1000);
            recentAlerts.removeIf(a -> a.getCreatedAt() < oneHourAgo);
            bundleQueue.removeIf(a -> a.getCreatedAt() < oneHourAgo);
            lastCleanup = now;
        }
    }
}
```

```java
/**

@Service
public class NotificationDeliveryService {

    private final PushNotificationService pushService;
    
    public NotificationDeliveryService(
        TwilioConfig twilioConfig,
        SendGridConfig sendGridConfig,
        PushNotificationService pushService) {
        this.twilioConfig = twilioConfig;
        this.sendGridConfig = sendGridConfig;
        this.pushService = pushService;
        
        // Initialize Twilio
        Twilio.init(twilioConfig.getAccountSid(), twilioConfig.getAuthToken());
    }
    
    /**
     * Send notification via appropriate channel
     */
    public DeliveryResult send(Notification notification) {
        
        try {
            switch (notification.getChannel()) {
                case SMS:
                    return sendSMS(notification);
                    
                case EMAIL:
                    return sendEmail(notification);
                    
                case PUSH_NOTIFICATION:
                    return sendPush(notification);
                    
                case PAGER:
                    return sendPager(notification);
                    
                case VOICE_CALL:
                    return initiateVoiceCall(notification);
                    
                default:
                    return DeliveryResult.failure("Unsupported channel");
            }
        } catch (Exception e) {
            return DeliveryResult.failure(e.getMessage());
        }
    }
    
    /**
     * Send SMS via Twilio
     */
    private DeliveryResult sendSMS(Notification notification) {
        
        User user = getUserById(notification.getUserId());
        if (user.getPhoneNumber() == null) {
            return DeliveryResult.failure("No phone number");
        }
        
        Message message = Message.creator(
            new PhoneNumber(user.getPhoneNumber()),
            new PhoneNumber(twilioConfig.getFromNumber()),
            notification.getMessage()
        ).create();
        
        return DeliveryResult.success(message.getSid());
    }
    
    /**
     * Send email via SendGrid
     */
    private DeliveryResult sendEmail(Notification notification) {
        
        User user = getUserById(notification.getUserId());
        if (user.getEmail() == null) {
            return DeliveryResult.failure("No email address");
        }
        
        Email from = new Email(sendGridConfig.getFromEmail());
        Email to = new Email(user.getEmail());
        String subject = "Clinical Alert: " + extractAlertType(notification.getMessage());
        Content content = new Content("text/plain", notification.getMessage());
        
        Mail mail = new Mail(from, subject, to, content);
        
        SendGrid sg = new SendGrid(sendGridConfig.getApiKey());
        Request request = new Request();
        
        try {
            request.setMethod(Method.POST);
            request.setEndpoint("mail/send");
            request.setBody(mail.build());
            Response response = sg.api(request);
            
            if (response.getStatusCode() >= 200 && response.getStatusCode() < 300) {
                return DeliveryResult.success(response.getHeaders().get("X-Message-Id"));
            } else {
                return DeliveryResult.failure("SendGrid error: " + response.getStatusCode());
            }
        } catch (IOException e) {
            return DeliveryResult.failure(e.getMessage());
        }
    }
    
    /**
     * Send push notification
     */
    private DeliveryResult sendPush(Notification notification) {
        return pushService.sendPushNotification(notification);
    }
    
    /**
     * Send to pager (via SMS gateway or dedicated pager service)
     */
    private DeliveryResult sendPager(Notification notification) {
        
        User user = getUserById(notification.getUserId());
        if (user.getPagerNumber() == null) {
            return DeliveryResult.failure("No pager number");
        }
        
        // Pagers use special SMS gateway format
        String pagerGateway = twilioConfig.getPagerGateway();
        String fullNumber = user.getPagerNumber() + "@" + pagerGateway;
        
        Message message = Message.creator(
            new PhoneNumber(fullNumber),
            new PhoneNumber(twilioConfig.getFromNumber()),
            notification.getMessage()
        ).create();
        
        return DeliveryResult.success(message.getSid());
    }
    
    /**
     * Initiate voice call for critical escalations
     */
    private DeliveryResult initiateVoiceCall(Notification notification) {
        
        User user = getUserById(notification.getUserId());
        if (user.getPhoneNumber() == null) {
            return DeliveryResult.failure("No phone number");
        }
        
        Call call = Call.creator(
            new PhoneNumber(user.getPhoneNumber()),
            new PhoneNumber(twilioConfig.getFromNumber()),
            new URI(twilioConfig.getVoiceCallbackUrl() + "?message=" + 
                    URLEncoder.encode(notification.getMessage(), "UTF-8"))
        ).create();
        
        return DeliveryResult.success(call.getSid());
    }
    
    private User getUserById(String userId) {
        // Fetch from database or cache
        return userRepository.findById(userId);
    }
    
    private String extractAlertType(String message) {
        // Extract alert type from message for email subject
        if (message.contains("Sepsis")) return "Sepsis Alert";
        if (message.contains("Mortality")) return "Mortality Risk";
        return "Patient Alert";
    }
}

 * Delivery result

@Data

public class DeliveryResult {
    private boolean success;
    private String messageId;
    private String errorMessage;
    
    public static DeliveryResult success(String messageId) {
        return new DeliveryResult(true, messageId, null);
    }
    
    public static DeliveryResult failure(String errorMessage) {
        return new DeliveryResult(false, null, errorMessage);
```

```java
    public void afterConnectionEstablished(WebSocketSession session) throws Exception {
        sessions.add(session);
        System.out.println("WebSocket connection established: " + session.getId());
        
        // Send initial connection message
        WebSocketMessage welcomeMsg = WebSocketMessage.builder()
            .type("CONNECTION_ESTABLISHED")
            .timestamp(System.currentTimeMillis())
            .data(Map.of("sessionId", session.getId()))
            .build();
        
        session.sendMessage(new TextMessage(objectMapper.writeValueAsString(welcomeMsg)));
    }
    
    @Override
    public void afterConnectionClosed(WebSocketSession session, CloseStatus status) {
        sessions.remove(session);
        System.out.println("WebSocket connection closed: " + session.getId());
    }
    
    @Override
    protected void handleTextMessage(WebSocketSession session, TextMessage message) throws Exception {
        // Handle incoming messages from client
        String payload = message.getPayload();
        System.out.println("Received message from " + session.getId() + ": " + payload);
        
        // Example: Client can subscribe to specific patient updates
        WebSocketMessage clientMsg = objectMapper.readValue(payload, WebSocketMessage.class);
        
        if ("SUBSCRIBE_PATIENT".equals(clientMsg.getType())) {
            String patientId = (String) clientMsg.getData().get("patientId");
            session.getAttributes().put("subscribedPatientId", patientId);
        }
    }
    
    /**
     * Broadcast message to all connected clients
     */
    public void broadcast(WebSocketMessage message) {
        String json;
        try {
            json = objectMapper.writeValueAsString(message);
        } catch (Exception e) {
            e.printStackTrace();
            return;
        }
        
        sessions.forEach(session -> {
            try {
                if (session.isOpen()) {
                    session.sendMessage(new TextMessage(json));
                }
            } catch (Exception e) {
                e.printStackTrace();
            }
        });
    }
    
    /**
     * Send message to specific session
     */
    public void sendToSession(String sessionId, WebSocketMessage message) {
        sessions.stream()
            .filter(s -> s.getId().equals(sessionId))
            .findFirst()
            .ifPresent(session -> {
                try {
                    String json = objectMapper.writeValueAsString(message);
                    session.sendMessage(new TextMessage(json));
                } catch (Exception e) {
                    e.printStackTrace();
                }
            });
    }
    
    /**
     * Initialize Kafka listeners for real-time updates
     */
    private void initializeKafkaListeners() {
        
        // Listen to KPI updates
        kafkaConsumer.subscribe("analytics-patient-census", this::handleCensusUpdate);
        kafkaConsumer.subscribe("analytics-alert-metrics", this::handleAlertMetricsUpdate);
        kafkaConsumer.subscribe("analytics-ml-performance", this::handleMLPerformanceUpdate);
        
        // Listen to new alerts
        kafkaConsumer.subscribe("ehr-alerts-final", this::handleNewAlert);
        
        // Listen to ML predictions
        kafkaConsumer.subscribe("ehr-ml-predictions", this::handleMLPrediction);

            .data(parseJson(message))
            .build();
        
        broadcast(wsMessage);
    }

        WebSocketMessage wsMessage = WebSocketMessage.builder()
            .type("ML_PERFORMANCE_UPDATE")
            .timestamp(System.currentTimeMillis())
            .data(parseJson(message))
            .build();
        
        broadcast(wsMessage);

    private void handleNewAlert(String message) {
        Map<String, Object> alert = parseJson(message);
        
        WebSocketMessage wsMessage = WebSocketMessage.builder()
            .type("NEW_ALERT")
            .timestamp(System.currentTimeMillis())
            .data(alert)
            .build();
        
        // Broadcast to all sessions
        broadcast(wsMessage);
        
        // Send targeted notification to sessions subscribed to this patient
        String patientId = (String) alert.get("patientId");
        sessions.stream()
            .filter(s -> patientId.equals(s.getAttributes().get("subscribedPatientId")))
            .forEach(s -> {
                WebSocketMessage targetedMsg = WebSocketMessage.builder()
                    .type("PATIENT_ALERT")
                    .timestamp(System.currentTimeMillis())
                    .data(alert)
                    .build();
                sendToSession(s.getId(), targetedMsg);
            });
    }
    
    private void handleMLPrediction(String message) {
        Map<String, Object> prediction = parseJson(message);
        
        WebSocketMessage wsMessage = WebSocketMessage.builder()
            .type("ML_PREDICTION")
            .timestamp(System.currentTimeMillis())
            .data(prediction)
            .build();

        String patientId = (String) prediction.get("patientId");
        sessions.stream()
            .filter(s -> patientId.equals(s.getAttributes().get("subscribedPatientId")))
            .forEach(s -> sendToSession(s.getId(), wsMessage));
    }

            return objectMapper.readValue(json, Map.class);
        } catch (Exception e) {
            e.printStackTrace();
            return Map.of("error", "Failed to parse JSON");
```

```javascript
// useWebSocket.js


import { useEffect, useRef, useState, useCallback } from 'react';


/**
 * Custom React hook for WebSocket connection
 * Handles automatic reconnection and message handling
 */
export const useWebSocket = (url, onMessage) => {
  const [isConnected, setIsConnected] = useState(false);
  const [lastMessage, setLastMessage] = useState(null);
  const websocketRef = useRef(null);
  const reconnectTimeoutRef = useRef(null);
  const reconnectAttemptsRef = useRef(0);


  const MAX_RECONNECT_ATTEMPTS = 5;
  const RECONNECT_INTERVAL = 3000; // 3 seconds


  const connect = useCallback(() => {
    try {
      const ws = new WebSocket(url);


      ws.onopen = () => {
        console.log('WebSocket connected');
        setIsConnected(true);
        reconnectAttemptsRef.current = 0;
      };


      ws.onmessage = (event) => {
        const message = JSON.parse(event.data);
        setLastMessage(message);
        
        if (onMessage) {
          onMessage(message);
        }
      };


      ws.onerror = (error) => {
        console.error('WebSocket error:', error);
      };


      ws.onclose = () => {
        console.log('WebSocket disconnected');
        setIsConnected(false);
        websocketRef.current = null;


        // Attempt to reconnect
        if (reconnectAttemptsRef.current < MAX_RECONNECT_ATTEMPTS) {
          reconnectAttemptsRef.current += 1;
          console.log(`Reconnecting... Attempt ${reconnectAttemptsRef.current}`);
          
          reconnectTimeoutRef.current = setTimeout(() => {
            connect();
          }, RECONNECT_INTERVAL);
        } else {
          console.error('Max reconnection attempts reached');
        }
      };


      websocketRef.current = ws;
    } catch (error) {
      console.error('Failed to create WebSocket connection:', error);
    }
  }, [url, onMessage]);


  useEffect(() => {
    connect();


    return () => {
      if (reconnectTimeoutRef.current) {
        clearTimeout(reconnectTimeoutRef.current);
      }
      if (websocketRef.current) {
        websocketRef.current.close();
      }
    };
  }, [connect]);


  const sendMessage = useCallback((message) => {
    if (websocketRef.current && websocketRef.current.readyState === WebSocket.OPEN) {
      websocketRef.current.send(JSON.stringify(message));
    } else {
      console.warn('WebSocket is not connected');
    }
  }, []);


  const subscribeToPatient = useCallback((patientId) => {
    sendMessage({
      type: 'SUBSCRIBE_PATIENT',
      data: { patientId }
    });
  }, [sendMessage]);


  return {
    isConnected,
    lastMessage,
    sendMessage,
    subscribeToPatient
  };
};
```

```java
    private final JdbcTemplate jdbcTemplate;

        this.jdbcTemplate = jdbcTemplate;

    /**
     * Export patients to CSV
     */
    public byte[] exportPatientsToCsv(String departmentId, Long startTime, Long endTime) 
        throws IOException {
        
        String query = 
            "SELECT " +
            "  patient_id, patient_name, age, gender, room, " +
            "  overall_risk_score, risk_category, active_alert_count, " +
            "  admission_time, length_of_stay " +
            "FROM patient_current_state " +
            "WHERE department_id = ? " +
            "  AND admission_time BETWEEN ? AND ? " +
            "ORDER BY overall_risk_score DESC";
        
        List<Map<String, Object>> patients = jdbcTemplate.queryForList(
            query, departmentId, startTime, endTime
        );
        
        // Write to CSV
        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
        OutputStreamWriter writer = new OutputStreamWriter(outputStream);
        CSVWriter csvWriter = new CSVWriter(writer);
        
        // Write header
        String[] header = {
            "Patient ID", "Name", "Age", "Gender", "Room",
            "Risk Score", "Risk Category", "Active Alerts",
            "Admission Time", "Length of Stay (hours)"
        };
        csvWriter.writeNext(header);
        
        // Write data
        for (Map<String, Object> patient : patients) {
            String[] row = {
                getString(patient, "patient_id"),
                getString(patient, "patient_name"),
                getString(patient, "age"),
                getString(patient, "gender"),
                getString(patient, "room"),
                getString(patient, "overall_risk_score"),
                getString(patient, "risk_category"),
                getString(patient, "active_alert_count"),
                getString(patient, "admission_time"),
                getString(patient, "length_of_stay")
            };
            csvWriter.writeNext(row);
        }
        
        csvWriter.close();
        return outputStream.toByteArray();
    }
    
    /**
     * Export alerts to CSV
     */
    public byte[] exportAlertsToCsv(String departmentId, Long startTime, Long endTime)
        throws IOException {
        
        String query = 
            "SELECT " +
            "  alert_id, patient_id, patient_name, alert_type, " +
            "  severity, message, status, created_at, acknowledged_at, acknowledged_by " +
            "FROM alerts " +
            "WHERE department = ? " +
            "  AND created_at BETWEEN ? AND ? " +
            "ORDER BY created_at DESC";
        
        List<Map<String, Object>> alerts = jdbcTemplate.queryForList(
            query, departmentId, startTime, endTime
        );
        
        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
        OutputStreamWriter writer = new OutputStreamWriter(outputStream);
        CSVWriter csvWriter = new CSVWriter(writer);
        
        // Write header
        String[] header = {
            "Alert ID", "Patient ID", "Patient Name", "Alert Type",
            "Severity", "Message", "Status", "Created At", 
            "Acknowledged At", "Acknowledged By"
        };
        csvWriter.writeNext(header);
        
        // Write data
        for (Map<String, Object> alert : alerts) {
            String[] row = {
                getString(alert, "alert_id"),
                getString(alert, "patient_id"),
                getString(alert, "patient_name"),
                getString(alert, "alert_type"),
                getString(alert, "severity"),
                getString(alert, "message"),
                getString(alert, "status"),
                getString(alert, "created_at"),
                getString(alert, "acknowledged_at"),
                getString(alert, "acknowledged_by")
            };
            csvWriter.writeNext(row);
        }
        
        csvWriter.close();
        return outputStream.toByteArray();
    }
    
    /**
     * Export ML predictions to JSON
     */
    public String exportPredictionsToJson(
        String departmentId, String modelType, Long startTime, Long endTime) {
        
        String query = 
            "SELECT " +
            "  prediction_id, patient_id, model_type, probability, " +
            "  risk_category, confidence, prediction_timestamp " +
            "FROM ml_predictions " +
            "WHERE department = ? " +
            "  AND model_type = ? " +
            "  AND prediction_timestamp BETWEEN ? AND ? " +
            "ORDER BY prediction_timestamp DESC";
        
        List<Map<String, Object>> predictions = jdbcTemplate.queryForList(
            query, departmentId, modelType, startTime, endTime
        );
        
        try {
            return objectMapper.writerWithDefaultPrettyPrinter()
                .writeValueAsString(predictions);
        } catch (Exception e) {
            throw new RuntimeException("Failed to convert to JSON", e);
        }
    }

    /**
```
     * Send report email
     */
    private void sendReportEmail(
        List<String> recipients,
        String subject,
        String body,
        Map<String, byte[]> attachments) {
        
        try {
            MimeMessage message = mailSender.createMimeMessage();
            MimeMessageHelper helper = new MimeMessageHelper(message, true);
            
            helper.setFrom("reports@hospital.com");
            helper.setTo(recipients.toArray(new String[0]));
            helper.setSubject(subject);
            helper.setText(body);
            
            // Add attachments if present
            if (attachments != null && !attachments.isEmpty()) {
                for (Map.Entry<String, byte[]> attachment : attachments.entrySet()) {
                    helper.addAttachment(
                        attachment.getKey(),
                        () -> new ByteArrayInputStream(attachment.getValue())
                    );
                }
            }
            
            mailSender.send(message);
            
        } catch (Exception e) {
            System.err.println("Failed to send email: " + e.getMessage());
            e.printStackTrace();
        }
    }

```sql
-- Patient analytics table
CREATE TABLE patient_current_state (

    length_of_stay DECIMAL(10,2),
    last_updated BIGINT,
    INDEX idx_department (department_id),
    INDEX idx_risk_score (overall_risk_score DESC),
    INDEX idx_risk_category (risk_category)
);


-- Alert history table
CREATE TABLE alerts (
    alert_id VARCHAR(100) PRIMARY KEY,
    patient_id VARCHAR(50),
    patient_name VARCHAR(200),
    alert_type VARCHAR(100),
    severity VARCHAR(20),
    message TEXT,
    department VARCHAR(50),
    room VARCHAR(50),
    status VARCHAR(20),
    created_at BIGINT,
    acknowledged_at BIGINT,
    acknowledged_by VARCHAR(100),
    resolved_at BIGINT,
    INDEX idx_patient (patient_id),
    INDEX idx_created_at (created_at DESC),
    INDEX idx_status (status),
    INDEX idx_severity (severity)
);


-- ML predictions table
CREATE TABLE ml_predictions (
    prediction_id VARCHAR(100) PRIMARY KEY,
    patient_id VARCHAR(50),
    model_type VARCHAR(50),
    probability DECIMAL(5,4),
    risk_category VARCHAR(20),
    confidence DECIMAL(5,4),
    prediction_timestamp BIGINT,
    department VARCHAR(50),
    INDEX idx_patient (patient_id),
    INDEX idx_model_type (model_type),
    INDEX idx_timestamp (prediction_timestamp DESC)
);


-- Department summary view
CREATE MATERIALIZED VIEW department_summary_view AS
SELECT 
    department_id,
    department_name,
    COUNT(*) as patient_count,
    SUM(CASE WHEN risk_category IN ('HIGH', 'CRITICAL') THEN 1 ELSE 0 END) as high_risk_count,
    SUM(active_alert_count) as active_alerts,
    AVG(overall_risk_score) as avg_risk_score,
    MAX(last_updated) as last_updated
FROM patient_current_state
GROUP BY department_id, department_name;


-- Refresh materialized view every minute
CREATE OR REPLACE FUNCTION refresh_department_summary()
RETURNS void AS $$
BEGIN
    REFRESH MATERIALIZED VIEW CONCURRENTLY department_summary_view;
END;
$$ LANGUAGE plpgsql;
```
