// Full medication ordering flow across 5 phases
public MedicationOrder orderMedicationWithFullChecks(
    Patient patient, String medicationName, VitalSigns vitals, LabResults labs) {
    
    // Phase 6: Get medication from database
    Medication medication = medicationDatabase.getMedication(medicationName);
    
    // Phase 4: Check renal function for dose adjustment
    double crCl = labs.getCreatinineClearance();
    
    // Phase 6: Calculate dose
    DoseRecommendation doseRec = doseCalculator.calculateDose(
        medication, patient.getWeight(), crCl, patient.getAge()
    );
    
    // Phase 6: Safety checks
    List<DrugInteraction> interactions = interactionChecker.check(
        medication, patient.getCurrentMedications()
    );
    
    if (interactions.stream().anyMatch(i -> i.getSeverity() == Severity.CONTRAINDICATED)) {
        throw new MedicationSafetyException("Contraindicated drug interaction detected");
    }
    
    // Phase 6: Check allergies
    if (allergyChecker.hasAllergy(patient, medication)) {
        // Phase 6: Find therapeutic substitution
        Medication substitute = substitutionEngine.findSubstitute(
            medication, patient.getAllergies()
        );
        medication = substitute;
    }
    
    // Phase 7: Attach evidence
    List<String> citationIds = evidenceRepository.getCitationsForMedication(
        medication.getMedicationId()
    );
    
    // Phase 8: Export to EHR via FHIR
    MedicationOrder order = new MedicationOrder(medication, doseRec);
    String fhirOrderId = fhirService.exportMedicationRequest(order);
    
    return order;
}

/**
 * MASTER INTEGRATION: All 8 phases working together
 * Scenario: Patient presents with chest pain via EHR
 */
@PostMapping("/cds-services/order-select")
public CdsHooksResponse handleChestPainPresentation(CdsHooksRequest request) {
    // Phase 8 (FHIR): Import patient data from EHR
    String patientId = request.getContext().getPatientId();
    Patient patient = fhirService.importPatientFromFHIR(patientId);
    VitalSigns vitals = fhirService.importVitalSignsFromFHIR(patientId);
    LabResults labs = fhirService.importLabsFromFHIR(patientId);
    List<MedicationOrder> currentMeds = fhirService.importMedicationsFromFHIR(patientId);
    
    // Phase 1: Identify applicable protocol
    Protocol chestPainProtocol = protocolEngine.getProtocol("chest_pain_evaluation");
    
    // Phase 4: Check if diagnostic tests ordered
    boolean ecgOrdered = diagnosticEngine.isTestOrdered(patient, "ecg_12_lead");
    if (!ecgOrdered) {
        // Return CDS Hook card recommending ECG
        return CdsHooksResponse.warning(
            "ECG Recommended",
            "12-lead ECG should be obtained within 10 minutes for chest pain evaluation"
        );
    }
    
    // Phase 2: Calculate HEART score
    ClinicalScore heartScore = scoringEngine.calculateScore("HEART", patient, vitals, labs);
    
    // Phase 8 (Predictive): Calculate cardiac event risk
    RiskScore cardiacRisk = predictiveEngine.calculateMortalityRisk(patient, vitals, labs);
    
    // Phase 5: Reference guideline
    ClinicalGuideline accGuideline = guidelineLibrary.getGuideline("acc_aha_nstemi_2023");
    
    // Phase 7: Get evidence
    List<Citation> evidence = evidenceRepository.getCitationsForProtocol("chest_pain_evaluation");
    
    // Phase 6: Check if medications ordered
    String draftOrderId = request.getContext().getDraftOrders();
    if (draftOrderId != null) {
        MedicationOrder draftOrder = parseDraftOrder(draftOrderId);
        
        // Phase 6: Check drug interactions
        List<DrugInteraction> interactions = interactionChecker.check(
            draftOrder.getMedication(), currentMeds
        );
        
        if (interactions.stream().anyMatch(i -> i.getSeverity() == Severity.MAJOR)) {
            // Return CDS Hook warning card
            return CdsHooksResponse.critical(
                "Major Drug Interaction",
                interactions.get(0).getDescription(),
                interactions.get(0).getManagementStrategy()
            );
        }
        
        // Phase 6: Check renal dose adjustment
        if (labs.getCreatinineClearance() < 30.0) {
            DoseRecommendation adjustedDose = doseCalculator.adjustForRenalFunction(
                draftOrder.getMedication(), labs.getCreatinineClearance()
            );
            
            if (adjustedDose.isDoseReductionRequired()) {
                return CdsHooksResponse.warning(
                    "Renal Dose Adjustment Required",
                    String.format("Recommend %s (reduced from %s due to CrCl %.1f mL/min)",
                                adjustedDose.getRecommendedDose(),
                                draftOrder.getDose(),
                                labs.getCreatinineClearance())
                );
            }
        }
    }
    
    // Phase 8 (Pathways): Suggest starting clinical pathway
    if (heartScore.getValue() >= 4) {
        // Intermediate to high risk
        return CdsHooksResponse.info(
            "Clinical Pathway Recommended",
            String.format("HEART Score: %d (Intermediate Risk). Consider starting Chest Pain Pathway", 
                        heartScore.getValue()),
            Arrays.asList(
                CdsHooksSuggestion.create("Start Chest Pain Pathway", 
                    () -> pathwayEngine.startPathway("chest_pain_evaluation", patientId)),
                CdsHooksSuggestion.create("Admit for Observation",
                    () -> createAdmissionOrder(patient, "Observation Unit"))
            )
        );
    }
    
    // Phase 8 (Population Health): Check for care gaps
    List<CareGap> gaps = populationHealthService.identifyCareGaps(patient);
    if (!gaps.isEmpty()) {
        CareGap mostUrgent = gaps.stream()
            .max(Comparator.comparing(CareGap::getPriority))
            .orElse(null);
        
        if (mostUrgent != null) {
            return CdsHooksResponse.info(
                "Care Gap Identified",
                mostUrgent.getDescription(),
                Arrays.asList(
                    CdsHooksSuggestion.create("Address Care Gap", mostUrgent.getRecommendedAction())
                )
            );
        }
    }
    
    // No alerts - safe to proceed
    return CdsHooksResponse.empty();
}

            labs
        );

            // PHASE 8: PREDICTIVE ANALYTICS - Calculate sepsis risk
            // ═══════════════════════════════════════════════════════

            System.out.println("   Risk Level: " + sepsisRisk.getRiskLevel());
            System.out.println("   Key Factors: " + sepsisRisk.getFeatureImportance());
            
            if (sepsisRisk.getRiskLevel().ordinal() >= RiskLevel.HIGH.ordinal()) {

                // PHASE 1: PROTOCOLS - Activate sepsis protocol

/**

@PostMapping("/cds-services/order-select")
public CdsHooksResponse handleChestPainOrderSelect(@RequestBody CdsHooksRequest request) {
    
    // Phase 8 (FHIR): Import context
    String patientId = request.getContext().getPatientId();
    Patient patient = fhirIntegrationService.importPatientFromFHIR(patientId);
    VitalSigns vitals = fhirIntegrationService.importVitalSignsFromFHIR(patientId);
    LabResults labs = fhirIntegrationService.importLabsFromFHIR(patientId);
    
    List<CdsHooksCard> cards = new ArrayList<>();
    
    // Phase 2: Calculate HEART score
    ClinicalScore heartScore = scoringEngine.calculateScore(ScoreType.HEART, patient, vitals, labs);
    
    // Phase 8 (Predictive): Calculate cardiac risk
    RiskScore cardiacRisk = predictiveEngine.calculateMortalityRisk(patient, vitals, labs);
    
    if (heartScore.getValue() >= 7 || cardiacRisk.getRiskLevel() == RiskLevel.HIGH) {
        // HIGH RISK - Critical card
        CdsHooksCard highRiskCard = new CdsHooksCard();
        highRiskCard.setSummary("🔴 HIGH CARDIAC RISK - Immediate Evaluation Required");
        highRiskCard.setDetail(String.format(
            "HEART Score: %d (High Risk)\n" +
            "30-day MACE Risk: %.1f%%\n\n" +
            "Recommendations:\n" +
            "• Admit for cardiac monitoring\n" +
            "• Serial troponins (0h, 3h, 6h)\n" +
            "• Cardiology consultation\n" +
            "• Consider coronary angiography",
            heartScore.getValue(),
            cardiacRisk.getScore() * 100
        ));
        highRiskCard.setIndicator(CdsHooksIndicator.CRITICAL);
        
        // Phase 8 (Pathways): Suggest starting pathway
        highRiskCard.addSuggestion(
            CdsHooksSuggestion.create(
                "Start High-Risk Chest Pain Pathway",
                () -> pathwayEngine.startPathway("chest_pain_high_risk", patientId)
            )
        );
        
        // Phase 7 (Evidence): Add supporting evidence link
        Citation heartScoreStudy = evidenceRepository.getCitationByPMID("20542329");
        highRiskCard.addLink(new CdsHooksLink(
            "Evidence: HEART Score Validation",
            "https://pubmed.gov/" + heartScoreStudy.getPmid(),
            "SMART"
        ));
        
        cards.add(highRiskCard);
        
    } else if (heartScore.getValue() >= 4) {
        // INTERMEDIATE RISK - Warning card
        CdsHooksCard intermediateCard = new CdsHooksCard();
        intermediateCard.setSummary("🟡 Intermediate Cardiac Risk - Observation Recommended");
        intermediateCard.setDetail(String.format(
            "HEART Score: %d (Intermediate Risk)\n" +
            "30-day MACE Risk: 12-20%%\n\n" +
            "Recommendations:\n" +
            "• Admit to observation unit\n" +
            "• Serial troponins (0h, 3h)\n" +
            "• Consider stress testing",
            heartScore.getValue()
        ));
        intermediateCard.setIndicator(CdsHooksIndicator.WARNING);
        
        cards.add(intermediateCard);
    }
    
    // Phase 6: Check if aspirin ordered
    String draftOrdersBundle = request.getContext().getDraftOrders();
    if (draftOrdersBundle != null) {
        Bundle draftOrders = parseFHIRBundle(draftOrdersBundle);
        boolean aspirinOrdered = draftOrders.getEntry().stream()
            .anyMatch(e -> medicationContainsAspirin((MedicationRequest) e.getResource()));
        
        if (!aspirinOrdered && heartScore.getValue() >= 4) {
            CdsHooksCard aspirinCard = new CdsHooksCard();
            aspirinCard.setSummary("💊 Consider Aspirin");
            aspirinCard.setDetail(
                "Aspirin 162-325mg recommended for suspected ACS\n" +
                "(contraindicated if active bleeding or aspirin allergy)"
            );
            aspirinCard.setIndicator(CdsHooksIndicator.INFO);
            
            // Phase 6: Add order suggestion
            aspirinCard.addSuggestion(
                CdsHooksSuggestion.create(
                    "Order Aspirin 325mg PO x1",
                    createMedicationRequestAction("aspirin_325mg", patient)
                )
            );
            
            cards.add(aspirinCard);
        }
    }
    
    CdsHooksResponse response = new CdsHooksResponse();
    response.setCards(cards);
    return response;
}
