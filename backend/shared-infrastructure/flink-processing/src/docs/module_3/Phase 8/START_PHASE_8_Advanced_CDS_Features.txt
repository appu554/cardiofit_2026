    FromFHIR(String patientId, LocalDateTime since) {
        // Search for Observation resources
        Bundle bundle = fhirClient
            .search()
            .forResource(Observation.class)
            .where(Observation.PATIENT.hasId(patientId))
            .where(Observation.CATEGORY.exactly().code("laboratory"))
            .where(Observation.DATE.afterOrEquals().day(since.toString()))
            .returnBundle(Bundle.class)
            .execute();
        
        LabResults labResults = new LabResults();
        labResults.setPatientId(patientId);
        
        // Process each observation
        for (Bundle.BundleEntryComponent entry : bundle.getEntry()) {
            Observation obs = (Observation) entry.getResource();
            
            // Extract LOINC code and value
            String loincCode = obs.getCode().getCodingFirstRep().getCode();
            Quantity value = obs.getValueQuantity();
            
            // Map to internal lab result
            mapLoincToLabResult(labResults, loincCode, value);
        }
        
        labResults.setCollectionTime(LocalDateTime.now());
        return labResults;
    }

    /**
     * Import medications from FHIR MedicationRequest
     */
    public List<MedicationOrder> importMedicationsFromFHIR(String patientId) {
        Bundle bundle = fhirClient
            .search()
            .forResource(MedicationRequest.class)
            .where(MedicationRequest.PATIENT.hasId(patientId))
            .where(MedicationRequest.STATUS.exactly().code("active"))
            .returnBundle(Bundle.class)
            .execute();
        
        List<MedicationOrder> medications = new ArrayList<>();
        
        for (Bundle.BundleEntryComponent entry : bundle.getEntry()) {
            MedicationRequest medRequest = (MedicationRequest) entry.getResource();
            
            // Extract medication details
            String medicationName = extractMedicationName(medRequest);
            Dosage dosage = medRequest.getDosageInstructionFirstRep();
            
            MedicationOrder order = MedicationOrder.builder()
                .orderId(medRequest.getIdElement().getIdPart())
                .patientId(patientId)
                .medicationName(medicationName)
                .dose(extractDose(dosage))
                .route(dosage.getRoute().getCodingFirstRep().getDisplay())
                .frequency(dosage.getTiming().getCode().getText())
                .orderedAt(convertFHIRDateTime(medRequest.getAuthoredOn()))
                .status(OrderStatus.ACTIVE)
                .build();
            
            medications.add(order);
        }
        
        return medications;
    }
    
    /**
     * Import conditions/diagnoses from FHIR Condition
     */
    public List<Condition> importConditionsFromFHIR(String patientId) {
        Bundle bundle = fhirClient
            .search()
            .forResource(org.hl7.fhir.r4.model.Condition.class)
            .where(org.hl7.fhir.r4.model.Condition.PATIENT.hasId(patientId))
            .returnBundle(Bundle.class)
            .execute();
        
        List<Condition> conditions = new ArrayList<>();
        
        for (Bundle.BundleEntryComponent entry : bundle.getEntry()) {
            org.hl7.fhir.r4.model.Condition fhirCondition = 
                (org.hl7.fhir.r4.model.Condition) entry.getResource();
            
            // Extract ICD-10 code
            String icd10Code = fhirCondition.getCode().getCodingFirstRep().getCode();
            String conditionName = fhirCondition.getCode().getCodingFirstRep().getDisplay();
            
            Condition condition = Condition.builder()
                .conditionId(fhirCondition.getIdElement().getIdPart())
                .code(icd10Code)
                .name(conditionName)
                .onsetDate(convertFHIRDate(fhirCondition.getOnsetDateTimeType().getValue()))
                .status(fhirCondition.getClinicalStatus().getCodingFirstRep().getCode())
                .build();
            
            conditions.add(condition);
        }
        
        return conditions;
    }
    
    /**
     * Export CDS recommendation to FHIR ServiceRequest
     */
    public String exportRecommendationToFHIR(ProtocolRecommendation recommendation) {
        ServiceRequest serviceRequest = new ServiceRequest();
        
        // Set patient reference
        serviceRequest.setSubject(new Reference("Patient/" + recommendation.getPatientId()));
        
        // Set recommendation as code
        CodeableConcept code = new CodeableConcept();
        code.setText(recommendation.getRecommendation());
        serviceRequest.setCode(code);
        
        // Set intent and status
        serviceRequest.setIntent(ServiceRequest.ServiceRequestIntent.PROPOSAL);
        serviceRequest.setStatus(ServiceRequest.ServiceRequestStatus.DRAFT);
        
        // Add rationale
        Annotation note = new Annotation();
        note.setText(recommendation.getRationale());
        serviceRequest.addNote(note);
        
        // Add priority
        serviceRequest.setPriority(convertToPriority(recommendation.getUrgency()));
        
        // Create on FHIR server
        MethodOutcome outcome = fhirClient
            .create()
            .resource(serviceRequest)
            .execute();
        
        return outcome.getId().getIdPart();
    }
    
    /**
     * CDS Hooks implementation
     * Triggered when clinician opens order entry screen
     */
    @PostMapping("/cds-services/order-select")
    public CdsHooksResponse handleOrderSelect(@RequestBody CdsHooksRequest request) {
        // Extract patient context
        String patientId = request.getContext().getPatientId();
        String draftOrderId = request.getContext().getDraftOrders();
        
        // Get patient data from FHIR
        Patient patient = importPatientFromFHIR(patientId);
        LabResults labs = importLabsFromFHIR(patientId, LocalDateTime.now().minusDays(7));
        List<MedicationOrder> currentMeds = importMedicationsFromFHIR(patientId);
        
        // Run CDS logic
        List<Alert> alerts = new ArrayList<>();
        
        // Check for drug interactions
        if (draftOrderId != null) {
            MedicationOrder draftOrder = parseDraftOrder(draftOrderId);
            alerts.addAll(checkDrugInteractions(draftOrder, currentMeds));
        }
        
        // Check for contraindications
        alerts.addAll(checkContraindications(patient, labs));
        
        // Check for duplicate therapy
        alerts.addAll(checkDuplicateTherapy(draftOrderId, currentMeds));
        
        // Convert to CDS Hooks response
        return buildCdsHooksResponse(alerts);
    }
    
    /**
     * Build CDS Hooks response with cards
     */
    private CdsHooksResponse buildCdsHooksResponse(List<Alert> alerts) {
        CdsHooksResponse response = new CdsHooksResponse();
        
        for (Alert alert : alerts) {
            CdsHooksCard card = new CdsHooksCard();
            card.setSummary(alert.getMessage());
            card.setDetail(alert.getDetails());
            card.setSource(new CdsHooksSource("Hospital CDS System", "v1.0"));
            card.setIndicator(convertToIndicator(alert.getSeverity()));
            
            // Add suggestions
            if (alert.getRecommendations() != null) {
                for (String recommendation : alert.getRecommendations()) {
                    CdsHooksSuggestion suggestion = new CdsHooksSuggestion();
                    suggestion.setLabel(recommendation);
                    card.addSuggestion(suggestion);
                }
            }
            
            response.addCard(card);
        }
        
        return response;
    }

    public String getSMARTAuthorizationUrl(String clientId, String redirectUri) {
        return String.format(
            "https://fhir.hospital.org/oauth/authorize?" +
            "response_type=code&" +
            "client_id=%s&" +
            "redirect_uri=%s&" +
            "scope=patient/*.read launch/patient&" +
            "state=%s",
            clientId,
            URLEncoder.encode(redirectUri, StandardCharsets.UTF_8),
            UUID.randomUUID().toString()
        );
    }
    
    /**
     * Exchange authorization code for access token
     */
    public SMARTToken exchangeCodeForToken(String code, String clientId, 
                                          String clientSecret, String redirectUri) {
        // HTTP POST to token endpoint
        MultiValueMap<String, String> params = new LinkedMultiValueMap<>();
        params.add("grant_type", "authorization_code");
        params.add("code", code);
        params.add("redirect_uri", redirectUri);
        params.add("client_id", clientId);
        params.add("client_secret", clientSecret);
        
        ResponseEntity<SMARTToken> response = restTemplate.postForEntity(
            "https://fhir.hospital.org/oauth/token",
            params,
            SMARTToken.class
        );
        
        return response.getBody();
    }
