public class TestOrderingRules {
    public List<TestRecommendation> getTestPanel(
        String protocolSpecialty,
        PatientContext context) {
        
        switch (protocolSpecialty) {
            case "SEPSIS":
                return getSepsisPanel();
            case "CARDIAC":
                return getCardiacPanel();
            case "RESPIRATORY":
                return getRespiratoryPanel();
            default:
                return Collections.emptyList();
        }
    }
    
    private List<TestRecommendation> getSepsisPanel() {
        return Arrays.asList(
            "Serum Lactate",
            "Blood Culture x2",

            "Comprehensive Metabolic Panel",
            "Procalcitonin",
            "Chest X-Ray",
            "Urinalysis"
        ).stream()
         .map(this::createPanelRecommendation)
         .collect(Collectors.toList());
    }
}

@DisplayName("Integration: ROHAN sepsis patient gets correct test recommendations")
public void testROHANCompleteWorkflow() {
    // 1. Create ROHAN patient
    PatientContext rohan = createPatientROHAN();
    
    // 2. Match protocols
    Protocol sepsisProtocol = protocolMatcher.getBestMatch(rohan);
    assertEquals("SEPSIS-SSC-2021", sepsisProtocol.getId());
    
    // 3. Get test recommendations
    List<TestRecommendation> tests = 
        testRecommender.recommendTests(rohan, sepsisProtocol);
    
    // 4. Verify Hour-1 Bundle tests
    assertTrue(hasTest(tests, "Serum Lactate"), "Must have lactate");
    assertTrue(hasTest(tests, "Blood Culture"), "Must have blood cultures");
    
    // 5. Verify priorities
    tests.stream()
        .filter(t -> t.getTestName().equals("Serum Lactate"))
        .findFirst()
        .ifPresent(lactate -> {
            assertEquals("P0", lactate.getPriority());
            assertTrue(lactate.getTimeframeMinutes() <= 60);
            assertEquals("STAT", lactate.getUrgency());
        });
    
    // 6. Verify imaging appropriate
    TestRecommendation cxr = getTest(tests, "Chest X-Ray");
    assertNotNull(cxr);
    assertEquals("USUALLY_APPROPRIATE", cxr.getAppropriateness());
    
    // 7. Verify safety checks passed
    tests.forEach(test -> {
        if (!test.isSafeToOrder()) {
            fail("Test " + test.getTestName() + " failed safety checks");
        }
    });
    
    // 8. Print summary
    System.out.println("\n=== ROHAN Test Recommendations ===");
    System.out.println("Total tests: " + tests.size());
    tests.stream()
        .sorted(Comparator.comparing(TestRecommendation::getPriority))
        .forEach(test -> {
            System.out.println(String.format(
                "  [%s] %s - %s (Complete within %d min)",
                test.getPriority(),
                test.getTestName(),
                test.getUrgency(),
                test.getTimeframeMinutes()
            ));
        });
}

## 🚨 **Common Pitfalls to Avoid**


1. **Don't perfect every test on Day 1**
   - Get 10-15 core tests working first
   - Add details to remaining tests throughout week


2. **Don't skip clinical validation**
   - Reference ranges must be accurate
   - Appropriateness criteria must match guidelines
   - Get physician sign-off


3. **Don't hardcode values**
   - Use YAML for all test definitions
   - Keep code generic and reusable


4. **Don't forget safety checks**
   - Contrast + renal impairment = dangerous
   - Radiation + pregnancy = cautious
   - Always check contraindications


5. **Don't ignore performance**
   - Cache loaded tests
   - Index by test name/ID
   - Avoid redundant YAML parsing
