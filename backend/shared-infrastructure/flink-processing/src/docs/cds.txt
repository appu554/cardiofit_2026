### **2. ProtocolTrigger.java**


```java
package com.cds.knowledgebase.model;


import lombok.Data;
import java.util.List;
import java.util.Map;


@Data
public class ProtocolTrigger {
    private String type; // ALERT, VITAL_SIGN, LAB_VALUE, SCORE, COMBINED
    private String operator; // AND, OR
    private List<TriggerCondition> conditions;
    
    @Data
    public static class TriggerCondition {
        private String source; // e.g., "alert", "vital", "lab", "score"
        private String field;  // e.g., "NEWS2", "lactate", "alert_type"
        private String operator; // >=, <=, ==, >, <, CONTAINS
        private Object value;
        private String unit;
        
        public boolean evaluate(PatientContext context) {
            Object actualValue = context.getValue(source, field);
            if (actualValue == null) return false;
            
            switch (operator) {
                case ">=":
                    return compareNumeric(actualValue, value) >= 0;
                case ">":
                    return compareNumeric(actualValue, value) > 0;
                case "<=":
                    return compareNumeric(actualValue, value) <= 0;
                case "<":
                    return compareNumeric(actualValue, value) < 0;
                case "==":
                    return actualValue.equals(value);
                case "CONTAINS":
                    return actualValue.toString().contains(value.toString());
                default:
                    return false;
            }
        }
        
        private int compareNumeric(Object actual, Object expected) {
            double actualNum = Double.parseDouble(actual.toString());
            double expectedNum = Double.parseDouble(expected.toString());
            return Double.compare(actualNum, expectedNum);
        }
    }
    
    public boolean evaluate(PatientContext context) {
        if ("AND".equals(operator)) {
            return conditions.stream().allMatch(c -> c.evaluate(context));
        } else { // OR
            return conditions.stream().anyMatch(c -> c.evaluate(context));
        }
    }
}
```

```java
package com.cds.knowledgebase.loader;


import com.cds.knowledgebase.model.Protocol;
import org.yaml.snakeyaml.Yaml;
import org.yaml.snakeyaml.constructor.Constructor;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;


import java.io.InputStream;
import java.util.HashMap;
import java.util.Map;


public class ProtocolLoader {
    private static final Logger logger = LoggerFactory.getLogger(ProtocolLoader.class);
    private final Map<String, Protocol> protocolCache = new HashMap<>();
    private final Yaml yaml;
    
    public ProtocolLoader() {
        this.yaml = new Yaml(new Constructor(Protocol.class));
    }
    
    /**
     * Load all protocols from resources directory
     */
    public void loadAllProtocols() {
        logger.info("Loading protocols from knowledge base...");
        
        String[] protocolFiles = {
            "sepsis-bundle-ssc-2021.yaml",
            "respiratory-failure.yaml",
            "stemi-protocol.yaml"
        };
        
        for (String filename : protocolFiles) {
            try {
                Protocol protocol = loadProtocol(filename);
                protocolCache.put(protocol.getId(), protocol);
                logger.info("Loaded protocol: {} (version: {})", 
                    protocol.getName(), protocol.getVersion());
            } catch (Exception e) {
                logger.error("Failed to load protocol: {}", filename, e);
            }
        }
        
        logger.info("Total protocols loaded: {}", protocolCache.size());
    }
    
    /**
     * Load single protocol from YAML file
     */
    public Protocol loadProtocol(String filename) throws Exception {
        String resourcePath = "/knowledge-base/protocols/" + filename;
        
        try (InputStream inputStream = getClass().getResourceAsStream(resourcePath)) {
            if (inputStream == null) {
                throw new IllegalArgumentException("Protocol file not found: " + filename);
            }
            
            Protocol protocol = yaml.load(inputStream);
            
            // Validate protocol
            validateProtocol(protocol);
            
            return protocol;
        }
    }
    
    /**
     * Validate protocol structure
     */
    private void validateProtocol(Protocol protocol) throws Exception {
        if (protocol.getId() == null || protocol.getId().isEmpty()) {
            throw new IllegalArgumentException("Protocol ID is required");
        }
        
        if (protocol.getTriggers() == null || protocol.getTriggers().isEmpty()) {
            throw new IllegalArgumentException("Protocol must have at least one trigger");
        }
        
        if (protocol.getActions() == null || protocol.getActions().isEmpty()) {
            throw new IllegalArgumentException("Protocol must have at least one action");
        }
        
        logger.debug("Protocol validated: {}", protocol.getId());
    }
    
    /**
     * Get protocol by ID
     */
    public Protocol getProtocol(String protocolId) {
        return protocolCache.get(protocolId);
    }
    
    /**
     * Get all loaded protocols
     */
    public Map<String, Protocol> getAllProtocols() {
        return new HashMap<>(protocolCache);
    }
    
    /**
     * Reload specific protocol
     */
    public void reloadProtocol(String filename) throws Exception {
        Protocol protocol = loadProtocol(filename);
        protocolCache.put(protocol.getId(), protocol);
        logger.info("Reloaded protocol: {}", protocol.getName());
    }
}
```

```java
package com.cds.knowledgebase.matcher;


import com.cds.knowledgebase.model.Protocol;
import com.cds.knowledgebase.model.PatientContext;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;


import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;


public class ProtocolMatcher {
    private static final Logger logger = LoggerFactory.getLogger(ProtocolMatcher.class);
    private final Map<String, Protocol> protocols;
    
    public ProtocolMatcher(Map<String, Protocol> protocols) {
        this.protocols = protocols;
    }
    
    /**
     * Find all protocols that match patient context
     */
    public List<Protocol> matchProtocols(PatientContext context) {
        logger.debug("Matching protocols for patient: {}", context.getPatientId());
        
        List<Protocol> matchedProtocols = protocols.values().stream()
            .filter(protocol -> protocol.shouldActivate(context))
            .collect(Collectors.toList());
        
        logger.info("Matched {} protocols for patient {}", 
            matchedProtocols.size(), context.getPatientId());
        
        matchedProtocols.forEach(p -> 
            logger.debug("  - {}", p.getName()));
        
        return matchedProtocols;
    }
    
    /**
     * Get highest priority protocol match
     */
    public Protocol getBestMatch(PatientContext context) {
        List<Protocol> matches = matchProtocols(context);
        
        if (matches.isEmpty()) {
            return null;
        }
        
        // Priority: sepsis > cardiac > respiratory > others
        return matches.stream()
            .sorted((p1, p2) -> comparePriority(p1, p2))
            .findFirst()
            .orElse(null);
    }
    
    private int comparePriority(Protocol p1, Protocol p2) {
        Map<String, Integer> priorityMap = Map.of(
            "SEPSIS", 1,
            "CARDIAC", 2,
            "RESPIRATORY", 3
        );
        
        int priority1 = priorityMap.getOrDefault(p1.getSpecialty(), 99);
        int priority2 = priorityMap.getOrDefault(p2.getSpecialty(), 99);
        
        return Integer.compare(priority1, priority2);
    }
}
```

```java
package com.cds.knowledgebase.builder;


import com.cds.knowledgebase.model.Protocol;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;


import java.util.ArrayList;
import java.util.List;
import java.util.stream.Collectors;


public class ActionBuilder {
    private static final Logger logger = LoggerFactory.getLogger(ActionBuilder.class);
    private final ContraindicationChecker contraindicationChecker;
    
    public ActionBuilder(ContraindicationChecker contraindicationChecker) {
        this.contraindicationChecker = contraindicationChecker;
    }
    
    /**
     * Build recommendations from protocol actions
     */
    public List<Recommendation> buildRecommendations(
            Protocol protocol, 
            PatientContext context) {
        
        logger.info("Building recommendations from protocol: {}", protocol.getName());
        
        List<Recommendation> recommendations = new ArrayList<>();
        
        for (ProtocolAction action : protocol.getActions()) {
            // Check contraindications
            if (contraindicationChecker.hasContraindication(action, context)) {
                logger.debug("Action {} contraindicated, finding alternative", 
                    action.getId());
                
                ProtocolAction alternative = findAlternative(action, context);
                if (alternative != null) {
                    recommendations.add(createRecommendation(alternative, context, true));
                }
            } else {
                recommendations.add(createRecommendation(action, context, false));
            }
        }
        
        // Sort by priority
        recommendations.sort((r1, r2) -> 
            comparePriority(r1.getPriority(), r2.getPriority()));
        
        logger.info("Generated {} recommendations", recommendations.size());
        return recommendations;
    }
    
    private Recommendation createRecommendation(
            ProtocolAction action, 
            PatientContext context,
            boolean isAlternative) {
        
        Recommendation rec = new Recommendation();
        rec.setId(generateRecommendationId());
        rec.setType(action.getType());
        rec.setPriority(action.getPriority());
        rec.setTitle(action.getTitle());
        rec.setDescription(action.getDescription());
        rec.setRationale(buildRationale(action, context, isAlternative));
        rec.setTimeframeMinutes(action.getTimeframeMinutes());
        rec.setEvidence(action.getEvidenceReferences());
        rec.setEvidenceLevel(action.getEvidenceLevel());
        
        // Add specific details based on action type
        if ("MEDICATION".equals(action.getType())) {
            rec.setMedicationDetails(buildMedicationDetails(action, context));
        } else if ("LAB_TEST".equals(action.getType())) {
            rec.setLabDetails(action.getLabTest());
        } else if ("IMAGING".equals(action.getType())) {
            rec.setImagingDetails(action.getImaging());
        }
        
        return rec;
    }
    
    private String buildRationale(
            ProtocolAction action, 
            PatientContext context,
            boolean isAlternative) {
        
        StringBuilder rationale = new StringBuilder();
        rationale.append(action.getRationale());
        
        if (isAlternative) {
            rationale.append(" (Alternative therapy due to contraindication)");
        }
        
        // Add patient-specific context
        if (context.hasRenalImpairment()) {
            rationale.append(" Dose adjusted for renal function (eGFR: ")
                    .append(context.getEgfr())
                    .append(" mL/min).");
        }
        
        return rationale.toString();
    }
    
    private MedicationDetails buildMedicationDetails(
            ProtocolAction action, 
            PatientContext context) {
        
        var medOrder = action.getMedication();
        MedicationDetails details = new MedicationDetails();
        
        details.setMedicationName(medOrder.getMedicationName());
        details.setDose(calculateDose(medOrder, context));
        details.setRoute(medOrder.getRoute());
        details.setFrequency(medOrder.getFrequency());
        details.setDuration(medOrder.getDuration());
        
        return details;
    }
    
    private String calculateDose(
            ProtocolAction.MedicationOrder medOrder, 
            PatientContext context) {
        
        String baseDose = medOrder.getDose();
        
        // Check if renal adjustment needed
        if (context.hasRenalImpairment() && 
            medOrder.getRenalAdjustment() != null) {
            
            double egfr = context.getEgfr();
            if (egfr < 30) {
                return medOrder.getRenalAdjustment().get("severe");
            } else if (egfr < 60) {
                return medOrder.getRenalAdjustment().get("moderate");
            }
        }
        
        return baseDose;
    }
    
    private ProtocolAction findAlternative(
            ProtocolAction action, 
            PatientContext context) {
        
        // For medication actions with alternatives
        if ("MEDICATION".equals(action.getType()) && 
            action.getMedication().getAlternatives() != null) {
            
            for (String altMedId : action.getMedication().getAlternatives()) {

                    // Create alternative action
                    return createAlternativeAction(action, altMedId);
                }
            }
        }
        
        return null;
    }

        Map<String, Integer> priorityOrder = Map.of(
```

```java
package com.cds.knowledgebase.builder;


import com.cds.knowledgebase.model.ProtocolAction;
import com.cds.knowledgebase.model.PatientContext;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;


import java.util.List;
import java.util.Set;


public class ContraindicationChecker {
    private static final Logger logger = LoggerFactory.getLogger(ContraindicationChecker.class);
    
    /**
     * Check if action has contraindications for patient
     */
    public boolean hasContraindication(ProtocolAction action, PatientContext context) {
        if ("MEDICATION".equals(action.getType())) {
            return hasMedicationContraindication(action, context);
        }
        
        return false;
    }
    
    private boolean hasMedicationContraindication(
            ProtocolAction action, 
            PatientContext context) {
        
        String medicationName = action.getMedication().getMedicationName();
        
        // Check allergies
        if (hasAllergy(medicationName, context)) {
            logger.info("Patient allergic to {}", medicationName);
            return true;
        }
        
        // Check other contraindications
        List<String> contraindications = action.getContraindications();
        if (contraindications != null) {
            for (String contraindication : contraindications) {
                if (hasCondition(contraindication, context)) {
                    logger.info("Patient has contraindication: {}", contraindication);
                    return true;
                }
            }
        }
        
        return false;
    }
    
    private boolean hasAllergy(String medicationName, PatientContext context) {
        Set<String> allergies = context.getAllergies();
        
        // Check exact match
        if (allergies.contains(medicationName.toLowerCase())) {
            return true;
        }
        
        // Check drug class (simplified - would need full drug class database)
        if (medicationName.toLowerCase().contains("cillin") && 
            allergies.contains("penicillin")) {
            return true;
        }
        
        if (medicationName.toLowerCase().contains("cef") && 
            allergies.contains("cephalosporin")) {

        }
        
        return false;

        Set<String> conditions = context.getMedicalConditions();
```
