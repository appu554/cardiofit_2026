## 🎯 **1. Architecture Overview**


### **1.1 Knowledge Base Components**


```
Clinical Knowledge Base
├── Protocols (Treatment pathways)
├── Medications (Drug database)
├── Diagnostics (Tests, imaging, procedures)
├── Guidelines (Clinical guidelines)
├── Evidence (Research citations)
├── Rules (Decision rules)
├── Scoring Systems (NEWS2, qSOFA, SIRS)
└── Reference Data (ICD codes, LOINC codes, RxNorm)
```


### **1.2 Design Principles**


**Externalized Configuration**
- No hardcoded clinical logic in code
- All protocols in YAML/JSON files
- Easy to update without code changes
- Version controlled in Git


**Evidence-Based**
- Every recommendation linked to source
- Evidence levels documented
- Regular updates as guidelines evolve


**Modular & Composable**
- Reusable components
- Protocols can reference other protocols
- Actions can be shared across protocols


**Clinician-Friendly**
- Human-readable format
- Clear structure
- Extensive comments
- Examples provided

## 📋 **3. Protocol Definition Framework**


### **3.1 Complete Protocol Template**


**File:** `templates/protocol-template.yaml`


```yaml
# ==============================================================================

# ==============================================================================
# This template provides the structure for defining clinical protocols
# Copy this template and customize for your specific protocol
# ==============================================================================


metadata:
  protocol_id: "PROTOCOL_ID_UPPERCASE"              # Unique identifier
  name: "Human-Readable Protocol Name"             # Display name
  version: "1.0"                                    # Semantic versioning
  effective_date: "2025-01-01"                     # When protocol becomes active
  review_date: "2026-01-01"                        # Next review date
  status: "active"                                  # active | draft | deprecated
  
  # Classification
  category: "CATEGORY"                              # sepsis | cardiac | respiratory | metabolic | neurological
  specialty: "SPECIALTY"                            # emergency | critical_care | cardiology | pulmonology
  urgency: "high"                                   # critical | high | medium | low
  
  # Authorship
  developed_by:
    organization: "Organization Name"
    committee: "Clinical Committee"
    lead_author: "Dr. Jane Smith"
    contributors:
      - "Dr. John Doe"
      - "Dr. Sarah Johnson"
  
  # Evidence base
  evidence_source:
    primary_guideline:
      name: "Guideline Name"
      organization: "Professional Society"
      year: 2021
      url: "https://guideline-url.org"
    evidence_level: "A"                             # A | B | C | Expert Opinion
    class_of_recommendation: "I"                    # I | IIa | IIb | III
  
  # Validation
  validated_by:
    - name: "Dr. Clinical Validator"
      role: "Chief Medical Officer"
      date: "2025-01-15"
  
  last_updated: "2025-01-15"
  last_updated_by: "System Administrator"
  
  # Usage tracking
  usage_count: 0
  success_rate: null
  average_outcome_score: null


# ==============================================================================
# TRIGGER CRITERIA
# ==============================================================================
# Define when this protocol should be activated
# Supports: ALL_OF, ANY_OF, nested conditions


trigger_criteria:
  evaluation_logic: "ALL_OF"                        # ALL_OF | ANY_OF
  
  conditions:
    # Condition 1: Example - SIRS Criteria
    - condition_id: "SIRS_CRITERIA"
      condition_type: "SCORING_SYSTEM"
      description: "SIRS criteria ≥2"
      parameters:
        scoring_system: "SIRS"

        operator: ">="                              # >= | > | <= | < | == | !=

        - vital_signs: ["temperature", "heart_rate", "respiratory_rate"]
        - labs: ["2524-7"]                          # LOINC for lactate
      weight: 1.0                                   # Importance weight

    - condition_id: "ELEVATED_LACTATE"
      condition_type: "LAB_VALUE"
      description: "Lactate >2.0 mmol/L"
      parameters:
        loinc_code: "2524-7"
        threshold: 2.0
        operator: ">"
        unit: "mmol/L"
      data_requirements:
        - labs: ["2524-7"]
      weight: 1.5                                   # Higher importance
      
    # Condition 3: Example - Nested ANY_OF
    - condition_id: "INFECTION_MARKERS"
      condition_type: "ANY_OF"
      description: "Suspected infection"
      sub_conditions:
        - indicator: "FEVER"
          parameter: "temperature"
          threshold: 38.0
          operator: ">"
          unit: "celsius"
        
        - indicator: "HYPOTHERMIA"
          parameter: "temperature"
          threshold: 36.0
          operator: "<"
          unit: "celsius"
        
        - indicator: "LEUKOCYTOSIS"
          parameter: "wbc"
          loinc_code: "6690-2"
          threshold: 12.0
          operator: ">"
          unit: "K/uL"
        
        - indicator: "LEUKOPENIA"
          parameter: "wbc"
          loinc_code: "6690-2"
          threshold: 4.0
          operator: "<"
          unit: "K/uL"
      weight: 1.0
  
  # Additional context filters
  context_filters:
    min_age: 18                                     # Minimum age (years)
    max_age: null                                   # Maximum age (null = no limit)
    exclude_conditions:                             # Don't activate if patient has:
      - "DNR_COMFORT_CARE"
      - "END_STAGE_DISEASE"
    require_location: null                          # null | ICU | ED | WARD
    time_restrictions: null                         # null | daytime_only | 24_7


# ==============================================================================
# CONFIDENCE SCORING
# ==============================================================================
# How confident are we that this protocol applies?


confidence_scoring:
  base_score: 0.80                                  # Starting confidence (0-1)
  
  # Modifiers adjust confidence based on clinical context
  modifiers:
    - condition: "LACTATE_SEVERELY_ELEVATED"
      description: "Lactate >4.0 mmol/L"
      parameters:
        loinc_code: "2524-7"
        threshold: 4.0
        operator: ">"
      adjustment: +0.10
      rationale: "Severe hyperlactatemia strongly suggests septic shock"
    
    - condition: "MULTIPLE_ORGAN_DYSFUNCTION"
      description: "≥2 organ systems failing"
      parameters:
        count_threshold: 2
        organ_systems: ["respiratory", "renal", "cardiac", "hepatic"]
      adjustment: +0.10
      rationale: "Multi-organ failure increases sepsis likelihood"
    
    - condition: "RECENT_NEGATIVE_CULTURES"
      description: "Negative blood cultures in last 48h"
      parameters:
        time_window_hours: 48
        test_type: "blood_culture"
        result: "negative"

  activation_threshold: 0.70
  
  # Require human confirmation if confidence below this

# ==============================================================================

  bundle_deadline_hours: 3                          # Overall deadline
  
  time_windows:
    # Hour 0: Immediate actions (recognition)

        - "ASSESS_HEMODYNAMICS"

    - window_name: "hour_1"

      critical: true
      mortality_impact: "Each hour delay increases mortality by 7.6%"

        - "ADMINISTER_BROAD_SPECTRUM_ANTIBIOTICS"
        - "INITIATE_FLUID_RESUSCITATION"
      
      # Alerts if deadline approaching
      alerts:
        - time_remaining_minutes: 30

# ==============================================================================
