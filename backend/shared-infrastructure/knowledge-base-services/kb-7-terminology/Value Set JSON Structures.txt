First, let's define the standard structure all Value Sets will follow:


```json
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "title": "KB7 Value Set Schema",
  "description": "FHIR R4-compatible Value Set structure optimized for KB7 Bridge",
  "type": "object",
  "required": ["resourceType", "id", "url", "name", "status", "compose", "expansion"],
  "properties": {
    "resourceType": { "const": "ValueSet" },
    "id": { "type": "string", "pattern": "^vs-[a-z0-9-]+$" },
    "url": { "type": "string", "format": "uri" },
    "version": { "type": "string" },
    "name": { "type": "string" },

    "experimental": { "type": "boolean" },
    "date": { "type": "string", "format": "date-time" },
    "publisher": { "type": "string" },
    "description": { "type": "string" },
    "compose": { "$ref": "#/definitions/compose" },
```

## Part 3: Value Set Loader Service


### Go Implementation


```go
// valuesets/loader.go
package valuesets


import (
        "context"
        "encoding/json"
        "fmt"
        "os"
        "path/filepath"
        "sync"
)


// ValueSet represents a FHIR R4 ValueSet
type ValueSet struct {
        ResourceType string    `json:"resourceType"`
        ID           string    `json:"id"`
        URL          string    `json:"url"`
        Version      string    `json:"version"`
        Name         string    `json:"name"`
        Title        string    `json:"title"`
        Status       string    `json:"status"`
        Description  string    `json:"description"`
        Purpose      string    `json:"purpose"`
        Compose      *Compose  `json:"compose,omitempty"`
        Expansion    *Expansion `json:"expansion,omitempty"`
}


// Compose defines the value set composition
type Compose struct {
        Include []ConceptSet `json:"include"`
        Exclude []ConceptSet `json:"exclude,omitempty"`
}


// ConceptSet defines a set of concepts from a system
type ConceptSet struct {
        System  string    `json:"system"`
        Concept []Concept `json:"concept,omitempty"`
        Filter  []Filter  `json:"filter,omitempty"`
}


// Concept represents a single coded concept
type Concept struct {
        Code    string `json:"code"`
        Display string `json:"display"`
}


// Filter for ECL-style filtering
type Filter struct {
        Property string `json:"property"`
        Op       string `json:"op"`
        Value    string `json:"value"`
}


// Expansion contains the expanded codes
type Expansion struct {
        Identifier string              `json:"identifier"`
        Timestamp  string              `json:"timestamp"`
        Total      int                 `json:"total"`
        Contains   []ExpansionContains `json:"contains"`
}


// ExpansionContains represents an expanded code
type ExpansionContains struct {
        System  string `json:"system"`
        Code    string `json:"code"`
        Display string `json:"display"`
}


// ValueSetRegistry manages all loaded value sets
type ValueSetRegistry struct {
        mu        sync.RWMutex
        valueSets map[string]*ValueSet
        codeIndex map[string]map[string][]string // system -> code -> []valueSetIDs
}


// NewValueSetRegistry creates a new registry
func NewValueSetRegistry() *ValueSetRegistry {
        return &ValueSetRegistry{
                valueSets: make(map[string]*ValueSet),
                codeIndex: make(map[string]map[string][]string),
        }
}


// LoadFromDirectory loads all value sets from a directory
func (r *ValueSetRegistry) LoadFromDirectory(dir string) error {
        files, err := filepath.Glob(filepath.Join(dir, "vs-*.json"))
        if err != nil {
                return fmt.Errorf("failed to list value set files: %w", err)
        }


        for _, file := range files {
                if err := r.LoadFromFile(file); err != nil {
                        return fmt.Errorf("failed to load %s: %w", file, err)
                }
        }


        return nil
}


// LoadFromFile loads a single value set from a JSON file
func (r *ValueSetRegistry) LoadFromFile(filename string) error {
        data, err := os.ReadFile(filename)
        if err != nil {
                return err
        }


        var vs ValueSet
        if err := json.Unmarshal(data, &vs); err != nil {
                return err
        }


        return r.Register(&vs)
}


// Register adds a value set to the registry
func (r *ValueSetRegistry) Register(vs *ValueSet) error {
        r.mu.Lock()
        defer r.mu.Unlock()


        r.valueSets[vs.ID] = vs


        // Build code index from expansion
        if vs.Expansion != nil {
                for _, code := range vs.Expansion.Contains {
                        if r.codeIndex[code.System] == nil {
                                r.codeIndex[code.System] = make(map[string][]string)
                        }
                        r.codeIndex[code.System][code.Code] = append(
                                r.codeIndex[code.System][code.Code],
                                vs.ID,
                        )
                }
        }


        return nil
}


// Get retrieves a value set by ID
func (r *ValueSetRegistry) Get(id string) (*ValueSet, bool) {
        r.mu.RLock()
        defer r.mu.RUnlock()
        vs, ok := r.valueSets[id]
        return vs, ok
}


// GetByURL retrieves a value set by URL
func (r *ValueSetRegistry) GetByURL(url string) (*ValueSet, bool) {
        r.mu.RLock()
        defer r.mu.RUnlock()
        for _, vs := range r.valueSets {
                if vs.URL == url {
                        return vs, true
                }
        }
        return nil, false
}


// Contains checks if a code is in a value set (O(1) lookup)
func (r *ValueSetRegistry) Contains(valueSetID, system, code string) bool {
        r.mu.RLock()
        defer r.mu.RUnlock()


        vs, ok := r.valueSets[valueSetID]
        if !ok || vs.Expansion == nil {
                return false
        }


        for _, c := range vs.Expansion.Contains {
                if c.System == system && c.Code == code {
                        return true
                }
        }
        return false
}


// FindValueSetsForCode finds all value sets containing a code
func (r *ValueSetRegistry) FindValueSetsForCode(system, code string) []string {
        r.mu.RLock()
        defer r.mu.RUnlock()


        if systemCodes, ok := r.codeIndex[system]; ok {
                if vsIDs, ok := systemCodes[code]; ok {
                        return vsIDs
                }
        }
        return nil
}


// Expand returns all codes in a value set
func (r *ValueSetRegistry) Expand(ctx context.Context, valueSetID string) ([]ExpansionContains, error) {
        r.mu.RLock()
        defer r.mu.RUnlock()


        vs, ok := r.valueSets[valueSetID]
        if !ok {
                return nil, fmt.Errorf("value set not found: %s", valueSetID)
        }


        if vs.Expansion == nil {
                return nil, fmt.Errorf("value set has no expansion: %s", valueSetID)
        }


        return vs.Expansion.Contains, nil
}


// List returns all registered value set IDs
func (r *ValueSetRegistry) List() []string {
        r.mu.RLock()
        defer r.mu.RUnlock()


        ids := make([]string, 0, len(r.valueSets))
        for id := range r.valueSets {
                ids = append(ids, id)
        }
        return ids
}


// Stats returns registry statistics
func (r *ValueSetRegistry) Stats() map[string]interface{} {
        r.mu.RLock()
        defer r.mu.RUnlock()


        totalCodes := 0
        for _, vs := range r.valueSets {
                if vs.Expansion != nil {
                        totalCodes += vs.Expansion.Total
                }
        }


        return map[string]interface{}{
                "value_set_count": len(r.valueSets),
                "total_codes":     totalCodes,
                "systems_indexed": len(r.codeIndex),
        }
}
```
