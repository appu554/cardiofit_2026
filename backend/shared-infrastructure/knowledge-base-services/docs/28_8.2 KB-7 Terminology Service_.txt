```sql
WITH RECURSIVE hierarchy AS (
  SELECT 
    system, code, preferred_term, 
    parent_codes, 0 as level,
    ARRAY[code] as path
  FROM concepts
  WHERE array_length(parent_codes, 1) IS NULL
  
  UNION ALL
  
  SELECT 
    c.system, c.code, c.preferred_term,
    c.parent_codes, h.level + 1,
    h.path || c.code
  FROM concepts c
  JOIN hierarchy h ON c.parent_codes @> ARRAY[h.code]
  WHERE h.level < 10
)
SELECT * FROM hierarchy
WITH DATA;


CREATE UNIQUE INDEX ON concept_hierarchy(system, code);
```

### Cache Infrastructure


```typescript
// cache/terminology-cache.ts
interface CacheLayer {
  name: string;
  ttl: number;
  maxSize: string;
  priority: number;
}


class MultiLayerTerminologyCache {
  private layers: Map<string, CacheLayer> = new Map([
    ['L1_HOT', { name: 'memory', ttl: 300, maxSize: '100MB', priority: 1 }],
    ['L2_WARM', { name: 'redis', ttl: 3600, maxSize: '1GB', priority: 2 }],
    ['L3_COLD', { name: 'redis_cluster', ttl: 86400, maxSize: '10GB', priority: 3 }]
  ]);


  private invalidationStrategy: CacheInvalidationStrategy;
  private stats: CacheStatistics;


  async get(key: string, loader?: () => Promise<any>): Promise<any> {
    // Try each layer in priority order
    for (const [layerName, layer] of this.layers) {
      const value = await this.getFromLayer(layerName, key);
      if (value) {
        this.stats.recordHit(layerName);
        // Promote to higher layers
        await this.promote(key, value, layer.priority);
        return value;
      }
    }


    this.stats.recordMiss();
    
    if (loader) {
      const value = await loader();
      await this.setAllLayers(key, value);
      return value;
    }
    
    return null;
  }


  async invalidate(pattern: InvalidationPattern): Promise<void> {
    const keys = await this.findKeys(pattern);
    
    // Smart invalidation based on pattern type
    switch (pattern.type) {
      case 'concept':
        await this.invalidateConcept(pattern);
        break;
      case 'valueSet':
        await this.invalidateValueSet(pattern);
        break;
      case 'expansion':
        await this.invalidateExpansion(pattern);
        break;
      case 'cascade':
        await this.cascadeInvalidation(pattern);
        break;
    }
    
    // Publish invalidation event
    await this.publishInvalidation(pattern);
  }


  private async cascadeInvalidation(pattern: InvalidationPattern): Promise<void> {
    // Find dependent cached items
    const dependencies = await this.findDependencies(pattern);
    
    // Invalidate in correct order
    for (const dep of dependencies.sort((a, b) => a.priority - b.priority)) {
      await this.invalidate(dep);
    }
  }
}
```

### High-Performance API Implementation


```typescript
// api/terminology-service.ts
import { FastifyInstance } from 'fastify';
import { Pool } from 'pg';


export class TerminologyService {
  private db: Pool;
  private cache: MultiLayerTerminologyCache;
  private validator: TerminologyValidator;
  private expansionEngine: ExpansionEngine;


  async registerRoutes(app: FastifyInstance) {
    // Batch operations for performance
    app.post('/api/v1/batch/validate', {
      schema: {
        body: {
          type: 'object',
          properties: {
            validations: {
              type: 'array',
              items: {
                type: 'object',
                properties: {
                  system: { type: 'string' },
                  code: { type: 'string' },
                  display: { type: 'string' },
                  valueSet: { type: 'string' }
                }
              },
              maxItems: 1000
            }
          }
        }
      }
    }, async (request, reply) => {
      const { validations } = request.body;
      
      // Build single query for all validations
      const query = `
        WITH input_codes AS (
          SELECT * FROM jsonb_to_recordset($1::jsonb) AS x(
            system text, 
            code text, 
            display text
          )
        )
        SELECT 
          i.system,
          i.code,
          i.display,
          c.preferred_term,
          c.active,
          CASE 
            WHEN c.code IS NOT NULL THEN true 
            ELSE false 
          END as valid
        FROM input_codes i
        LEFT JOIN concepts c ON 
          c.system = i.system AND 
          c.code = i.code AND
          c.active = true
      `;
      
      const results = await this.db.query(query, [JSON.stringify(validations)]);
      
      return reply.send({
        results: results.rows,
        timestamp: new Date().toISOString()
      });
    });


    // High-performance value set expansion
    app.post('/fhir/ValueSet/$expand', async (request, reply) => {
      const { url, filter, offset = 0, count = 1000, includeDesignations = false } = request.body;
      
      // Generate deterministic cache key
      const cacheKey = this.generateExpansionCacheKey({
        url, filter, offset, count, includeDesignations
      });
      
      // Check cache
      let expansion = await this.cache.get(cacheKey);
      
      if (!expansion) {
        // Check persisted expansions
        expansion = await this.getPersistedExpansion(url, cacheKey);
        
        if (!expansion) {
          // Perform expansion
          expansion = await this.expansionEngine.expand({
            url, filter, offset, count, includeDesignations
          });
          
          // Persist if not too large
          if (expansion.total < 10000) {
            await this.persistExpansion(expansion, cacheKey);
          }
        }
        
        // Cache result
        await this.cache.set(cacheKey, expansion);
      }
      
      return reply.send(expansion);
    });


    // SNOMED expression validation
    app.post('/api/v1/snomed/validate', async (request, reply) => {
      const { expression } = request.body;
      
      const validator = new SNOMEDExpressionValidator(this.db);
      const result = await validator.validate(expression);
      
      if (result.valid) {
        // Store normalized form
        await this.db.query(`
          INSERT INTO snomed_expressions 
          (expression_hash, expression, normal_form, focus_concepts, refinements, validation_status)
          VALUES ($1, $2, $3, $4, $5, 'valid')
          ON CONFLICT (expression_hash) DO NOTHING
        `, [
          result.normalizedHash,
          expression,
          result.normalForm,
          result.focusConcepts,
          result.refinements
        ]);
      }
      
      return reply.send(result);
    });
  }


  private async getPersistedExpansion(url: string, paramsHash: string) {
    const result = await this.db.query(`
      SELECT 
        e.id,
        e.total,
        e.generated_at,
        e.expires_at,
        json_agg(
          json_build_object(
            'system', ec.system,
            'code', ec.code,
            'display', ec.display,
            'designation', ec.designation
          ) ORDER BY ec.system, ec.code
        ) as contains
      FROM value_set_expansions e
      JOIN expansion_contains ec ON ec.expansion_id = e.id
      WHERE e.value_set_id = (SELECT id FROM value_sets WHERE url = $1)
        AND e.params_hash = $2
        AND (e.expires_at IS NULL OR e.expires_at > NOW())
      GROUP BY e.id
    `, [url, paramsHash]);
    
    return result.rows[0];
  }
}
```

### SNOMED Expression Validator


```typescript
// validators/snomed-validator.ts
export class SNOMEDExpressionValidator {
  private parser: SNOMEDParser;
  private db: Pool;


  async validate(expression: string): Promise<ValidationResult> {
    try {
      // Parse expression
      const ast = this.parser.parse(expression);
      
      // Validate all concepts exist
      const conceptValidation = await this.validateConcepts(ast.getAllConcepts());
      if (!conceptValidation.valid) {
        return conceptValidation;
      }
      
      // Validate relationships
      const relationshipValidation = await this.validateRelationships(ast);
      if (!relationshipValidation.valid) {
        return relationshipValidation;
      }
      
      // Check semantic coherence
      const semanticValidation = await this.validateSemantics(ast);
      if (!semanticValidation.valid) {
        return semanticValidation;
      }
      
      // Generate normal form
      const normalForm = this.generateNormalForm(ast);
      const hash = crypto.createHash('sha256').update(normalForm).digest('hex');
      
      return {
        valid: true,
        normalForm,
        normalizedHash: hash,
        focusConcepts: ast.focusConcepts,
        refinements: ast.refinements
      };
      
    } catch (error) {
      return {
        valid: false,
        error: error.message
      };
    }
  }


  private async validateConcepts(concepts: string[]): Promise<ValidationResult> {
    const result = await this.db.query(`
      SELECT code 
      FROM concepts 
      WHERE system = 'SNOMED' 
        AND code = ANY($1::text[])
        AND active = true
    `, [concepts]);
    
    const foundCodes = new Set(result.rows.map(r => r.code));
    const missing = concepts.filter(c => !foundCodes.has(c));
    
    if (missing.length > 0) {
      return {
        valid: false,
        error: `Invalid concepts: ${missing.join(', ')}`
      };
    }
    
    return { valid: true };
  }


  private async validateRelationships(ast: SNOMEDExpression): Promise<ValidationResult> {
    // Validate each attribute relationship is valid for the concept
    for (const refinement of ast.refinements) {
      const valid = await this.db.query(`
        SELECT 1
        FROM concept_relationships
        WHERE system_uri = 'http://snomed.info/sct'
          AND source_code = $1
          AND relationship = $2
          AND target_code = $3
      `, [refinement.concept, refinement.relationship, refinement.value]);
      
      if (valid.rows.length === 0) {
        return {
          valid: false,
          error: `Invalid relationship: ${refinement.concept} ${refinement.relationship} ${refinement.value}`
        };
      }
    }
    
    return { valid: true };
  }
}
```

```typescript
// security/license-enforcement.ts
export class LicenseEnforcement {
  private licenseCache = new Map<string, LicenseInfo>();
  
  async checkAccess(
    userId: string,
    system: string,
    operation: 'lookup' | 'export' | 'bulk',
    codeCount: number = 1
  ): Promise<AccessDecision> {
    const license = await this.getUserLicense(userId);
    const systemLicense = await this.getSystemLicense(system);
    
    // Check system-specific restrictions
    if (systemLicense.type === 'restricted') {
      if (!license.systems.includes(system)) {
        await this.audit({
          userId,
          system,
          operation,
          result: 'denied',
          reason: 'no_system_license'
        });
        
        return {
          allowed: false,
          reason: 'No license for this terminology system'
        };
      }
    }
    
    // Check operation limits
    const limits = license.limits[operation];
    if (limits) {
      const usage = await this.getUsage(userId, operation);
      
      if (usage.count + codeCount > limits.daily) {
        return {
          allowed: false,
          reason: 'Daily limit exceeded',
          limit: limits.daily,
          current: usage.count
        };
      }
    }
    
    // Check rate limits
    const rateLimited = await this.checkRateLimit(userId, license.tier);
    if (rateLimited) {
      return {
        allowed: false,
        reason: 'Rate limit exceeded',
        retryAfter: rateLimited.retryAfter
      };
    }
    
    // Audit successful access
    await this.audit({
      userId,
      system,
      operation,
      codeCount,
      result: 'allowed'
    });
    
    return { allowed: true };
  }
  
  private async checkRateLimit(userId: string, tier: string): Promise<any> {
    const limits = {
      'basic': { rps: 10, burst: 20 },
      'professional': { rps: 100, burst: 200 },
      'enterprise': { rps: 1000, burst: 2000 }
    };
    
    const limit = limits[tier];
    const key = `rate:${userId}`;
    
    // Token bucket algorithm
    const current = await redis.get(key);
    const tokens = current ? JSON.parse(current) : { count: limit.burst, lastRefill: Date.now() };
    
    // Refill tokens
    const now = Date.now();
    const elapsed = (now - tokens.lastRefill) / 1000;
    tokens.count = Math.min(limit.burst, tokens.count + (elapsed * limit.rps));
    tokens.lastRefill = now;
    
    if (tokens.count < 1) {
      const retryAfter = Math.ceil((1 - tokens.count) / limit.rps);
      return { limited: true, retryAfter };
    }
    
    tokens.count--;
    await redis.set(key, JSON.stringify(tokens), 'EX', 60);
    
    return null;
  }
}
```
