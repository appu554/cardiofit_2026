```typescript
// kb4/engine/safety_rule_engine.ts


interface RuleAST {
  op: 'AND' | 'OR' | 'NOT' | 'COMPARE' | 'IN' | 'EXISTS';
  args: (RuleAST | Value)[];
}


interface CompiledRule {
  id: string;
  type: 'absolute' | 'relative' | 'duplicate' | 'monitoring';
  priority: number;  // Lower = higher priority
  ast: RuleAST;
  compiledFn: Function;  // JIT compiled
  requiredFields: Set<string>;
  action: 'PASS' | 'WARN' | 'VETO';
  monitoring?: MonitoringRequirement[];
  evidence: EvidenceReference[];
}


export class SafetyRuleEngine {
  private ruleCache: Map<string, CompiledRule[]> = new Map();
  private jitCompiler: JITCompiler;
  private explainer: RuleExplainer;
  private validator: RuleValidator;
  
  async evaluateSafety(
    request: SafetyCheckRequest
  ): Promise<SafetyCheckResponse> {
    const startTime = process.hrtime.bigint();
    
    const results: TherapySafetyResult[] = await Promise.all(
      request.therapies.map(therapy => 
        this.evaluateTherapy(therapy, request.patient_context)
      )
    );
    
    // Generate evidence envelope
    const envelope = this.generateEvidenceEnvelope(
      results,
      request,
      process.hrtime.bigint() - startTime
    );
    
    // Emit safety event
    await this.emitSafetyEvent(request, results, envelope);
    
    return {
      transaction_id: request.transaction_id,
      results,
      evidence_envelope: envelope
    };
  }
  
  private async evaluateTherapy(
    therapy: ProposedTherapy,
    context: PatientContext
  ): Promise<TherapySafetyResult> {
    // Get compiled rules
    const rules = await this.getCompiledRules(therapy.drug_code);
    
    // Check cache
    const cacheKey = this.generateCacheKey(therapy, context, rules);
    const cached = await this.checkCache(cacheKey);
    if (cached) {
      return cached;
    }
    
    // Evaluate rules in priority order
    const findings: SafetyFinding[] = [];
    let finalStatus: SafetyStatus = 'PASS';
    
    // Phase 1: Absolute contraindications (fail-fast)
    for (const rule of rules.filter(r => r.type === 'absolute')) {
      const result = await this.evaluateRule(rule, context);
      if (result.triggered) {
        findings.push(this.createFinding(rule, result));
        finalStatus = 'VETO';
        break; // Fail fast on absolute contraindication
      }
    }
    
    // Phase 2: Duplicate therapy (if not vetoed)
    if (finalStatus !== 'VETO') {
      for (const rule of rules.filter(r => r.type === 'duplicate')) {
        const result = await this.evaluateRule(rule, context);
        if (result.triggered) {
          findings.push(this.createFinding(rule, result));
          finalStatus = 'VETO';
          break;
        }
      }
    }
    
    // Phase 3: Relative contraindications (accumulate warnings)
    if (finalStatus !== 'VETO') {
      for (const rule of rules.filter(r => r.type === 'relative')) {
        const result = await this.evaluateRule(rule, context);
        if (result.triggered) {
          findings.push(this.createFinding(rule, result));
          if (finalStatus === 'PASS') {
            finalStatus = 'WARN';
          }
        }
      }
    }
    
    // Phase 4: Monitoring requirements (always evaluate)
    const monitoringFindings = await this.evaluateMonitoring(
      rules.filter(r => r.type === 'monitoring'),
      context
    );
    findings.push(...monitoringFindings);
    
    const result: TherapySafetyResult = {
      therapy_id: therapy.therapy_id,
      drug_code: therapy.drug_code,
      status: finalStatus,
      findings,
      explanation: this.explainer.explain(findings, context)
    };
    
    // Cache result
    await this.cacheResult(cacheKey, result, rules);
    
    return result;
  }
  
  private async evaluateRule(
    rule: CompiledRule,
    context: PatientContext
  ): Promise<RuleEvaluation> {
    try {
      // Use JIT-compiled function for performance
      const triggered = rule.compiledFn(context);
      
      return {
        triggered,
        confidence: this.calculateConfidence(rule, context),
        reasoning: triggered ? 
          this.explainer.explainRule(rule, context) : null
      };
    } catch (error) {
      // Log but don't fail - safety first
      console.error(`Rule evaluation error: ${rule.id}`, error);
      
      // Fail safe - assume rule triggers on error
      return {
        triggered: true,
        confidence: 0.5,
        reasoning: `Rule evaluation failed - assuming unsafe`,
        error: error.message
      };
    }
  }
  
  private calculateConfidence(
    rule: CompiledRule,
    context: PatientContext
  ): number {
    // Calculate confidence based on data completeness
    const requiredFields = Array.from(rule.requiredFields);
    const availableFields = requiredFields.filter(field => 
      this.hasValue(context, field)
    );
    
    const completeness = availableFields.length / requiredFields.length;
    
    // Adjust for data recency
    const recencyFactor = this.calculateRecencyFactor(context, requiredFields);
    
    return completeness * recencyFactor;
  }
}
```

```typescript
// kb4/engine/rule_compiler.ts


export class RuleCompiler {
  private optimizer: RuleOptimizer;
  private sandboxer: Sandboxer;
  
  compileYAMLToBundle(yaml: string): CompiledBundle {
    // Parse YAML
    const parsed = parseYAML(yaml);
    
    // Validate schema
    this.validateSchema(parsed);
    
    // Extract rules
    const rules = parsed.rules.map(rule => {
      // Parse predicate to AST
      const ast = this.parsePredicateToAST(rule.predicate);
      
      // Optimize AST
      const optimized = this.optimizer.optimize(ast);
      
      // Extract required fields
      const requiredFields = this.extractRequiredFields(optimized);
      
      // JIT compile to function
      const compiledFn = this.compileToFunction(optimized);
      
      // Validate compiled function
      this.validateCompiledFunction(compiledFn, rule.test_cases);
      
      return {
        id: rule.id,
        type: rule.type,
        priority: this.calculatePriority(rule),
        ast: optimized,
        compiledFn,
        requiredFields,
        action: rule.action,
        monitoring: rule.monitoring,
        evidence: rule.evidence_refs
      };
    });
    
    return {
      bundle_id: `${parsed.meta.drug_class}-${parsed.meta.version}`,
      version: parsed.meta.version,
      rules,
      checksum: this.calculateChecksum(rules)
    };
  }
  
  private compileToFunction(ast: RuleAST): Function {
    // Generate JavaScript code from AST
    const jsCode = this.generateJSCode(ast);
    
    // Wrap in sandboxed function
    const sandboxed = this.sandboxer.wrap(jsCode);
    
    // JIT compile
    return new Function('context', sandboxed);
  }
  
  private generateJSCode(ast: RuleAST): string {
    switch (ast.op) {
      case 'AND':
        return `(${ast.args.map(a => this.generateJSCode(a)).join(' && ')})`;
      
      case 'OR':
        return `(${ast.args.map(a => this.generateJSCode(a)).join(' || ')})`;
      
      case 'NOT':
        return `!(${this.generateJSCode(ast.args[0])})`;
      
      case 'COMPARE':
        const [left, op, right] = ast.args;
        return `(${this.generateJSCode(left)} ${op} ${this.generateJSCode(right)})`;
      
      case 'IN':
        const [value, list] = ast.args;
        return `${this.generateJSCode(list)}.includes(${this.generateJSCode(value)})`;
      
      case 'EXISTS':
        return `(${this.generateJSCode(ast.args[0])} !== undefined)`;
      
      default:
        // Leaf node - variable or constant
        if (typeof ast === 'string' && ast.startsWith('context.')) {
          return ast;
        }
        return JSON.stringify(ast);
    }
  }
  
  private validateCompiledFunction(
    fn: Function,
    testCases: TestCase[]
  ): void {
    for (const testCase of testCases) {
      const result = fn(testCase.context);
      if (result !== testCase.expected) {
        throw new Error(
          `Compilation validation failed: ${testCase.name}\n` +
          `Expected: ${testCase.expected}, Got: ${result}`
        );
      }
    }
  }
}
```

```typescript
// kb4/signals/signal_detector.ts


export class SafetySignalDetector {
  private statisticalEngine: StatisticalEngine;
  private baselineCalculator: BaselineCalculator;
  private alerter: SignalAlerter;
  
  async detectSignals(): Promise<DetectedSignal[]> {
    const signals: DetectedSignal[] = [];
    
    // Get all active drugs
    const drugs = await this.getActiveDrugs();
    
    for (const drug of drugs) {
      // Calculate baseline rates
      const baseline = await this.baselineCalculator.calculate(drug.code, {
        window: '90 days',
        minSampleSize: 100
      });
      
      // Get recent events
      const recent = await this.getRecentEvents(drug.code, '7 days');
      
      // Statistical process control
      const spcResult = this.statisticalEngine.performSPC({
        baseline: baseline.rate,
        current: recent.rate,
        sampleSize: recent.count,
        confidenceLevel: 0.99
      });
      
      if (spcResult.isSignificant) {
        const signal: DetectedSignal = {
          signal_id: generateId(),
          drug_code: drug.code,
          signal_type: this.classifySignal(spcResult),
          baseline_rate: baseline.rate,
          current_rate: recent.rate,
          z_score: spcResult.zScore,
          p_value: spcResult.pValue,
          detection_window: '7 days',
          sample_size: recent.count,
          confidence_level: 0.99,
          status: 'detected',
          detected_at: new Date()
        };
        
        signals.push(signal);
        
        // Alert if critical
        if (signal.z_score > 3 || signal.signal_type === 'adverse_event') {
          await this.alerter.sendCriticalAlert(signal);
        }
      }
      
      // Check for override patterns
      const overridePattern = await this.detectOverridePattern(drug.code);
      if (overridePattern) {
        signals.push(overridePattern);
      }
    }
    
    return signals;
  }
  
  private async detectOverridePattern(
    drugCode: string
  ): Promise<DetectedSignal | null> {
    // Query override audit
    const overrides = await this.db.query(`
      SELECT 
        COUNT(*) as total,
        COUNT(*) FILTER (WHERE override_level = 'L3') as l3_count,
        COUNT(*) FILTER (WHERE adverse_event = true) as adverse_count
      FROM override_audit oa
      JOIN safety_alerts sa ON oa.event_id = sa.event_id
      WHERE sa.drug_code = $1
        AND oa.authorized_at > NOW() - INTERVAL '30 days'
    `, [drugCode]);
    
    const result = overrides.rows[0];
    
    // Check for concerning patterns
    if (result.l3_count > 5 || result.adverse_count > 0) {
      return {
        signal_id: generateId(),
        drug_code: drugCode,
        signal_type: 'override_pattern',
        metadata: {
          l3_overrides: result.l3_count,
          adverse_events: result.adverse_count,
          total_overrides: result.total
        },
        status: 'detected',
        priority: 'high'
      };
    }
    
    return null;
  }
  
  private classifySignal(spcResult: SPCResult): SignalType {
    if (spcResult.trend === 'increasing' && spcResult.metric === 'veto_rate') {
      return 'safety_concern';
    }
    if (spcResult.trend === 'increasing' && spcResult.metric === 'adverse_event') {
      return 'adverse_event';
    }
    if (spcResult.trend === 'increasing' && spcResult.metric === 'override_rate') {
      return 'override_pattern';
    }
    return 'anomaly';
  }
}
```

```typescript
// kb4/cache/intelligent_cache.ts


export class SafetyCache {
  private redis: Redis;
  private dependencyTracker: DependencyTracker;
  private invalidator: CacheInvalidator;
  
  async get(key: string): Promise<CachedResult | null> {
    // Check Redis first
    const cached = await this.redis.get(key);
    if (!cached) return null;
    
    const parsed = JSON.parse(cached);
    
    // Validate dependencies haven't changed
    const valid = await this.validateDependencies(parsed.dependencies);
    if (!valid) {
      await this.invalidate(key);
      return null;
    }
    
    // Check expiry
    if (new Date(parsed.expires_at) < new Date()) {
      await this.invalidate(key);
      return null;
    }
    
    return parsed.result;
  }
  
  async set(
    key: string,
    result: TherapySafetyResult,
    rules: CompiledRule[]
  ): Promise<void> {
    // Track dependencies
    const dependencies = {
      rules: rules.map(r => ({ id: r.id, version: r.version })),
      fields: this.extractUsedFields(rules),
      timestamp: new Date()
    };
    
    // Calculate TTL based on risk
    const ttl = this.calculateTTL(result.status);
    
    const cacheEntry = {
      result,
      dependencies,
      expires_at: new Date(Date.now() + ttl * 1000),
      created_at: new Date()
    };
    
    // Store in Redis
    await this.redis.setex(key, ttl, JSON.stringify(cacheEntry));
    
    // Store in PostgreSQL for longer-term analysis
    await this.storeCacheEntry(key, cacheEntry);
  }
  
  private calculateTTL(status: SafetyStatus): number {
    // Shorter TTL for higher risk outcomes
    switch (status) {
      case 'VETO':
        return 60;    // 1 minute
      case 'WARN':
        return 180;   // 3 minutes
      case 'PASS':
        return 300;   // 5 minutes
      default:
        return 60;    // Default to shortest
    }
  }
  
  async invalidatePattern(pattern: string): Promise<void> {
    // Get all keys matching pattern
    const keys = await this.redis.keys(pattern);
    
    if (keys.length > 0) {
      // Batch delete
      await this.redis.del(...keys);
      
      // Log invalidation
      await this.logInvalidation(pattern, keys.length);
    }
  }
  
  async invalidateByRule(ruleId: string): Promise<void> {
    // Find all cache entries using this rule
    const affected = await this.db.query(`
      SELECT cache_key
      FROM rule_evaluation_cache
      WHERE $1 = ANY(depends_on_rules)
        AND expires_at > NOW()
    `, [ruleId]);
    
    // Invalidate all affected entries
    for (const row of affected.rows) {
      await this.invalidate(row.cache_key);
    }
  }
  
  private async validateDependencies(deps: Dependencies): Promise<boolean> {
    // Check if any rules have been updated
    for (const rule of deps.rules) {
      const current = await this.getRuleVersion(rule.id);
      if (current !== rule.version) {
        return false;
      }
    }
    
    // Check if profile has been updated
    const profileVersion = await this.getProfileVersion(deps.drug_code);
    if (profileVersion !== deps.profile_version) {
      return false;
    }
    
    return true;
  }
}
```

```typescript
// kb4/overrides/override_manager.ts


export class OverrideManager {
  private stateMachine: OverrideStateMachine;
  private authorizer: OverrideAuthorizer;
  private notifier: OverrideNotifier;
  
  async requestOverride(
    eventId: string,
    level: OverrideLevel,
    context: OverrideContext
  ): Promise<OverrideResult> {
    // Get current state
    const currentState = await this.getCurrentState(eventId);
    
    // Validate transition
    if (!this.canTransition(currentState, level)) {
      throw new Error(`Invalid transition from ${currentState} to ${level}`);
    }
    
    switch (level) {
      case 'L1':
        return this.handleL1Override(eventId, context);
      
      case 'L2':
        return this.handleL2Override(eventId, context);
      
      case 'L3':
        return this.handleL3Override(eventId, context);
      
      default:
        throw new Error(`Unknown override level: ${level}`);
    }
  }
  
  private async handleL1Override(
    eventId: string,
    context: OverrideContext
  ): Promise<OverrideResult> {
    // L1: Simple acknowledgment
    const override = {
      override_id: generateId(),
      event_id: eventId,
      override_level: 'L1',
      acknowledged_by: context.user_id,
      acknowledged_at: new Date()
    };
    
    await this.saveOverride(override);
    
    return {
      success: true,
      override_id: override.override_id,
      level: 'L1',
      message: 'Warning acknowledged'
    };
  }
  
  private async handleL2Override(
    eventId: string,
    context: OverrideContext
  ): Promise<OverrideResult> {
    // L2: Requires justification
    if (!context.justification || context.justification.length < 50) {
      throw new Error('L2 override requires detailed justification (min 50 chars)');
    }
    
    const override = {
      override_id: generateId(),
      event_id: eventId,
      override_level: 'L2',
      previous_level: 'L1',
      justification: context.justification,
      justified_by: context.user_id,
      justified_at: new Date()
    };
    
    await this.saveOverride(override);
    
    // Notify pharmacy
    await this.notifier.notifyPharmacy(override);
    
    return {
      success: true,
      override_id: override.override_id,
      level: 'L2',
      message: 'Override recorded with justification'
    };
  }
  
  private async handleL3Override(
    eventId: string,
    context: OverrideContext
  ): Promise<OverrideResult> {
    // L3: Break glass - requires dual authorization
    
    // Verify break glass token
    if (!await this.verifyBreakGlassToken(context.break_glass_token)) {
      throw new Error('Invalid break glass token');
    }
    
    // Check dual authorization
    if (!context.secondary_authorizer) {
      throw new Error('L3 override requires secondary authorization');
    }
    
    // Verify both authorizers have appropriate privileges
    const authorized = await this.authorizer.verifyDualAuth(
      context.user_id,
      context.secondary_authorizer
    );
    
    if (!authorized) {
      throw new Error('Insufficient privileges for L3 override');
    }
    
    const override = {
      override_id: generateId(),
      event_id: eventId,
      override_level: 'L3',
      previous_level: 'L2',
      justification: context.justification,
      break_glass_token: context.break_glass_token,
      authorized_by: [context.user_id, context.secondary_authorizer],
      authorized_at: new Date()
    };
    
    await this.saveOverride(override);
    
    // Critical notifications
    await Promise.all([
      this.notifier.notifySafetyCommittee(override),
      this.notifier.notifyChiefMedicalOfficer(override),
      this.notifier.pageOnCall(override)
    ]);
    
    // Schedule follow-up
    await this.scheduleFollowUp(override);
    
    return {
      success: true,
      override_id: override.override_id,
      level: 'L3',
      message: 'CRITICAL: Absolute contraindication overridden',
      warnings: [
        'This action has been logged and reported',
        'Patient must be closely monitored',
        '30-day outcome report required'
      ]
    };
  }
  
  private async scheduleFollowUp(override: Override): Promise<void> {
    // Schedule 24h, 7d, and 30d follow-ups
    const followUpDates = [1, 7, 30].map(days =>
```

## Shadow → Canary → Production Rollout


```typescript
// deployment/rollout_orchestrator.ts


export class KB4RolloutOrchestrator {
  
  async executeRollout(version: string): Promise<RolloutResult> {
    const stages = [
      this.shadowStage,
      this.canaryStage,
      this.productionStage
    ];
    
    for (const stage of stages) {
      const result = await stage(version);
      
      if (!result.success) {
        await this.rollback(version, stage.name);
        throw new Error(`Rollout failed at ${stage.name}: ${result.error}`);
      }
      
      // Gate between stages
      await this.waitForApproval(stage.name);
    }
    
    return { success: true, version };
  }
  
  private async shadowStage(version: string): Promise<StageResult> {
    // Deploy in shadow mode - receives traffic but doesn't enforce
    await this.deploy(version, {
      mode: 'shadow',
      percentage: 100,
      duration: '2 weeks'
    });
    
    // Collect metrics
    const metrics = await this.collectShadowMetrics();
    
    // Compare with production
    const analysis = await this.analyzeShadowResults(metrics);
    
    if (analysis.discrepancyRate > 0.01) {
      return {
        success: false,
        error: `Shadow discrepancy rate ${analysis.discrepancyRate} exceeds threshold`
      };
    }
    
    return { success: true };
  }
  
  private async canaryStage(version: string): Promise<StageResult> {
    // Start with 1% traffic
    await this.deploy(version, {
      mode: 'canary',
      percentage: 1,
      duration: '24h'
    });
    
    // Progressive rollout
    const percentages = [1, 5, 10, 25, 50];
    
    for (const pct of percentages) {
      await this.updateCanary(version, pct);
      
      // Monitor for 1 hour at each level
      const healthy = await this.monitorCanary(version, '1h');
      
      if (!healthy) {
        return {
          success: false,
          error: `Canary failed at ${pct}% traffic`
        };
      }
    }
    
    return { success: true };
  }
}
```
