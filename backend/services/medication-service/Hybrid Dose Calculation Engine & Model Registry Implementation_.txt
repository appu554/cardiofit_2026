# Deep Dive: Rule Engine & Model Registry Implementation


## Part 1: Rule Engine - High-Performance Interpreter for Standard Dosing Logic


### 1.1 Architecture Overview


The Rule Engine is a high-performance, TOML-driven calculation system that processes dosing rules for ~95% of medications. It's designed for sub-millisecond latency while maintaining clinical safety and auditability.

// ============== Rule Engine Implementation ==============


impl RuleEngine {
    pub fn new() -> Self {
        Self {
            rule_cache: Arc::new(RwLock::new(HashMap::new())),
            parser: TomlParser::new(),
            executor: RuleExecutor::new(),
            validator: RuleValidator::new(),
            metrics: MetricsCollector::new(),
            feature_flags: FeatureFlags::default(),
        }
    }
    
    pub async fn calculate_dose(
        &self,
        request: DoseCalculationRequest
    ) -> Result<DoseCalculationResponse> {
        let start = std::time::Instant::now();
        
        // Step 1: Load and compile rules (with caching)
        let rules = self.load_rules(&request.drug_id).await?;
        
        // Step 2: Validate input against rule requirements
        self.validator.validate_request(&request, &rules)?;
        
        // Step 3: Execute rule chain
        let execution_context = ExecutionContext::from_request(request.clone());
        let result = self.executor.execute(&rules, execution_context)?;
        
        // Step 4: Apply safety limits
        let safe_result = self.apply_safety_limits(result, &rules.safety_limits)?;
        
        // Step 5: Generate audit trail
        let audit = self.create_audit_trail(&request, &safe_result, &rules);
        
        // Step 6: Record metrics
        self.metrics.record_calculation(
            &request.drug_id,
            start.elapsed(),
            &safe_result
        );
        
        Ok(DoseCalculationResponse {
            dose: safe_result,
            audit_trail: audit,
            warnings: self.validator.generate_warnings(&safe_result, &rules),
            calculation_time_ms: start.elapsed().as_millis() as u32,
        })
    }
    
    async fn load_rules(&self, drug_id: &str) -> Result<CompiledRuleSet> {
        // Check cache first
        {
            let cache = self.rule_cache.read().unwrap();
            if let Some(rules) = cache.get(drug_id) {
                if self.is_cache_valid(rules) {
                    self.metrics.record_cache_hit(drug_id);
                    return Ok(rules.clone());
                }
            }
        }
        
        // Load from storage and compile
        let toml_content = self.load_toml_from_storage(drug_id).await?;
        let parsed_rules = self.parser.parse(&toml_content)?;
        let compiled_rules = self.compile_rules(parsed_rules)?;
        
        // Update cache
        {
            let mut cache = self.rule_cache.write().unwrap();
            cache.insert(drug_id.to_string(), compiled_rules.clone());
        }
        
        self.metrics.record_cache_miss(drug_id);
        Ok(compiled_rules)
    }
    
    fn compile_rules(&self, parsed: ParsedToml) -> Result<CompiledRuleSet> {
        let mut adjustment_chain = Vec::new();
        
        // Compile renal adjustments
        if let Some(renal) = parsed.adjustments.renal {
            adjustment_chain.push(self.compile_renal_adjustment(renal)?);
        }
        
        // Compile hepatic adjustments  
        if let Some(hepatic) = parsed.adjustments.hepatic {
            adjustment_chain.push(self.compile_hepatic_adjustment(hepatic)?);
        }
        
        // Sort by priority for correct application order
        adjustment_chain.sort_by_key(|a| a.priority);
        
        Ok(CompiledRuleSet {
            drug_id: parsed.metadata.drug_id,
            version: parsed.metadata.version,
            indication_rules: self.compile_indication_rules(parsed.dosing)?,
            adjustment_chain,
            safety_limits: parsed.safety,
            metadata: parsed.metadata,
            compiled_at: Utc::now(),
            checksum: self.calculate_checksum(&parsed),
        })
    }
}


// ============== Rule Executor ==============


pub struct RuleExecutor {
    expression_evaluator: ExpressionEvaluator,
    performance_monitor: PerformanceMonitor,
}


impl RuleExecutor {
    pub fn execute(
        &self,
        rules: &CompiledRuleSet,
        mut context: ExecutionContext
    ) -> Result<DoseResult> {
        // Step 1: Get base dose for indication
        let indication_rule = rules.indication_rules
            .get(&context.indication)
            .ok_or_else(|| anyhow::anyhow!("No rule for indication: {}", context.indication))?;
        
        let mut dose = indication_rule.initial_dose.typical;
        context.add_step(format!("Base dose for {}: {} mg", context.indication, dose));
        
        // Step 2: Apply adjustment chain
        for adjustment in &rules.adjustment_chain {
            if self.should_apply_adjustment(adjustment, &context)? {
                let old_dose = dose;
                dose = self.apply_adjustment(dose, adjustment, &context)?;
                
                context.add_step(format!(
                    "Applied {} adjustment: {} mg -> {} mg (factor: {:.2})",
                    adjustment.name, old_dose, dose, dose / old_dose
                ));
            }
        }
        
        // Step 3: Apply frequency modifications
        let frequency = self.determine_frequency(
            &indication_rule.frequency,
            &context
        )?;
        
        // Step 4: Check for titration requirements
        let titration = if context.is_initial_dose {
            indication_rule.titration.clone()
        } else {
            None
        };
        
        Ok(DoseResult {
            amount_mg: dose,
            frequency,
            route: indication_rule.route.clone(),
            titration_schedule: titration,
            adjustments_applied: context.steps.clone(),
            confidence_score: self.calculate_confidence(&context),
        })
    }
    
    fn should_apply_adjustment(
        &self,
        adjustment: &AdjustmentRule,
        context: &ExecutionContext
    ) -> Result<bool> {
        for condition in &adjustment.conditions {
            if !self.evaluate_condition(condition, context)? {
                return Ok(false);
            }
        }
        Ok(true)
    }
    
    fn evaluate_condition(
        &self,
        condition: &Condition,
        context: &ExecutionContext
    ) -> Result<bool> {
        match condition {
            Condition::Between { field, min, max } => {
                let value = context.get_numeric_field(field)?;
                Ok(value >= *min && value <= *max)
            },
            Condition::GreaterThan { field, value } => {
                let field_value = context.get_numeric_field(field)?;
                Ok(field_value > *value)
            },
            Condition::LessThan { field, value } => {
                let field_value = context.get_numeric_field(field)?;
                Ok(field_value < *value)
            },
            Condition::Equals { field, value } => {
                let field_value = context.get_string_field(field)?;
                Ok(field_value == *value)
            },
            Condition::In { field, values } => {
                let field_value = context.get_string_field(field)?;
                Ok(values.contains(&field_value))
            },
            Condition::And(conditions) => {
                for cond in conditions {
                    if !self.evaluate_condition(cond, context)? {
                        return Ok(false);
                    }
                }
                Ok(true)
            },
            Condition::Or(conditions) => {
                for cond in conditions {
                    if self.evaluate_condition(cond, context)? {
                        return Ok(true);
                    }
                }
                Ok(false)
            },
        }
    }
    
    fn apply_adjustment(
        &self,
        current_dose: f64,
        adjustment: &AdjustmentRule,
        context: &ExecutionContext
    ) -> Result<f64> {
        match &adjustment.factor {
            AdjustmentFactor::Multiply(factor) => Ok(current_dose * factor),
            AdjustmentFactor::Add(amount) => Ok(current_dose + amount),
            AdjustmentFactor::Replace(new_dose) => Ok(*new_dose),
            AdjustmentFactor::Table(entries) => {
                for entry in entries {
                    if self.evaluate_condition(&entry.condition, context)? {
                        return Ok(current_dose * entry.factor);
                    }
                }
                Ok(current_dose) // No matching entry, keep current
            },
            AdjustmentFactor::Formula(formula) => {
                self.expression_evaluator.evaluate(formula, current_dose, context)
            },
        }
    }
}


// ============== TOML Parser with Validation ==============


pub struct TomlParser {
    schema_validator: SchemaValidator,
}


impl TomlParser {
    pub fn parse(&self, content: &str) -> Result<ParsedToml> {
        // Parse TOML
        let value: toml::Value = toml::from_str(content)
            .context("Failed to parse TOML")?;
        
        // Validate against schema
        self.schema_validator.validate(&value)?;
        
        // Deserialize into structured format
        let parsed: ParsedToml = toml::from_str(content)
            .context("Failed to deserialize TOML")?;
        
        // Validate clinical logic
        self.validate_clinical_logic(&parsed)?;
        
        Ok(parsed)
    }
    
    fn validate_clinical_logic(&self, parsed: &ParsedToml) -> Result<()> {
        // Ensure min <= typical <= max for all dose ranges
        for (_, rule) in &parsed.dosing {
            if rule.initial_dose.min > rule.initial_dose.typical ||
               rule.initial_dose.typical > rule.initial_dose.max {
                bail!("Invalid dose range for {}", parsed.metadata.drug_id);
            }
        }
        
        // Ensure adjustment factors are reasonable (0.1 to 10x)
        for adj in &parsed.adjustments.all() {
            if let AdjustmentFactor::Multiply(factor) = adj.factor {
                if factor < 0.1 || factor > 10.0 {
                    bail!("Adjustment factor {} out of reasonable range", factor);
                }
            }
        }
        
        Ok(())
    }
}


// ============== Performance Optimizations ==============


pub struct PerformanceOptimizer {
    // Pre-compiled expressions for faster evaluation
    compiled_expressions: HashMap<String, CompiledExpression>,
    // Memoization cache for repeated calculations
    memoization_cache: Arc<RwLock<LruCache<u64, f64>>>,
}


impl PerformanceOptimizer {
    pub fn optimize_rule_set(&mut self, rules: &mut CompiledRuleSet) {
        // Pre-compile all formula expressions
        for adjustment in &mut rules.adjustment_chain {
            if let AdjustmentFactor::Formula(formula) = &adjustment.factor {
                let compiled = self.compile_expression(formula);
                self.compiled_expressions.insert(
                    format!("{}_{}", rules.drug_id, adjustment.name),
                    compiled
                );
            }
        }
        
        // Sort adjustments by dependency order
        self.optimize_adjustment_order(&mut rules.adjustment_chain);
        
        // Pre-calculate common paths
        self.pre_calculate_common_scenarios(rules);
    }
    
    fn optimize_adjustment_order(&self, adjustments: &mut Vec<AdjustmentRule>) {
        // Ensure independent adjustments are applied first
        // This allows for better parallelization
        adjustments.sort_by(|a, b| {
            let a_deps = self.count_dependencies(a);
            let b_deps = self.count_dependencies(b);
            a_deps.cmp(&b_deps).then(a.priority.cmp(&b.priority))
        });
    }
}

        // Check for contraindications
        self.check_contraindications(request, rules)?;
        
        // Validate drug interactions
        self.validate_interactions(request, rules)?;
        
        Ok(())
    }
    
    fn validate_value_ranges(&self, request: &DoseCalculationRequest) -> Result<()> {
        // Age validation
        if request.patient.age < 0.0 || request.patient.age > 150.0 {
            bail!("Invalid age: {}", request.patient.age);
        }
        
        // Weight validation
        if request.patient.weight < 0.5 || request.patient.weight > 500.0 {
            bail!("Invalid weight: {}", request.patient.weight);
        }
        
        // Lab value validation
        if let Some(egfr) = request.labs.egfr {
            if egfr < 0.0 || egfr > 200.0 {
                bail!("Invalid eGFR: {}", egfr);
            }
        }
        
        Ok(())
    }
}

// ============== Complex Adjustment Patterns ==============


pub struct AdvancedAdjustments {
    pub conditional_chains: Vec<ConditionalChain>,
    pub interaction_matrix: InteractionMatrix,
    pub population_models: HashMap<String, PopulationModel>,
}


#[derive(Debug, Clone)]
pub struct ConditionalChain {
    pub name: String,
    pub conditions: Vec<ChainCondition>,
    pub adjustments: Vec<ChainedAdjustment>,
}


#[derive(Debug, Clone)]
pub struct ChainCondition {
    pub if_condition: Condition,
    pub then_apply: AdjustmentFactor,
    pub else_apply: Option<AdjustmentFactor>,
}


impl AdvancedAdjustments {
    pub fn apply_complex_adjustments(
        &self,
        base_dose: f64,
        context: &ExecutionContext
    ) -> Result<f64> {
        let mut dose = base_dose;
        
        // Apply conditional chains
        for chain in &self.conditional_chains {
            dose = self.apply_conditional_chain(dose, chain, context)?;
        }
        
        // Apply interaction adjustments
        dose = self.apply_interaction_adjustments(dose, context)?;
        
        // Apply population-specific models
        if let Some(population) = self.identify_population(context) {
            dose = self.apply_population_model(dose, population, context)?;
        }
        
        Ok(dose)
    }
    
    fn apply_conditional_chain(
        &self,
        dose: f64,
        chain: &ConditionalChain,
        context: &ExecutionContext
    ) -> Result<f64> {
        let mut current_dose = dose;
        
        for condition in &chain.conditions {
            let applies = self.evaluate_condition(&condition.if_condition, context)?;
            
            if applies {
                current_dose = self.apply_factor(current_dose, &condition.then_apply)?;
            } else if let Some(else_factor) = &condition.else_apply {
                current_dose = self.apply_factor(current_dose, else_factor)?;
            }
        }
        
        Ok(current_dose)
    }
}


// ============== Parallel Rule Processing ==============


use rayon::prelude::*;


pub struct ParallelRuleEngine {
    thread_pool: rayon::ThreadPool,
    rule_partitioner: RulePartitioner,
}


impl ParallelRuleEngine {
    pub fn execute_parallel(
        &self,
        rules: &CompiledRuleSet,
        context: ExecutionContext
    ) -> Result<DoseResult> {
        // Partition independent adjustments for parallel execution
        let (independent, dependent) = self.rule_partitioner.partition(&rules.adjustment_chain);
        
        // Execute independent adjustments in parallel
        let parallel_results: Vec<f64> = independent
            .par_iter()
            .map(|adjustment| {
                self.calculate_adjustment_factor(adjustment, &context)
            })
            .collect::<Result<Vec<_>>>()?;
        
        // Combine parallel results
        let combined_factor: f64 = parallel_results.iter().product();
        
        // Apply dependent adjustments sequentially
        let mut dose = rules.indication_rules
            .get(&context.indication)
            .map(|r| r.initial_dose.typical)
            .unwrap_or(0.0) * combined_factor;
        
        for adjustment in dependent {
            dose = self.apply_adjustment_sequential(dose, adjustment, &context)?;
        }
        
        Ok(DoseResult {
            amount_mg: dose,
            ..Default::default()
        })
    }
}

        }
        
        // Store in registry
        let mut models = self.models.write().unwrap();
        models.insert(model_id.clone(), model.clone());
        
        // Update version tracking
        self.versioning.register_version(model.model_id(), model.version());
        
        log::info!("Registered model: {} v{}", model.model_id(), model.version());
    }
    
    pub async fn calculate(
        &self,
        model_id: &str,
        version: Option<&str>,
        input: ModelInput
    ) -> Result<ModelOutput> {
        let start = std::time::Instant::now();
        
        // Get the appropriate model version
        let model = self.get_model(model_id, version)?;
        
        // Validate inputs
        model.validate_inputs(&input)?;
        
        // Check performance requirements
        self.check_resources(&model.performance_profile())?;
        
        // Execute in sandbox for safety
        let output = self.sandbox.execute_model(&*model, input).await?;
        
        // Validate output
        self.validator.validate_output(&output, &model.precision_requirements())?;
        
        // Record metrics
        self.performance_monitor.record_execution(
            model_id,
            start.elapsed(),
            &output
        );
        
        Ok(output)
    }
    
    fn get_model(
        &self,
        model_id: &str,
        version: Option<&str>
    ) -> Result<Arc<dyn CompiledModel>> {
        let models = self.models.read().unwrap();
        
        let key = if let Some(v) = version {
            format!("{}_{}", model_id, v)
        } else {
            // Get latest version
            let latest = self.versioning.get_latest(model_id)?;
            format!("{}_{}", model_id, latest)
        };
        
        models.get(&key)
            .cloned()
            .ok_or_else(|| anyhow::anyhow!("Model not found: {}", key))
    }
}

impl VancomycinAUCModel {
    fn calculate_auc_targeted_dose(
        &self,
        pk_params: &PKParameters,
        target_auc: f64,
        input: &ModelInput
    ) -> Result<DoseRecommendation> {
        // Implement iterative AUC calculation
        let mut dose = self.estimate_initial_dose(pk_params, target_auc);
        let mut iterations = 0;
        const MAX_ITERATIONS: u32 = 100;
        const TOLERANCE: f64 = 0.01;
        
        loop {
            iterations += 1;
            if iterations > MAX_ITERATIONS {
                bail!("Failed to converge after {} iterations", MAX_ITERATIONS);
            }
            
            // Calculate predicted AUC
            let predicted_auc = self.predict_auc(dose, pk_params)?;
            
            // Check convergence
            let error = (predicted_auc - target_auc).abs() / target_auc;
            if error < TOLERANCE {
                break;
            }
            
            // Adjust dose using Newton-Raphson method
            let gradient = self.calculate_gradient(dose, pk_params)?;
            dose = dose - (predicted_auc - target_auc) / gradient;
            
            // Apply safety constraints
            dose = dose.max(self.safety_limits.min_dose)
                       .min(self.safety_limits.max_dose);
        }
        
        Ok(DoseRecommendation {
            dose_mg: dose,
            frequency: self.determine_frequency(input.clinical.creatinine_clearance),
            loading_dose_mg: self.calculate_loading_dose(pk_params, target_auc),
            infusion_duration_hours: self.calculate_infusion_time(dose),
        })
    }
    
    fn predict_auc(&self, dose: f64, pk: &PKParameters) -> Result<f64> {
        // Implement trapezoidal AUC calculation
        let ke = pk.elimination_constant;
        let vd = pk.volume_distribution;
        let tau = 12.0; // Dosing interval in hours
        
        // Calculate peak concentration
        let cmax = dose / vd;
        
        // Calculate trough concentration
        let cmin = cmax * (-ke * tau).exp();
        
        // Calculate AUC using log-trapezoidal method
        let auc = (dose / pk.clearance) * 24.0 / tau;
        
        Ok(auc)
    }
}

impl CarboplatinCalvertModel {
    fn calculate_gfr(&self, patient: &PatientData, clinical: &ClinicalData) -> Result<f64> {
        let scr = clinical.serum_creatinine.unwrap();
        
        // Cockcroft-Gault formula
        let gfr = match patient.sex {
            Sex::Male => {
                ((140.0 - patient.age) * patient.weight) / (72.0 * scr)
            },
            Sex::Female => {
                0.85 * ((140.0 - patient.age) * patient.weight) / (72.0 * scr)
            },
        };
        
        // Cap GFR at 125 mL/min for dosing purposes
        Ok(gfr.min(125.0))
    }
    
    fn apply_safety_caps(&self, dose: f64, gfr: f64) -> f64 {
        // Apply maximum dose caps based on GFR
        let max_dose = if gfr > 125.0 {
            800.0  // Maximum dose cap
        } else if gfr > 60.0 {
            750.0
        } else {
            600.0
        };
        
        dose.min(max_dose)
    }
}


// ============== Model Sandbox for Safe Execution ==============


pub struct ModelSandbox {
    resource_limiter: ResourceLimiter,
    timeout_ms: u64,
}


impl ModelSandbox {
    pub async fn execute_model(
        &self,
        model: &dyn CompiledModel,
        input: ModelInput
    ) -> Result<ModelOutput> {
        // Create isolated execution context
        let execution_task = tokio::spawn(async move {
            // Set resource limits
            self.resource_limiter.apply_limits()?;
            
            // Execute with timeout
            tokio::time::timeout(
                Duration::from_millis(self.timeout_ms),
                model.calculate(input)
            ).await
            .map_err(|_| anyhow::anyhow!("Model execution timeout"))?
        });
        
        // Monitor execution
        let monitor_task = tokio::spawn(async move {
            self.monitor_execution(execution_task.id()).await
        });
        
        // Wait for completion
        let result = execution_task.await??;
        monitor_task.abort();
        
        Ok(result)
    }
    
    async fn monitor_execution(&self, task_id: TaskId) -> Result<()> {
        let mut interval = tokio::time::interval(Duration::from_millis(100));
        
        loop {
            interval.tick().await;
            
            // Check memory usage
            if self.get_memory_usage(task_id) > self.resource_limiter.max_memory_mb {
                return Err(anyhow::anyhow!("Memory limit exceeded"));
            }
            
            // Check CPU usage
            if self.get_cpu_usage(task_id) > self.resource_limiter.max_cpu_percent {
                log::warn!("High CPU usage in model execution");
            }
        }
    }
}

pub struct ModelHotLoader {

    registry: Arc<RwLock<ModelRegistry>>,

impl ModelHotLoader {
    pub async fn watch_for_updates(&mut self) -> Result<()> {
        let mut rx = self.watcher.watch("./models").await?;
        
        while let Some(event) = rx.recv().await {
            match event {
                FileEvent::Modified(path) => {
                    self.handle_model_update(path).await?;
                },
                FileEvent::Created(path) => {
                    self.handle_new_model(path).await?;
                },
                _ => {}
            }
        }
        
        Ok(())
    }
    
    async fn handle_model_update(&mut self, path: PathBuf) -> Result<()> {
        // Compile the updated model
        let compiled = self.compiler.compile(&path).await?;
        
        // Validate the new version
        let validator = ModelValidator::new();
        validator.validate_model(&*compiled)?;
        
        // Hot-swap in registry
        let mut registry = self.registry.write().unwrap();
        registry.hot_swap(compiled)?;
        
        log::info!("Hot-loaded updated model from {:?}", path);
        Ok(())
    }
}
