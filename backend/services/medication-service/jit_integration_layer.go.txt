// ==================== JIT Safety Service Client ====================

// JITSafetyClient provides interface to the Rust JIT Safety Engine
type JITSafetyClient struct {
    baseURL    string
    httpClient *retryablehttp.Client
    logger     *zap.Logger
    timeout    time.Duration
}

// NewJITSafetyClient creates a new JIT safety client
func NewJITSafetyClient(baseURL string, logger *zap.Logger) *JITSafetyClient {
    retryClient := retryablehttp.NewClient()
    retryClient.RetryMax = 3
    retryClient.RetryWaitMin = 100 * time.Millisecond
    retryClient.RetryWaitMax = 1 * time.Second
    retryClient.Logger = nil // Use our own logger
    
    return &JITSafetyClient{
        baseURL:    baseURL,
        httpClient: retryClient,
        logger:     logger,
        timeout:    5 * time.Second,
    }
}

// ==================== Request/Response Models ====================

// RunJITSafetyCheck performs JIT safety verification for a selected candidate
func (c *JITSafetyClient) RunJITSafetyCheck(
    ctx context.Context,
    candidate recommendation.CandidateProposal,
    patientContext recommendation.PatientContext,
    proposedDoseMg float64,
) (*JITSafetyResponse, error) {
    // Convert to DTOs
    request := c.buildRequest(candidate, patientContext, proposedDoseMg)
    
    // Marshal request
    reqBody, err := json.Marshal(request)
    if err != nil {
        c.logger.Error("Failed to marshal JIT request", zap.Error(err))
        return nil, fmt.Errorf("marshal request: %w", err)
    }
    
    // Create HTTP request
    httpReq, err := retryablehttp.NewRequestWithContext(
        ctx,
        "POST",
        fmt.Sprintf("%s/api/v1/safety/jit-check", c.baseURL),
        bytes.NewReader(reqBody),
    )
    if err != nil {
        return nil, fmt.Errorf("create request: %w", err)
    }
    
    httpReq.Header.Set("Content-Type", "application/json")
    httpReq.Header.Set("X-Request-ID", patientContext.RequestID)
    
    // Add timeout
    ctx, cancel := context.WithTimeout(ctx, c.timeout)
    defer cancel()
    
    // Execute request
    resp, err := c.httpClient.Do(httpReq.WithContext(ctx))
    if err != nil {
        c.logger.Error("JIT safety check request failed", 
            zap.Error(err),
            zap.String("drug_id", candidate.Drug.ID),
        )
        return nil, fmt.Errorf("execute request: %w", err)
    }
    defer resp.Body.Close()
    
    // Read response
    respBody, err := io.ReadAll(resp.Body)
    if err != nil {
        return nil, fmt.Errorf("read response: %w", err)
    }
    
    // Check status code
    if resp.StatusCode != http.StatusOK {
        c.logger.Error("JIT safety check returned error",
            zap.Int("status_code", resp.StatusCode),
            zap.String("response", string(respBody)),
        )
        return nil, fmt.Errorf("JIT check failed with status %d: %s", resp.StatusCode, string(respBody))
    }
    
    // Parse response
    var jitResp JITSafetyResponse
    if err := json.Unmarshal(respBody, &jitResp); err != nil {
        return nil, fmt.Errorf("unmarshal response: %w", err)
    }
    
    c.logger.Info("JIT safety check completed",
        zap.String("drug_id", candidate.Drug.ID),
        zap.String("action", jitResp.Action.Type),
        zap.Float64("safety_score", jitResp.Score.Overall),
        zap.Int("findings", len(jitResp.Findings)),
    )
    
    return &jitResp, nil
}

func (c *JITSafetyClient) buildRequest(
    candidate recommendation.CandidateProposal,
    patientContext recommendation.PatientContext,
    proposedDoseMg float64,
) JITSafetyRequest {
    // Build drug candidate DTO
    drugCandidate := DrugCandidateDTO{
        DrugID:           candidate.Drug.ID,
        Name:             candidate.Drug.Name,
        GenericName:      candidate.Drug.GenericName,
        ProposedDoseMg:   proposedDoseMg,
        ProposedDoseUnit: "mg",
        Route:            "Oral", // Would come from drug metadata
        Frequency: FrequencyDTO{
            TimesPerDay: 1, // Would be calculated based on drug
        },
        Provenance: candidate.Provenance,
    }
    
    // Build patient context DTO
    patientContextDTO := PatientContextDTO{
        PatientID:    patientContext.PatientID,
        AgeYears:     int(patientContext.Age),
        Sex:          c.mapBiologicalSex(patientContext.Sex),
        WeightKg:     patientContext.Weight,
        HeightCm:     175.0, // Would come from patient data
        PregnancyStatus: PregnancyStatusDTO{
            Status: c.mapPregnancyStatus(patientContext.IsPregnant),
        },
        Breastfeeding: patientContext.IsBreastfeeding,
        Labs:          c.buildLabResults(patientContext),
        Conditions:    c.extractConditions(patientContext),
        ActiveMedications: c.buildActiveMedications(patientContext.ActiveMeds),
        Allergies:     c.buildAllergies(patientContext.Allergies),
        KBVersions:    patientContext.LocalKBVersions,
        Timestamp:     patientContext.Timestamp,
    }
    
    // Check for recent procedures
    if procedures := c.extractRecentProcedures(patientContext); len(procedures) > 0 {
        patientContextDTO.RecentProcedures = procedures
    }
    
    return JITSafetyRequest{
        Candidate:      drugCandidate,
        PatientContext: patientContextDTO,
        RequestID:      patientContext.RequestID,
    }
}

func (c *JITSafetyClient) mapBiologicalSex(sex string) string {
    // Map from patient context to DTO format
    switch sex {
    case "M", "Male":
        return "Male"
    case "F", "Female":
        return "Female"
    default:
        return "Other"
    }
}

func (c *JITSafetyClient) mapPregnancyStatus(isPregnant bool) string {
    if isPregnant {
        return "pregnant"
    }
    return "not_pregnant"
}

func (c *JITSafetyClient) buildLabResults(pc recommendation.PatientContext) LabResultsDTO {
    labs := LabResultsDTO{}
    
    if pc.eGFR > 0 {
        labs.EgfrMlMin = &LabValueDTO{
            Value:     pc.eGFR,
            Timestamp: time.Now().Add(-24 * time.Hour), // Would come from actual lab timestamp
            Unit:      "mL/min/1.73m2",
        }
    }
    
    // Map other labs from patient context
    // This would be expanded based on actual patient data structure
    
    return labs
}

func (c *JITSafetyClient) extractConditions(pc recommendation.PatientContext) []string {
    conditions := []string{}
    
    // Extract ICD-10 codes from patient flags
    for flag, value := range pc.PatientFlags {
        if value {
            // Map flags to ICD-10 codes
            switch flag {
            case "DIABETES":
                conditions = append(conditions, "E11")
            case "HYPERTENSION":
                conditions = append(conditions, "I10")
            case "HEART_FAILURE":
                conditions = append(conditions, "I50")
            case "CHRONIC_KIDNEY_DISEASE":
                conditions = append(conditions, "N18")
            }
        }
    }
    
    return conditions
}

func (c *JITSafetyClient) buildActiveMedications(meds []recommendation.ActiveMedication) []ActiveMedicationDTO {
    activeMeds := make([]ActiveMedicationDTO, 0, len(meds))
    
    for _, med := range meds {
        activeMeds = append(activeMeds, ActiveMedicationDTO{
            DrugID:      med.DrugID,
            Name:        med.Name,
            GenericName: med.Name, // Would parse generic from name
            DoseMg:      100.0, // Would come from parsed dose
            Route:       "Oral",
            Frequency: FrequencyDTO{
                TimesPerDay: med.FrequencyPerDay,
            },
            StartDate:        med.StartDate,
            TherapeuticClass: []string{med.Class},
            MechanismOfAction: []string{}, // Would be populated from KB
        })
    }
    
    return activeMeds
}

func (c *JITSafetyClient) buildAllergies(allergies map[string]bool) []AllergyDTO {
    allergyList := []AllergyDTO{}
    
    for allergen, hasAllergy := range allergies {
        if hasAllergy {
            allergyList = append(allergyList, AllergyDTO{
                Allergen:     allergen,
                ReactionType: "Unknown", // Would come from patient data
                Severity:     "Moderate", // Would come from patient data
            })
        }
    }
    
    return allergyList
}

func (c *JITSafetyClient) extractRecentProcedures(pc recommendation.PatientContext) []ProcedureDTO {
    procedures := []ProcedureDTO{}
    
    // Check for recent surgery flag
    for flag, value := range pc.PatientFlags {
        if value && strings.Contains(flag, "RECENT_SURGERY") {
            procedures = append(procedures, ProcedureDTO{
                Name:        "Surgery",
                Code:        "00000", // Would be actual CPT code
                Date:        time.Now().Add(-3 * 24 * time.Hour),
                RequiresNPO: true,
            })
        }
    }
    
    return procedures
}

// ==================== Safety Action Interpreter ====================

// InterpretSafetyAction converts JIT safety response to actionable recommendation
func InterpretSafetyAction(
    response *JITSafetyResponse,
    candidate recommendation.CandidateProposal,
) recommendation.SafetyVerifiedProposal {
    proposal := recommendation.SafetyVerifiedProposal{
        Candidate:       candidate,
        SafetyScore:     response.Score.Overall,
        CanProceed:      false,
        RequiresReview:  false,
        Monitoring:      []string{},
        Warnings:        []string{},
        Contraindicated: false,
    }
    
    // Interpret action type
    switch response.Action.Type {
    case "Proceed":
        proposal.CanProceed = true
        
    case "ProceedWithMonitoring":
        proposal.CanProceed = true
        proposal.Monitoring = response.Action.Parameters
        
    case "AdjustDose":
        proposal.CanProceed = true
        proposal.RequiresReview = true
        if response.Action.RecommendedDoseMg != nil {
            proposal.AdjustedDoseMg = response.Action.RecommendedDoseMg
        }
        
    case "HoldForClinician":
        proposal.RequiresReview = true
        proposal.ReviewUrgency = response.Action.Urgency
        
    case "AbortAndSwitch":
        proposal.Contraindicated = true
        proposal.AlternativeIDs = response.Action.AlternativeDrugIDs
        
    case "RequireSpecialistReview":
        proposal.RequiresReview = true
        proposal.SpecialistType = response.Action.Specialty
    }
    
    // Extract warnings from findings
    for _, finding := range response.Findings {
        if finding.Severity == "Warning" || finding.Severity == "Major" {
            proposal.Warnings = append(proposal.Warnings, finding.Message)
        }
    }
    
    // Extract recommendations
    for _, rec := range response.Recommendations {
        if rec.Priority == "Critical" || rec.Priority == "High" {
            proposal.CriticalActions = append(proposal.CriticalActions, rec.Description)
        }
    }
    
    return proposal
}

// SafetyVerifiedProposal represents a candidate that has passed JIT safety checks
type SafetyVerifiedProposal struct {
    Candidate         recommendation.CandidateProposal
    SafetyScore       float64
    CanProceed        bool
    RequiresReview    bool
    ReviewUrgency     *string
    SpecialistType    *string
    Monitoring        []string
    Warnings          []string
    CriticalActions   []string
    Contraindicated   bool
    AlternativeIDs    []string
    AdjustedDoseMg    *float64
}
