    pub async fn generate_schedule(
        &self,
        request: TitrationRequest
    ) -> Result<TitrationSchedule> {
        // Step 1: Select appropriate strategy
        let strategy = self.select_strategy(&request)?;
        
        // Step 2: Generate base schedule
        let mut schedule = strategy.generate_base_schedule(&request)?;
        
        // Step 3: Personalize based on patient factors
        self.personalize_schedule(&mut schedule, &request.patient)?;
        
        // Step 4: Apply safety constraints
        self.safety_checker.apply_safety_constraints(&mut schedule, &request)?;
        
        // Step 5: Optimize for adherence
        self.optimize_for_adherence(&mut schedule, &request.patient)?;
        
        // Step 6: Add monitoring plan
        schedule.monitoring_plan = self.generate_monitoring_plan(&schedule, &request)?;
        
        // Step 7: Create decision points
        schedule.decision_points = self.create_decision_points(&schedule)?;
        
        // Step 8: Validate complete schedule
        self.validate_schedule(&schedule)?;
        
        Ok(schedule)
    }
    
    pub async fn evaluate_progression(
        &self,
        schedule: &TitrationSchedule,
        current_data: PatientData
    ) -> Result<ProgressionDecision> {
        let current_phase = self.get_current_phase(schedule)?;
        
        // Check if minimum duration met
        if !self.minimum_duration_met(schedule, &current_phase)? {
            return Ok(ProgressionDecision::ContinueCurrentPhase);
        }
        
        // Evaluate progression criteria
        let criteria_met = self.progression_analyzer.evaluate_criteria(
            &current_phase.progression_criteria,
            &current_data
        )?;
        
        // Check safety parameters
        let safety_check = self.safety_checker.check_progression_safety(
            schedule,
            &current_data
        )?;
        
        // Generate recommendation
        let decision = match (criteria_met, safety_check) {
            (true, SafetyStatus::Safe) => {
                ProgressionDecision::ProgressToNextPhase
            },
            (true, SafetyStatus::Caution) => {
                ProgressionDecision::RequireClinicalReview {
                    reason: "Safety parameters require review".to_string(),
                    suggested_action: SuggestedAction::SlowTitration,
                }
            },
            (false, _) => {
                if self.can_extend_phase(&current_phase)? {
                    ProgressionDecision::ExtendCurrentPhase {
                        additional_days: 7,
                        reason: "Criteria not yet met".to_string(),
                    }
                } else {
                    ProgressionDecision::RequireClinicalReview {
                        reason: "Maximum phase duration reached".to_string(),
                        suggested_action: SuggestedAction::ConsiderAlternative,
                    }
                }
            },
            (_, SafetyStatus::Unsafe) => {
                ProgressionDecision::HaltTitration {
                    reason: safety_check.get_reason(),
                    recommended_action: "Reduce dose or discontinue".to_string(),
                }
            },
        };
        
        // Log decision
        self.log_progression_decision(schedule, &decision, &current_data).await?;
        
        Ok(decision)
    }
    
    fn personalize_schedule(
        &self,
        schedule: &mut TitrationSchedule,
        patient: &PatientData
    ) -> Result<()> {
        // Adjust for age
        if patient.age > 75.0 {
            self.apply_geriatric_modifications(schedule)?;
        } else if patient.age < 18.0 {
            self.apply_pediatric_modifications(schedule)?;
        }
        
        // Adjust for renal function
        if patient.egfr < 30.0 {
            self.apply_renal_modifications(schedule)?;
        }
        
        // Adjust for hepatic function
        if patient.hepatic_impairment.is_some() {
            self.apply_hepatic_modifications(schedule)?;
        }
        
        // Adjust for polypharmacy
        if patient.concurrent_medications.len() > 5 {
            self.apply_polypharmacy_modifications(schedule)?;
        }
        
        Ok(())
    }
    
    fn apply_geriatric_modifications(&self, schedule: &mut TitrationSchedule) -> Result<()> {
        for phase in &mut schedule.phases {
            // Slower titration for elderly
            phase.duration_days = (phase.duration_days as f64 * 1.5) as u32;
            
            // Lower dose increments
            if phase.phase_number > 1 {
                phase.dose_mg *= 0.75;
            }
            
            // More frequent monitoring
            phase.monitoring_requirements.push(MonitoringRequirement {
                parameter: "orthostatic_vitals".to_string(),
                frequency_days: 3,
                critical_action: "Hold if SBP drop > 20mmHg".to_string(),
            });
        }
        Ok(())
    }
}

impl SmartTitrationOptimizer {
    pub async fn optimize_schedule(
        &self,
        base_schedule: TitrationSchedule,
        patient_profile: &PatientProfile
    ) -> Result<OptimizedSchedule> {
        // Predict optimal titration speed
        let predicted_tolerance = self.ml_predictor.predict_tolerance(
            &base_schedule,
            patient_profile
        ).await?;
        
        // Analyze adherence patterns
        let adherence_risk = self.adherence_analyzer.assess_risk(
            patient_profile
        )?;
        
        // Adjust schedule based on predictions
        let mut optimized = base_schedule.clone();
        
        if predicted_tolerance.score > 0.8 && adherence_risk.is_low() {
            // Fast-track titration for well-tolerating, adherent patients
            self.apply_fast_track(&mut optimized)?;
        } else if predicted_tolerance.score < 0.4 || adherence_risk.is_high() {
            // Conservative approach for at-risk patients
            self.apply_conservative(&mut optimized)?;
        }
        
        // Add personalized monitoring
        optimized.monitoring_plan = self.generate_personalized_monitoring(
            &optimized,
            patient_profile,
            &predicted_tolerance
        )?;
        
        Ok(OptimizedSchedule {
            schedule: optimized,
            confidence_score: predicted_tolerance.confidence,
            expected_time_to_target: self.calculate_time_to_target(&optimized),
            risk_factors: self.identify_risk_factors(patient_profile),
        })
    }
    
    fn apply_fast_track(&self, schedule: &mut TitrationSchedule) -> Result<()> {
        // Reduce phase durations by 30%
        for phase in &mut schedule.phases {
            phase.duration_days = (phase.duration_days as f64 * 0.7) as u32;
            phase.can_skip = phase.phase_number > 1; // Allow skipping intermediate phases
        }
        
        // Add rapid titration flag
        schedule.safety_parameters.allow_rapid_titration = true;
        
        Ok(())
    }
    
    fn apply_conservative(&self, schedule: &mut TitrationSchedule) -> Result<()> {
        // Increase phase durations by 50%
        for phase in &mut schedule.phases {
            phase.duration_days = (phase.duration_days as f64 * 1.5) as u32;
            
            // Add intermediate doses
            if phase.phase_number > 1 {
                phase.dose_mg *= 0.75;
            }
            
            // Increase monitoring frequency
            for req in &mut phase.monitoring_requirements {
                req.frequency_days = (req.frequency_days as f64 * 0.5) as u32;
            }
        }
        
        Ok(())
    }

impl ClinicalDecisionSupport {
    pub fn evaluate_titration_decision(
        &self,
        schedule: &TitrationSchedule,
        patient_data: &PatientData,
        clinical_context: &ClinicalContext
    ) -> Result<TitrationDecision> {
        let mut decision = TitrationDecision::default();
        
        // Evaluate clinical rules
        let rule_results = self.rule_engine.evaluate_all(
            schedule,
            patient_data,
            clinical_context
        )?;
        
        // Generate alerts if needed
        for result in &rule_results {
            if result.severity == Severity::High {
                decision.alerts.push(self.alert_system.create_alert(result)?);
            }
        }
        
        // Generate recommendations
        decision.recommendations = self.recommendation_engine.generate(
            schedule,
            patient_data,
            &rule_results
        )?;
        
        // Determine action
        decision.recommended_action = self.determine_action(&rule_results)?;
        
        // Add supporting evidence
        decision.evidence = self.gather_evidence(&rule_results);
        
        Ok(decision)
    }
    
    fn determine_action(&self, results: &[RuleResult]) -> Result<RecommendedAction> {
        let high_risk_count = results.iter()
            .filter(|r| r.severity == Severity::High)
            .count();
        
        let medium_risk_count = results.iter()
            .filter(|r| r.severity == Severity::Medium)
            .count();
        
        if high_risk_count > 0 {
            Ok(RecommendedAction::HaltTitration)
        } else if medium_risk_count > 2 {
            Ok(RecommendedAction::SlowTitration)
        } else {
            Ok(RecommendedAction::ContinueAsPlanned)
        }
    }
}

        // Step 2: Analyze drug-drug interactions

        )?;
        
        // Step 3: Calculate temporal risk factors
        let temporal_risk = self.temporal_analyzer.analyze_temporal_patterns(

        )?;
        
        // Step 4: Apply population risk models
        let population_risk = self.population_risk.calculate_risk(
            patient,
            medications,
            clinical_context
        )?;
        
        // Step 5: Synthesize cumulative risk
        let cumulative_score = self.synthesize_risk_score(
            &individual_risks,
            &interaction_risk,
            &temporal_risk,
            &population_risk
        )?;
        
        // Step 6: Generate risk mitigation strategies
        let mitigation_strategies = self.generate_mitigation_strategies(
            &cumulative_score,
            medications,
            patient
        )?;
        
        Ok(CumulativeRiskProfile {
            overall_risk_score: cumulative_score,
            risk_categories: self.categorize_risks(&cumulative_score),
            individual_medication_risks: individual_risks,
            interaction_risks: interaction_risk,
            temporal_risks: temporal_risk,
            population_based_risk: population_risk,
            mitigation_strategies,
            confidence_interval: self.calculate_confidence_interval(&cumulative_score),
            recommendations: self.generate_recommendations(&cumulative_score),
        })
    }
    
    fn synthesize_risk_score(
        &self,
        individual: &[MedicationRisk],
        interactions: &InteractionRiskProfile,
        temporal: &TemporalRiskProfile,
        population: &PopulationRiskScore
    ) -> Result<CumulativeRiskScore> {
        // Weighted combination of risk factors
        let weights = RiskWeights {
            individual: 0.3,
            interaction: 0.3,
            temporal: 0.2,
            population: 0.2,
        };
        
        let weighted_score = 
            individual.iter().map(|r| r.score).sum::<f64>() * weights.individual +
            interactions.overall_score * weights.interaction +
            temporal.risk_score * weights.temporal +
            population.percentile_risk * weights.population;
        
        // Apply non-linear scaling for multiple high risks
        let adjusted_score = self.apply_risk_amplification(
            weighted_score,
            individual,
            interactions
        )?;
        
        Ok(CumulativeRiskScore {
            raw_score: weighted_score,
            adjusted_score,
            risk_level: self.determine_risk_level(adjusted_score),
            contributing_factors: self.identify_top_contributors(
                individual,
                interactions,
                temporal,
                population
            ),
        })
    }

impl RiskAwareTitrationEngine {
    pub async fn generate_risk_adjusted_schedule(
        &self,
        request: TitrationRequest,
        all_medications: Vec<Medication>
    ) -> Result<RiskAdjustedTitrationSchedule> {
        // Generate base titration schedule
        let base_schedule = self.titration_engine.generate_schedule(request.clone()).await?;
        
        // Assess cumulative risk including the new medication
        let mut medications_with_new = all_medications.clone();
        medications_with_new.push(Medication {
            drug_id: request.drug_id.clone(),
            dose: base_schedule.phases[0].dose_mg,
            frequency: base_schedule.phases[0].frequency.clone(),
        });
        
        let risk_profile = self.risk_assessment.assess_cumulative_risk(
            &request.patient,
            &medications_with_new,
            &request.clinical_context
        ).await?;
        
        // Adjust schedule based on risk
        let adjusted_schedule = self.adjust_schedule_for_risk(
            base_schedule,
            &risk_profile
        )?;
        
        Ok(RiskAdjustedTitrationSchedule {
            schedule: adjusted_schedule,
            risk_profile,
            safety_checkpoints: self.create_safety_checkpoints(&adjusted_schedule, &risk_profile),
            monitoring_intensification: self.determine_monitoring_intensity(&risk_profile),
        })
    }
    
    fn adjust_schedule_for_risk(
        &self,
        mut schedule: TitrationSchedule,
        risk: &CumulativeRiskProfile
    ) -> Result<TitrationSchedule> {
        match risk.overall_risk_score.risk_level {
            RiskLevel::Low => {
                // No adjustment needed
            },
            RiskLevel::Medium => {
                // Extend phase durations by 25%
                for phase in &mut schedule.phases {
                    phase.duration_days = (phase.duration_days as f64 * 1.25) as u32;
                }
            },
            RiskLevel::High => {
                // Conservative titration
                for phase in &mut schedule.phases {
                    phase.duration_days *= 2;
                    phase.dose_mg *= 0.75;
                    phase.monitoring_requirements.push(MonitoringRequirement {
                        parameter: "comprehensive_metabolic_panel".to_string(),
                        frequency_days: 7,
                        critical_action: "Review all values before progression".to_string(),
                    });
                }
            },
            RiskLevel::VeryHigh => {
                // Require specialist consultation
                schedule.phases[0].progression_criteria.require_clinician_review = true;
                for phase in &mut schedule.phases {
                    phase.progression_criteria.require_clinician_review = true;
                }
            },
        }
        
        Ok(schedule)
    }

                clinical_context: request.clinical.clone(),
                preferences: TitrationPreferences::default(),
            };
            
            let all_medications = self.get_patient_medications(&request.patient.id).await?;
            
            let risk_adjusted_schedule = titration_engine
                .generate_risk_adjusted_schedule(titration_request, all_medications)
                .await?;
            
            response.titration_schedule = Some(risk_adjusted_schedule.schedule);
            response.cumulative_risk = Some(risk_adjusted_schedule.risk_profile);
        }
        
        // Assess cumulative risk if requested
        if assess_cumulative_risk && !include_titration {
            let risk_engine = CumulativeRiskAssessment::new();
            let all_medications = self.get_patient_medications(&request.patient.id).await?;
            
            let risk_profile = risk_engine.assess_cumulative_risk(
                &request.patient,
                &all_medications,
                &request.clinical
            ).await?;
            
            response.cumulative_risk = Some(risk_profile);
        }
        
        // Generate integrated recommendations
        response.integrated_recommendations = self.generate_integrated_recommendations(
            &response.primary_dose,
            &response.titration_schedule,
            &response.cumulative_risk
        )?;
        
        Ok(response)
    }
}
