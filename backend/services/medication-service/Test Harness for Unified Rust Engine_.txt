        let request = DoseCalculationRequest {
            drug_id: "rivaroxaban".to_string(), // Adding another anticoagulant!
            indication: "dvt_prophylaxis".to_string(),
            patient,
            clinical: ClinicalContext {
                platelet_count: Some(180000),
                hemoglobin: Some(10.5),
                history_of_bleeding: false,
                ..Default::default()
            },
            constraints: DosingConstraints::default(),
            include_titration: false,
            assess_cumulative_risk: true,
        };
        
        let response = ctx.engine
            .calculate_with_titration_and_risk(request, false, true)
            .await;
        
        // Should likely fail or have extreme warnings
        if response.is_ok() {
            let resp = response.unwrap();
            
            assert!(resp.cumulative_risk.is_some());
            let risk = resp.cumulative_risk.unwrap();
            
            assert!(matches!(risk.overall_risk_score.risk_level, 
                RiskLevel::VeryHigh | RiskLevel::Unacceptable));
            assert!(risk.bleeding_risk_score > 0.8, "Very high bleeding risk");
            
            // Should recommend against
            assert!(resp.integrated_recommendations.iter()
                .any(|r| r.recommendation_type == RecommendationType::AvoidCombination));
            assert!(resp.integrated_recommendations.iter()
                .any(|r| r.message.contains("bleeding")));
            
            // Multiple interaction warnings
            assert!(risk.interaction_risks.significant_interactions.len() >= 3);
            assert!(risk.interaction_risks.overall_score > 0.9);
        } else {
            // May refuse to calculate due to extreme risk
            let error = response.unwrap_err();
            assert!(error.to_string().contains("UNACCEPTABLE_RISK") ||
                   error.to_string().contains("CONTRAINDICATED"));
        }
    }
    
    // ==================== SCENARIO 10: Performance & Load Testing ====================
    
    #[tokio::test]
    async fn test_scenario_10_performance_bulk_calculations() {
        let ctx = TestContext::new();
        use std::time::Instant;
        
        // Generate 1000 diverse requests
        let mut requests = Vec::new();
        for i in 0..1000 {
            let patient_type = match i % 5 {
                0 => PatientProfile::HealthyAdult,
                1 => PatientProfile::ElderlyWithCKD,
                2 => PatientProfile::PediatricComplex,
                3 => PatientProfile::PregnantSecondTrimester,
                _ => PatientProfile::HealthyAdult,
            };
            
            let drug = match i % 10 {
                0 => "lisinopril",
                1 => "metformin",
                2 => "metoprolol",
                3 => "amoxicillin",
                4 => "warfarin",
                5 => "vancomycin",  // Model-based
                6 => "gentamicin",   // Model-based
                7 => "amlodipine",
                8 => "simvastatin",
                _ => "omeprazole",
            };
            
            requests.push(DoseCalculationRequest {
                drug_id: drug.to_string(),
                indication: "test_indication".to_string(),
                patient: ctx.create_patient(patient_type),
                clinical: ClinicalContext::default(),
                constraints: DosingConstraints::default(),
            });
        }
        
        // Measure performance
        let start = Instant::now();
        let mut results = Vec::new();
        
        for request in requests {
            let result = ctx.engine.calculate(request).await;
            results.push(result);
        }
        
        let elapsed = start.elapsed();
        
        // Performance assertions
        assert!(results.iter().filter(|r| r.is_ok()).count() > 990, 
            "At least 99% success rate");
        
        let avg_time_ms = elapsed.as_millis() / 1000;
        assert!(avg_time_ms < 50, "Average time per calculation < 50ms");
        
        // Analyze response times
        let response_times: Vec<u32> = results.iter()
            .filter_map(|r| r.as_ref().ok())
            .map(|r| r.calculation_time_ms)
            .collect();
        
        let p50 = percentile(&response_times, 50.0);
        let p95 = percentile(&response_times, 95.0);
        let p99 = percentile(&response_times, 99.0);
        
        println!("Performance Metrics:");
        println!("  P50: {}ms", p50);
        println!("  P95: {}ms", p95);
        println!("  P99: {}ms", p99);
        
        assert!(p50 < 5, "P50 latency < 5ms for rule-based");
        assert!(p95 < 100, "P95 latency < 100ms including models");
        assert!(p99 < 200, "P99 latency < 200ms");
    }

        runner.generate_report()
    }
    
    async fn run_test<F, Fut>(&mut self, name: &str, test_fn: F)
    where
        F: FnOnce() -> Fut,
        Fut: Future<Output = ()>,
    {
        print!("Running {:<50}", format!("{}...", name));
        let start = Instant::now();
        
        let result = panic::catch_unwind(AssertUnwindSafe(|| {
            let runtime = tokio::runtime::Runtime::new().unwrap();
            runtime.block_on(test_fn())
        }));
        
        let duration = start.elapsed();
        
        match result {
            Ok(_) => {
                println!("✅ PASSED ({:.2}ms)", duration.as_millis());
                self.results.push(TestResult::Pass { name: name.to_string(), duration });
            },
            Err(e) => {
                println!("❌ FAILED");
                let error = if let Some(s) = e.downcast_ref::<String>() {
                    s.clone()
                } else if let Some(s) = e.downcast_ref::<&str>() {
                    s.to_string()
                } else {
                    "Unknown error".to_string()
                };
                self.results.push(TestResult::Fail { 
                    name: name.to_string(), 
                    error,
                    duration 
                });
            }
        }
    }
    
    fn generate_report(self) -> TestReport {
        let total_duration = self.start_time.elapsed();
        let passed = self.results.iter().filter(|r| matches!(r, TestResult::Pass { .. })).count();
        let failed = self.results.iter().filter(|r| matches!(r, TestResult::Fail { .. })).count();
        
        println!("\n" + &"=".repeat(80));
        println!("📊 TEST RESULTS SUMMARY\n");
        println!("Total Tests: {}", self.results.len());
        println!("✅ Passed: {} ({:.1}%)", passed, (passed as f64 / self.results.len() as f64) * 100.0);
        println!("❌ Failed: {} ({:.1}%)", failed, (failed as f64 / self.results.len() as f64) * 100.0);
        println!("⏱️  Total Time: {:.2}s", total_duration.as_secs_f64());
        
        if failed > 0 {
            println!("\n❌ FAILED TESTS:");
            for result in &self.results {
                if let TestResult::Fail { name, error, .. } = result {
                    println!("  - {}: {}", name, error);
                }
            }
        }
        
        println!("\n" + &"=".repeat(80));
        
        TestReport {
            total_tests: self.results.len(),
            passed,
            failed,
            duration: total_duration,
            results: self.results,
            success: failed == 0,
        }
    }
}
