# Medication Microservice: Implementation Guidelines & Technical Roadmap

## Executive Summary

This document provides comprehensive implementation guidelines for building the Medication Microservice as the **Domain Expert for Pharmaceutical Intelligence**. The implementation follows a 12-week phased approach, focusing on the Calculate → Validate → Commit pattern with clear separation between business logic and safety validation.

# Dose calculation service implementation
class DoseCalculationService:
    def __init__(self):
        self.calculation_strategies = {
            DosingType.WEIGHT_BASED: WeightBasedCalculator(),
            DosingType.BSA_BASED: BSACalculator(),
            DosingType.AUC_BASED: AUCCalculator(),
            DosingType.FIXED: FixedDoseCalculator(),
            DosingType.TIERED: TieredDoseCalculator()
        }
    
    async def calculate_dose(
        self,
        medication: Medication,
        patient_context: PatientContext,
        indication: str
    ) -> DoseProposal:
        """
        Calculate appropriate dose based on patient factors
        NO safety validation - just mathematical calculations
        """
        # Determine dosing strategy
        dosing_type = self._determine_dosing_type(medication, indication)
        calculator = self.calculation_strategies[dosing_type]
        
        # Perform calculation
        calculated_dose = await calculator.calculate(
            medication=medication,
            weight_kg=patient_context.weight_kg,
            height_cm=patient_context.height_cm,
            age_years=patient_context.age_years,
            indication=indication
        )
        
        # Apply rounding rules
        rounded_dose = self._apply_rounding_rules(
            calculated_dose,
            medication.rounding_rules
        )
        
        # Create proposal
        return DoseProposal(
            medication=medication,
            calculated_dose=calculated_dose,
            rounded_dose=rounded_dose,
            calculation_method=dosing_type,
            calculation_factors={
                "weight_kg": patient_context.weight_kg,
                "bsa_m2": calculator.bsa if hasattr(calculator, 'bsa') else None,
                "age_years": patient_context.age_years
            }
        )

class WeightBasedCalculator:
    async def calculate(self, medication, weight_kg, **kwargs):
        """Weight-based dosing calculations"""
        dose_per_kg = medication.dosing_info.dose_per_kg
        
        # Check for ideal body weight adjustment
        if medication.dosing_info.use_ideal_body_weight:
            weight_kg = self._calculate_ideal_body_weight(
                weight_kg, 
                kwargs.get('height_cm'),
                kwargs.get('gender')
            )
        
        # Calculate base dose
        calculated_dose = dose_per_kg * weight_kg
        
        # Apply maximum dose cap if exists
        if medication.dosing_info.max_dose:
            calculated_dose = min(calculated_dose, medication.dosing_info.max_dose)
        
        return calculated_dose

class BSACalculator:
    async def calculate(self, medication, weight_kg, height_cm, **kwargs):
        """BSA-based dosing for chemotherapy and other drugs"""
        # Calculate BSA using preferred formula
        self.bsa = self._calculate_bsa(weight_kg, height_cm, 
                                       medication.bsa_formula)
        
        # Apply dose per m²
        dose_per_m2 = medication.dosing_info.dose_per_m2
        calculated_dose = dose_per_m2 * self.bsa
        
        # Apply BSA cap if specified (e.g., cap at 2.0 m²)
        if medication.dosing_info.bsa_cap:
            capped_bsa = min(self.bsa, medication.dosing_info.bsa_cap)
            calculated_dose = dose_per_m2 * capped_bsa
        
        return calculated_dose
    
    def _calculate_bsa(self, weight_kg, height_cm, formula="mosteller"):
        """Calculate BSA using specified formula"""
        if formula == "mosteller":
            return math.sqrt((height_cm * weight_kg) / 3600)
        elif formula == "dubois":
            return 0.007184 * (weight_kg ** 0.425) * (height_cm ** 0.725)
        # Add other formulas as needed

#### 2.2 Renal & Hepatic Adjustments

class RenalDoseAdjustmentService:
    async def adjust_for_renal_function(
        self,
        base_dose: float,
        medication: Medication,
        renal_function: RenalFunction
    ) -> AdjustedDose:
        """
        Adjust dose based on renal function
        Uses CrCl or eGFR as appropriate for the medication
        """
        if not medication.requires_renal_adjustment:
            return AdjustedDose(dose=base_dose, adjusted=False)
        
        # Determine which measure to use
        if medication.renal_adjustment.use_crcl:
            clearance = renal_function.creatinine_clearance
        else:
            clearance = renal_function.egfr
        
        # Find appropriate adjustment tier
        adjustment_factor = self._get_adjustment_factor(
            clearance,
            medication.renal_adjustment.tiers
        )
        
        adjusted_dose = base_dose * adjustment_factor
        
        return AdjustedDose(
            dose=adjusted_dose,
            adjusted=True,
            adjustment_reason=f"Renal adjustment for CrCl {clearance}",
            adjustment_factor=adjustment_factor
        )

class FormularyManagementService:
    def __init__(self, formulary_repo, insurance_client):
        self.formulary_repo = formulary_repo
        self.insurance_client = insurance_client
        self.cache = FormularyCache()
    
    async def check_formulary_status(
        self,
        medication: Medication,
        insurance_plan_id: str,
        pharmacy_id: Optional[str] = None
    ) -> FormularyStatus:
        """
        Check medication formulary status across multiple formularies
        """
        # Check cache first
        cache_key = f"{medication.rxnorm_code}:{insurance_plan_id}"
        cached_status = await self.cache.get(cache_key)
        if cached_status:
            return cached_status
        
        # Check system formulary
        system_status = await self.formulary_repo.get_system_status(
            medication.rxnorm_code
        )
        
        # Check insurance formulary
        insurance_status = await self.insurance_client.check_coverage(
            medication=medication,
            plan_id=insurance_plan_id
        )
        
        # Combine statuses
        formulary_status = FormularyStatus(
            medication=medication,
            system_tier=system_status.tier,
            insurance_tier=insurance_status.tier,
            requires_prior_auth=insurance_status.requires_pa,
            quantity_limits=insurance_status.quantity_limits,
            step_therapy_required=insurance_status.step_therapy,
            estimated_copay=insurance_status.copay,
            preferred_alternatives=await self._get_alternatives(
                medication, insurance_plan_id
            )
        )
        
        # Cache result
        await self.cache.set(cache_key, formulary_status, ttl=3600)
        
        return formulary_status
    
    async def find_therapeutic_alternatives(
        self,
        medication: Medication,
        reason: AlternativeReason,
        patient_context: PatientContext
    ) -> List[TherapeuticAlternative]:
        """
        Find therapeutic alternatives for a medication
        """
        alternatives = []
        
        # Get medications in same therapeutic class
        same_class_meds = await self.formulary_repo.find_by_therapeutic_class(
            medication.therapeutic_class
        )
        
        for alt_med in same_class_meds:
            if alt_med.medication_id == medication.medication_id:
                continue
            
            # Check if alternative is appropriate
            if reason == AlternativeReason.COST:
                if alt_med.average_cost >= medication.average_cost:
                    continue
            elif reason == AlternativeReason.FORMULARY:
                status = await self.check_formulary_status(
                    alt_med,
                    patient_context.insurance_plan_id
                )
                if status.insurance_tier > 2:  # Not preferred
                    continue
            
            # Calculate equivalent dose
            equivalent_dose = await self._calculate_equivalent_dose(
                medication, alt_med, patient_context
            )
            
            alternatives.append(TherapeuticAlternative(
                medication=alt_med,
                equivalent_dose=equivalent_dose,
                conversion_ratio=equivalent_dose.ratio,
                evidence_level=self._get_evidence_level(medication, alt_med),
                considerations=self._get_switching_considerations(
                    medication, alt_med
                )
            ))
        
        # Sort by preference (formulary tier, then cost)
        return sorted(alternatives, key=lambda x: (
            x.medication.formulary_tier,
            x.medication.average_cost
        ))

class MedicationSearchService:
    def __init__(self, search_client: ElasticsearchClient):
        self.search_client = search_client
        self.search_strategies = {
            SearchType.NAME: NameSearchStrategy(),
            SearchType.INDICATION: IndicationSearchStrategy(),
            SearchType.CLASS: TherapeuticClassSearchStrategy()
        }
    
    async def search_medications(
        self,
        query: str,
        search_type: SearchType = SearchType.NAME,
        filters: Optional[SearchFilters] = None
    ) -> SearchResults:
        """
        Intelligent medication search with fuzzy matching and ranking
        """
        # Clean and prepare query
        processed_query = self._preprocess_query(query)
        
        # Get search strategy
        strategy = self.search_strategies[search_type]
        
        # Build Elasticsearch query
        es_query = strategy.build_query(processed_query, filters)
        
        # Add formulary boosting
        if filters and filters.prefer_formulary:
            es_query = self._add_formulary_boost(es_query)
        
        # Execute search
        results = await self.search_client.search(
            index="medications",
            body=es_query,
            size=filters.limit if filters else 20
        )
        
        # Process results
        medications = []
        for hit in results['hits']['hits']:
            med = self._map_to_medication(hit['_source'])
            med.search_score = hit['_score']
            med.search_highlights = hit.get('highlight', {})
            medications.append(med)
        
        return SearchResults(
            medications=medications,
            total_count=results['hits']['total']['value'],
            query=query,
            filters=filters
        )
    
    def _preprocess_query(self, query: str) -> str:
        """Preprocess query for better matching"""
        # Handle common abbreviations
        query = self._expand_abbreviations(query)
        
        # Handle phonetic similarities
        query = self._add_phonetic_variants(query)
        
        # Handle brand/generic synonyms
        query = self._expand_synonyms(query)
        
        return query

class NameSearchStrategy:
    def build_query(self, query: str, filters: Optional[SearchFilters]):
        """Build Elasticsearch query for name-based search"""
        return {
            "bool": {
                "should": [
                    # Exact match on brand name
                    {"match": {"brand_name": {"query": query, "boost": 3}}},
                    # Exact match on generic name
                    {"match": {"generic_name": {"query": query, "boost": 3}}},
                    # Fuzzy match on brand name
                    {"match": {"brand_name": {
                        "query": query,
                        "fuzziness": "AUTO",
                        "boost": 1.5
                    }}},
                    # Fuzzy match on generic name
                    {"match": {"generic_name": {
                        "query": query,
                        "fuzziness": "AUTO",
                        "boost": 1.5
                    }}},
                    # Phonetic match
                    {"match": {"brand_name.phonetic": {"query": query}}},
                    {"match": {"generic_name.phonetic": {"query": query}}}
                ],
                "minimum_should_match": 1,
                "filter": self._build_filters(filters)
            }
        }

class MedicationProposalService:
    def __init__(
        self,
        dose_calc_service: DoseCalculationService,
        formulary_service: FormularyManagementService,
        context_client: MedicationContextClient
    ):
        self.dose_calc_service = dose_calc_service
        self.formulary_service = formulary_service
        self.context_client = context_client
    
    async def propose_medication(
        self,
        request: MedicationProposalRequest
    ) -> MedicationProposal:
        """
        Generate medication proposal with all business logic
        NO safety validation performed here
        """
        # Fetch business context for calculations
        dosing_context = await self.context_client.get_dosing_context(
            patient_id=request.patient_id,
            context_type=self._determine_context_type(request)
        )
        
        # Find medication
        medication = await self.medication_repo.find_by_id(
            request.medication_id
        )
        
        # Calculate dose
        dose_proposal = await self.dose_calc_service.calculate_dose(
            medication=medication,
            patient_context=dosing_context,
            indication=request.indication
        )
        
        # Check formulary
        formulary_status = await self.formulary_service.check_formulary_status(
            medication=medication,
            insurance_plan_id=dosing_context.insurance_plan
        )
        
        # Select appropriate formulation
        formulation = self._select_formulation(
            medication=medication,
            route=request.route,
            dose=dose_proposal.rounded_dose,
            patient_preferences=dosing_context.preferences
        )
        
        # Calculate quantity and day supply
        quantity_calc = self._calculate_quantity(
            dose=dose_proposal.rounded_dose,
            frequency=request.frequency,
            duration=request.duration,
            formulation=formulation
        )
        
        # Build proposal
        proposal = MedicationProposal(
            proposal_id=uuid.uuid4(),
            patient_id=request.patient_id,
            medication=medication,
            dose=dose_proposal.rounded_dose,
            dose_unit=formulation.dose_unit,
            route=formulation.route,
            frequency=request.frequency,
            duration=request.duration,
            quantity=quantity_calc.quantity,
            refills=request.refills,
            indication=request.indication,
            formulary_status=formulary_status,
            calculation_details=dose_proposal.calculation_factors,
            proposed_at=datetime.utcnow(),
            proposal_context={
                "calculation_method": dose_proposal.calculation_method,
                "formulary_tier": formulary_status.insurance_tier,
                "estimated_copay": formulary_status.estimated_copay
            }
        )
        
        # Store proposal for later commit
        await self.proposal_repo.save(proposal)
        
        return proposal

class ProtocolManagementService:
    def __init__(self):
        self.protocol_repo = ProtocolRepository()
        self.cycle_calculator = CycleCalculator()
        self.dose_modifier = DoseModificationEngine()
    
    async def initiate_protocol(
        self,
        protocol_id: str,
        patient_id: str,
        start_date: date
    ) -> ProtocolEnrollment:
        """
        Initiate a clinical protocol for a patient
        """
        # Load protocol definition
        protocol = await self.protocol_repo.get_by_id(protocol_id)
        
        # Create enrollment
        enrollment = ProtocolEnrollment(
            enrollment_id=uuid.uuid4(),
            patient_id=patient_id,
            protocol=protocol,
            start_date=start_date,
            current_cycle=1,
            current_day=1,
            status=ProtocolStatus.ACTIVE,
            dose_modifications=[]
        )
        
        # Calculate first cycle schedule
        first_cycle = await self.cycle_calculator.calculate_cycle(
            protocol=protocol,
            cycle_number=1,
            start_date=start_date
        )
        
        enrollment.scheduled_medications = first_cycle.medications
        
        # Save enrollment
        await self.enrollment_repo.save(enrollment)
        
        return enrollment
    
    async def calculate_protocol_medications(
        self,
        enrollment: ProtocolEnrollment,
        cycle_day: int
    ) -> List[ProtocolMedication]:
        """
        Calculate medications due for a specific protocol day
        """
        medications = []
        
        # Get medications scheduled for this day
        day_meds = [
            med for med in enrollment.scheduled_medications
            if cycle_day in med.administration_days
        ]
        
        for scheduled_med in day_meds:
            # Get base medication
            medication = await self.medication_repo.get_by_id(
                scheduled_med.medication_id
            )
            
            # Calculate base dose
            base_dose = await self._calculate_protocol_dose(
                medication=medication,
                protocol_dosing=scheduled_med.dosing,
                patient_context=await self._get_patient_context(
                    enrollment.patient_id
                )
            )
            
            # Apply any dose modifications
            if enrollment.dose_modifications:
                modified_dose = await self.dose_modifier.apply_modifications(
                    base_dose=base_dose,
                    modifications=enrollment.dose_modifications,
                    medication=medication
                )
            else:
                modified_dose = base_dose
            
            medications.append(ProtocolMedication(
                medication=medication,
                dose=modified_dose,
                route=scheduled_med.route,
                infusion_time=scheduled_med.infusion_time,
                pre_medications=scheduled_med.pre_medications,
                administration_notes=scheduled_med.notes
            ))
        
        return medications

class DoseModificationEngine:
    async def apply_modifications(
        self,
        base_dose: float,
        modifications: List[DoseModification],
        medication: Medication
    ) -> float:
        """
        Apply protocol dose modifications based on toxicity or other factors
        """
        current_dose = base_dose
        
        for modification in modifications:
            if modification.applies_to_medication(medication):
                if modification.modification_type == ModificationType.PERCENTAGE:
                    current_dose = current_dose * (modification.percentage / 100)
                elif modification.modification_type == ModificationType.ABSOLUTE:
                    current_dose = modification.absolute_dose
                elif modification.modification_type == ModificationType.HOLD:
                    return 0  # Medication held
        
        return current_dose
```

class OutboxEventPublisher:
    def __init__(self, db_session, kafka_producer):
        self.db_session = db_session
        self.kafka_producer = kafka_producer
        self.running = False
    
    async def publish_event(
        self,
        aggregate_id: UUID,
        aggregate_type: str,
        event_type: str,
        event_data: dict
    ):
        """
        Save event to outbox table (same transaction as business logic)
        """
        outbox_event = OutboxEvent(
            event_id=uuid.uuid4(),
            aggregate_id=aggregate_id,
            aggregate_type=aggregate_type,
            event_type=event_type,
            event_data=event_data,
            created_at=datetime.utcnow()
        )
        
        self.db_session.add(outbox_event)
        # Note: commit happens in the service layer with business logic
    
    async def start_publisher(self):
        """
        Background process to publish events from outbox to Kafka
        """
        self.running = True
        while self.running:
            try:
                # Fetch unpublished events
                unpublished = await self._fetch_unpublished_events(limit=100)
                
                for event in unpublished:
                    # Publish to Kafka
                    await self.kafka_producer.send(
                        topic=self._get_topic_for_event(event.event_type),
                        key=str(event.aggregate_id),
                        value=json.dumps({
                            "eventId": str(event.event_id),
                            "aggregateId": str(event.aggregate_id),
                            "aggregateType": event.aggregate_type,
                            "eventType": event.event_type,
                            "eventData": event.event_data,
                            "timestamp": event.created_at.isoformat()
                        })
                    )
                    
                    # Mark as published
                    event.published_at = datetime.utcnow()
                    await self.db_session.commit()
                
                # Sleep if no events
                if not unpublished:
                    await asyncio.sleep(1)
                    
            except Exception as e:
                logger.error(f"Outbox publisher error: {e}")
                await asyncio.sleep(5)

class MedicationReconciliationService:
    def __init__(self):
        self.matcher = MedicationMatcher()
        self.conflict_resolver = ConflictResolver()
    
    async def reconcile_medications(
        self,
        home_medications: List[Medication],
        admission_medications: List[Medication],
        reconciliation_context: ReconciliationContext
    ) -> ReconciliationResult:
        """
        Perform medication reconciliation between home and admission meds
        """
        result = ReconciliationResult()
        
        # Match medications
        matches = await self.matcher.match_medications(
            home_medications,
            admission_medications
        )
        
        # Process matches
        for match in matches:
            if match.is_exact_match:
                result.continued_medications.append(
                    ContinuedMedication(
                        medication=match.medication,
                        source="home",
                        no_changes=True
                    )
                )
            elif match.is_therapeutic_match:
                # Therapeutic substitution
                result.substitutions.append(
                    TherapeuticSubstitution(
                        home_medication=match.home_medication,
                        admission_medication=match.admission_medication,
                        reason=match.substitution_reason
                    )
                )
            elif match.has_dose_change:
                # Dose modification
                result.modifications.append(
                    DoseModification(
                        medication=match.medication,
                        home_dose=match.home_dose,
                        admission_dose=match.admission_dose,
                        reason=match.modification_reason
                    )
                )
        
        # Find unmatched medications
        result.discontinued = self._find_discontinued(
            home_medications, matches
        )
        result.new_medications = self._find_new(
            admission_medications, matches
        )
        
        # Resolve conflicts
        if result.has_conflicts():
            result.conflicts = await self.conflict_resolver.resolve(
                result.get_conflicts(),
                reconciliation_context
            )
        
        return result

class MedicationMatcher:
    async def match_medications(
        self,
        list1: List[Medication],
        list2: List[Medication]
    ) -> List[MedicationMatch]:
        """
        Match medications between two lists using various strategies
        """
        matches = []
        
        for med1 in list1:
            best_match = None
            best_score = 0
            
            for med2 in list2:
                score = await self._calculate_match_score(med1, med2)
                
                if score > best_score:
                    best_score = score
                    best_match = med2
            
            if best_score > 0.8:  # High confidence match
                matches.append(MedicationMatch(
                    medication1=med1,
                    medication2=best_match,
                    match_score=best_score,
                    match_type=self._determine_match_type(
                        med1, best_match, best_score
                    )
                ))
        
        return matches
    
    async def _calculate_match_score(
        self,
        med1: Medication,
        med2: Medication
    ) -> float:
        """Calculate similarity score between two medications"""
        score = 0.0
        
        # Exact RxNorm match
        if med1.rxnorm_code == med2.rxnorm_code:
            return 1.0
        
        # Generic name match
        if med1.generic_name.lower() == med2.generic_name.lower():
            score += 0.6
        
        # Therapeutic class match
        if med1.therapeutic_class == med2.therapeutic_class:
            score += 0.3
        
        # Similar dose form
        if self._similar_dose_form(med1.dose_form, med2.dose_form):
            score += 0.1
        
        return score
```

class MedicationCacheManager:
    def __init__(self, redis_client):
        self.redis = redis_client
        self.cache_ttls = {
            'medication': 3600,      # 1 hour
            'formulary': 1800,       # 30 minutes
            'calculation': 300,      # 5 minutes
            'protocol': 7200,        # 2 hours
            'search': 600           # 10 minutes
        }
    
    async def get_or_calculate(
        self,
        key: str,
        calculator: Callable,
        ttl_type: str = 'calculation'
    ):
        """
        Get from cache or calculate if missing
        """
        # Try cache first
        cached = await self.redis.get(key)
        if cached:
            return json.loads(cached)
        
        # Calculate if not in cache
        result = await calculator()
        
        # Cache result
        ttl = self.cache_ttls.get(ttl_type, 300)
        await self.redis.set(
            key,
            json.dumps(result, default=str),
            ex=ttl
        )
        
        return result
    
    async def invalidate_medication_cache(self, medication_id: str):
        """Invalidate all caches related to a medication"""
        patterns = [
            f"medication:{medication_id}:*",
            f"formulary:*:{medication_id}",
            f"search:*"  # Invalidate search results
        ]
        
        for pattern in patterns:
            keys = await self.redis.keys(pattern)
            if keys:
                await self.redis.delete(*keys)
